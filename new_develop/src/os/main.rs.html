35fn clear_bss() {
36    unsafe extern "C" {
37        fn sbss();
38        fn ebss();
39    }
40    (sbss as usize..ebss as usize).for_each(|a| unsafe {
41        (a as *mut u8).write_volatile(0);
42    });
43}
