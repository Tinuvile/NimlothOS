1//! # 陷阱和中断处理模块
2//!
3//! 处理所有从用户态到内核态的陷阱 (trap)，包括系统调用、异常和中断。
4//! 陷阱是 RISC-V 架构中用户态与内核态交互的核心机制。
5//!
6//! ## 处理的陷阱类型
7//!
8//! - **系统调用** (`UserEnvCall`): 用户程序请求内核服务
9//! - **时钟中断** (`SupervisorTimer`): 实现抢占式多任务调度
10//! - **页面异常** (`StoreFault`, `StorePageFault`): 内存访问违规
11//! - **非法指令** (`IllegalInstruction`): 执行无效指令
12//!
13//! ## 执行流程
14//!
15//! 1. **陷阱触发**: 用户态程序执行 `ecall` 或发生异常/中断
16//! 2. **硬件切换**: CPU 自动切换到 S 模式，跳转到 `stvec` 指定的处理程序
17//! 3. **上下文保存**: `__alltraps` 保存所有寄存器到陷阱上下文
18//! 4. **处理分发**: `trap_handler` 根据陷阱类型执行相应处理
19//! 5. **上下文恢复**: `__restore` 恢复寄存器并返回用户态
20//!
21//! ## 寄存器使用
22//!
23//! - `stvec`: 陷阱向量寄存器，指向陷阱处理入口 `__alltraps`
24//! - `scause`: 陷阱原因寄存器，标识陷阱类型和具体原因
25//! - `stval`: 陷阱值寄存器，包含相关的地址或值信息
26//! - `sstatus`: 状态寄存器，控制中断使能和特权级
27//! - `sepc`: 异常程序计数器，指向触发陷阱的指令地址
28

48/// 初始化陷阱处理系统
49///
50/// 设置内核态的陷阱处理入口点，配置陷阱向量寄存器 (`stvec`)
51/// 指向内核陷阱处理函数。这是陷阱处理系统的初始化函数。
52///
53/// ## 初始化内容
54///
55/// - 设置 `stvec` 寄存器指向 `trap_from_kernel`
56/// - 配置直接模式 (`TrapMode::Direct`)
57/// - 为内核态陷阱处理做准备
58///
59/// ## 调用时机
60///
61/// 应在系统启动早期调用，在启用中断之前完成陷阱系统初始化。
62///
63/// ## Note
64///
65/// 此函数只设置内核态陷阱入口，用户态陷阱入口会在任务切换时
66/// 通过 `set_user_trap_entry()` 动态设置。
67pub fn init() {
68    set_kernel_trap_entry();
69}

71/// 启用时钟中断
72///
73/// 在监督者中断使能寄存器 (`sie`) 中启用时钟中断位，
74/// 允许时钟中断触发陷阱进入内核进行任务调度。
75///
76/// ## 功能
77///
78/// - 设置 `sie.STIE` 位，启用监督者时钟中断
79/// - 配合时钟设备实现抢占式多任务调度
80/// - 必须在时钟设备配置完成后调用
81///
82/// ## Safety
83///
84/// 此函数是安全的，但内部使用 `unsafe` 访问 CSR 寄存器。
85pub fn enable_timer_interrupt() {
86    unsafe {
87        sie::set_stimer();
88    }
89}

91/// 设置内核态陷阱入口
92///
93/// 配置 `stvec` 寄存器指向内核陷阱处理函数 `trap_from_kernel`。
94/// 当内核态发生陷阱时，硬件会跳转到此函数执行。
95///
96/// ## 安全性
97///
98/// 使用 `unsafe` 直接操作 CSR 寄存器，这是特权操作。
99fn set_kernel_trap_entry() {
100    unsafe {
101        stvec::write(trap_from_kernel as usize, TrapMode::Direct);
102    }
103}

105/// 设置用户态陷阱入口
106///
107/// 配置 `stvec` 寄存器指向用户态陷阱处理入口 `TRAMPOLINE`。
108/// 当用户态发生陷阱时，硬件会跳转到 Trampoline 页面执行。
109///
110/// ## Trampoline 机制
111///
112/// Trampoline 页面包含 `__alltraps` 和 `__restore` 汇编代码，
113/// 负责保存和恢复完整的用户态上下文。
114///
115/// ## 安全性
116///
117/// 使用 `unsafe` 直接操作 CSR 寄存器，这是特权操作。
118fn set_user_trap_entry() {
119    unsafe {
120        stvec::write(TRAMPOLINE as usize, TrapMode::Direct);
121    }
122}

124/// 内核态陷阱处理函数
125///
126/// 当内核态发生陷阱时的处理函数。在当前的简化实现中，
127/// 内核态不应该发生陷阱，因此直接触发 panic。
128///
129/// ## 设计原理
130///
131/// - 内核代码应该是可信的，不应该产生异常
132/// - 如果内核态发生陷阱，说明存在严重的内核 bug
133/// - 立即停止系统运行，避免进一步的损坏
134///
135/// ## 可能的陷阱原因
136///
137/// - 内核代码访问无效内存地址
138/// - 内核代码执行无效指令
139/// - 硬件故障或配置错误
140///
141/// ## 属性说明
142///
143/// - `#[unsafe(no_mangle)]`: 防止函数名被混淆，确保链接器能找到
144/// - `-> !`: 函数永不返回，因为会触发 panic
145#[unsafe(no_mangle)]
146pub fn trap_from_kernel() -> ! {
147    panic!("a trap from kernel!");
148}

150/// 陷阱处理主函数
151///
152/// 所有用户态陷阱的统一处理入口，根据陷阱原因分发到相应的处理逻辑。
153/// 这是陷阱处理系统的核心函数，处理系统调用、异常和中断。
154///
155/// ## 处理流程
156///
157/// 1. **设置内核陷阱入口**: 防止处理过程中的嵌套陷阱
158/// 2. **获取陷阱信息**: 读取 `scause` 和 `stval` 寄存器
159/// 3. **分发处理**: 根据陷阱类型调用相应的处理逻辑
160/// 4. **返回用户态**: 调用 `trap_return()` 恢复用户执行
161///
162/// ## 支持的陷阱类型
163///
164/// - **系统调用** (`UserEnvCall`): 处理用户程序的系统调用请求
165/// - **内存异常** (`StoreFault`, `StorePageFault`, `LoadFault`, `LoadPageFault`): 处理内存访问违规
166/// - **非法指令** (`IllegalInstruction`): 处理无效指令执行
167/// - **时钟中断** (`SupervisorTimer`): 处理抢占式调度
168///
169/// ## 错误处理
170///
171/// - 对于致命异常（内存违规、非法指令），终止当前任务
172/// - 对于未知陷阱类型，触发 panic
173///
174/// ## 属性说明
175///
176/// - `#[unsafe(no_mangle)]`: 防止函数名被混淆，汇编代码需要调用此函数
177/// - `-> !`: 函数永不返回，总是通过 `trap_return()` 返回用户态
178#[unsafe(no_mangle)]
179pub fn trap_handler() -> ! {
180    set_kernel_trap_entry();
181    let cx = current_trap_cx();
182    let scause = scause::read();
183    let stval = stval::read();
184    match scause.cause() {
185        Trap::Exception(Exception::UserEnvCall) => {
186            // 系统调用处理
187            cx.sepc += 4; // 跳过 ecall 指令
188            cx.x[10] = syscall(cx.x[17], [cx.x[10], cx.x[11], cx.x[12]]) as usize;
189        }
190        Trap::Exception(Exception::StoreFault)
191        | Trap::Exception(Exception::StorePageFault)
192        | Trap::Exception(Exception::LoadFault)
193        | Trap::Exception(Exception::LoadPageFault) => {
194            // 存储异常：非法内存访问
195            println!(
196                "[kernel] PageFault in application, bad addr = {:#x}, bad instruction = {:#x}, kernel killed it.",
197                stval, cx.sepc
198            );
199            exit_current_and_run_next();
200        }
201        Trap::Exception(Exception::IllegalInstruction) => {
202            // 非法指令异常：执行了无效的指令
203            println!(
204                "[kernel] IllegalInstruction in application, bad instruction = {:#x}, kernel killed it.",
205                cx.sepc
206            );
207            exit_current_and_run_next();
208        }
209        Trap::Interrupt(Interrupt::SupervisorTimer) => {
210            // 时钟中断：实现抢占式调度
211            set_next_trigger();
212            suspend_current_and_run_next();
213        }
214        _ => {
215            // 未处理的陷阱类型
216            panic!(
217                "Unsupported trap {:?}, stval = {:#x}!",
218                scause.cause(),
219                stval,
220            );
221        }
222    }
223    trap_return();
224}

226/// 陷阱返回函数
227///
228/// 陷阱处理完成后返回用户态的函数，负责恢复用户态执行环境并
229/// 跳转回用户程序继续执行。这是陷阱处理的最后阶段。
230///
231/// ## 执行流程
232///
233/// 1. **设置用户陷阱入口**: 配置 `stvec` 指向 Trampoline
234/// 2. **准备返回参数**: 获取陷阱上下文地址和用户页表标识符
235/// 3. **跳转到 Trampoline**: 通过内联汇编跳转到 `__restore`
236/// 4. **恢复用户状态**: `__restore` 恢复所有寄存器并执行 `sret`
237///
238/// ## 地址空间切换
239///
240/// 函数执行过程中会发生地址空间切换：
241/// - 开始：内核地址空间
242/// - 跳转到 Trampoline：仍在内核地址空间（Trampoline 在两个地址空间都有映射）
243/// - `__restore` 执行 `sret`：切换到用户地址空间
244///
245/// ## Trampoline 机制
246///
247/// ```text
248/// 地址空间切换过程:
249/// ┌─────────────────┐    ┌─────────────────┐    ┌─────────────────┐
250/// │  Kernel Space   │    │   Trampoline    │    │   User Space    │
251/// │                 │───>│    (Shared)     │───>│                 │
252/// │  trap_return()  │    │   __restore     │    │  User Program   │
253/// └─────────────────┘    └─────────────────┘    └─────────────────┘
254/// ```
255///
256/// ## 内联汇编
257///
258/// 使用内联汇编执行关键的跳转操作：
259/// - `fence.i`: 指令缓存同步
260/// - `jr {restore_va}`: 跳转到 `__restore` 函数
261/// - 传递参数：`a0` = 陷阱上下文地址，`a1` = 用户页表标识符
262///
263/// ## 安全性
264///
265/// - 使用 `unsafe` 执行特权操作和内联汇编
266/// - 地址计算确保跳转到正确的 Trampoline 位置
267/// - 参数传递确保 `__restore` 获得正确的上下文信息
268///
269/// ## 属性说明
270///
271/// - `#[unsafe(no_mangle)]`: 防止函数名被混淆，可能被其他代码调用
272/// - `-> !`: 函数永不返回，总是跳转到用户程序执行
273#[unsafe(no_mangle)]
274pub fn trap_return() -> ! {
275    set_user_trap_entry();
276    let trap_cx_ptr = TRAP_CONTEXT;
277    let user_satp = current_user_token();
278    unsafe extern "C" {
279        fn __alltraps();
280        fn __restore();
281    }
282    let restore_va = __restore as usize - __alltraps as usize + TRAMPOLINE;
283    unsafe {
284        asm!(
285            "fence.i",
286            "jr {restore_va}",
287            restore_va = in(reg) restore_va,
288            in("a0") trap_cx_ptr,
289            in("a1") user_satp,
290            options(noreturn),
291        );
292    }
