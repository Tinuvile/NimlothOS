10/// 系统调用：进程退出
11///
12/// 实现 `exit(2)` 系统调用，终止当前正在运行的应用程序。
13/// 该函数会清理当前任务的资源，并调度下一个就绪任务运行。
14///
15/// ## Arguments
16///
17/// * `exit_code` - 进程退出码，用于向父进程或系统报告执行状态
18///   - 0 通常表示成功完成
19///   - 非零值通常表示出现错误或异常情况
20///
21/// ## Returns
22///
23/// 该函数实际上不会返回，因为调用进程会被终止。
24/// 返回类型 `isize` 仅用于与系统调用接口保持一致。
25///
26/// ## Behavior
27///
28/// 1. 记录进程退出信息到内核日志
29/// 2. 将当前任务标记为已退出状态
30/// 3. 调度并切换到下一个就绪任务
31/// 4. 如果没有其他任务，系统将处理所有任务完成的情况
32///
33/// ## Panics
34///
35/// 如果任务切换函数意外返回，会触发 panic（这在正常情况下不应该发生）
36///
37/// ## Examples
38///
39/// 从用户态调用：
40/// ```c
41/// exit(0);  // 正常退出
42/// exit(1);  // 异常退出
43/// ```
44pub fn sys_exit(exit_code: i32) -> isize {
45    println!("[kernel] Application exited with code {}", exit_code);
46    exit_current_and_run_next();
47    panic!("Unreachable in sys_exit!");
48}

50/// 系统调用：让出 CPU 时间片
51///
52/// 实现 `sched_yield(2)` 系统调用，当前任务主动让出 CPU，
53/// 允许调度器选择其他就绪任务运行。这是一种协作式多任务的实现方式。
54///
55/// ## Returns
56///
57/// 成功时返回 0。当任务重新获得 CPU 时间片时，会从此处继续执行。
58///
59/// ## Behavior
60///
61/// 1. 将当前任务状态从 `Running` 改为 `Ready`
62/// 2. 调度器选择下一个就绪任务运行
63/// 3. 执行任务上下文切换
64/// 4. 当前任务将来重新被调度时，从此函数返回
65///
66/// ## Use Cases
67///
68/// - 长时间运行的任务主动让出 CPU，提高系统响应性
69/// - 等待某些条件满足时的忙等待优化
70/// - 实现用户空间的协作式多任务
71///
72/// ## Examples
73///
74/// 从用户态调用：
75/// ```c
76/// while (condition) {
77///     yield();  // 避免长时间占用 CPU
78/// }
79/// ```
80pub fn sys_yield() -> isize {
81    suspend_current_and_run_next();
82    0
83}

85/// 系统调用：获取系统时间
86///
87/// 实现获取系统运行时间的系统调用，返回自系统启动以来经过的毫秒数。
88/// 这是 `gettimeofday(2)` 的简化版本。
89///
90/// ## Returns
91///
92/// 返回系统启动以来经过的毫秒数（作为 `isize` 类型）
93///
94/// ## Precision
95///
96/// 时间精度取决于系统时钟频率和定时器实现，通常为毫秒级精度。
97///
98/// ## Use Cases
99///
100/// - 测量代码执行时间
101/// - 实现定时器和超时机制
102/// - 性能分析和基准测试
103/// - 随机数生成的种子
104///
105/// ## Examples
106///
107/// 从用户态调用：
108/// ```c
109/// long start_time = get_time();
110/// // 执行某些操作
111/// long end_time = get_time();
112/// long elapsed = end_time - start_time;  // 计算耗时
113/// ```
114///
115/// ## Note
116///
117/// 返回的时间值可能会在长时间运行后溢出，调用者应该考虑处理这种情况。
118pub fn sys_get_time() -> isize {
119    get_time_ms() as isize
120}

122/// 系统调用：调整程序断点（堆管理）
123///
124/// 实现 `sbrk(2)` 系统调用，用于动态调整当前进程的堆大小。
125/// 这是实现 `malloc`、`free` 等动态内存分配函数的基础系统调用。
126///
127/// ## Arguments
128///
129/// * `size` - 堆大小的变化量（字节）
130///   - **正数**: 扩展堆空间，增加可用内存
131///   - **负数**: 收缩堆空间，释放内存
132///   - **零**: 查询当前程序断点位置，不改变堆大小
133///
134/// ## Returns
135///
136/// - **成功**: 返回调整前的程序断点地址（旧的堆顶地址）
137/// - **失败**: 返回 -1，可能的失败原因：
138///   - 内存不足，无法分配新的物理页面
139///   - 试图收缩到堆底部以下
140///   - 地址空间操作失败
141///
142/// ## 堆管理机制
143///
144/// ```text
145/// 进程地址空间中的堆布局：
146/// ┌─────────────────┐ ← heap_bottom (fixed)
147/// │                 │
148/// │  Allocated Heap │   ← Memory usable by user
149/// │     Space       │
150/// ├─────────────────┤ ← program_brk (variable, old value)
151/// │                 │
152/// │   New Space     │   ← New space when size > 0
153/// │                 │
154/// ├─────────────────┤ ← program_brk (variable, new value)
155/// │   Unallocated   │
156/// │      Space      │
157/// └─────────────────┘
158/// ```
159///
160/// ## 实现细节
161///
162/// 1. **委托处理**: 调用 `change_program_brk()` 执行实际的堆调整
163/// 2. **页面对齐**: 底层自动处理页面边界对齐
164/// 3. **物理内存**: 自动分配/释放物理页帧
165/// 4. **地址空间**: 更新进程的虚拟地址空间映射
166///
167/// ## 错误处理
168///
169/// - 返回 -1 表示操作失败
170/// - 用户程序应该检查返回值并适当处理错误
171/// - 失败后堆状态保持不变
172///
173/// ## 使用场景
174///
175/// - **动态内存分配**: `malloc()` 的底层实现
176/// - **内存池管理**: 自定义内存分配器
177/// - **垃圾收集**: 动态调整堆大小
178/// - **性能优化**: 预分配大块内存减少系统调用次数
179///
180/// ## Examples
181///
182/// 从用户态调用：
183/// ```c
184/// // 扩展堆空间 1KB
185/// void *old_brk = sbrk(1024);
186/// if (old_brk == (void*)-1) {
187///     // 分配失败，处理错误
188///     return -1;
189/// }
190///
191/// // 查询当前堆顶
192/// void *current_brk = sbrk(0);
193///
194/// // 收缩堆空间 512 字节
195/// void *prev_brk = sbrk(-512);
196/// ```
197///
198/// ## Security Notes
199///
200/// - 系统会自动验证内存访问权限
201/// - 防止堆与栈重叠
202/// - 确保不会访问其他进程的内存
203///
204/// ## Performance Considerations
205///
206/// - 频繁的小幅调整可能导致性能问题
207/// - 建议批量分配以减少系统调用开销
208/// - 页面分配/释放涉及内核态切换，有一定开销
209pub fn sys_sbrk(size: i32) -> isize {
210    if let Some(old_brk) = change_program_brk(size) {
211        old_brk as isize
212    } else {
213        -1
214    }
