12/// 任务控制块 (Task Control Block, TCB)
13///
14/// 存储单个任务的所有必要信息，包括任务状态和执行上下文。
15/// 每个应用程序对应一个任务控制块。
16///
17/// ## 设计原则
18///
19/// - **最小化设计**: 只包含任务调度所需的核心信息
20/// - **状态分离**: 将任务状态与上下文分开存储，便于管理
21/// - **复制语义**: 实现 `Copy` trait，支持高效的值拷贝
22///
23/// ## 扩展性
24///
25/// 未来版本可能会添加更多字段：
26/// - 任务优先级
27/// - 执行时间统计
28/// - 内存使用信息
29/// - 进程 ID 和父子关系
30pub struct TaskControlBlock {
31    /// 任务当前状态
32    ///
33    /// 标识任务在其生命周期中的当前阶段，用于调度决策。
34    pub task_status: TaskStatus,
35
36    /// 任务上下文
37    ///
38    /// 保存任务的 CPU 寄存器状态，用于任务切换时的状态保存和恢复。
39    pub task_cx: TaskContext,
40
41    /// 任务的内存地址空间
42    ///
43    /// 包含该任务的完整虚拟地址空间，包括代码段、数据段、堆、栈等。
44    /// 每个任务都有独立的地址空间，实现进程间的内存隔离。
45    pub memory_set: MemorySet,
46
47    /// 陷阱上下文的物理页号
48    ///
49    /// 指向存储该任务陷阱上下文的物理页面。陷阱上下文存储在用户地址空间
50    /// 的固定虚拟地址 `TRAP_CONTEXT`，此字段记录其对应的物理页号，
51    /// 便于内核直接访问和修改。
52    pub trap_cx_ppn: PhysPageNum,
53
54    /// 基础大小（用户栈顶地址）
55    ///
56    /// 记录用户程序的初始栈顶地址，也作为堆空间的起始参考点。
57    /// 在任务创建时设置，通常不会改变。
58    #[allow(unused)]
59    pub base_size: usize,
60
61    /// 堆底部地址
62    ///
63    /// 用户程序堆空间的起始地址，通常等于 `base_size`。
64    /// 堆从此地址开始向高地址方向增长，是 `sbrk` 系统调用的参考基准。
65    pub heap_bottom: usize,
66
67    /// 程序断点（当前堆顶）
68    ///
69    /// 当前用户程序堆空间的结束地址，即堆顶位置。
70    /// 通过 `sbrk` 系统调用可以调整此值来扩展或收缩堆空间。
71    /// 范围：`[heap_bottom, program_brk)` 为已分配的堆空间。
72    pub program_brk: usize,
73}

75/// 任务状态枚举
76///
77/// 定义任务在其生命周期中可能处于的各种状态。
78/// 任务状态转换遵循特定的规则，确保系统调度的正确性。
79///
80/// ## 状态转换图
81///
82/// ```text
83/// ┌─────────┐    load     ┌───────┐    schedule   ┌─────────┐
84/// │ Uninit  │ ──────────> │ Ready │ ────────────> │ Running │
85/// └─────────┘             └───────┘               └─────────┘
86///                            ^                         │
87///                            │          yield/         │ exit/
88///                            │         timeout         │ error
89///                            └─────────────────────────┘
90///                                                      │
91///                                                      v
92///                                              ┌─────────────┐
93///                                              │   Exited    │
94///                                              └─────────────┘
95/// ```
96///
97/// ## 状态说明
98///
99/// - **Uninit**: 任务尚未初始化，仅在系统启动时短暂存在
100/// - **Ready**: 任务已准备就绪，等待被调度执行
101/// - **Running**: 任务正在 CPU 上执行
102/// - **Exited**: 任务已完成执行，不会再被调度
103#[derive(Copy, Clone, PartialEq)]
104pub enum TaskStatus {
105    /// 未初始化状态
106    ///
107    /// 任务刚创建时的初始状态，表示任务控制块已分配但尚未完成初始化。
108    /// 这个状态在系统启动过程中短暂存在，很快会转换为 Ready 状态。
109    Uninit,
110
111    /// 就绪状态
112    ///
113    /// 任务已准备好执行，正在等待获得 CPU 时间片。
114    /// 处于此状态的任务会被调度器考虑进行调度。
115    Ready,
116
117    /// 运行状态
118    ///
119    /// 任务正在 CPU 上执行。在单 CPU 系统中，同时只能有一个任务处于此状态。
120    /// 任务可能因为时间片用完、主动让出或异常而离开此状态。
121    Running,
122
123    /// 已退出状态
124    ///
125    /// 任务已完成执行并退出，不会再被调度。
126    /// 任务可能因为正常结束、调用 exit() 或发生致命错误而进入此状态。
127    Exited,
128}

130impl TaskControlBlock {
131    /// 创建新的任务控制块
132    ///
133    /// 从 ELF 文件数据创建一个完整的任务控制块，包括解析 ELF 文件、
134    /// 构建用户地址空间、分配内核栈、初始化陷阱上下文等所有必要步骤。
135    ///
136    /// ## Arguments
137    ///
138    /// * `elf_data` - 应用程序的 ELF 文件二进制数据
139    /// * `app_id` - 应用程序标识符，用于分配独立的内核栈
140    ///
141    /// ## Returns
142    ///
143    /// 完全初始化的任务控制块，可以被调度执行
144    ///
145    /// ## 初始化流程
146    ///
147    /// 1. **解析 ELF**: 调用 `MemorySet::from_elf()` 构建用户地址空间
148    /// 2. **获取陷阱上下文**: 通过虚拟地址转换获取陷阱上下文的物理页号
149    /// 3. **分配内核栈**: 在内核地址空间中为该任务分配独立的内核栈
150    /// 4. **创建任务上下文**: 设置任务上下文指向 `trap_return`
151    /// 5. **初始化陷阱上下文**: 设置用户程序入口、栈指针和内核环境
152    /// 6. **设置堆管理**: 初始化堆底部和程序断点
153    ///
154    /// ## 内存布局
155    ///
156    /// ```text
157    /// 用户地址空间:
158    /// ┌─────────────────┐ ← TRAMPOLINE
159    /// │   Trampoline    │
160    /// ├─────────────────┤ ← TRAP_CONTEXT  
161    /// │  Trap Context   │ ← trap_cx_ppn 指向这里
162    /// ├─────────────────┤
163    /// │   User Stack    │ ← user_sp
164    /// ├─────────────────┤
165    /// │   ELF Sections  │ ← entry_point
166    /// └─────────────────┘
167    ///
168    /// 内核地址空间:
169    /// ┌─────────────────┐ ← kernel_stack_top
170    /// │  Kernel Stack   │ ← 每个任务独立的内核栈
171    /// └─────────────────┘ ← kernel_stack_bottom
172    /// ```
173    ///
174    /// ## 字段初始化
175    ///
176    /// - `task_status`: 设置为 `Ready` 状态
177    /// - `task_cx`: 指向 `trap_return` 的任务上下文
178    /// - `memory_set`: 从 ELF 构建的完整用户地址空间
179    /// - `trap_cx_ppn`: 陷阱上下文的物理页号
180    /// - `base_size`: 用户栈顶地址（堆的初始大小）
181    /// - `heap_bottom`: 堆的起始地址
182    /// - `program_brk`: 程序断点（当前堆顶）
183    ///
184    /// ## Examples
185    ///
186    /// ```rust
187    /// let app_data = get_app_data(0);
188    /// let task = TaskControlBlock::new(app_data, 0);
189    /// // 任务现在可以被调度执行
190    /// ```
191    pub fn new(elf_data: &[u8], app_id: usize) -> Self {
192        let (memory_set, user_sp, entry_point) = MemorySet::from_elf(elf_data);
193        let trap_cx_ppn = memory_set
194            .translate(VirtAddr::from(TRAP_CONTEXT).into())
195            .unwrap()
196            .ppn();
197        let task_status = TaskStatus::Ready;
198
199        let (kernel_stack_bottom, kernel_stack_top) = kernel_stack_position(app_id);
200        KERNEL_SPACE.exclusive_access().insert_framed_area(
201            kernel_stack_bottom.into(),
202            kernel_stack_top.into(),
203            MapPermission::R | MapPermission::W,
204        );
205        let task_control_block = Self {
206            task_status,
207            task_cx: TaskContext::goto_trap_return(kernel_stack_top),
208            memory_set,
209            trap_cx_ppn,
210            base_size: user_sp,
211            heap_bottom: user_sp,
212            program_brk: user_sp,
213        };
214        let trap_cx = task_control_block.get_trap_cx();
215        *trap_cx = TrapContext::app_init_context(
216            entry_point,
217            user_sp,
218            KERNEL_SPACE.exclusive_access().token(),
219            kernel_stack_top,
220            trap_handler as usize,
221        );
222        task_control_block
223    }

225    /// 获取任务的陷阱上下文
226    ///
227    /// 通过陷阱上下文的物理页号获取该任务陷阱上下文的可变引用。
228    /// 陷阱上下文存储在任务的用户地址空间中的固定虚拟地址。
229    ///
230    /// ## Returns
231    ///
232    /// 指向任务陷阱上下文的可变引用，生命周期为 `'static`
233    ///
234    /// ## 实现原理
235    ///
236    /// 1. **物理页号**: `trap_cx_ppn` 存储陷阱上下文所在的物理页号
237    /// 2. **直接访问**: 通过物理页号直接获取物理内存的可变引用
238    /// 3. **类型转换**: 将物理页面解释为 `TrapContext` 结构体
239    ///
240    /// ## 使用场景
241    ///
242    /// - **系统调用处理**: 读取和修改系统调用参数和返回值
243    /// - **异常处理**: 访问触发异常时的寄存器状态
244    /// - **任务初始化**: 设置新任务的初始陷阱上下文
245    /// - **调试和诊断**: 检查任务的执行状态
246    ///
247    /// ## Safety
248    ///
249    /// 此函数返回 `'static` 生命周期的引用，调用者需要确保：
250    /// - 不会在任务销毁后继续使用该引用
251    /// - 同一时间只有一个任务在访问其陷阱上下文
252    ///
253    /// ## Examples
254    ///
255    /// ```rust
256    /// // 获取当前任务的陷阱上下文
257    /// let trap_cx = task.get_trap_cx();
258    ///
259    /// // 读取系统调用参数
260    /// let syscall_id = trap_cx.x[17];
261    /// let arg0 = trap_cx.x[10];
262    ///
263    /// // 设置系统调用返回值
264    /// trap_cx.x[10] = result as usize;
265    /// ```
266    pub fn get_trap_cx(&self) -> &'static mut TrapContext {
267        self.trap_cx_ppn.get_mut()
268    }

270    /// 获取用户地址空间的页表标识符
271    ///
272    /// 返回该任务用户地址空间的页表标识符（`satp` 寄存器值），
273    /// 用于地址空间切换和陷阱处理。
274    ///
275    /// ## Returns
276    ///
277    /// 用户地址空间的页表标识符，包含页表模式、ASID 和根页表物理页号
278    ///
279    /// ## 使用场景
280    ///
281    /// - **陷阱返回**: 在 `trap_return` 中切换回用户地址空间
282    /// - **任务切换**: 保存和恢复任务的地址空间标识符
283    /// - **调试工具**: 获取任务的地址空间信息
284    /// - **内存管理**: 识别不同任务的地址空间
285    ///
286    /// ## 实现细节
287    ///
288    /// 此函数是 `self.memory_set.token()` 的便捷包装，
289    /// 提供了任务级别的接口来访问底层的内存管理功能。
290    ///
291    /// ## 与内核地址空间的区别
292    ///
293    /// - **用户 token**: 每个任务都有独立的用户地址空间标识符
294    /// - **内核 token**: 所有任务共享同一个内核地址空间标识符
295    /// - **切换时机**: 在陷阱处理和任务切换时需要在两者间切换
296    ///
297    /// ## Examples
298    ///
299    /// ```rust
300    /// // 获取用户地址空间标识符
301    /// let user_token = task.get_user_token();
302    ///
303    /// // 在陷阱返回时使用
304    /// unsafe {
305    ///     satp::write(user_token);
306    ///     asm!("sfence.vma");
307    /// }
308    /// ```
309    pub fn get_user_token(&self) -> usize {
310        self.memory_set.token()
311    }

313    /// 修改程序断点（堆大小调整）
314    ///
315    /// 实现 `sbrk` 系统调用的核心功能，通过调整程序断点来动态改变
316    /// 用户程序的堆大小。支持堆的扩展和收缩操作。
317    ///
318    /// ## Arguments
319    ///
320    /// * `size` - 堆大小的变化量（字节）
321    ///   - 正数：扩展堆空间
322    ///   - 负数：收缩堆空间
323    ///   - 零：查询当前断点位置
324    ///
325    /// ## Returns
326    ///
327    /// - `Some(old_brk)` - 成功时返回调整前的程序断点地址
328    /// - `None` - 失败时返回 None（通常是内存不足或参数无效）
329    ///
330    /// ## 堆管理机制
331    ///
332    /// ```text
333    /// 堆内存布局:
334    /// ┌─────────────────┐ ← heap_bottom (固定)
335    /// │                 │
336    /// │   已分配堆空间   │
337    /// │                 │
338    /// ├─────────────────┤ ← program_brk (可变)
339    /// │                 │
340    /// │   未分配空间     │
341    /// │                 │
342    /// └─────────────────┘
343    /// ```
344    ///
345    /// ## 操作流程
346    ///
347    /// 1. **计算新断点**: `new_brk = current_brk + size`
348    /// 2. **边界检查**: 确保新断点不低于 `heap_bottom`
349    /// 3. **内存操作**:
350    ///    - 扩展：调用 `memory_set.append_to()` 分配新页面
351    ///    - 收缩：调用 `memory_set.shrink_to()` 释放页面
352    /// 4. **更新断点**: 成功后更新 `program_brk` 字段
353    ///
354    /// ## 错误情况
355    ///
356    /// - **堆收缩过度**: 新断点低于 `heap_bottom`
357    /// - **内存不足**: 无法分配足够的物理页面
358    /// - **地址空间冲突**: 与其他内存区域重叠
359    ///
360    /// ## 并发安全
361    ///
362    /// 此函数需要 `&mut self`，确保同一时间只有一个线程能修改
363    /// 任务的堆状态，避免竞争条件。
364    ///
365    /// ## Examples
366    ///
367    /// ```rust
368    /// // 扩展堆空间 4KB
369    /// if let Some(old_brk) = task.change_program_brk(4096) {
370    ///     println!("Heap extended from {:#x}", old_brk);
371    /// }
372    ///
373    /// // 收缩堆空间 2KB  
374    /// if let Some(old_brk) = task.change_program_brk(-2048) {
375    ///     println!("Heap shrunk from {:#x}", old_brk);
376    /// }
377    ///
378    /// // 查询当前断点
379    /// if let Some(current_brk) = task.change_program_brk(0) {
380    ///     println!("Current program break: {:#x}", current_brk);
381    /// }
382    /// ```
383    #[allow(unused)]
384    pub fn change_program_brk(&mut self, size: i32) -> Option<usize> {
385        let old_break = self.program_brk;
386        let new_brk = self.program_brk as isize + size as isize;
387        if new_brk < self.heap_bottom as isize {
388            return None;
389        }
390        let result = if size < 0 {
391            self.memory_set
392                .shrink_to(VirtAddr(self.heap_bottom), VirtAddr(new_brk as usize))
393        } else {
394            self.memory_set
395                .append_to(VirtAddr(self.heap_bottom), VirtAddr(new_brk as usize))
396        };
397        if result {
398            self.program_brk = new_brk as usize;
399            Some(old_break)
400        } else {
401            None
402        }
403    }
