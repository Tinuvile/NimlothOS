1//! # 任务管理模块
2//!
3//! 实现协作式多任务调度系统，支持多个用户应用程序的并发执行。
4//! 采用时间片轮转调度算法，通过上下文切换在不同任务间切换。
5//!
6//! ## 核心组件
7//!
8//! - [`TaskManager`] - 任务管理器，负责所有任务的生命周期管理
9//! - [`TaskControlBlock`] - 任务控制块，存储单个任务的状态信息
10//! - [`TaskContext`] - 任务上下文，保存 CPU 寄存器状态
11//! - [`TaskStatus`] - 任务状态枚举
12//!
13//! ## 调度策略
14//!
15//! 使用简单的时间片轮转 (Round Robin) 调度：
16//! - 每个任务获得固定的时间片
17//! - 时间片用完或主动让出时切换到下一个就绪任务
18//! - 任务按 ID 顺序循环调度
19//!
20//! ## 任务状态转换
21//!
22//! ```text
23//! Uninit -> Ready -> Running -> Exited
24//!             ^         |
25//!             |         v  
26//!             +--- Ready (yield)
27//! ```

47/// 任务管理器
48///
49/// 负责管理系统中所有任务的生命周期，包括任务创建、调度和销毁。
50/// 使用 [`UPSafeCell`] 提供线程安全的内部可变性。
51///
52/// ## 设计特点
53///
54/// - **静态任务数组**: 预先分配固定数量的任务槽位
55/// - **轮转调度**: 使用简单的轮转算法选择下一个任务
56/// - **状态管理**: 跟踪每个任务的执行状态
57/// - **上下文保存**: 保存和恢复任务的 CPU 状态
58pub struct TaskManager {
59    /// 系统中的应用程序数量
60    num_app: usize,
61    /// 任务管理器的内部状态，使用 UPSafeCell 保证线程安全
62    inner: UPSafeCell<TaskManagerInner>,
63}
64
65/// 任务管理器内部状态
66///
67/// 包含实际的任务数组和当前运行任务的索引。
68/// 该结构体被 [`UPSafeCell`] 包装以提供安全的并发访问。
69pub struct TaskManagerInner {
70    /// 所有任务的控制块数组
71    ///
72    /// 数组大小固定为 [`MAX_APP_NUM`]，每个槽位对应一个应用程序。
73    tasks: Vec<TaskControlBlock>,
74
75    /// 当前正在运行的任务 ID
76    ///
77    /// 用作 `tasks` 数组的索引，指向当前获得 CPU 时间片的任务。
78    current_task: usize,
79}

81lazy_static! {
82    /// 全局任务管理器实例
83    ///
84    /// 系统启动时创建的单例任务管理器，负责管理所有用户应用程序。
85    /// 使用 `lazy_static!` 确保在首次访问时才进行初始化。
86    ///
87    /// ## 初始化过程
88    ///
89    /// 待补充
90    pub static ref TASK_MANAGER: TaskManager = {
91        let num_app = get_num_app();
92        let mut tasks: Vec<TaskControlBlock> = Vec::new();
93        println!("init TaskManager, num_app: {}", num_app);
94        for i in 0..num_app {
95            tasks.push(TaskControlBlock::new(get_app_data(i), i));
96        }
97        TaskManager {
98            num_app,
99            inner: unsafe {
100                UPSafeCell::new(TaskManagerInner {
101                    tasks,
102                    current_task: 0,
103                })
104            },
105        }
106    };
107}

109impl TaskManager {
110    /// 启动第一个任务
111    ///
112    /// 系统初始化完成后调用此函数开始任务调度。
113    /// 该函数会将第一个任务（ID 为 0）设置为运行状态，
114    /// 然后执行上下文切换跳转到用户任务。
115    ///
116    /// ## 执行流程
117    ///
118    /// 1. 获取任务管理器内部状态的独占访问
119    /// 2. 将第一个任务状态设置为 `Running`
120    /// 3. 获取任务上下文指针
121    /// 4. 释放内部状态锁
122    /// 5. 执行上下文切换到第一个任务
123    ///
124    /// ## Panics
125    ///
126    /// 如果上下文切换意外返回，会触发 panic（正常情况下不应该发生）
127    fn run_first_task(&self) -> ! {
128        let mut inner = self.inner.exclusive_access();
129        let task0 = &mut inner.tasks[0];
130        task0.task_status = TaskStatus::Running;
131        let next_task_cx_ptr = &task0.task_cx as *const TaskContext;
132        drop(inner);
133        let mut _unused = TaskContext::zero_init();
134        unsafe {
135            __switch(&mut _unused as *mut _, next_task_cx_ptr);
136        }
137        panic!("unreachable in run_first_task!");
138    }
139
140    /// 将当前任务标记为挂起状态
141    ///
142    /// 将当前正在运行的任务状态从 `Running` 改为 `Ready`，
143    /// 表示该任务可以被重新调度执行。通常在任务主动让出 CPU 时调用。
144    fn mark_current_suspended(&self) {
145        let mut inner = self.inner.exclusive_access();
146        let current_task = inner.current_task;
147        inner.tasks[current_task].task_status = TaskStatus::Ready;
148    }
149
150    /// 将当前任务标记为已退出状态
151    ///
152    /// 将当前正在运行的任务状态设置为 `Exited`，
153    /// 表示该任务已经完成执行，不会再被调度。
154    fn mark_current_exited(&self) {
155        let mut inner = self.inner.exclusive_access();
156        let current = inner.current_task;
157        inner.tasks[current].task_status = TaskStatus::Exited;
158    }
159
160    /// 查找下一个可运行的任务
161    ///
162    /// 使用轮转调度算法在当前任务之后查找下一个状态为 `Ready` 的任务。
163    /// 搜索顺序是从当前任务的下一个开始，循环遍历所有任务。
164    ///
165    /// ## Returns
166    ///
167    /// - `Some(task_id)` - 找到的下一个就绪任务的 ID
168    /// - `None` - 没有找到就绪任务
169    ///
170    /// ## 调度算法
171    ///
172    /// 轮转调度 (Round Robin)：
173    /// ```text
174    /// 当前任务ID = 2, 任务总数 = 4
175    /// 搜索顺序: 3 -> 0 -> 1 -> (回到2)
176    /// ```
177    fn find_next_task(&self) -> Option<usize> {
178        let inner = self.inner.exclusive_access();
179        let current = inner.current_task;
180
181        (current + 1..current + self.num_app + 1)
182            .map(|id| id % self.num_app)
183            .find(|id| inner.tasks[*id].task_status == TaskStatus::Ready)
184    }
185
186    /// 切换到下一个任务
187    ///
188    /// 查找下一个就绪任务并执行任务切换。如果找到了就绪任务，
189    /// 会保存当前任务的上下文，恢复目标任务的上下文，然后跳转执行。
190    ///
191    /// ## 执行流程
192    ///
193    /// 1. 查找下一个就绪任务
194    /// 2. 如果找到：
195    ///    - 将目标任务状态设置为 `Running`
196    ///    - 更新当前任务索引
197    ///    - 执行上下文切换
198    /// 3. 如果没找到：触发 panic（所有任务都已完成）
199    ///
200    /// ## Panics
201    ///
202    /// 当没有任何就绪任务时会触发 panic，表示所有应用程序都已完成执行
203    fn run_next_task(&self) {
204        if let Some(next) = self.find_next_task() {
205            let mut inner = self.inner.exclusive_access();
206            let current = inner.current_task;
207            inner.tasks[next].task_status = TaskStatus::Running;
208            inner.current_task = next;
209            let current_task_cx_ptr = &mut inner.tasks[current].task_cx as *mut TaskContext;
210            let next_task_cx_ptr = &inner.tasks[next].task_cx as *const TaskContext;
211            drop(inner);
212            unsafe {
213                __switch(current_task_cx_ptr, next_task_cx_ptr);
214            }
215        } else {
216            println!("All applications completed!");
217            shutdown(false);
218        }
219    }
220
221    /// 获取当前任务的用户地址空间标识符
222    ///
223    /// 返回当前正在运行任务的用户地址空间页表标识符，
224    /// 用于陷阱返回时切换到用户地址空间。
225    ///
226    /// ## Returns
227    ///
228    /// 当前任务的用户地址空间页表标识符（`satp` 寄存器值）
229    fn get_current_token(&self) -> usize {
230        let inner = self.inner.exclusive_access();
231        let current = inner.current_task;
232        inner.tasks[current].get_user_token()
233    }
234
235    /// 获取当前任务的陷阱上下文
236    ///
237    /// 返回当前正在运行任务的陷阱上下文的可变引用，
238    /// 用于系统调用处理和异常处理。
239    ///
240    /// ## Returns
241    ///
242    /// 当前任务陷阱上下文的可变引用
243    fn get_current_trap_cx(&self) -> &mut TrapContext {
244        let inner = self.inner.exclusive_access();
245        let current = inner.current_task;
246        inner.tasks[current].get_trap_cx()
247    }
248
249    /// 修改当前任务的程序断点
250    ///
251    /// 调整当前正在运行任务的堆大小，实现 `sbrk` 系统调用的功能。
252    /// 这是任务管理器级别的堆管理接口。
253    ///
254    /// ## Arguments
255    ///
256    /// * `size` - 堆大小的变化量（字节）
257    ///   - 正数：扩展堆空间
258    ///   - 负数：收缩堆空间
259    ///   - 零：查询当前断点位置
260    ///
261    /// ## Returns
262    ///
263    /// - `Some(old_brk)` - 成功时返回调整前的程序断点地址
264    /// - `None` - 失败时返回 None
265    pub fn change_current_program_brk(&self, size: i32) -> Option<usize> {
266        let mut inner = self.inner.exclusive_access();
267        let current = inner.current_task;
268        inner.tasks[current].change_program_brk(size)
269    }
270}

321/// 挂起当前任务并切换到下一个任务
322///
323/// 协作式任务调度的主要接口。当任务主动让出 CPU 时调用，
324/// 会将当前任务标记为就绪状态，然后切换到下一个就绪任务。
325///
326/// ## 使用场景
327///
328/// - 实现 `yield()` 系统调用
329/// - 任务等待某些条件时主动让出 CPU
330/// - 时钟中断处理中的抢占式调度
331///
332/// ## 执行流程
333///
334/// 1. 将当前任务状态设置为 `Ready`
335/// 2. 查找下一个就绪任务
336/// 3. 执行上下文切换
337pub fn suspend_current_and_run_next() {
338    mark_current_suspended();
339    run_next_task();
340}

342/// 终止当前任务并切换到下一个任务
343///
344/// 任务退出时的调度接口。将当前任务标记为已退出状态，
345/// 然后切换到下一个就绪任务继续执行。
346///
347/// ## 使用场景
348///
349/// - 实现 `exit()` 系统调用
350/// - 任务因异常终止时的清理
351/// - 应用程序正常结束时的处理
352///
353/// ## 执行流程
354///
355/// 1. 将当前任务状态设置为 `Exited`
356/// 2. 查找下一个就绪任务
357/// 3. 执行上下文切换（如果有就绪任务）
358/// 4. 如果没有就绪任务，系统结束运行
359pub fn exit_current_and_run_next() {
360    mark_current_exited();
361    run_next_task();
362}

427/// 修改当前任务的程序断点
428///
429/// 调整当前任务的堆大小的公共接口，实现 `sbrk` 系统调用。
430/// 这是用户程序动态内存管理的核心接口。
431///
432/// ## Arguments
433///
434/// * `size` - 堆大小的变化量（字节）
435///   - **正数**: 扩展堆空间，分配更多内存
436///   - **负数**: 收缩堆空间，释放内存
437///   - **零**: 查询当前程序断点位置
438///
439/// ## Returns
440///
441/// - `Some(old_brk)` - 成功时返回调整前的程序断点地址
442/// - `None` - 失败时返回 None，可能的原因：
443///   - 内存不足，无法分配新页面
444///   - 试图收缩到堆底部以下
445///   - 地址空间操作失败
446///
447/// ## 实现原理
448///
449/// 1. 委托给任务管理器的相应方法
450/// 2. 任务管理器找到当前任务
451/// 3. 调用任务控制块的堆管理方法
452/// 4. 底层通过内存集合进行实际的页面分配/释放
453///
454/// ## 使用场景
455///
456/// - **`sbrk` 系统调用**: 用户程序动态调整堆大小
457/// - **内存分配器**: `malloc`/`free` 的底层实现
458/// - **垃圾收集器**: 动态调整堆空间
459///
460/// ## Examples
461///
462/// ```rust
463/// // 扩展堆空间
464/// if let Some(old_brk) = change_program_brk(4096) {
465///     println!("Heap expanded from {:#x}", old_brk);
466/// } else {
467///     println!("Failed to expand heap");
468/// }
469///
470/// // 查询当前断点
471/// if let Some(current_brk) = change_program_brk(0) {
472///     println!("Current program break: {:#x}", current_brk);
473/// }
474/// ```
475#[allow(unused)]
476pub fn change_program_brk(size: i32) -> Option<usize> {
477    TASK_MANAGER.change_current_program_brk(size)
478}
