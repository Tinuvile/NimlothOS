32pub fn suspend_current_and_run_next() {
33    let task = take_current_task().unwrap();
34    let mut task_inner = task.inner_exclusive_access();
35    let task_cx_ptr = &mut task_inner.task_cx as *mut TaskContext;
36    task_inner.task_status = TaskStatus::Ready;
37    drop(task_inner);
38    add_task(task);
39    schedule(task_cx_ptr);
40}

42pub fn exit_current_and_run_next(exit_code: i32) {
43    let task = take_current_task().unwrap();
44    let mut inner = task.inner_exclusive_access();
45    inner.task_status = TaskStatus::Zombie;
46    inner.exit_code = exit_code;
47    {
48        let mut initproc_inner = INITPROC.inner_exclusive_access();
49        for child in inner.children.iter() {
50            child.inner_exclusive_access().parent = Some(Arc::downgrade(&INITPROC));
51            initproc_inner.children.push(child.clone());
52        }
53    }
54    inner.children.clear();
55    inner.memory_set.recycle_data_pages();
56    drop(inner);
57    drop(task);
58    let mut _unused = TaskContext::zero_init();
59    schedule(&mut _unused as *mut _);
60}
