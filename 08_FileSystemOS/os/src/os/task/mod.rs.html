1//! # 任务管理模块
2//!
3//! 提供内核中的任务/进程管理与调度功能，涵盖任务上下文保存与切换、
4//! 调度队列维护、PID 管理、内核栈管理以及处理器当前任务状态管理。
5//!
6//! ## 模块组织
7//!
8//! - [`context`]   - 任务上下文 `TaskContext` 的保存与恢复
9//! - [`manager`]   - 就绪队列管理与基本调度（FIFO）
10//! - [`pid`]       - 进程 ID 分配与回收、内核栈管理
11//! - [`processor`] - 当前处理器状态、当前任务获取、调度入口
12//! - [`switch`]    - 低层上下文切换实现（汇编封装）
13//! - [`task`]      - 任务控制块 `TaskControlBlock` 及其内部结构
14//!
15//! ## 公开接口（re-exports）
16//!
17//! - 类型：[`TaskContext`]
18//! - 函数：[`add_task`], [`run_tasks`], [`schedule`], [`current_task`],
19//!   [`current_trap_cx`], [`current_user_token`], [`take_current_task`],
20//!   [`add_initproc`], [`suspend_current_and_run_next`], [`exit_current_and_run_next`]
21//! - 常量：[`IDLE_PID`], [`INITPROC`]
22//!
23//! ## 调度模型
24//!
25//! - 调度策略：基于就绪队列的 FIFO 调度
26//! - 切换路径：`run_tasks()` 选择下一个任务 → `__switch` 切到任务 →
27//!   任务因时间片到期/主动让出/阻塞 → `schedule()` 切回调度器
28//!
29//! ## 初始化与启动
30//!
31//! - 初始进程：[`INITPROC`]（从内置应用镜像加载 `initproc`）
32//! - 空闲进程：PID 为 [`IDLE_PID`]（值为 0）的特殊进程，负责系统空闲时的处理
33//! - 启动流程：调用 [`add_initproc()`] 将初始进程加入就绪队列，随后
34//!   通过 [`run_tasks()`] 进入主调度循环
35//!
36//! ## 任务生命周期管理
37//!
38//! - 任务挂起：[`suspend_current_and_run_next()`] 将当前任务状态置为 `Ready` 并重新入队
39//! - 任务退出：[`exit_current_and_run_next()`] 处理任务结束、孤儿进程托管和资源回收
40//! - 空闲进程退出：当 PID 为 [`IDLE_PID`] 的进程退出时，根据退出码决定系统关机行为
41//!
42//! ## 使用示例
43//!
44//! ```rust
45//! // 启动阶段：注册初始进程并进入调度循环
46//! task::add_initproc();
47//! // run_tasks() 在本工程由处理器模块统一驱动
48//! ```
49//!

70lazy_static! {
71    /// 初始进程（initproc）
72    ///
73    /// 调用open_file函数，打开initproc文件，并读取文件内容，创建TaskControlBlock
74    pub static ref INITPROC: Arc<TaskControlBlock> = Arc::new({
75        let inode = open_file("initproc", OpenFlags::RDONLY).unwrap();
76        let data = inode.read_all();
77        TaskControlBlock::new(data.as_slice())
78    });
79}

81/// 将初始进程加入就绪队列
82///
83/// 在系统启动阶段调用，把 [`INITPROC`] 推入调度器的就绪队列，等待
84/// 调度器选择并运行。
85pub fn add_initproc() {
86    add_task(INITPROC.clone());
87}
88
89/// 让出当前任务并切换到下一个就绪任务
90///
91/// 将当前任务状态从 `Running` 置为 `Ready`，重新放回就绪队列，然后通过
92/// [`schedule()`] 切换回调度器上下文，由调度器选择下一个任务运行。
93///
94/// ## 行为
95/// - 保存当前任务上下文
96/// - 更新任务状态为 `Ready`
97/// - 重新入队就绪队列
98/// - 触发上下文切换回调度器
99pub fn suspend_current_and_run_next() {
100    let task = take_current_task().unwrap();
101    let mut task_inner = task.inner_exclusive_access();
102    let task_cx_ptr = &mut task_inner.task_cx as *mut TaskContext;
103    task_inner.task_status = TaskStatus::Ready;
104    drop(task_inner);
105    add_task(task);
106    schedule(task_cx_ptr);
107}
108
109/// 空闲进程的 PID
110///
111/// 值为 0 的特殊 PID，用于标识系统中的空闲进程。当空闲进程退出时，
112/// 系统会根据其退出码决定是否关机：非零退出码触发错误关机，零退出码
113/// 触发正常关机。
114pub const IDLE_PID: usize = 0;
115
116/// 结束当前任务并切换到下一个任务
117///
118/// 将当前任务标记为 `Zombie`，记录退出码，并进行"孤儿进程"托管：
119/// 将其所有子进程的父指针重定向到 [`INITPROC`]。随后清空子进程列表、
120/// 释放任务私有地址空间的区域元数据（不主动取消映射），最后切换回
121/// 调度器，由调度器继续运行其他任务。
122///
123/// ## 特殊处理
124/// - 如果当前任务 PID 为 [`IDLE_PID`]，则根据退出码决定系统关机行为：
125///   - 非零退出码：触发错误关机
126///   - 零退出码：触发正常关机
127///
128/// ## Arguments
129/// * `exit_code` - 任务退出码
130///
131/// ## 备注
132/// - 子进程在被重新托管后，退出回收将由 `initproc` 负责
133/// - 地址空间的底层页帧由 RAII 管理，任务生命周期结束时被回收
134pub fn exit_current_and_run_next(exit_code: i32) {
135    let task = take_current_task().unwrap();
136
137    let pid = task.getpid();
138    if pid == IDLE_PID {
139        println!(
140            "[kernel] Idle process exit with exit_code {} ...",
141            exit_code
142        );
143        if exit_code != 0 {
144            shutdown(true)
145        } else {
146            shutdown(false)
147        }
148    }
149
150    let mut inner = task.inner_exclusive_access();
151    inner.task_status = TaskStatus::Zombie;
152    inner.exit_code = exit_code;
153    {
154        let mut initproc_inner = INITPROC.inner_exclusive_access();
155        for child in inner.children.iter() {
156            child.inner_exclusive_access().parent = Some(Arc::downgrade(&INITPROC));
157            initproc_inner.children.push(child.clone());
158        }
159    }
160    inner.children.clear();
161    inner.memory_set.recycle_data_pages();
162    drop(inner);
163    drop(task);
164    let mut _unused = TaskContext::zero_init();
165    schedule(&mut _unused as *mut _);
166}
