1//! # 块设备抽象模块
2//!
3//! 提供块设备的抽象接口，定义了块设备的基本操作。块设备是文件系统的基础，
4//! 提供了以块为单位的读写操作，支持多种存储介质的统一抽象。
5//!
6//! ## 设计目标
7//!
8//! - **统一接口**：为不同的块设备实现提供统一的抽象接口
9//! - **类型安全**：通过 trait 约束确保类型安全和线程安全
10//! - **性能优化**：支持高效的块级读写操作
11//! - **扩展性**：支持内存块设备、磁盘块设备等多种实现
12//!
13//! ## 核心组件
14//!
15//! - [`BlockDevice`] - 块设备抽象 trait，定义了块设备的基本操作
16//!
17//! ## 块设备特性
18//!
19//! - **块大小固定**：所有块设备使用相同的块大小（通常为 4KB）
20//! - **随机访问**：支持任意块 ID 的读写操作
21//! - **原子操作**：单个块的读写操作是原子的
22//! - **线程安全**：支持多线程并发访问
23//!
24//! ## 使用示例
25//!
26//! ```rust
27//! use easy_fs::BlockDevice;
28//!
29//! // 实现内存块设备
30//! struct MemoryBlockDevice {
31//!     blocks: Vec<Vec<u8>>,
32//! }
33//!
34//! impl BlockDevice for MemoryBlockDevice {
35//!     fn read_block(&self, block_id: usize, buf: &mut [u8]) {
36//!         if let Some(block) = self.blocks.get(block_id) {
37//!             buf.copy_from_slice(&block[..buf.len()]);
38//!         }
39//!     }
40//!     
41//!     fn write_block(&self, block_id: usize, buf: &[u8]) {
42//!         if let Some(block) = self.blocks.get_mut(block_id) {
43//!             block[..buf.len()].copy_from_slice(buf);
44//!         }
45//!     }
46//! }
47//! ```
48//!
49
50use core::any::Any;

52/// 块设备抽象接口
53///
54/// 定义了块设备的基本操作，包括块读取和块写入。块设备是文件系统的底层存储
55/// 抽象，提供了以固定大小块为单位的随机访问能力。
56///
57/// ## 特性约束
58///
59/// - `Send` - 可以在线程间安全传递，支持跨线程使用
60/// - `Sync` - 可以在线程间安全共享引用，支持并发访问
61/// - `Any` - 支持运行时类型检查和动态分发
62///
63/// ## 实现要求
64///
65/// 实现者需要确保：
66/// - **块大小一致**：所有块的读写操作都使用相同的块大小
67/// - **原子性**：单个块的读写操作应该是原子的，要么完全成功，要么完全失败
68/// - **错误处理**：在设备错误时应该有适当的错误处理机制
69/// - **边界检查**：正确处理无效的块 ID 和缓冲区大小不匹配的情况
70///
71/// ## 块操作语义
72///
73/// - `read_block`: 从指定块读取数据到缓冲区
74/// - `write_block`: 将缓冲区数据写入指定块
75///
76/// ## 性能考虑
77///
78/// - 实现应该尽可能减少内存拷贝
79/// - 支持批量操作以提高性能
80/// - 考虑缓存策略以优化频繁访问的块
81///
82/// ## 注意事项
83///
84/// - 块 ID 的有效范围由具体实现决定
85/// - 缓冲区大小应该与块大小匹配以获得最佳性能
86/// - 实现应该处理边界情况和错误状态
87/// - 写入操作可能需要考虑持久化策略
88pub trait BlockDevice: Send + Sync + Any {
89    /// 从指定块读取数据
90    ///
91    /// 将块设备中指定块的数据读取到提供的缓冲区中。该操作是原子的，
92    /// 要么读取整个块的数据，要么失败。
93    ///
94    /// ## Arguments
95    /// * `block_id` - 要读取的块 ID，由实现定义有效范围
96    /// * `buf` - 用于存储读取数据的缓冲区
97    ///
98    /// ## 行为
99    /// - 从块设备读取整个块的数据
100    /// - 将数据写入提供的缓冲区
101    /// - 如果缓冲区大小小于块大小，只读取缓冲区能容纳的数据
102    /// - 如果缓冲区大小大于块大小，多余部分保持不变
103    ///
104    /// ## 错误处理
105    /// - 如果块 ID 无效，行为由具体实现定义（可能 panic 或静默失败）
106    /// - 如果设备不可用，应该通过 panic 或其他方式报告错误
107    /// - 读取失败时缓冲区内容未定义
108    ///
109    /// ## 性能说明
110    /// - 实现应该尽可能避免不必要的内存拷贝
111    /// - 可以考虑缓存机制以提高频繁访问块的性能
112    fn read_block(&self, block_id: usize, buf: &mut [u8]);
113
114    /// 向指定块写入数据
115    ///
116    /// 将缓冲区中的数据写入块设备中的指定块。该操作是原子的，
117    /// 要么完全写入成功，要么完全失败。
118    ///
119    /// ## Arguments
120    /// * `block_id` - 要写入的块 ID，由实现定义有效范围
121    /// * `buf` - 包含要写入数据的缓冲区
122    ///
123    /// ## 行为
124    /// - 将缓冲区数据写入指定的块
125    /// - 如果缓冲区大小小于块大小，块中未覆盖的部分保持不变
126    /// - 如果缓冲区大小大于块大小，只写入块能容纳的数据
127    /// - 写入操作完成后，数据立即可用于后续的读取操作
128    ///
129    /// ## 错误处理
130    /// - 如果块 ID 无效，行为由具体实现定义
131    /// - 如果设备不可写或不可用，应该通过 panic 或其他方式报告错误
132    /// - 写入操作应该是原子的，要么完全成功，要么完全失败
133    /// - 写入失败时块的内容应该保持不变
134    ///
135    /// ## 持久化
136    /// - 实现应该确保写入的数据在系统重启后仍然可用
137    /// - 可能需要考虑写入缓存和刷新策略
138    /// - 对于内存块设备，持久化行为由实现定义
139    fn write_block(&self, block_id: usize, buf: &[u8]);
140}
