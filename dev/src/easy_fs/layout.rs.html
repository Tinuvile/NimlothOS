1//! # 文件系统布局模块
2//!
3//! 定义了简单文件系统（Easy File System）的磁盘布局和数据结构。
4//! 包含超级块、磁盘 inode、目录项等核心数据结构的定义和操作。
5//!
6//! ## 文件系统布局
7//!
8//! ```text
9//! ┌─────────────────────────────────────────────────────────────┐
10//! │                        SuperBlock                           │
11//! ├─────────────────────────────────────────────────────────────┤
12//! │                       Inode Bitmap                          │
13//! ├─────────────────────────────────────────────────────────────┤
14//! │                        Inode Area                           │
15//! ├─────────────────────────────────────────────────────────────┤
16//! │                        Data Bitmap                          │
17//! ├─────────────────────────────────────────────────────────────┤
18//! │                         Data Area                           │
19//! └─────────────────────────────────────────────────────────────┘
20//! ```
21//!
22//! ## 核心组件
23//!
24//! - [`SuperBlock`] - 文件系统超级块，包含文件系统元数据
25//! - [`DiskInode`] - 磁盘 inode，管理文件的数据块分配
26//! - [`DirEntry`] - 目录项，用于目录文件的组织
27//!
28//! ## 数据块分配策略
29//!
30//! 采用三级索引结构：
31//! - **直接块**：前 28 个数据块直接存储在 inode 中
32//! - **一级间接块**：第 29-1024 个数据块通过一级间接块索引
33//! - **二级间接块**：第 1025 个及以后的数据块通过二级间接块索引
34//!
35//! ## 使用示例
36//!
37//! ```rust
38//! use easy_fs::layout::{SuperBlock, DiskInode, DirEntry};
39//!
40//! // 初始化超级块
41//! let mut sb = SuperBlock::default();
42//! sb.initialize(1024, 1, 8, 1, 100);
43//!
44//! // 创建文件 inode
45//! let mut inode = DiskInode::default();
46//! inode.initialize(DiskInodeType::File);
47//!
48//! // 创建目录项
49//! let entry = DirEntry::new("test.txt", 1);
50//! ```
51//!

56/// 文件系统魔数，用于标识 Easy File System
57const EFS_MAGIC: u32 = 0x3b800001;

59/// 文件名长度限制（不包括结尾的 null 字符）
60const NAME_LENGTH_LIMIT: usize = 27;

62/// 直接数据块数量
63const INODE_DIRECT_COUNT: usize = 28;
64
65/// 直接块边界
66const DIRECT_BOUND: usize = INODE_DIRECT_COUNT;
67
68/// 一级间接块能索引的数据块数量
69const INODE_INDIRECT1_COUNT: usize = BLOCK_SZ / 4;
70
71/// 一级间接块边界
72const INDIRECT1_BOUND: usize = DIRECT_BOUND + INODE_INDIRECT1_COUNT;
73
74/// 二级间接块能索引的数据块数量
75const INODE_INDIRECT2_COUNT: usize = INODE_INDIRECT1_COUNT * INODE_INDIRECT1_COUNT;
76
77/// 二级间接块边界
78const INDIRECT2_BOUND: usize = INDIRECT1_BOUND + INODE_INDIRECT2_COUNT;

80/// 目录项大小（字节）
81pub const DIRENT_SZ: usize = 32;

89/// 文件系统超级块
90///
91/// 存储文件系统的全局元数据，包括文件系统大小、各个区域的块数等信息。
92/// 超级块是文件系统的核心数据结构，用于文件系统的识别和初始化。
93///
94/// ## 布局结构
95///
96/// ```text
97/// ┌─────────┬─────────────────┬─────────────────┬─────────────────┬─────────────────┐
98/// │  Magic  │  Total Blocks   │  Inode Bitmap   │   Inode Area    │   Data Bitmap   │
99/// │  (4B)   │      (4B)       │      (4B)       │      (4B)       │       (4B)      │
100/// └─────────┴─────────────────┴─────────────────┴─────────────────┴─────────────────┘
101/// ```
102///
103/// ## 字段说明
104///
105/// - `magic` - 文件系统魔数，用于验证文件系统格式
106/// - `total_blocks` - 文件系统总块数
107/// - `inode_bitmap_blocks` - inode 位图占用的块数
108/// - `inode_area_blocks` - inode 区域占用的块数
109/// - `data_bitmap_blocks` - 数据位图占用的块数
110/// - `data_area_blocks` - 数据区域占用的块数
111#[repr(C)]
112pub struct SuperBlock {
113    magic: u32,
114    pub total_blocks: u32,
115    pub inode_bitmap_blocks: u32,
116    pub inode_area_blocks: u32,
117    pub data_bitmap_blocks: u32,
118    pub data_area_blocks: u32,
119}

121/// 磁盘 inode 结构
122///
123/// 存储文件或目录的元数据，包括文件大小、数据块分配信息等。
124/// 支持三级索引结构，能够管理大文件的数据块分配。
125///
126/// ## 数据块索引结构
127///
128/// ```text
129/// ┌───────────┬───────────────────┬────────────────┬────---───-─────┐
130/// │ Size (4B) │ Direct[28] (112B) │ Indirect1 (4B) │ Indirect2 (4B) │
131/// └───────────┴───────────────────┴────────────────┴───────---──────┘
132/// ```
133///
134/// ## 索引策略
135///
136/// - **直接块**：前 28 个数据块直接存储在 `direct` 数组中
137/// - **一级间接块**：通过 `indirect1` 指向的块存储块 ID 数组
138/// - **二级间接块**：通过 `indirect2` 指向的块存储一级间接块的块 ID 数组
139///
140/// ## 最大文件大小
141///
142/// 理论上支持的最大文件大小：
143/// - 直接块：28 × 512B = 14KB
144/// - 一级间接块：128 × 512B = 64KB
145/// - 二级间接块：128 × 128 × 512B = 8MB
146/// - 总计：约 8.1MB
147#[repr(C)]
148pub struct DiskInode {
149    pub size: u32,
150    pub direct: [u32; INODE_DIRECT_COUNT],
151    pub indirect1: u32,
152    pub indirect2: u32,
153    type_: DiskInodeType,
154}

156/// 磁盘 inode 类型
157///
158/// 标识 inode 对应的文件类型，用于区分普通文件和目录。
159#[derive(PartialEq)]
160pub enum DiskInodeType {
161    /// 普通文件
162    File,
163    /// 目录文件
164    Directory,
165}

241    /// 获取指定逻辑块号对应的物理块 ID
242    ///
243    /// 根据三级索引结构查找逻辑块号对应的物理块 ID。
244    /// 支持直接块、一级间接块和二级间接块的查找。
245    ///
246    /// ## Arguments
247    /// * `inner_id` - 逻辑块号（从 0 开始）
248    /// * `block_device` - 块设备引用
249    ///
250    /// ## Returns
251    /// 对应的物理块 ID
252    ///
253    /// ## 索引策略
254    /// - 0-27：直接块，从 `direct` 数组获取
255    /// - 28-1023：一级间接块，通过 `indirect1` 查找
256    /// - 1024+：二级间接块，通过 `indirect2` 查找
257    pub fn get_block_id(&self, inner_id: u32, block_device: &Arc<dyn BlockDevice>) -> u32 {
258        let inner_id = inner_id as usize;
259        if inner_id < INODE_DIRECT_COUNT {
260            self.direct[inner_id]
261        } else if inner_id < INDIRECT1_BOUND {
262            get_block_cache(self.indirect1 as usize, Arc::clone(block_device))
263                .lock()
264                .read(0, |indirect_block: &IndirectBlock| {
265                    indirect_block[inner_id - INODE_DIRECT_COUNT]
266                })
267        } else {
268            let last = inner_id - INDIRECT1_BOUND;
269            let indirect1 = get_block_cache(self.indirect2 as usize, Arc::clone(block_device))
270                .lock()
271                .read(0, |indirect2: &IndirectBlock| {
272                    indirect2[last / INODE_INDIRECT1_COUNT]
273                });
274            get_block_cache(indirect1 as usize, Arc::clone(block_device))
275                .lock()
276                .read(0, |indirect1: &IndirectBlock| {
277                    indirect1[last % INODE_INDIRECT1_COUNT]
278                })
279        }
280    }

282    /// 计算文件占用的数据块数量
283    ///

292    /// 根据文件大小计算数据块数量
293    ///
294    /// ## Arguments
295    /// * `size` - 文件大小（字节）
296    ///
297    /// ## Returns
298    /// 需要的数据块数量
299    fn _data_blocks(size: u32) -> u32 {
300        (size + BLOCK_SZ as u32 - 1) / BLOCK_SZ as u32
301    }

303    /// 计算文件占用的总块数（包括索引块）
304    ///
305    /// 计算文件占用的所有块数，包括数据块和索引块。
306    ///
307    /// ## Arguments
308    /// * `size` - 文件大小（字节）
309    ///
310    /// ## Returns
311    /// 文件占用的总块数
312    pub fn total_blocks(size: u32) -> u32 {
313        let data_blocks = Self::_data_blocks(size) as usize;
314        let mut total = data_blocks;
315        if data_blocks > INODE_DIRECT_COUNT {
316            total += 1;
317        }
318        if data_blocks > INDIRECT1_BOUND {
319            total += 1;
320            total +=
321                (data_blocks - INDIRECT1_BOUND + INODE_INDIRECT1_COUNT - 1) / INODE_INDIRECT1_COUNT;
322        }
323        total as u32
324    }

326    /// 计算扩展文件需要的新块数
327    ///
328    /// 计算将文件从当前大小扩展到新大小需要的新块数。
329    ///
330    /// ## Arguments
331    /// * `new_size` - 新的文件大小
332    ///

334    /// 需要的新块数

337    /// 如果 `new_size` 小于当前大小则 panic
338    pub fn blocks_num_needed(&self, new_size: u32) -> u32 {

341    }

343    /// 扩展文件大小并分配新块
344    ///
345    /// 将文件扩展到指定大小，并分配新的数据块。支持三级索引结构的
346    /// 块分配，自动处理直接块、一级间接块和二级间接块的分配。
347    ///
348    /// ## Arguments
349    /// * `new_size` - 新的文件大小
350    /// * `new_blocks` - 新分配的块 ID 列表
351    /// * `block_device` - 块设备引用
352    ///
353    /// ## 分配策略
354    /// 1. 优先使用直接块（前 28 个）
355    /// 2. 当直接块用完时，分配一级间接块
356    /// 3. 当一级间接块用完时，分配二级间接块
357    ///
358    /// ## 注意事项
359    /// - `new_blocks` 的长度必须等于 `blocks_num_needed(new_size)`
360    /// - 扩展操作是原子的，要么完全成功，要么完全失败
361    pub fn increase_size(
362        &mut self,
363        new_size: u32,
364        new_blocks: Vec<u32>,
365        block_device: &Arc<dyn BlockDevice>,
366    ) {
367        let mut current_blocks = self.data_blocks();
368        self.size = new_size;
369        let mut total_blocks = self.data_blocks();
370        let mut new_blocks = new_blocks.into_iter();
371
372        while current_blocks < total_blocks.min(INODE_DIRECT_COUNT as u32) {
373            self.direct[current_blocks as usize] = new_blocks.next().unwrap();
374            current_blocks += 1;
375        }
376
377        if total_blocks > INODE_DIRECT_COUNT as u32 {
378            if current_blocks == INODE_DIRECT_COUNT as u32 {
379                self.indirect1 = new_blocks.next().unwrap();
380            }
381            current_blocks -= INODE_DIRECT_COUNT as u32;
382            total_blocks -= INODE_DIRECT_COUNT as u32;
383        } else {
384            return;
385        }
386        get_block_cache(self.indirect1 as usize, Arc::clone(block_device))
387            .lock()
388            .modify(0, |indirect1: &mut IndirectBlock| {
389                while current_blocks < total_blocks.min(INODE_INDIRECT1_COUNT as u32) {
390                    indirect1[current_blocks as usize] = new_blocks.next().unwrap();
391                    current_blocks += 1;
392                }
393            });
394
395        if total_blocks > INODE_INDIRECT1_COUNT as u32 {
396            if current_blocks == INODE_INDIRECT1_COUNT as u32 {
397                self.indirect2 = new_blocks.next().unwrap();
398            }
399            current_blocks -= INODE_INDIRECT1_COUNT as u32;
400            total_blocks -= INODE_INDIRECT1_COUNT as u32;
401        } else {
402            return;
403        }
404        let mut a0 = current_blocks as usize / INODE_INDIRECT1_COUNT;
405        let mut b0 = current_blocks as usize % INODE_INDIRECT1_COUNT;
406        let a1 = total_blocks as usize / INODE_INDIRECT1_COUNT;
407        let b1 = total_blocks as usize % INODE_INDIRECT1_COUNT;
408        get_block_cache(self.indirect2 as usize, Arc::clone(block_device))
409            .lock()
410            .modify(0, |indirect2: &mut IndirectBlock| {
411                while (a0 < a1) || (a0 == a1 && b0 < b1) {
412                    if b0 == 0 {
413                        indirect2[a0] = new_blocks.next().unwrap();
414                    }
415                    get_block_cache(indirect2[a0] as usize, Arc::clone(block_device))
416                        .lock()
417                        .modify(0, |indirect1: &mut IndirectBlock| {
418                            indirect1[b0] = new_blocks.next().unwrap();
419                        });
420                    b0 += 1;
421                    if b0 == INODE_INDIRECT1_COUNT {
422                        b0 = 0;
423                        a0 += 1;
424                    }
425                }
426            });
427    }

429    /// 清空文件内容并回收所有数据块
430    ///
431    /// 将文件大小重置为 0，清空所有数据块引用，并返回所有被释放的块 ID。
432    /// 支持三级索引结构的块回收。
433    ///
434    /// ## Arguments
435    /// * `block_device` - 块设备引用
436    ///
437    /// ## Returns
438    /// 被释放的块 ID 列表
439    ///
440    /// ## 回收策略
441    /// 1. 回收直接块
442    /// 2. 回收一级间接块及其指向的数据块
443    /// 3. 回收二级间接块及其指向的所有块
444    pub fn clear_size(&mut self, block_device: &Arc<dyn BlockDevice>) -> Vec<u32> {
445        let mut v = Vec::new();
446        let mut data_blocks = self.data_blocks() as usize;
447        self.size = 0;
448        let mut current_blocks = 0usize;
449
450        while current_blocks < data_blocks.min(INODE_DIRECT_COUNT) {
451            v.push(self.direct[current_blocks]);
452            self.direct[current_blocks] = 0;
453            current_blocks += 1;
454        }
455
456        if data_blocks > INODE_DIRECT_COUNT {
457            v.push(self.indirect1);
458            data_blocks -= INODE_DIRECT_COUNT;
459            current_blocks = 0;
460        } else {
461            return v;
462        }
463        get_block_cache(self.indirect1 as usize, Arc::clone(block_device))
464            .lock()
465            .modify(0, |indirect1: &mut IndirectBlock| {
466                while current_blocks < data_blocks.min(INODE_INDIRECT1_COUNT) {
467                    v.push(indirect1[current_blocks]);
468                    current_blocks += 1;
469                }
470            });
471        self.indirect1 = 0;
472
473        if data_blocks > INODE_INDIRECT1_COUNT {
474            v.push(self.indirect2);
475            data_blocks -= INODE_INDIRECT1_COUNT;
476        } else {
477            return v;
478        }
479        assert!(data_blocks <= INODE_INDIRECT2_COUNT);
480        let a1 = data_blocks / INODE_INDIRECT1_COUNT;
481        let b1 = data_blocks % INODE_INDIRECT1_COUNT;
482        get_block_cache(self.indirect2 as usize, Arc::clone(block_device))
483            .lock()
484            .modify(0, |indirect2: &mut IndirectBlock| {
485                for entry in indirect2.iter_mut().take(a1) {
486                    v.push(*entry);
487                    get_block_cache(*entry as usize, Arc::clone(block_device))
488                        .lock()
489                        .modify(0, |indirect1: &mut IndirectBlock| {
490                            for entry in indirect1.iter() {
491                                v.push(*entry);
492                            }
493                        });
494                }
495                if b1 > 0 {
496                    v.push(indirect2[a1]);
497                    get_block_cache(indirect2[a1] as usize, Arc::clone(block_device))
498                        .lock()
499                        .modify(0, |indirect1: &mut IndirectBlock| {
500                            for entry in indirect1.iter().take(b1) {
501                                v.push(*entry);
502                            }
503                        });
504                }
505            });
506        self.indirect2 = 0;
507        v
508    }

510    /// 从指定偏移量读取文件数据
511    ///
512    /// 从文件的指定偏移量开始读取数据到缓冲区中。支持跨块读取，
513    /// 自动处理块边界和数据对齐。
514    ///
515    /// ## Arguments
516    /// * `offset` - 读取起始偏移量
517    /// * `buf` - 数据缓冲区
518    /// * `block_device` - 块设备引用
519    ///
520    /// ## Returns
521    /// 实际读取的字节数
522    ///
523    /// ## 读取策略
524    /// 1. 计算起始块号和结束块号
525    /// 2. 逐块读取数据
526    /// 3. 处理块内偏移和跨块读取
527    ///
528    /// ## 边界处理
529    /// - 如果偏移量超出文件大小，返回 0
530    /// - 如果缓冲区大小超出文件剩余部分，只读取到文件末尾
531    pub fn read_at(
532        &self,
533        offset: usize,
534        buf: &mut [u8],
535        block_device: &Arc<dyn BlockDevice>,
536    ) -> usize {
537        let mut start = offset;
538        let end = (offset + buf.len()).min(self.size as usize);
539        if start >= end {
540            return 0;
541        }
542
543        let mut start_block = start / BLOCK_SZ;
544        let mut read_size = 0usize;
545        loop {
546            let mut end_current_block = (start / BLOCK_SZ + 1) * BLOCK_SZ;
547            end_current_block = end_current_block.min(end);
548
549            let block_read_size = end_current_block - start;
550            let dst = &mut buf[read_size..read_size + block_read_size];
551            get_block_cache(
552                self.get_block_id(start_block as u32, block_device) as usize,
553                Arc::clone(block_device),
554            )
555            .lock()
556            .read(0, |data_block: &DataBlock| {
557                let src = &data_block[start % BLOCK_SZ..start % BLOCK_SZ + block_read_size];
558                dst.copy_from_slice(src);
559            });
560            read_size += block_read_size;
561            if end_current_block == end {
562                break;
563            }
564            start_block += 1;
565            start = end_current_block;
566        }
567        read_size
568    }

570    /// 向指定偏移量写入文件数据
571    ///
572    /// 将缓冲区中的数据写入文件的指定偏移量位置。支持跨块写入，
573    /// 自动处理块边界和数据对齐。
574    ///
575    /// ## Arguments
576    /// * `offset` - 写入起始偏移量
577    /// * `buf` - 要写入的数据
578    /// * `block_device` - 块设备引用
579    ///
580    /// ## Returns
581    /// 实际写入的字节数
582    ///
583    /// ## 写入策略
584    /// 1. 计算起始块号和结束块号
585    /// 2. 逐块写入数据
586    /// 3. 处理块内偏移和跨块写入
587    ///
588    /// ## 注意事项
589    /// - 写入操作会修改文件内容
590    /// - 如果写入位置超出当前文件大小，需要先扩展文件
591    /// - 写入是原子的，要么完全成功，要么完全失败
592    pub fn write_at(
593        &mut self,
594        offset: usize,
595        buf: &[u8],
596        block_device: &Arc<dyn BlockDevice>,
597    ) -> usize {
598        let mut start = offset;
599        let end = (offset + buf.len()).min(self.size as usize);
600        assert!(start <= end);
601
602        let mut start_block = start / BLOCK_SZ;
603        let mut write_size = 0usize;
604
605        loop {
606            let mut end_current_block = (start / BLOCK_SZ + 1) * BLOCK_SZ;
607            end_current_block = end_current_block.min(end);
608
609            let block_write_size = end_current_block - start;
610            get_block_cache(
611                self.get_block_id(start_block as u32, block_device) as usize,
612                Arc::clone(block_device),
613            )
614            .lock()
615            .modify(0, |data_block: &mut DataBlock| {
616                let src = &buf[write_size..write_size + block_write_size];
617                let dst = &mut data_block[start % BLOCK_SZ..start % BLOCK_SZ + block_write_size];
618                dst.copy_from_slice(src);
619            });
620            write_size += block_write_size;
621            if end_current_block == end {
622                break;
623            }
624            start_block += 1;
625            start = end_current_block;
626        }
627        write_size
628    }
629}

631/// 目录项结构
632///
633/// 用于在目录文件中存储文件和子目录的信息。每个目录项包含文件名
634/// 和对应的 inode 号，支持目录的层次结构组织。
635///
636/// ## 布局结构
637///
638/// ```text
639/// ┌────────────┬─────────────┐
640/// │ Name (28B) │ Inode (4B)  │
641/// └────-───────┴─────────────┘
642/// ```
643///
644/// ## 字段说明
645///
646/// - `name` - 文件名，固定长度 28 字节，以 null 结尾
647/// - `inode_number` - 对应的 inode 号
648///
649/// ## 大小限制
650///
651/// - 文件名最大长度：27 个字符（不包括结尾的 null）
652/// - 目录项总大小：32 字节
653#[repr(C)]
654pub struct DirEntry {
655    name: [u8; NAME_LENGTH_LIMIT + 1],
656    inode_number: u32,
657}
658
659impl DirEntry {
660    /// 创建空的目录项
661    ///
662    /// 创建一个未使用的目录项，文件名为空，inode 号为 0。
663    ///
664    /// ## Returns
665    /// 空的目录项
666    pub fn empty() -> Self {
667        Self {
668            name: [0u8; NAME_LENGTH_LIMIT + 1],
669            inode_number: 0,
670        }
671    }
672
673    /// 创建新的目录项
674    ///
675    /// 使用指定的文件名和 inode 号创建目录项。
676    ///
677    /// ## Arguments
678    /// * `name` - 文件名
679    /// * `inode_number` - 对应的 inode 号
680    ///
681    /// ## 注意事项
682    /// - 文件名长度不能超过 27 个字符
683    /// - 如果文件名过长，会被截断
684    pub fn new(name: &str, inode_number: u32) -> Self {
685        let mut bytes = [0u8; NAME_LENGTH_LIMIT + 1];
686        bytes[..name.len()].copy_from_slice(name.as_bytes());
687        Self {
688            name: bytes,
689            inode_number,
690        }
691    }
692
693    /// 获取目录项的字节表示
694    ///
695    /// 将目录项转换为字节切片，用于磁盘存储。
696    ///
697    /// ## Returns
698    /// 目录项的字节表示
699    ///
700    /// ## Safety
701    /// 返回的字节切片直接对应结构体的内存布局
702    pub fn as_bytes(&self) -> &[u8] {
703        unsafe { core::slice::from_raw_parts(self as *const _ as usize as *const u8, DIRENT_SZ) }
704    }
705
706    /// 获取目录项的可变字节表示
707    ///
708    /// 将目录项转换为可变字节切片，用于磁盘写入。
709    ///
710    /// ## Returns
711    /// 目录项的可变字节表示
712    ///
713    /// ## Safety
714    /// 返回的字节切片直接对应结构体的内存布局
715    pub fn as_bytes_mut(&mut self) -> &mut [u8] {
716        unsafe { core::slice::from_raw_parts_mut(self as *mut _ as usize as *mut u8, DIRENT_SZ) }
717    }
718
719    /// 获取文件名
720    ///
721    /// 从目录项中提取文件名，自动处理 null 结尾。
722    ///
723    /// ## Returns
724    /// 文件名字符串
725    ///
726    /// ## 注意事项
727    /// - 文件名以 null 字符结尾
728    /// - 返回的字符串不包含 null 字符
729    pub fn name(&self) -> &str {
730        let len = (0usize..).find(|i| self.name[*i] == 0).unwrap();
731        core::str::from_utf8(&self.name[..len]).unwrap()
732    }
733
734    /// 获取 inode 号
735    ///
736    /// ## Returns
737    /// 对应的 inode 号
738    pub fn inode_number(&self) -> u32 {
739        self.inode_number
740    }
741}
