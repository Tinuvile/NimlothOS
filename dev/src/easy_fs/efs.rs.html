1//! # Easy File System 核心模块
2//!
3//! 实现了简单文件系统（Easy File System）的核心功能，包括文件系统的创建、
4//! 打开、inode 和数据块的管理等。该模块是文件系统的核心，负责协调各个
5//! 组件的工作。
6//!
7//! ## 文件系统结构
8//!
9//! ```text
10//! ┌─────────────────────────────────────────────────────────────┐
11//! │                        SuperBlock                           │
12//! ├─────────────────────────────────────────────────────────────┤
13//! │                       Inode Bitmap                          │
14//! ├─────────────────────────────────────────────────────────────┤
15//! │                        Inode Area                           │
16//! ├─────────────────────────────────────────────────────────────┤
17//! │                        Data Bitmap                          │
18//! ├─────────────────────────────────────────────────────────────┤
19//! │                         Data Area                           │
20//! └─────────────────────────────────────────────────────────────┘
21//! ```
22//!
23//! ## 核心组件
24//!
25//! - [`EasyFileSystem`] - 文件系统主体，管理所有文件系统操作
26//! - 位图管理：inode 位图和数据位图，用于资源分配
27//! - 区域管理：inode 区域和数据区域的位置计算
28//!
29//! ## 主要功能
30//!
31//! - **文件系统创建**：格式化块设备，创建文件系统结构
32//! - **文件系统打开**：从现有块设备加载文件系统
33//! - **资源分配**：inode 和数据块的分配与回收
34//! - **根目录管理**：提供根目录的访问接口
35//!
36//! ## 使用示例
37//!
38//! ```rust
39//! use easy_fs::{EasyFileSystem, BlockDevice};
40//!
41//! // 创建新的文件系统
42//! let efs = EasyFileSystem::create(block_device, 1024, 1);
43//!
44//! // 打开现有文件系统
45//! let efs = EasyFileSystem::open(block_device);
46//!
47//! // 获取根目录
48//! let root_inode = EasyFileSystem::root_inode(&efs);
49//! ```
50//!

60/// Easy File System 主体结构
61///
62/// 管理整个文件系统的状态和操作，包括位图管理、区域定位、资源分配等。
63/// 该结构是文件系统的核心，所有文件系统操作都通过它进行。
64///
65/// ## 字段说明
66///
67/// - `block_device` - 底层块设备，提供存储能力
68/// - `inode_bitmap` - inode 位图，管理 inode 的分配状态
69/// - `data_bitmap` - 数据位图，管理数据块的分配状态
70/// - `inode_area_start_block` - inode 区域的起始块号
71/// - `data_area_start_block` - 数据区域的起始块号
72///
73/// ## 生命周期
74///
75/// 文件系统的生命周期包括：
76/// 1. **创建阶段**：格式化块设备，初始化所有数据结构
77/// 2. **运行阶段**：处理文件操作，管理资源分配
78/// 3. **关闭阶段**：同步缓存，确保数据持久化
79pub struct EasyFileSystem {
80    pub block_device: Arc<dyn BlockDevice>,
81    pub inode_bitmap: Bitmap,
82    pub data_bitmap: Bitmap,
83    inode_area_start_block: u32,
84    data_area_start_block: u32,
85}

87impl EasyFileSystem {
88    /// 创建新的文件系统
89    ///
90    /// 在指定的块设备上创建并格式化一个新的 Easy File System。
91    /// 该操作会清空块设备上的所有数据，并初始化文件系统结构。
92    ///
93    /// ## Arguments
94    /// * `block_device` - 要格式化的块设备
95    /// * `total_blocks` - 文件系统总块数
96    /// * `inode_bitmap_blocks` - inode 位图占用的块数
97    ///
98    /// ## 创建过程
99    /// 1. 计算各个区域的大小和位置
100    /// 2. 初始化位图结构
101    /// 3. 清空所有数据块
102    /// 4. 创建超级块
103    /// 5. 创建根目录 inode
104    /// 6. 同步所有缓存到磁盘
105    ///
106    /// ## Returns
107    /// 新创建的文件系统实例，包装在 `Arc<Mutex<>>` 中以支持并发访问
108    ///
109    /// ## 注意事项
110    /// - 此操作会清空块设备上的所有现有数据
111    /// - 创建完成后会自动分配根目录 inode（ID 为 0）
112    /// - 文件系统创建后立即可用
113    pub fn create(
114        block_device: Arc<dyn BlockDevice>,
115        total_blocks: u32,
116        inode_bitmap_blocks: u32,
117    ) -> Arc<Mutex<Self>> {
118        let inode_bitmap = Bitmap::new(1, inode_bitmap_blocks as usize);
119        let inode_num = inode_bitmap.maximum();
120        let inode_area_blocks =
121            ((inode_num * core::mem::size_of::<DiskInode>() + BLOCK_SZ - 1) / BLOCK_SZ) as u32;
122        let inode_total_blocks = inode_bitmap_blocks + inode_area_blocks;
123        let data_total_blocks = total_blocks - 1 - inode_total_blocks;
124        let data_bitmap_blocks = (data_total_blocks + 4096) / 4097;
125        let data_area_blocks = data_total_blocks - data_bitmap_blocks;
126        let data_bitmap = Bitmap::new(
127            (1 + inode_total_blocks) as usize,
128            data_bitmap_blocks as usize,
129        );
130        let mut efs = Self {
131            block_device: Arc::clone(&block_device),
132            inode_bitmap,
133            data_bitmap,
134            inode_area_start_block: 1 + inode_bitmap_blocks,
135            data_area_start_block: 1 + inode_total_blocks + data_bitmap_blocks,
136        };
137        for i in 0..total_blocks {
138            get_block_cache(i as usize, Arc::clone(&block_device))
139                .lock()
140                .modify(0, |data_block: &mut DataBlock| {
141                    for byte in data_block.iter_mut() {
142                        *byte = 0;
143                    }
144                });
145        }
146        get_block_cache(0, Arc::clone(&block_device)).lock().modify(
147            0,
148            |super_block: &mut SuperBlock| {
149                super_block.initialize(
150                    total_blocks,
151                    inode_bitmap_blocks,
152                    inode_area_blocks,
153                    data_bitmap_blocks,
154                    data_area_blocks,
155                );
156            },
157        );
158        assert_eq!(efs.alloc_inode(), 0);
159        let (root_inode_block_id, root_inode_offset) = efs.get_disk_inode_pos(0);
160        get_block_cache(root_inode_block_id as usize, Arc::clone(&block_device))
161            .lock()
162            .modify(root_inode_offset, |disk_inode: &mut DiskInode| {
163                disk_inode.initialize(DiskInodeType::Directory);
164            });
165        block_cache_sync_all();
166        Arc::new(Mutex::new(efs))
167    }

169    /// 打开现有文件系统
170    ///
171    /// 从指定的块设备上加载已存在的 Easy File System。
172    /// 该操作会读取超级块信息，并恢复文件系统的状态。
173    ///
174    /// ## Arguments
175    /// * `block_device` - 包含文件系统的块设备
176    ///
177    /// ## 加载过程
178    /// 1. 读取块设备第 0 块的超级块
179    /// 2. 验证文件系统魔数
180    /// 3. 根据超级块信息重建位图结构
181    /// 4. 计算各个区域的位置
182    ///
183    /// ## Returns
184    /// 加载的文件系统实例，包装在 `Arc<Mutex<>>` 中以支持并发访问
185    ///
186    /// ## Panics
187    /// 如果块设备不包含有效的 Easy File System 则 panic
188    ///
189    /// ## 注意事项
190    /// - 块设备必须包含有效的文件系统
191    /// - 加载过程不会修改块设备上的数据
192    /// - 文件系统加载后立即可用
193    pub fn open(block_device: Arc<dyn BlockDevice>) -> Arc<Mutex<Self>> {
194        get_block_cache(0, Arc::clone(&block_device))
195            .lock()
196            .read(0, |super_block: &SuperBlock| {
197                assert!(super_block.is_valid(), "Error loading EFS!");
198                let inode_total_blocks =
199                    super_block.inode_bitmap_blocks + super_block.inode_area_blocks;
200                let efs = Self {
201                    block_device,
202                    inode_bitmap: Bitmap::new(1, super_block.inode_bitmap_blocks as usize),
203                    data_bitmap: Bitmap::new(
204                        (1 + inode_total_blocks) as usize,
205                        super_block.data_bitmap_blocks as usize,
206                    ),
207                    inode_area_start_block: 1 + super_block.inode_bitmap_blocks,
208                    data_area_start_block: 1 + inode_total_blocks + super_block.data_bitmap_blocks,
209                };
210                Arc::new(Mutex::new(efs))
211            })
212    }

214    /// 获取指定 inode 在磁盘上的位置
215    ///
216    /// 根据 inode ID 计算对应的磁盘 inode 在块设备上的具体位置。
217    /// 返回块号和块内偏移量。
218    ///
219    /// ## Arguments
220    /// * `inode_id` - inode 的 ID
221    ///
222    /// ## Returns
223    /// `(block_id, offset)` 元组，表示 inode 所在的块号和块内偏移量
224    ///
225    /// ## 计算原理
226    /// - 每个块可以存储多个 inode（取决于 `DiskInode` 的大小）
227    /// - 块号 = inode 区域起始块号 + inode_id / 每块 inode 数量
228    /// - 偏移量 = (inode_id % 每块 inode 数量) × inode 大小
229    pub fn get_disk_inode_pos(&self, inode_id: u32) -> (u32, usize) {
230        let inode_size = core::mem::size_of::<DiskInode>();
231        let inodes_per_block = (BLOCK_SZ / inode_size) as u32;
232        let block_id = self.inode_area_start_block + inode_id / inodes_per_block;
233        (
234            block_id,
235            (inode_id % inodes_per_block) as usize * inode_size,
236        )
237    }

239    /// 获取数据块在磁盘上的实际块号
240    ///
241    /// 将逻辑数据块 ID 转换为在块设备上的实际块号。
242    /// 数据块 ID 是相对于数据区域的，需要加上数据区域的起始块号。
243    ///
244    /// ## Arguments
245    /// * `data_block_id` - 逻辑数据块 ID
246    ///
247    /// ## Returns
248    /// 数据块在块设备上的实际块号
249    pub fn get_data_block_id(&self, data_block_id: u32) -> u32 {
250        self.data_area_start_block + data_block_id
251    }

253    /// 分配一个新的 inode
254    ///
255    /// 从 inode 位图中分配一个可用的 inode ID。
256    /// 该操作会更新位图并返回新分配的 inode ID。
257    ///
258    /// ## Returns
259    /// 新分配的 inode ID
260    ///
261    /// ## Panics
262    /// 如果没有可用的 inode 则 panic
263    ///
264    /// ## 注意事项
265    /// - 分配的 inode 需要后续初始化才能使用
266    /// - inode ID 从 0 开始连续分配
267    /// - 根目录的 inode ID 固定为 0
268    pub fn alloc_inode(&mut self) -> u32 {
269        self.inode_bitmap.alloc(&self.block_device).unwrap() as u32
270    }

272    /// 分配一个新的数据块
273    ///
274    /// 从数据位图中分配一个可用的数据块，并返回其在块设备上的实际块号。
275    /// 该操作会更新位图并返回新分配的数据块号。
276    ///
277    /// ## Returns
278    /// 新分配的数据块在块设备上的实际块号
279    ///
280    /// ## Panics
281    /// 如果没有可用的数据块则 panic
282    ///
283    /// ## 注意事项
284    /// - 返回的是绝对块号，可以直接用于块设备操作
285    /// - 分配的数据块内容未初始化
286    /// - 数据块分配后需要手动初始化内容
287    pub fn alloc_data(&mut self) -> u32 {
288        self.data_bitmap.alloc(&self.block_device).unwrap() as u32 + self.data_area_start_block
289    }

327    /// 释放指定的数据块
328    ///
329    /// 将指定的数据块标记为可用，并清空其内容。
330    /// 该操作会更新位图并重置数据块内容。
331    ///
332    /// ## Arguments
333    /// * `block_id` - 要释放的数据块在块设备上的实际块号
334    ///
335    /// ## 操作过程
336    /// 1. 清空数据块的所有字节为 0
337    /// 2. 更新数据位图，标记块为可用
338    /// 3. 将绝对块号转换为相对块号进行位图操作
339    ///
340    /// ## 注意事项
341    /// - `block_id` 必须是绝对块号（相对于整个块设备）
342    /// - 释放操作会清空数据块内容
343    /// - 释放后的数据块可以重新分配使用
344    pub fn dealloc_data(&mut self, block_id: u32) {
345        get_block_cache(block_id as usize, Arc::clone(&self.block_device))
346            .lock()
347            .modify(0, |data_block: &mut DataBlock| {
348                data_block.iter_mut().for_each(|p| {
349                    *p = 0;
350                })
351            });
352        self.data_bitmap.dealloc(
353            &self.block_device,
354            (block_id - self.data_area_start_block) as usize,
355        )
356    }

358    /// 获取文件系统的根目录 inode
359    ///
360    /// 创建并返回根目录的 inode 对象，提供对根目录的访问接口。
361    /// 根目录是文件系统的入口点，所有其他文件和目录都从根目录开始访问。
362    ///
363    /// ## Arguments
364    /// * `efs` - 文件系统实例的引用
365    ///
366    /// ## Returns
367    /// 根目录的 inode 对象
368    ///
369    /// ## 根目录特性
370    /// - 根目录的 inode ID 固定为 0
371    /// - 根目录在文件系统创建时自动初始化
372    /// - 根目录始终存在且可访问
373    /// - 根目录支持子文件和子目录的创建
374    ///
375    /// ## 使用示例
376    /// ```rust
377    /// let root = EasyFileSystem::root_inode(&efs);
378    /// let file = root.create("test.txt").unwrap();
379    /// ```
380    pub fn root_inode(efs: &Arc<Mutex<Self>>) -> Inode {
381        let block_device = Arc::clone(&efs.lock().block_device);
382        let (block_id, block_offset) = efs.lock().get_disk_inode_pos(0);
383        Inode::new(block_id, block_offset, Arc::clone(efs), block_device)
384    }
