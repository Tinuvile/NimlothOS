1//! # SBI (Supervisor Binary Interface) 调用模块
2//!
3//! 提供与 RISC-V SBI 固件的交互接口，实现操作系统内核与底层固件之间的通信。
4//! SBI 是 RISC-V 架构中监督者模式软件与机器模式固件之间的标准接口。
5//!
6//! ## 支持的 SBI 扩展
7//!
8//! ### Legacy 扩展 (0.1 版本)
9//! - 控制台输入输出
10//! - 时钟设置
11//! - IPI (处理器间中断)
12//! - TLB 管理
13//! - 系统关闭
14//!
15//! ### 系统复位扩展 (SRST)
16//! - 系统关闭和重启功能
17//! - 支持不同的复位类型和原因
18//!
19//! ## 调用约定
20//!
21//! SBI 调用使用 `ecall` 指令从 S 模式陷入 M 模式：
22//! - `a7` 寄存器：扩展 ID (EID)
23//! - `a6` 寄存器：函数 ID (FID)  
24//! - `a0-a2` 寄存器：函数参数
25//! - 返回值通过 `a0` (错误码) 和 `a1` (返回值) 寄存器传递
26

63/// SBI 调用返回值结构
64///
65/// 封装 SBI 调用的返回信息，包含错误码和实际返回值。
66/// 遵循 SBI 规范的返回值约定。
67#[derive(Debug, Clone, Copy)]
68#[allow(dead_code)]
69pub struct SbiRet {
70    /// 错误码，0 表示成功，负值表示错误
71    pub error: usize,
72    /// 实际返回值，具体含义取决于调用的 SBI 函数
73    pub value: usize,
74}

76/// 系统复位扩展 ID (SRST)
77///
78/// SRST 扩展提供标准化的系统复位功能，支持关闭和重启。
79/// 扩展 ID 为 ASCII "SRST" 的数值表示。
80const SRST_EXTENSION: usize = 0x53525354;
81
82/// 系统复位函数 ID
83const SBI_SYSTEM_RESET: usize = 0;

85/// 系统复位类型枚举
86///
87/// 定义不同类型的系统复位操作，用于指示固件执行何种复位行为。
88#[repr(usize)]
89#[allow(dead_code)]
90enum SystemResetType {
91    /// 正常关闭系统
92    Shutdown = 0,
93    /// 冷重启 - 完全断电重启
94    ColdReboot = 1,
95    /// 热重启 - 不断电重启
96    WarmReboot = 2,
97}

99/// 系统复位原因枚举
100///
101/// 说明执行系统复位的原因，用于日志记录和故障分析。
102#[repr(usize)]
103#[allow(dead_code)]
104enum SystemResetReason {
105    /// 无特定原因
106    NoReason = 0,
107    /// 系统故障导致的复位
108    SystemFailure = 1,
109}

111/// 执行 SBI 调用
112///
113/// 这是所有 SBI 接口的底层实现，通过 `ecall` 指令从监督者模式
114/// 陷入机器模式，调用 SBI 固件提供的服务。
115///
116/// ## Arguments
117///
118/// * `eid` - 扩展 ID，标识要调用的 SBI 扩展
119/// * `fid` - 函数 ID，标识扩展内的具体函数
120/// * `arg0` - 第一个参数
121/// * `arg1` - 第二个参数  
122/// * `arg2` - 第三个参数
123///
124/// ## Returns
125///
126/// 返回 [`SbiRet`] 结构，包含错误码和返回值
127///
128/// ## Safety
129///
130/// 使用内联汇编执行 `ecall` 指令，假设 SBI 固件正确实现。
131///
132/// ## 寄存器约定
133///
134/// - `a7` (x17): 扩展 ID
135/// - `a6` (x16): 函数 ID
136/// - `a0-a2` (x10-x12): 函数参数
137/// - 返回：`a0` 为错误码，`a1` 为返回值
138#[inline(always)]
139fn sbi_call(eid: usize, fid: usize, arg0: usize, arg1: usize, arg2: usize) -> SbiRet {
140    let (error, value);
141    unsafe {
142        core::arch::asm!(
143            "ecall",
144            inlateout("a0") arg0 => error,
145            inlateout("a1") arg1 => value,
146            in("a2") arg2,
147            in("a6") fid,
148            in("a7") eid,
149            options(nostack, preserves_flags)
150        );
151    }
152    SbiRet { error, value }
153}

155/// 控制台字符输出
156///
157/// 通过 SBI 接口向控制台输出单个字符，是控制台输出的基础函数。
158///
159/// ## Arguments
160///
161/// * `c` - 要输出的字符，以 `usize` 形式传递（实际为 ASCII 码值）
162///
163/// ## Usage
164///
165/// ```rust
166/// console_putchar('H' as usize);  // 输出字符 'H'
167/// console_putchar(65);            // 输出字符 'A' (ASCII 65)
168/// ```
169///
170/// ## Note
171///
172/// - 该函数是同步的，会阻塞直到字符输出完成
173/// - 不处理换行符的特殊转换
174/// - 是 `print!` 和 `println!` 宏的底层实现
175pub fn console_putchar(c: usize) {
176    sbi_call(SBI_CONSOLE_PUTCHAR, 0, c, 0, 0);
177}

179/// 系统关闭
180///
181/// 通过 SBI 系统复位扩展安全关闭系统。该函数会请求固件关闭系统，
182/// 正常情况下不会返回。
183///
184/// ## 关闭流程
185///
186/// 1. 调用 SRST 扩展的系统复位函数
187/// 2. 指定复位类型为关闭 (`Shutdown`)
188/// 3. 指定复位原因为无特定原因 (`NoReason`)
189/// 4. 固件执行关闭操作
190///
191/// ## Returns
192///
193/// 该函数的返回类型为 `!`（never type），表示正常情况下不会返回。
194/// 如果 SBI 调用失败或固件不支持关闭功能，会触发 panic。
195///
196/// ## Panics
197///
198/// 如果系统未能正常关闭（SBI 调用返回），会触发 panic 以确保系统不会
199/// 处于未定义状态。
200///
201/// ## Usage
202///
203/// ```rust
204/// // 在 panic 处理程序中关闭系统
205/// shutdown(); // 此后不会执行到任何代码
206/// ```
207pub fn shutdown() -> ! {
208    let _ = sbi_call(
209        SRST_EXTENSION,
210        SBI_SYSTEM_RESET,
211        SystemResetType::Shutdown as usize,
212        SystemResetReason::NoReason as usize,
213        0,
214    );
215    panic!("It should have shutdown !")
216}

218/// 设置时钟中断触发时间
219///
220/// 通过 SBI 接口配置时钟中断在指定时间触发，用于实现抢占式任务调度。
221/// 时间值基于系统的时钟周期计数。
222///
223/// ## Arguments
224///
225/// * `time` - 时钟中断触发的绝对时间，单位为时钟周期
226///
227/// ## 工作原理
228///
229/// 1. SBI 固件会将传入的时间值与当前时钟进行比较
230/// 2. 当系统时钟达到指定时间时，触发时钟中断
231/// 3. 中断会导致 CPU 跳转到中断处理程序
232///
233/// ## Usage
234///
235/// ```rust
236/// let current_time = get_time();
237/// let next_trigger = current_time + CLOCK_FREQ / 100;  // 10ms 后触发
238/// set_timer(next_trigger);
239/// ```
240///
241/// ## Note
242///
243/// - 时间值必须是绝对时间，不是相对时间
244/// - 如果指定的时间已经过去，中断会立即触发
245/// - 每次处理时钟中断后都需要重新设置下次触发时间
246pub fn set_timer(time: usize) {
247    sbi_call(SBI_SET_TIMER, 0, time, 0, 0);
248}
