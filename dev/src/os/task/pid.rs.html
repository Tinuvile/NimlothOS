1//! # 进程 ID 和内核栈管理模块
2//!
3//! 本模块提供了进程标识符 (PID) 的分配、回收以及内核栈的生命周期管理功能。
4//! 是操作系统任务管理的核心基础设施之一。
5//!
6//! ## 核心功能
7//!
8//! ### 进程 ID 管理
9//! - **PID 分配**: 为新创建的进程分配唯一的数字标识符
10//! - **PID 回收**: 回收已终止进程的 PID，实现资源复用
11//! - **重复分配检测**: 防止同一 PID 被重复回收，确保系统一致性
12//! - **RAII 管理**: 通过 `PidHandle` 自动管理 PID 的生命周期
13//!
14//! ### 内核栈管理
15//! - **栈空间分配**: 为每个进程分配独立的内核栈空间
16//! - **虚拟内存映射**: 将内核栈映射到内核地址空间
17//! - **自动清理**: 进程销毁时自动回收内核栈资源
18//! - **栈顶操作**: 支持在内核栈顶部压入数据
19//!
20//! ## 设计原理
21//!
22//! ### PID 分配策略
23//! 采用混合分配策略，优先复用已回收的 PID：
24//! 1. 检查回收池中是否有可用的 PID
25//! 2. 如果回收池为空，则分配新的递增 PID
26//! 3. 通过 `PidHandle` 提供 RAII 风格的自动回收
27//!
28//! ### 内核栈布局
29//! 每个进程的内核栈在虚拟地址空间中的布局：
30//!
31//! ```text
32//! 高地址 TRAMPOLINE (0x3ffffff000)
33//!         ↓
34//!     ┌─────────────────────────────────┐
35//!     │        Guard Page               │ ← 4KB 保护页
36//!     ├─────────────────────────────────┤
37//!     │     Process 0 Kernel Stack      │ ← 8KB 栈空间
38//!     ├─────────────────────────────────┤
39//!     │        Guard Page               │ ← 4KB 保护页
40//!     ├─────────────────────────────────┤
41//!     │     Process 1 Kernel Stack      │ ← 8KB 栈空间
42//!     ├─────────────────────────────────┤
43//!     │           ...                   │
44//!         ↓
45//! 低地址
46//! ```
47//!
48//! ## 安全保证
49//!
50//! - **内存安全**: 所有内存操作都经过虚拟内存管理器验证
51//! - **PID 唯一性**: 防止 PID 重复分配和重复回收
52//! - **栈隔离**: 每个进程的内核栈完全独立，避免相互干扰
53//! - **自动清理**: 通过 RAII 确保资源不会泄漏
54//!
55//! ## 使用示例
56//!
57//! ```rust
58//! // 分配新的进程 ID
59//! let pid_handle = pid_alloc();
60//! println!("Allocated PID: {}", pid_handle.0);
61//!
62//! // 创建对应的内核栈
63//! let kernel_stack = KernelStack::new(&pid_handle);
64//!
65//! // 在栈顶压入数据
66//! let trap_context_ptr = kernel_stack.push_on_top(TrapContext::default());
67//!
68//! // 变量离开作用域时，PID 和内核栈会自动回收
69//! ```

80/// 进程 ID 句柄
81///
82/// 对进程标识符的封装，提供 RAII 风格的自动生命周期管理。当 `PidHandle`
83/// 离开作用域时，会自动将持有的 PID 归还给分配器，实现资源的自动回收。
84///
85/// ## 设计特性
86///
87/// - **唯一所有权**: 每个 `PidHandle` 唯一拥有一个 PID
88/// - **移动语义**: 支持所有权转移，避免意外的重复释放
89/// - **自动回收**: 通过 `Drop` trait 实现自动资源管理
90/// - **零开销抽象**: 编译时优化为原始 usize 访问
91///
92/// ## 内部表示
93///
94/// 内部存储一个 `usize` 类型的 PID 值，该值保证在系统范围内唯一。
95/// PID 从 0 开始分配，依次递增，回收的 PID 会被优先重新分配。
96///
97/// ## Examples
98///
99/// ```rust
100/// // 分配新的 PID
101/// let pid_handle = pid_alloc();
102/// let pid_value = pid_handle.0;
103///
104/// // 转移所有权
105/// let task = Task::new(pid_handle);
106///
107/// // 当 task 被销毁时，PID 会自动回收
108/// ```
109pub struct PidHandle(pub usize);
110
111/// 进程 ID 分配器
112///
113/// 管理系统中所有进程 ID 的分配和回收，采用高效的混合分配策略。
114/// 优先复用已回收的 PID，在回收池为空时分配新的递增 PID。
115///
116/// ## 分配策略
117///
118/// ### 分配顺序
119/// 1. **优先回收**: 首先检查回收池中是否有可用的 PID
120/// 2. **递增分配**: 回收池为空时，分配新的连续 PID
121/// 3. **唯一性保证**: 确保系统中不存在重复的活跃 PID
122///
123/// ### 数据结构
124/// - `current`: 下一个要分配的新 PID 值（单调递增）
125/// - `recycled`: 已回收的 PID 池，使用向量实现的栈结构
126///
127/// ## 算法复杂度
128///
129/// - **分配操作**: O(1) - 无论是从回收池还是新分配
130/// - **回收操作**: O(n) - 需要检查重复回收（n 为回收池大小）
131/// - **空间复杂度**: O(r) - r 为已回收但未重新分配的 PID 数量
132///
133/// ## 线程安全
134///
135/// 本结构体本身不提供线程安全保证，需要配合 `UPSafeCell` 等同步原语使用。
136/// 全局实例 `PID_ALLOCATOR` 通过互斥访问确保线程安全。
137pub struct PidAllocator {
138    /// 下一个要分配的新 PID（从 0 开始递增）
139    current: usize,
140    /// 回收的 PID 池，后进先出的栈结构
141    recycled: Vec<usize>,
142}
143
144impl PidAllocator {
145    /// 创建新的 PID 分配器
146    ///
147    /// 初始化一个空的分配器，PID 从 0 开始分配，回收池为空。
148    /// 这是系统启动时调用的构造函数。
149    ///
150    /// ## Returns
151    ///
152    /// 返回初始化完成的 `PidAllocator` 实例
153    ///
154    /// ## Examples
155    ///
156    /// ```rust
157    /// let allocator = PidAllocator::new();
158    /// assert_eq!(allocator.current, 0);
159    /// assert!(allocator.recycled.is_empty());
160    /// ```
161    pub fn new() -> Self {
162        PidAllocator {
163            current: 0,
164            recycled: Vec::new(),
165        }
166    }
167
168    /// 分配一个新的进程 ID
169    ///
170    /// 根据分配策略返回一个可用的 PID。优先从回收池中取出已释放的 PID，
171    /// 如果回收池为空，则分配下一个连续的新 PID。
172    ///
173    /// ## 分配逻辑
174    ///
175    /// ```text
176    /// 检查回收池
177    ///     ↓
178    /// 有可用PID? ───Yes──→ 返回回收的PID
179    ///     ↓ No
180    /// 分配新PID (current)
181    ///     ↓
182    /// current += 1
183    ///     ↓
184    /// 返回新分配的PID
185    /// ```
186    ///
187    /// ## Returns
188    ///
189    /// 返回包装在 `PidHandle` 中的唯一 PID
190    ///
191    /// ## Examples
192    ///
193    /// ```rust
194    /// let mut allocator = PidAllocator::new();
195    ///
196    /// // 分配前几个PID
197    /// let pid1 = allocator.alloc(); // PID = 0
198    /// let pid2 = allocator.alloc(); // PID = 1
199    /// let pid3 = allocator.alloc(); // PID = 2
200    ///
201    /// // 回收 PID 1
202    /// allocator.dealloc(1);
203    ///
204    /// // 下次分配会复用回收的 PID
205    /// let pid4 = allocator.alloc(); // PID = 1 (复用)
206    /// ```
207    pub fn alloc(&mut self) -> PidHandle {
208        if let Some(pid) = self.recycled.pop() {
209            PidHandle(pid)
210        } else {
211            self.current += 1;
212            PidHandle(self.current - 1)
213        }
214    }
215
216    /// 回收一个进程 ID
217    ///
218    /// 将已使用完毕的 PID 放回回收池中，供后续分配复用。包含完整的
219    /// 安全检查以防止重复回收和无效回收。
220    ///
221    /// ## 安全检查
222    ///
223    /// 1. **有效性检查**: 确保 PID 小于当前分配的最大值
224    /// 2. **重复检查**: 防止同一 PID 被多次回收
225    /// 3. **系统一致性**: 维护分配器内部状态的正确性
226    ///
227    /// ## Arguments
228    ///
229    /// * `pid` - 要回收的进程 ID，必须是之前通过 `alloc()` 分配的有效 PID
230    ///
231    /// ## Panics
232    ///
233    /// * 如果 `pid >= current`，表示试图回收未分配的 PID
234    /// * 如果 `pid` 已经在回收池中，表示重复回收
235    ///
236    /// ## Examples
237    ///
238    /// ```rust
239    /// let mut allocator = PidAllocator::new();
240    ///
241    /// // 分配 PID
242    /// let handle1 = allocator.alloc(); // PID = 0
243    /// let handle2 = allocator.alloc(); // PID = 1
244    ///
245    /// // 正确回收
246    /// allocator.dealloc(0); // ✓ 正确
247    ///
248    /// // 以下操作会 panic
249    /// // allocator.dealloc(0);  // ✗ 重复回收
250    /// // allocator.dealloc(10); // ✗ 未分配的 PID
251    /// ```
252    ///
253    /// ## 实现注意事项
254    ///
255    /// 回收检查的时间复杂度为 O(n)，其中 n 是回收池大小。在高频率的
256    /// 进程创建/销毁场景下可能成为性能瓶颈，但对于系统安全性是必要的。
257    pub fn dealloc(&mut self, pid: usize) {
258        assert!(pid < self.current);
259        assert!(
260            !self.recycled.iter().any(|ppid| *ppid == pid),
261            "pid {} has been deallocated!",
262            pid
263        );
264        self.recycled.push(pid);
265    }
266}

294/// 分配新的进程 ID
295///
296/// 系统级的 PID 分配接口，为新创建的进程分配一个唯一的标识符。
297/// 该函数是进程创建流程中的关键步骤，通常在任务创建时调用。
298///
299/// ## 实现细节
300///
301/// 1. **获取锁**: 通过 `exclusive_access()` 获取全局分配器的互斥访问权
302/// 2. **执行分配**: 调用分配器的 `alloc()` 方法
303/// 3. **返回句柄**: 返回包装了 PID 的 `PidHandle`
304/// 4. **自动释放锁**: 函数结束时自动释放互斥访问权
305///
306/// ## Returns
307///
308/// 返回一个 `PidHandle`，包含新分配的唯一 PID。该句柄实现了 RAII，
309/// 当其离开作用域时会自动将 PID 归还给分配器。
310///
311/// ## 线程安全
312///
313/// 此函数是线程安全的，可以在多个执行上下文中安全调用。
314/// 内部的互斥访问机制确保不会产生竞态条件。
315///
316/// ## Examples
317///
318/// ```rust
319/// use crate::task::pid_alloc;
320///
321/// // 为新进程分配 PID
322/// let pid_handle = pid_alloc();
323/// println!("分配了 PID: {}", pid_handle.0);
324///
325/// // 创建任务时使用
326/// let task = TaskControlBlock::new(
327///     app_data,
328///     pid_handle, // 所有权转移给任务
329/// );
330///
331/// // 当 task 被销毁时，PID 会自动回收
332/// ```
333///
334/// ## 性能注意事项
335///
336/// - **锁竞争**: 在高并发情况下可能产生短暂的锁竞争
337/// - **分配成本**: 单次调用的成本非常低（O(1) 或 O(n)，取决于回收池状态）
338pub fn pid_alloc() -> PidHandle {
339    PID_ALLOCATOR.exclusive_access().alloc()
340}

342/// `PidHandle` 的自动资源管理
343///
344/// 实现 `Drop` trait 为 `PidHandle` 提供 RAII 风格的自动资源管理。
345/// 当 `PidHandle` 离开作用域时，会自动将持有的 PID 归还给全局分配器。
346///
347/// ## RAII 原理
348///
349/// RAII (Resource Acquisition Is Initialization) 确保：
350/// - **获取即初始化**: PID 在 `PidHandle` 创建时分配
351/// - **销毁即释放**: PID 在 `PidHandle` 销毁时自动回收
352/// - **异常安全**: 即使发生 panic，资源也会被正确清理
353/// - **无手动管理**: 开发者无需手动调用回收函数
354///
355/// ## 自动清理时机
356///
357/// `drop()` 方法会在以下情况下自动调用：
358/// 1. **变量离开作用域**: 正常的语句块结束
359/// 2. **显式销毁**: 调用 `drop(handle)` 显式销毁
360/// 3. **重新赋值**: 变量被赋予新值时
361/// 4. **程序终止**: 程序正常或异常终止时
362/// 5. **panic 展开**: 即使发生 panic，也会正确清理
363///
364/// ## 实现细节
365///
366/// ```rust
367/// fn drop(&mut self) {
368///     // 1. 获取全局分配器的互斥访问权
369///     // 2. 调用 dealloc() 回收 PID
370///     // 3. 自动释放互斥访问权
371///     PID_ALLOCATOR.exclusive_access().dealloc(self.0);
372/// }
373/// ```
374///
375/// ## 线程安全
376///
377/// `drop()` 方法是线程安全的，即使在多线程环境下也可以安全调用。
378/// 内部使用的互斥访问机制确保操作原子性。
379///
380/// ## 性能特性
381///
382/// - **低开销**: `drop()` 操作非常轻量，主要开销是互斥锁获取
383/// - **O(n) 复杂度**: 因为需要检查重复回收，时间复杂度与回收池大小相关
384/// - **无内存分配**: 回收过程不需要额外的内存分配
385impl Drop for PidHandle {
386    fn drop(&mut self) {
387        PID_ALLOCATOR.exclusive_access().dealloc(self.0);
388    }
389}

391/// 内核栈管理器
392///
393/// 为每个进程分配和管理独立的内核栈空间。内核栈用于存储系统调用、中断处理
394/// 和任务切换时的临时数据。每个内核栈在虚拟地址空间中占用固定的位置。
395///
396/// ## 设计特性
397///
398/// ### 栈空间隔离
399/// - **独立映射**: 每个进程拥有完全独立的内核栈空间
400/// - **保护页**: 相邻栈之间有保护页防止栈溢出
401/// - **固定大小**: 每个栈的大小为 `KERNEL_STACK_SIZE` (通常 8KB)
402/// - **预定位置**: 栈位置由 PID 唯一确定，便于快速定位
403///
404/// ### 虚拟内存布局
405///
406/// ```text
407/// TRAMPOLINE (高地址)
408///     ↓
409/// ┌─────────────────┐ ← PID 0 栈顶
410/// │   Kernel Stack  │
411/// │     (PID 0)     │ ← 8KB
412/// └─────────────────┘ ← PID 0 栈底
413/// ┌─────────────────┐ ← 保护页 (4KB)
414/// └─────────────────┘
415/// ┌─────────────────┐ ← PID 1 栈顶
416/// │   Kernel Stack  │
417/// │     (PID 1)     │ ← 8KB  
418/// └─────────────────┘ ← PID 1 栈底
419/// ┌─────────────────┐ ← 保护页 (4KB)
420/// └─────────────────┘
421///     ↓
422/// (低地址方向)
423/// ```
424///
425/// ## RAII 资源管理
426///
427/// - **自动分配**: 构造时自动分配并映射虚拟内存
428/// - **自动回收**: 析构时自动解映射并释放内存
429/// - **异常安全**: 即使发生异常也能正确清理资源
430/// - **零泄漏**: 通过 Rust 的所有权系统保证无内存泄漏
431///
432/// ## 线程安全
433///
434/// 内核栈的分配和回收涉及全局内存管理器，通过互斥访问保证线程安全。
435/// 但单个 `KernelStack` 实例本身不支持跨线程共享。
436pub struct KernelStack {
437    /// 关联的进程 ID，用于确定内核栈在虚拟地址空间中的位置
438    pid: usize,
439}
440
441impl KernelStack {
442    /// 为指定进程创建新的内核栈
443    ///
444    /// 根据进程 ID 计算内核栈的虚拟地址范围，并在内核地址空间中建立映射。
445    /// 内核栈具有读写权限，用于存储系统调用和中断处理时的临时数据。
446    ///
447    /// ## Arguments
448    ///
449    /// * `pid_handle` - 进程 ID 句柄的引用，用于确定栈的虚拟地址位置
450    ///
451    /// ## Returns
452    ///
453    /// 返回初始化完成的 `KernelStack` 实例
454    ///
455    /// ## 实现过程
456    ///
457    /// 1. **地址计算**: 通过 `kernel_stack_position()` 计算栈的起始和结束地址
458    /// 2. **内存映射**: 在内核地址空间中建立虚拟到物理的页面映射
459    /// 3. **权限设置**: 设置页面的读写权限 (R|W)
460    /// 4. **实例创建**: 创建并返回 `KernelStack` 实例
461    ///
462    /// ## 内存分配
463    ///
464    /// - **按需分配**: 物理页面在访问时才真正分配（延迟分配）
465    /// - **连续虚拟地址**: 保证栈空间在虚拟地址上连续
466    /// - **页面对齐**: 所有地址都按页边界对齐
467    ///
468    /// ## Panics
469    ///
470    /// 如果虚拟内存映射失败（如地址冲突或内存不足），函数会 panic
471    ///
472    /// ## Examples
473    ///
474    /// ```rust
475    /// // 分配进程 ID
476    /// let pid_handle = pid_alloc();
477    ///
478    /// // 创建对应的内核栈
479    /// let kernel_stack = KernelStack::new(&pid_handle);
480    ///
481    /// // 现在可以使用内核栈进行系统调用处理
482    /// let stack_top = kernel_stack.get_top();
483    /// println!("Kernel stack top: 0x{:x}", stack_top);
484    /// ```
485    pub fn new(pid_handle: &PidHandle) -> Self {
486        let pid = pid_handle.0;
487        let (kernel_stack_bottom, kernel_stack_top) = kernel_stack_position(pid);
488        KERNEL_SPACE.exclusive_access().insert_framed_area(
489            kernel_stack_bottom.into(),
490            kernel_stack_top.into(),
491            MapPermission::R | MapPermission::W,
492        );
493        KernelStack { pid: pid_handle.0 }
494    }
495
496    /// 在内核栈顶压入数据
497    ///
498    /// 在内核栈的顶部压入一个值，并返回指向该值的可变指针。这是一个泛型方法，
499    /// 可以压入任何实现了 `Sized` trait 的类型。主要用于在任务切换时保存上下文信息。
500    ///
501    /// ## Type Parameters
502    ///
503    /// * `T` - 要压入的数据类型，必须实现 `Sized` trait
504    ///
505    /// ## Arguments
506    ///
507    /// * `value` - 要压入栈顶的值，通过值传递（移动语义）
508    ///
509    /// ## Returns
510    ///
511    /// 返回指向栈顶压入值的可变原始指针 `*mut T`
512    ///
513    /// ## 栈操作过程
514    ///
515    /// ```text
516    /// 压入前:                   压入后:
517    /// ┌──────────────┐         ┌──────────────┐
518    /// │              │         │   New Data   │ ← Returned pointer
519    /// │              │         ├──────────────┤ ← New stack top
520    /// │              │         │              │
521    /// │ Stack Space  │         │ Original     │
522    /// │              │         │ Content      │
523    /// │              │         │              │
524    /// └──────────────┘         └──────────────┘
525    /// ↑ 栈顶                   ↑ 栈底
526    /// ```
527    ///
528    /// ## Safety
529    ///
530    /// 此方法使用 `unsafe` 代码直接操作内存，但通过以下方式确保安全：
531    /// - 栈顶地址由 `get_top()` 正确计算
532    /// - 减去数据大小确保不会越界
533    /// - 目标地址在已映射的内核栈范围内
534    ///
535    /// ## 使用场景
536    ///
537    /// - **保存上下文**: 在任务切换时保存 `TrapContext`
538    /// - **传递参数**: 向新任务传递初始化参数
539    /// - **临时存储**: 存储系统调用期间的临时数据
540    ///
541    /// ## Examples
542    ///
543    /// ```rust
544    /// // 创建内核栈
545    /// let kernel_stack = KernelStack::new(&pid_handle);
546    ///
547    /// // 压入陷阱上下文
548    /// let trap_ctx = TrapContext::init_app_context(entry_point, user_sp);
549    /// let ctx_ptr = kernel_stack.push_on_top(trap_ctx);
550    ///
551    /// // 现在可以通过 ctx_ptr 访问压入的数据
552    /// unsafe {
553    ///     (*ctx_ptr).sepc = new_entry_point;
554    /// }
555    /// ```
556    ///
557    /// ## 注意事项
558    ///
559    /// - **栈溢出风险**: 频繁压入大对象可能导致栈溢出
560    /// - **生命周期**: 返回的指针的有效性与 `KernelStack` 实例相关
561    /// - **并发安全**: 不支持多线程同时操作同一个内核栈
562    #[allow(unused)]
563    pub fn push_on_top<T>(&self, value: T) -> *mut T
564    where
565        T: Sized,
566    {
567        let kernel_stack_top = self.get_top();
568        let ptr_mut = (kernel_stack_top - core::mem::size_of::<T>()) as *mut T;
569        unsafe {
570            *ptr_mut = value;
571        }
572        ptr_mut
573    }

575    /// 获取内核栈顶地址
576    ///
577    /// 返回当前进程内核栈的顶部虚拟地址。栈顶是栈空间的最高地址，
578    /// 新数据从这里开始向下增长。
579    ///
580    /// ## Returns
581    ///
582    /// 内核栈顶的虚拟地址（`usize` 类型）
583    ///
584    /// ## 地址计算
585    ///
586    /// 栈顶地址通过以下公式计算：
587    /// ```text
588    /// stack_top = TRAMPOLINE - (KERNEL_STACK_SIZE + PAGE_SIZE) * pid
589    /// ```
590    ///
591    /// 其中：
592    /// - `TRAMPOLINE`: 蹦床页的起始地址（虚拟地址空间最高处）
593    /// - `KERNEL_STACK_SIZE`: 单个内核栈的大小（通常 8KB）
594    /// - `PAGE_SIZE`: 页面大小（通常 4KB），用作保护页
595    /// - `pid`: 当前进程的 ID
596    ///
597    /// ## 使用场景
598    ///
599    /// - **栈指针初始化**: 设置任务切换时的栈指针
600    /// - **数据压栈**: 计算压入数据后的新地址
601    /// - **栈空间检查**: 验证栈使用情况和剩余空间
602    /// - **调试诊断**: 输出栈地址信息进行问题排查
603    ///
604    /// ## Examples
605    ///
606    /// ```rust
607    /// let kernel_stack = KernelStack::new(&pid_handle);
608    /// let stack_top = kernel_stack.get_top();
609    ///
610    /// println!("Kernel stack top: 0x{:x}", stack_top);
611    ///
612    /// // 压入数据并计算新的栈顶
613    /// let data_ptr = kernel_stack.push_on_top(some_data);
614    /// let new_top = stack_top - core::mem::size_of::<SomeData>();
615    /// assert_eq!(data_ptr as usize, new_top);
616    /// ```
617    ///
618    /// ## 性能特性
619    ///
620    /// - **O(1) 复杂度**: 地址计算是简单的算术运算
621    /// - **无内存访问**: 不需要访问实际的物理内存
622    /// - **缓存友好**: 频繁调用不会产生缓存开销
623    pub fn get_top(&self) -> usize {
624        let (_, kernel_stack_top) = kernel_stack_position(self.pid);
625        kernel_stack_top
626    }
627}

629/// `KernelStack` 的自动资源清理
630///
631/// 实现 `Drop` trait 为内核栈提供 RAII 风格的自动资源管理。当 `KernelStack`
632/// 实例被销毁时，会自动清理对应的虚拟内存映射，防止内存泄漏。
633///
634/// ## 清理过程
635///
636/// 1. **地址计算**: 重新计算内核栈的底部虚拟地址
637/// 2. **地址转换**: 将 usize 地址转换为 `VirtAddr` 类型
638/// 3. **页面转换**: 将虚拟地址转换为虚拟页号 (VPN)
639/// 4. **映射移除**: 从内核地址空间中移除整个内核栈区域的映射
640/// 5. **物理页回收**: 底层的物理页面被自动回收到页面分配器
641///
642/// ## 自动触发时机
643///
644/// `drop()` 方法在以下情况下自动调用：
645/// - **作用域结束**: `KernelStack` 变量离开作用域
646/// - **显式销毁**: 调用 `drop(kernel_stack)` 显式销毁
647/// - **重新赋值**: 变量被赋予新值时，旧值被销毁
648/// - **容器清理**: 从 Vec 或其他容器中移除时
649/// - **程序终止**: 程序结束时自动清理所有资源
650///
651/// ## 内存管理
652///
653/// ```text
654/// 销毁前:                     销毁后:
655/// ┌─────────────────┐         ┌─────────────────┐
656/// │ Virtual Address │         │ Virtual Address │
657/// │ Space           │         │ Space           │
658/// │ ┌─────────────┐ │         │                 │
659/// │ │ Stack       │ │   -->   │  (Mapping       │
660/// │ │ Mapping     │ │         │   Removed)      │
661/// │ └─────────────┘ │         │                 │
662/// └─────────────────┘         └─────────────────┘
663///       ↓                            
664/// ┌─────────────────┐         ┌─────────────────┐
665/// │ Physical Memory │         │ Physical Memory │
666/// │ ┌─────────────┐ │         │ ┌─────────────┐ │
667/// │ │ Allocated   │ │   -->   │ │ Free Pages  │ │
668/// │ │ Pages       │ │         │ │             │ │
669/// │ └─────────────┘ │         │ └─────────────┘ │
670/// └─────────────────┘         └─────────────────┘
671/// ```
672///
673/// ## 线程安全
674///
675/// 通过 `KERNEL_SPACE.exclusive_access()` 获取互斥访问权，确保在多线程环境下
676/// 的内存管理操作是原子的和线程安全的。
677///
678/// ## 异常安全
679///
680/// 即使在 `drop()` 执行过程中发生 panic，内核地址空间的状态仍然保持一致，
681/// 不会出现部分清理的情况。
682///
683/// ## 性能特性
684///
685/// - **延迟回收**: 物理页面回收可能是延迟的，具体取决于内存管理器实现
686/// - **批量操作**: 移除整个区域的映射比逐页移除更高效
687/// - **无额外分配**: 清理过程不需要额外的内存分配
688///
689/// ## 调试信息
690///
691/// 在调试模式下，可以通过内存管理器的调试接口验证内核栈是否被正确清理。
692impl Drop for KernelStack {
693    fn drop(&mut self) {
694        let (kernel_stack_bottom, _) = kernel_stack_position(self.pid);
695        let kernel_stack_bottom_va: VirtAddr = kernel_stack_bottom.into();
696        KERNEL_SPACE
697            .exclusive_access()
698            .remove_area_with_start_vpn(kernel_stack_bottom_va.into());
699    }
700}

702/// 计算内核栈在虚拟地址空间中的位置
703///
704/// 根据进程 ID 计算对应内核栈的虚拟地址范围。内核栈在虚拟地址空间中从高地址向低地址排列，
705/// 每个栈之间通过保护页分隔，防止栈溢出时影响其他进程的内核栈。
706///
707/// ## Arguments
708///
709/// * `app_id` - 进程/应用程序的 ID，用于确定内核栈在地址空间中的位置
710///
711/// ## Returns
712///
713/// 返回一个元组 `(bottom, top)`:
714/// - `bottom`: 内核栈底部的虚拟地址（低地址）
715/// - `top`: 内核栈顶部的虚拟地址（高地址）
716///
717/// ## 地址计算公式
718///
719/// ```text
720/// stack_top = TRAMPOLINE - (KERNEL_STACK_SIZE + PAGE_SIZE) * app_id
721/// stack_bottom = stack_top - KERNEL_STACK_SIZE
722/// ```
723///
724/// 其中：
725/// - `TRAMPOLINE`: 跳板页的起始地址，通常为用户地址空间的最高地址
726/// - `KERNEL_STACK_SIZE`: 单个内核栈的大小（通常为 8KB）
727/// - `PAGE_SIZE`: 页面大小（通常为 4KB），用作保护页
728/// - `app_id`: 进程 ID，从 0 开始
729///
730/// ## 内存布局示意图
731///
732/// ```text
733/// 高地址 TRAMPOLINE
734///        ↓
735/// ┌─────────────────────┐ ← app_id=0 的栈顶
736/// │   Kernel Stack 0    │ ← 8KB
737/// └─────────────────────┘ ← app_id=0 的栈底
738/// ┌─────────────────────┐ ← 保护页 (4KB)
739/// └─────────────────────┘
740/// ┌─────────────────────┐ ← app_id=1 的栈顶
741/// │   Kernel Stack 1    │ ← 8KB
742/// └─────────────────────┘ ← app_id=1 的栈底
743/// ┌─────────────────────┐ ← 保护页 (4KB)
744/// └─────────────────────┘
745///        ↓
746///    (更多内核栈...)
747///        ↓
748/// 低地址
749/// ```
750///
751/// ## 设计考虑
752///
753/// ### 保护页机制
754/// 每个内核栈之间插入一个未映射的保护页，提供以下保护：
755/// - **栈溢出检测**: 访问保护页会触发页面错误
756/// - **隔离保护**: 防止一个进程的栈溢出影响其他进程
757/// - **调试辅助**: 栈溢出时能快速定位问题
758///
759/// ### 地址分配策略
760/// - **确定性布局**: 每个进程的内核栈位置完全由其 PID 决定
761/// - **快速计算**: O(1) 时间复杂度，无需查表或搜索
762/// - **内存对齐**: 所有地址都自然对齐到页边界
763/// - **向下增长**: 符合大多数架构的栈增长方向
764///
765/// ## 使用示例
766///
767/// ```rust
768/// // 计算进程 0 的内核栈位置
769/// let (bottom, top) = kernel_stack_position(0);
770/// println!("Process 0 kernel stack: 0x{:x} - 0x{:x}", bottom, top);
771/// assert_eq!(top - bottom, KERNEL_STACK_SIZE);
772///
773/// // 验证相邻进程的栈有保护页分隔
774/// let (bottom1, top1) = kernel_stack_position(1);
775/// assert!(top1 + PAGE_SIZE <= bottom); // 保护页存在
776/// ```
777///
778/// ## 限制和注意事项
779///
780/// - **地址空间限制**: 可分配的内核栈数量受虚拟地址空间大小限制
781/// - **PID 范围**: 过大的 `app_id` 可能导致地址下溢或冲突
782/// - **静态布局**: 内核栈大小在编译时确定，运行时无法调整
783///
784/// ## 性能特性
785///
786/// - **计算开销**: O(1) 时间复杂度，仅涉及简单算术运算
787/// - **无内存访问**: 纯计算函数，不访问内存或全局状态
788/// - **缓存友好**: 多次调用不会产生缓存未命中
789pub fn kernel_stack_position(app_id: usize) -> (usize, usize) {
790    let top = TRAMPOLINE - (KERNEL_STACK_SIZE + PAGE_SIZE) * app_id;
791    let bottom = top - KERNEL_STACK_SIZE;
792    (bottom, top)
