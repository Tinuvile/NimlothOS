1//! # 任务控制块和任务状态
2//!
3//! 定义任务的基本数据结构和状态枚举，用于任务管理和调度。
4
5use super::context::TaskContext;
6
7/// 任务控制块 (Task Control Block, TCB)
8///
9/// 存储单个任务的所有必要信息，包括任务状态和执行上下文。
10/// 每个应用程序对应一个任务控制块。
11///
12/// ## 设计原则
13///
14/// - **最小化设计**: 只包含任务调度所需的核心信息
15/// - **状态分离**: 将任务状态与上下文分开存储，便于管理
16/// - **复制语义**: 实现 `Copy` trait，支持高效的值拷贝
17///
18/// ## 扩展性
19///
20/// 未来版本可能会添加更多字段：
21/// - 任务优先级
22/// - 执行时间统计
23/// - 内存使用信息
24/// - 进程 ID 和父子关系
25#[derive(Copy, Clone)]
26pub struct TaskControlBlock {
27    /// 任务当前状态
28    ///
29    /// 标识任务在其生命周期中的当前阶段，用于调度决策。
30    pub task_status: TaskStatus,
31
32    /// 任务上下文
33    ///
34    /// 保存任务的 CPU 寄存器状态，用于任务切换时的状态保存和恢复。
35    pub task_cx: TaskContext,
36}
37
38/// 任务状态枚举
39///
40/// 定义任务在其生命周期中可能处于的各种状态。
41/// 任务状态转换遵循特定的规则，确保系统调度的正确性。
42///
43/// ## 状态转换图
44///
45/// ```text
46/// ┌─────────┐    load     ┌───────┐    schedule   ┌─────────┐
47/// │ Uninit  │ ──────────> │ Ready │ ────────────> │ Running │
48/// └─────────┘             └───────┘               └─────────┘
49///                            ^                         │
50///                            │          yield/         │ exit/
51///                            │         timeout         │ error
52///                            └─────────────────────────┘
53///                                                      │
54///                                                      v
55///                                              ┌─────────────┐
56///                                              │   Exited    │
57///                                              └─────────────┘
58/// ```
59///
60/// ## 状态说明
61///
62/// - **Uninit**: 任务尚未初始化，仅在系统启动时短暂存在
63/// - **Ready**: 任务已准备就绪，等待被调度执行
64/// - **Running**: 任务正在 CPU 上执行
65/// - **Exited**: 任务已完成执行，不会再被调度
66#[derive(Copy, Clone, PartialEq)]
67pub enum TaskStatus {
68    /// 未初始化状态
69    ///
70    /// 任务刚创建时的初始状态，表示任务控制块已分配但尚未完成初始化。
71    /// 这个状态在系统启动过程中短暂存在，很快会转换为 Ready 状态。
72    Uninit,
73
74    /// 就绪状态
75    ///
76    /// 任务已准备好执行，正在等待获得 CPU 时间片。
77    /// 处于此状态的任务会被调度器考虑进行调度。
78    Ready,
79
80    /// 运行状态
81    ///
82    /// 任务正在 CPU 上执行。在单 CPU 系统中，同时只能有一个任务处于此状态。
83    /// 任务可能因为时间片用完、主动让出或异常而离开此状态。
84    Running,
85
86    /// 已退出状态
87    ///
88    /// 任务已完成执行并退出，不会再被调度。
89    /// 任务可能因为正常结束、调用 exit() 或发生致命错误而进入此状态。
90    Exited,
91}
