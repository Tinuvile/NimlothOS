1//! # 任务控制块模块
2//!
3//! 提供任务控制块 (Task Control Block, TCB) 的实现，管理进程的完整生命周期。
4//! 支持进程创建、fork、exec、状态管理等核心操作，是操作系统进程管理的核心数据结构。
5//!
6//! ## 核心组件
7//!
8//! - [`TaskControlBlock`] - 任务控制块，包含进程的所有信息
9//! - [`TaskControlBlockInner`] - TCB 内部可变部分，受互斥锁保护
10//! - [`TaskStatus`] - 任务状态枚举，表示进程的运行状态
11//!
12//! ## 设计原理
13//!
14//! ### 分离设计
15//! TCB 采用内外分离的设计模式：
16//! - **不变部分**：PID 和内核栈在进程生命周期内保持不变
17//! - **可变部分**：任务状态、内存集合等需要互斥保护的数据
18//!
19//! ### 进程层次结构
20//! 支持完整的进程树结构：
21//! - **父子关系**：通过 `parent` 和 `children` 字段维护
22//! - **引用计数**：使用 `Arc` 和 `Weak` 防止循环引用
23//! - **孤儿进程处理**：父进程退出时子进程重新指向 init 进程
24//!
25//! ## 内存布局
26//!
27//! 每个进程的内存空间布局：
28//!
29//! ```text
30//! 高地址 TRAMPOLINE (0x3ffffff000)
31//! ┌──────────────────────────────────────────────────────┐
32//! │                   Trampoline                         │
33//! │                    (R+X)                             │
34//! ├──────────────────────────────────────────────────────┤
35//! │                 Trap Context                         │
36//! │                    (R+W)                             │
37//! ├──────────────────────────────────────────────────────┤
38//! │                  User Stack                          │
39//! │                   (R+W+U)                            │
40//! ├──────────────────────────────────────────────────────┤
41//! │               Program Sections                       │
42//! │            (.text/.data/.bss etc)                    │
43//! │              (Based on ELF flags)                    │
44//! └──────────────────────────────────────────────────────┘
45//! 低地址 (0x10000)
46//!
47//! 内核空间 - 每个进程独立的内核栈：
48//! ┌──────────────────────────────────────────────────────┐
49//! │             Process N Kernel Stack                   │
50//! ├──────────────────────────────────────────────────────┤
51//! │                  Guard Page                          │
52//! ├──────────────────────────────────────────────────────┤
53//! │           Process N-1 Kernel Stack                   │
54//! └──────────────────────────────────────────────────────┘
55//! ```
56//!
57//! ## 进程状态转换
58//!
59//! ```text
60//!           fork/exec
61//!         ┌──────────┐
62//!         │          │
63//!         ▼          │
64//!     ┌───────┐      │      Scheduler Selection
65//!     │ Ready │◄─────┘ ◄─────────────┐
66//!     └───┬───┘                      │
67//!         │                          │
68//!         │ Scheduler Selection      │ Timeslice Expire/Yield
69//!         │                          │
70//!         ▼                          │
71//!    ┌─────────┐                     │
72//!    │ Running │───────────────────-─┘
73//!    └────┬────┘
74//!         │
75//!         │ exit() System Call
76//!         │
77//!         ▼
78//!     ┌───────┐
79//!     │Zombie │ ◄── Wait for Parent wait()
80//!     └───────┘
81//! ```
82//!
83//! ## 并发安全
84//!
85//! - **互斥保护**：可变部分使用 `UPSafeCell` 保证线程安全
86//! - **原子操作**：PID 分配和内核栈管理通过全局锁保护
87//! - **引用计数**：`Arc` 确保进程对象的生命周期管理
88//!
89//! ## 使用示例
90//!
91//! ```rust
92//! // 创建新进程
93//! let elf_data = get_app_data(0);
94//! let task = Arc::new(TaskControlBlock::new(elf_data));
95//!
96//! // Fork 子进程
97//! let child_task = task.fork();
98//!
99//! // 执行新程序
100//! task.exec(new_elf_data);
101//!
102//! // 获取进程状态
103//! let inner = task.inner_exclusive_access();
104//! println!("Task status: {:?}", inner.task_status);
105//! ```

120/// 任务控制块 (Task Control Block)
121///
122/// 操作系统中每个进程的核心数据结构，包含进程运行所需的全部信息。
123/// TCB 采用内外分离设计，将不变和可变部分分开管理，提高并发访问效率。
124///
125/// ## 结构组成
126///
127/// ### 不变部分（直接字段）
128/// - `pid`: 进程标识符句柄，进程生命周期内唯一且不变
129/// - `kernel_stack`: 内核栈，用于系统调用和中断处理
130///
131/// ### 可变部分（受保护字段）
132/// - `inner`: 包含所有可变状态，使用 [`UPSafeCell`] 进行互斥保护
133///
134/// ## 设计优势
135///
136/// **性能优化**：
137/// - 不变字段可以直接访问，避免锁开销
138/// - 可变字段集中保护，减少锁的粒度和争用
139///
140/// **并发安全**：
141/// - [`UPSafeCell`] 提供单处理器环境下的安全可变访问
142/// - 通过 [`Arc`] 支持多处理器环境下的引用计数管理
143///
144/// **资源管理**：
145/// - PID 和内核栈通过 RAII 自动管理生命周期
146/// - 内存集合在进程退出时自动清理
147///
148/// ## 生命周期
149///
150/// ```text
151/// 创建阶段:
152/// ┌────────────-─┐    ┌─-────────────--┐    ┌────────────┐
153/// │ Allocate PID │───►│ Create K-Stack │───►│ Parse ELF  │
154/// └──────────-───┘    └-────────────-─-┘    └────────────┘
155///                                                 │
156///                                                 ▼
157/// ┌────────────┐    ┌─-────────-────┐      ┌──────────--───┐
158/// │ Init State │◄───│ Setup AddrSpc │  ◄─  │ Setup TrapCtx │
159/// └-───────────┘    └-─────────────-┘      └────────────--─┘
160///
161/// 运行阶段：
162/// Ready ──Schedule──► Running ──Timeslice/Yield──► Ready
163///                       │
164///                       │ exit()
165///                       ▼
166///                    Zombie ──wait()──► Destroy
167///
168/// 销毁阶段:
169/// ┌─────────────┐    ┌───────────-──┐    ┌──────────┐
170/// │ Free Memory │───►│ Free K-Stack │───►│ Free PID │
171/// └─────────────┘    └─────────────-┘    └──────────┘
172/// ```
173///
174/// ## 使用模式
175///
176/// TCB 通常包装在 [`Arc`] 中使用，支持多所有者场景：
177/// - 任务管理器持有引用进行调度
178/// - 父进程持有子进程引用进行管理
179/// - 处理器持有当前运行任务的引用
180///
181/// ## Examples
182///
183/// ```rust
184/// use alloc::sync::Arc;
185///
186/// // 创建新进程
187/// let elf_data = include_bytes!("user_program.elf");
188/// let task = Arc::new(TaskControlBlock::new(elf_data));
189///
190/// // 访问不变字段（无需锁）
191/// println!("Process PID: {}", task.getpid());
192///
193/// // 访问可变字段（需要获取锁）
194/// {
195///     let inner = task.inner_exclusive_access();
196///     println!("Task status: {:?}", inner.task_status);
197/// } // 锁在此处自动释放
198///
199/// // Fork 创建子进程
200/// let child_task = task.fork();
201/// ```
202pub struct TaskControlBlock {
203    /// 进程标识符句柄
204    ///
205    /// 包含系统唯一的进程 ID，通过 RAII 机制自动管理 PID 的分配和回收。
206    /// 在进程的整个生命周期中保持不变，可以安全地并发访问。
207    pub pid: PidHandle,
208
209    /// 内核栈
210    ///
211    /// 每个进程在内核空间中的独立栈空间，用于：
212    /// - 系统调用处理时的临时数据存储
213    /// - 中断和异常处理时的上下文保存
214    /// - 任务切换时的寄存器保存
215    ///
216    /// 通过 RAII 机制自动管理内核栈的分配、映射和回收。
217    pub kernel_stack: KernelStack,
218
219    /// 内部可变状态
220    ///
221    /// 包含所有需要在运行时修改的进程状态信息，使用 [`UPSafeCell`]
222    /// 提供线程安全的可变访问。包括任务状态、内存集合、上下文等。
223    inner: UPSafeCell<TaskControlBlockInner>,
224}
225
226/// 任务控制块内部可变状态
227///
228/// 包含进程的所有可变状态信息，需要在运行时进行修改的字段都集中在此结构中。
229/// 通过 [`UPSafeCell`] 进行互斥保护，确保并发访问的安全性。
230///
231/// ## 字段说明
232///
233/// ### 运行时状态
234/// - `task_status`: 进程当前状态（Ready/Running/Zombie）
235/// - `task_cx`: 任务上下文，保存寄存器状态用于任务切换
236/// - `exit_code`: 进程退出码，用于父进程获取子进程执行结果
237///
238/// ### 内存管理
239/// - `memory_set`: 进程的完整地址空间，包含所有内存映射区域
240/// - `trap_cx_ppn`: 陷阱上下文的物理页号，用于用户态/内核态切换
241/// - `base_size`: 进程初始堆栈大小，用于内存分配决策
242///
243/// ### 进程关系
244/// - `parent`: 父进程的弱引用，避免循环引用导致内存泄漏
245/// - `children`: 子进程列表，维护进程树结构
246///
247/// ## 设计考虑
248///
249/// **并发安全**：
250/// - 所有字段都受到外层 [`UPSafeCell`] 的保护
251/// - 访问时需要获取独占锁，避免数据竞争
252/// - 锁的持有时间应当尽可能短，避免性能影响
253///
254/// **内存管理**：
255/// - [`MemorySet`] 自动管理进程地址空间的生命周期
256/// - 使用 [`Arc`] 和 [`Weak`] 管理进程间的引用关系
257/// - 进程退出时自动清理相关资源
258///
259/// **状态一致性**：
260/// - 任务状态与上下文信息保持同步
261/// - 父子关系的双向引用保持一致
262/// - 内存映射与陷阱上下文匹配
263pub struct TaskControlBlockInner {
264    /// 任务当前状态
265    ///
266    /// 表示进程在操作系统中的当前状态，影响调度器的调度决策。
267    /// 状态转换遵循严格的状态机规则。
268    pub task_status: TaskStatus,
269
270    /// 任务上下文
271    ///
272    /// 保存进程在任务切换时需要恢复的寄存器状态，包括：
273    /// - 返回地址 (ra): 任务恢复时的执行地址
274    /// - 栈指针 (sp): 内核栈的栈顶地址
275    /// - 被调用者保存寄存器 (s0-s11): 函数调用约定要求保存的寄存器
276    pub task_cx: TaskContext,
277
278    /// 进程地址空间
279    ///
280    /// 管理进程的完整虚拟地址空间，包括：
281    /// - 代码段、数据段、堆段、栈段的映射
282    /// - 页表管理和地址转换
283    /// - 内存权限控制和保护
284    pub memory_set: MemorySet,
285
286    /// 陷阱上下文物理页号
287    ///
288    /// 指向保存陷阱上下文的物理页面，陷阱上下文包含：
289    /// - 用户态所有寄存器的值
290    /// - 系统调用参数和返回值
291    /// - 异常处理相关信息
292    pub trap_cx_ppn: PhysPageNum,
293
294    /// 进程基础内存大小
295    ///
296    /// 记录进程初始化时的内存使用情况，用于：
297    /// - 内存分配和回收的决策参考
298    /// - 进程资源使用统计
299    /// - 堆空间管理的基准值
300    pub base_size: usize,
301
302    /// 父进程引用
303    ///
304    /// 指向父进程的弱引用，用于维护进程树结构。使用 [`Weak`] 避免
305    /// 父子进程间的循环引用，防止内存泄漏。当父进程退出时，
306    /// 子进程会被重新指向 init 进程。
307    pub parent: Option<Weak<TaskControlBlock>>,
308
309    /// 子进程列表
310    ///
311    /// 维护当前进程的所有子进程，用于：
312    /// - 实现 wait 系统调用，等待子进程退出
313    /// - 进程退出时处理孤儿进程，重新指向 init 进程
314    /// - 信号传递和进程组管理
315    pub children: Vec<Arc<TaskControlBlock>>,
316
317    /// 进程退出码
318    ///
319    /// 记录进程的退出状态，供父进程通过 wait 系统调用获取。
320    /// 标准约定：0 表示正常退出，非零表示异常退出。
321    pub exit_code: i32,
322}

324/// 任务状态枚举
325///
326/// 定义进程在操作系统中的各种运行状态，遵循经典的进程状态模型。
327/// 状态转换受到严格的状态机规则约束，确保系统的一致性和可预测性。
328///
329/// ## 状态转换图
330///
331/// ```text
332///                Process Creation
333///                      │
334///                      ▼
335///               ┌─────────────┐
336///               │    Ready    │ ◄─────────────────┐
337///               │  (Ready)    │                   │
338///               └──────┬──────┘                   │
339///                      │                          │
340///                      │ Scheduler Selection      │
341///                      ▼                          │
342///               ┌─────────────┐                   │
343///               │   Running   │                   │
344///               │  (Running)  │                   │
345///               └──────┬──────┘                   │
346///                      │                          │
347///            ┌─────────┴─────────┐                │
348///            │                   │                │
349///            ▼                   ▼                │
350///   Timeslice Expire/Yield   exit() System Call   │
351///            │                   │                │
352///            └───────────────────┼───────────────-┘
353///                                ▼
354///                        ┌─────────────┐
355///                        │   Zombie    │
356///                        │  (Zombie)   │
357///                        └─────────────┘
358///                                │
359///                                │ Parent wait()
360///                                ▼
361///                        Process Destruction
362/// ```
363///
364/// ## 状态详细说明
365///
366/// ### Ready (就绪)
367/// - **含义**: 进程已准备好运行，等待 CPU 分配
368/// - **特征**: 所有运行条件已满足，仅等待调度器调度
369/// - **转入**: 进程创建完成、时间片用完、系统调用返回
370/// - **转出**: 被调度器选中执行
371///
372/// ### Running (运行)
373/// - **含义**: 进程正在 CPU 上执行
374/// - **特征**: 拥有 CPU 控制权，正在执行指令
375/// - **转入**: 从就绪队列被调度器选中
376/// - **转出**: 时间片用完、主动让出 CPU、进程退出
377///
378/// ### Zombie (僵尸)
379/// - **含义**: 进程已执行完毕，等待父进程收集退出信息
380/// - **特征**: 保留进程控制块，但不再调度执行
381/// - **转入**: 进程调用 exit 系统调用
382/// - **转出**: 父进程调用 wait 系统调用回收
383///
384/// ## 使用示例
385///
386/// ```rust
387/// // 创建进程后的状态检查
388/// let task = TaskControlBlock::new(elf_data);
389/// {
390///     let inner = task.inner_exclusive_access();
391///     assert_eq!(inner.task_status, TaskStatus::Ready);
392/// }
393///
394/// // 状态转换示例
395/// match current_status {
396///     TaskStatus::Ready => {
397///         // 可以被调度器选中
398///         scheduler.add_to_running_queue(task);
399///     }
400///     TaskStatus::Running => {
401///         // 正在执行，可能需要时间片管理
402///         if time_slice_expired() {
403///             task.suspend_and_yield();
404///         }
405///     }
406///     TaskStatus::Zombie => {
407///         // 等待父进程回收
408///         if parent.is_waiting() {
409///             parent.collect_child_exit_code(task.exit_code);
410///         }
411///     }
412/// }
413/// ```
414///
415/// ## 状态约束
416///
417/// - **互斥性**: 同一时刻进程只能处于一种状态
418/// - **有序性**: 状态转换必须遵循预定义的转换路径
419/// - **原子性**: 状态转换是原子操作，不存在中间状态
420/// - **一致性**: 状态转换与系统其他部分的状态保持一致
421#[derive(Copy, Clone, PartialEq, Debug)]
422pub enum TaskStatus {
423    /// 就绪状态
424    ///
425    /// 进程已准备好执行，所有资源已分配，等待调度器分配 CPU 时间。
426    /// 处于此状态的进程位于就绪队列中，按调度策略等待执行机会。
427    Ready,
428
429    /// 运行状态
430    ///
431    /// 进程正在 CPU 上执行指令，拥有处理器控制权。在单处理器系统中，
432    /// 同一时刻只有一个进程可以处于运行状态。
433    Running,
434
435    /// 僵尸状态
436    ///
437    /// 进程已执行完毕并退出，但进程控制块仍然保留，等待父进程
438    /// 通过 wait 系统调用收集其退出状态信息。
439    Zombie,
440}

442impl TaskControlBlockInner {
443    /// 获取陷阱上下文的可变引用
444    ///
445    /// 通过陷阱上下文物理页号获取指向 [`TrapContext`] 的可变引用。
446    /// 陷阱上下文用于在用户态和内核态之间切换时保存和恢复寄存器状态。
447    ///
448    /// ## Returns
449    ///
450    /// 返回陷阱上下文的可变引用，具有 `'static` 生命周期
451    ///
452    /// ## 安全性
453    ///
454    /// 返回的可变引用安全性基于以下保证：
455    /// - 陷阱上下文所在的物理页面在进程生命周期内有效
456    /// - 物理页号由内存集合管理，保证映射关系的正确性
457    /// - 同一时刻只有一个可变引用存在，避免别名引用
458    ///
459    /// ## 使用场景
460    ///
461    /// - **系统调用处理**: 读取和修改系统调用参数
462    /// - **异常处理**: 获取异常发生时的寄存器状态
463    /// - **进程初始化**: 设置新进程的初始寄存器状态
464    /// - **信号处理**: 修改进程的执行上下文
465    ///
466    /// ## Examples
467    ///
468    /// ```rust
469    /// let mut inner = task.inner_exclusive_access();
470    /// let trap_cx = inner.get_trap_cx();
471    ///
472    /// // 读取系统调用参数
473    /// let syscall_id = trap_cx.x[17];
474    /// let arg0 = trap_cx.x[10];
475    ///
476    /// // 设置系统调用返回值
477    /// trap_cx.x[10] = result;
478    /// ```
479    pub fn get_trap_cx(&self) -> &'static mut TrapContext {
480        self.trap_cx_ppn.get_mut()
481    }
482
483    /// 获取用户地址空间令牌
484    ///
485    /// 返回当前进程地址空间的页表令牌，用于地址空间切换和地址转换。
486    /// 令牌包含页表模式、ASID 和根页表物理页号信息。
487    ///
488    /// ## Returns
489    ///
490    /// 返回可直接写入 `satp` 寄存器的令牌值
491    ///
492    /// ## satp 寄存器格式
493    ///
494    /// ```text
495    /// 63      60 59           44 43                     0
496    /// ┌─────────┐┌────────────────┐┌────────────────────────────────────────────┐
497    /// │  MODE   ││      ASID      ││                   PPN                      │
498    /// │ (4bit)  ││    (16bit)     ││                 (44bit)                    │
499    /// └─────────┘└────────────────┘└────────────────────────────────────────────┘
500    /// ```
501    ///
502    /// ## 使用场景
503    ///
504    /// - **任务切换**: 在任务切换时保存当前地址空间信息
505    /// - **系统调用**: 为地址转换提供用户态页表信息
506    /// - **内存管理**: 访问用户态内存时指定地址空间

540    ///

542    /// 僵尸进程不会被调度执行，但保留 TCB 以供父进程获取退出信息。
543    ///
544    /// ## Returns
545    ///
546    /// - `true` - 任务为僵尸状态，等待父进程回收
547    /// - `false` - 任务不是僵尸状态，可能正在运行或就绪
548    ///
549    /// ## 使用场景
550    ///
551    /// - **父进程管理**: 父进程检查子进程是否退出
552    /// - **进程清理**: 系统定期清理僵尸进程
553    /// - **wait 系统调用**: 等待子进程退出的实现
554    /// - **调度器**: 过滤不可调度的进程
555    ///
556    /// ## Examples
557    ///
558    /// ```rust

580    /// 从 ELF 文件创建新的任务控制块
581    ///
582    /// 解析给定的 ELF 可执行文件，创建完整的进程控制块。
583    /// 包括分配 PID、创建内核栈、建立地址空间和初始化陷阱上下文。
584    ///
585    /// ## Arguments
586    ///
587    /// * `elf_data` - ELF 文件的二进制数据切片，必须是有效的 ELF 格式
588    ///
589    /// ## Returns
590    ///
591    /// 返回初始化完成的任务控制块，状态为 [`TaskStatus::Ready`]
592    ///
593    /// ## 初始化过程
594    ///
595    /// ```text
596    /// 1. 解析 ELF 文件
597    ///    │
598    ///    │ - 提取程序段（.text, .data, .bss 等）
599    ///    │ - 获取入口点地址和用户栈指针
600    ///    ▼
601    /// 2. 建立地址空间
602    ///    │
603    ///    │ - 创建用户态地址空间
604    ///    │ - 映射程序段、用户栈、Trampoline 等
605    ///    ▼
606    /// 3. 分配系统资源
607    ///    │
608    ///    │ - 分配 PID
609    ///    │ - 创建内核栈
610    ///    ▼
611    /// 4. 初始化上下文
612    ///    │
613    ///    │ - 设置任务上下文（指向 trap_return）
614    ///    │ - 设置陷阱上下文（用户态寄存器初始值）
615    ///    ▼
616    /// 5. 进程创建完成
617    /// ```
618    ///
619    /// ## 内存布局
620    ///
621    /// 创建后的进程具有以下内存布局：
622    ///
623    /// ```text
624    /// 用户地址空间:
625    /// 高地址 TRAMPOLINE
626    /// ┌────────────────────┐
627    /// │    Trampoline      │ ← 用户态/内核态切换代码
628    /// ├────────────────────┤
629    /// │   Trap Context     │ ← 用户态寄存器状态
630    /// ├────────────────────┤
631    /// │    User Stack      │ ← 用户态栈空间
632    /// ├────────────────────┤
633    /// │    Guard Page      │ ← 防止栈溢出
634    /// ├────────────────────┤
635    /// │  Program Sections  │ ← .text/.data/.bss
636    /// └────────────────────┘
637    /// 低地址 0x10000
638    ///
639    /// 内核地址空间:
640    /// ┌────────────────────┐
641    /// │   Kernel Stack     │ ← 系统调用处理时使用
642    /// └────────────────────┘
643    /// ```
644    ///
645    /// ## 初始状态
646    ///
647    /// 新创建的任务具有以下初始状态：
648    /// - **任务状态**: [`TaskStatus::Ready`] - 准备执行
649    /// - **父进程**: `None` - 无父进程关系
650    /// - **子进程**: 空列表 - 暂无子进程
651    /// - **退出码**: 0 - 默认退出码
652    ///
653    /// ## Panics
654    ///
655    /// 在以下情况下会触发 panic：
656    /// - ELF 文件格式错误或无效
657    /// - PID 分配失败（系统 PID 资源耗尽）
658    /// - 内核栈分配失败（内存不足）
659    /// - 地址空间创建失败（虚拟内存系统错误）
660    ///
661    /// ## Examples
662    ///
663    /// ```rust
664    /// use alloc::sync::Arc;
665    ///
666    /// // 从应用程序数据创建任务
667    /// let app_data = get_app_data(0);
668    /// let task = Arc::new(TaskControlBlock::new(app_data));
669    ///
670    /// // 检查初始状态
671    /// {
672    ///     let inner = task.inner_exclusive_access();
673    ///     assert_eq!(inner.task_status, TaskStatus::Ready);
674    ///     assert_eq!(inner.exit_code, 0);
675    ///     assert!(inner.parent.is_none());
676    ///     assert!(inner.children.is_empty());
677    /// }
678    ///
679    /// println!("Created task with PID: {}", task.getpid());
680    /// ```
681    ///
682    /// ## 相关方法
683    ///
684    /// - [`fork()`] - 从现有进程创建子进程
685    /// - [`exec()`] - 替换当前进程的可执行文件
686    pub fn new(elf_data: &[u8]) -> Self {
687        let (memory_set, user_sp, entry_point) = MemorySet::from_elf(elf_data);
688        let trap_cx_ppn = memory_set
689            .translate(VirtAddr::from(TRAP_CONTEXT).into())
690            .unwrap()
691            .ppn();
692        let pid_handle = pid_alloc();
693        let kernel_stack = KernelStack::new(&pid_handle);
694        let kernel_stack_top = kernel_stack.get_top();
695        let task_control_block = Self {
696            pid: pid_handle,
697            kernel_stack,
698            inner: unsafe {
699                UPSafeCell::new(TaskControlBlockInner {
700                    task_status: TaskStatus::Ready,
701                    task_cx: TaskContext::goto_trap_return(kernel_stack_top),
702                    memory_set,
703                    trap_cx_ppn,
704                    base_size: user_sp,
705                    parent: None,
706                    children: Vec::new(),
707                    exit_code: 0,
708                })
709            },
710        };
711        let trap_cx = task_control_block.inner_exclusive_access().get_trap_cx();
712        *trap_cx = TrapContext::app_init_context(
713            entry_point,
714            user_sp,
715            KERNEL_SPACE.exclusive_access().token(),
716            kernel_stack_top,
717            trap_handler as usize,
718        );
719        task_control_block
720    }

722    /// 获取内部状态的排他访问权
723    ///
724    /// 返回 [`TaskControlBlockInner`] 的排他可变引用，用于安全地读取和修改
725    /// 任务的可变状态。在同一时刻只能有一个访问者获得访问权。
726    ///
727    /// ## Returns
728    ///
729    /// 返回 [`RefMut<TaskControlBlockInner>`]，提供对内部状态的排他访问
730    ///
731    /// ## 并发安全
732    ///
733    /// 通过 [`UPSafeCell`] 的互斥机制保证：
734    /// - **排他性**: 同一时刻只能有一个访问者
735    /// - **未欣正风险**: 编译时检查并发访问合法性
736    /// - **自动释放**: 引用超出作用域时自动释放锁
737    ///
738    /// ## 使用模式
739    ///
740    /// 应该尽可能缩短锁的持有时间，避免阵塞其他线程：
741    ///
742    /// ```rust
743    /// // 推荐的使用方式：短时间持有锁
744    /// {
745    ///     let mut inner = task.inner_exclusive_access();
746    ///     inner.task_status = TaskStatus::Running;
747    ///     // 其他对 inner 的操作...
748    /// } // 锁在此处自动释放
749    ///
750    /// // 不推荐：长时间持有锁
751    /// let mut inner = task.inner_exclusive_access();
752    /// // 长时间的计算或 I/O 操作...
753    /// heavy_computation();
754    /// inner.task_status = TaskStatus::Ready;
755    /// ```
756    ///
757    /// ## 常用操作
758    ///
759    /// 通过排他访问可以执行以下操作：
760    /// - **状态管理**: 修改任务状态和退出码
761    /// - **上下文访问**: 获取陷阱上下文和任务上下文
762    /// - **内存管理**: 访问和修改地址空间
763    /// - **进程关系**: 管理父子进程关系
764    ///
765    /// ## Examples
766    ///
767    /// ```rust
768    /// // 修改任务状态
769    /// {
770    ///     let mut inner = task.inner_exclusive_access();
771    ///     inner.task_status = TaskStatus::Running;
772    /// }
773    ///
774    /// // 访问陷阱上下文
775    /// {
776    ///     let inner = task.inner_exclusive_access();
777    ///     let trap_cx = inner.get_trap_cx();
778    ///     let syscall_id = trap_cx.x[17];
779    /// }
780    ///
781    /// // 检查进程状态
782    /// {
783    ///     let inner = task.inner_exclusive_access();
784    ///     if inner.is_zombie() {
785    ///         println!("Process exited with code: {}", inner.exit_code);
786    ///     }
787    /// }
788    /// ```
789    ///
790    /// ## 性能考虑
791    ///
792    /// - **锁争用**: 高频率访问可能导致性能下降
793    /// - **死锁防范**: 避免在持有锁时获取其他锁
794    /// - **内存开销**: `RefMut` 本身有较小的运行时开销
795    pub fn inner_exclusive_access(&self) -> RefMut<'_, TaskControlBlockInner> {
796        self.inner.exclusive_access()
797    }

800    ///
801    /// 返回当前任务的进程标识符 (PID)。PID 在系统中全局唯一，
802    /// 用于识别和区分不同的进程。

805    ///
806    /// 返回 `usize` 类型的 PID 值，PID 从 0 开始分配
807    ///
808    /// ## 特性
809    ///
810    /// - **唯一性**: PID 在系统运行期间唯一标识一个进程
811    /// - **不变性**: PID 在进程生命周期内不变
812    /// - **递增性**: PID 按创建顺序递增分配（在回收重用前）
813    /// - **无锁访问**: 获取 PID 不需要获取任何锁
814    ///
815    /// ## 使用场景
816    ///
817    /// - **进程管理**: 系统调用中识别调用进程
818    /// - **调试输出**: 日志和调试信息中显示进程 ID
819    /// - **进程通信**: 信号、管道等进程间通信机制
820    /// - **资源管理**: 进程资源统计和限制
821    /// - **安全检查**: 权限验证和访问控制
822    ///
823    /// ## 与 POSIX 兼容性
824    ///
825    /// 返回的 PID 值遵循 POSIX 约定：
826    /// - PID 0 通常保留给系统调度器
827    /// - PID 1 通常是 init 进程
828    /// - 正数 PID 表示正常用户进程
829    ///
830    /// ## Examples
831    ///
832    /// ```rust

834    /// let current_pid = task.getpid();
835    /// println!("Current process PID: {}", current_pid);
836    ///
837    /// // 在系统调用中使用
838    /// match syscall_id {
839    ///     SYS_GETPID => {
840    ///         let pid = current_task().unwrap().getpid();

842    ///         pid as isize
843    ///     }
844    ///     _ => -1,
845    /// }
846    ///
847    /// // 进程管理中的使用
848    /// fn kill_process(target_pid: usize) -> bool {
849    ///     for task in &task_list {
850    ///         if task.getpid() == target_pid {
851    ///             // 找到目标进程，执行终止操作
852    ///             return terminate_task(task);
853    ///         }
854    ///     }
855    ///     false // 未找到目标进程
856    /// }
857    ///
858    /// // 调试输出中的使用
859    /// println!("Task {} entering syscall {}", task.getpid(), syscall_id);
860    /// ```
861    ///

876    /// Fork 系统调用实现：创建子进程
877    ///
878    /// 复制当前进程创建一个新的子进程，子进程继承父进程的地址空间、文件描述符等资源。
879    /// 这是 UNIX/Linux 系统中创建新进程的主要方式，遵循经典的 fork 语义。
880    ///
881    /// ## Arguments
882    ///
883    /// * `self` - 父进程的 Arc 引用，用于建立父子关系
884    ///
885    /// ## Returns
886    ///
887    /// 返回新创建的子进程的 [`Arc<TaskControlBlock>`]
888    ///
889    /// ## Fork 语义
890    ///
891    /// **父子进程差异**：
892    /// - **PID**: 子进程分配新的 PID
893    /// - **内核栈**: 子进程分配独立的内核栈
894    /// - **返回值**: 在父进程中返回子进程 PID，在子进程中返回 0
895    ///
896    /// **共享与复制**：
897    /// - **地址空间**: 子进程获得父进程地址空间的完整副本
898    /// - **寄存器状态**: 子进程继承父进程当前的所有寄存器值
899    /// - **文件描述符**: 子进程继承父进程打开的文件描述符
900    /// - **工作目录**: 子进程继承父进程的工作目录
901    ///
902    /// ## Fork 过程详解
903    ///
904    /// ```text
905    /// 1. 复制地址空间
906    ///    │
907    ///    │ - 创建新的内存集合
908    ///    │ - 逐页复制父进程的所有内存内容
909    ///    │ - 建立独立的页表结构
910    ///    ▼
911    /// 2. 分配系统资源
912    ///    │
913    ///    │ - 分配新的 PID
914    ///    │ - 创建独立的内核栈
915    ///    │ - 分配新的 TCB
916    ///    ▼
917    /// 3. 建立父子关系
918    ///    │
919    ///    │ - 子进程记录父进程弱引用
920    ///    │ - 父进程添加子进程到 children 列表
921    ///    │ - 维护进程树结构
922    ///    ▼
923    /// 4. 初始化子进程状态
924    ///    │
925    ///    │ - 设置任务状态为 Ready
926    ///    │ - 复制任务上下文
927    ///    │ - 调整内核栈指针
928    ///    ▼
929    /// 5. 子进程创建完成
930    /// ```
931    ///
932    /// ## 内存布局对比
933    ///
934    /// ```text
935    /// Fork 前 (父进程):
936    /// ┌─────────────────┐
937    /// │   Parent Task   │
938    /// │     (PID n)     │
939    /// ├─────────────────┤
940    /// │  Address Space  │ ← 原始地址空间
941    /// │    Virtual      │
942    /// │     Memory      │
943    /// └─────────────────┘
944    ///
945    /// Fork 后:
946    /// ┌─────────────────┐    ┌─────────────────┐
947    /// │   Parent Task   │    │   Child Task    │
948    /// │     (PID n)     │    │    (PID n+1)    │
949    /// ├─────────────────┤    ├─────────────────┤
950    /// │  Address Space  │    │  Address Space  │ ← 完整副本
951    /// │    (Original)   │    │     (Copy)      │
952    /// │     Memory      │    │     Memory      │
953    /// └─────────────────┘    └─────────────────┘
954    ///            │                      ▲
955    ///            └──────────────────────┘
956    ///                 父子关系
957    /// ```
958    ///
959    /// ## 系统调用返回值处理
960    ///
961    /// Fork 的一个重要特征是一次调用两次返回：
962    ///
963    /// ```text
964    /// 调用时机：                父进程返回          子进程返回
965    /// ┌──────────┐             ┌──────────┐        ┌──────────┐
966    /// │ fork()   │────────────►│child_pid │        │    0     │
967    /// │ SysCall  │             │          │        │          │
968    /// └──────────┘             └──────────┘        └──────────┘
969    /// ```
970    ///
971    /// 返回值设置在陷阱上下文中的 `x[10]` 寄存器（RISC-V ABI 返回值寄存器）。
972    ///
973    /// ## 使用示例
974    ///
975    /// ```rust
976    /// use alloc::sync::Arc;
977    ///
978    /// // 父进程执行 fork
979    /// let parent_task = Arc::new(TaskControlBlock::new(parent_elf));
980    /// let child_task = parent_task.fork();
981    ///
982    /// // 检查父子关系
983    /// {
984    ///     let child_inner = child_task.inner_exclusive_access();
985    ///     assert!(child_inner.parent.is_some());
986    /// }
987    ///
988    /// {
989    ///     let parent_inner = parent_task.inner_exclusive_access();
990    ///     assert_eq!(parent_inner.children.len(), 1);
991    ///     assert_eq!(parent_inner.children[0].getpid(), child_task.getpid());
992    /// }
993    ///
994    /// println!("Parent PID: {}, Child PID: {}",
995    ///          parent_task.getpid(), child_task.getpid());
996    /// ```
997    ///
998    /// ## 错误处理
999    ///
1000    /// Fork 可能在以下情况失败：
1001    /// - **内存不足**: 无法分配子进程的地址空间或内核栈
1002    /// - **PID 耗尽**: 系统 PID 资源已用完
1003    /// - **系统限制**: 达到进程数量或内存使用限制
1004    ///
1005    /// ## 性能考虑
1006    ///
1007    /// - **写时复制 (COW)**: 某些系统实现 COW 优化，本实现为完整复制
1008    /// - **内存开销**: 完整复制地址空间会消耗大量内存
1009    /// - **时间开销**: 复制过程的时间与父进程地址空间大小成正比
1010    ///
1011    /// ## RISC-V 特定处理
1012    ///
1013    /// - **内核栈指针**: 更新子进程陷阱上下文中的内核栈指针
1014    /// - **寄存器继承**: 子进程继承父进程的所有通用寄存器值
1015    /// - **特权模式**: 子进程在用户模式下开始执行
1016    ///
1017    /// ## 相关系统调用
1018    ///
1019    /// - [`exec()`] - 替换进程映像，通常与 fork 配合使用
1020    /// - [`wait()`] - 父进程等待子进程退出
1021    /// - [`exit()`] - 进程正常退出
1022    pub fn fork(self: &Arc<Self>) -> Arc<Self> {
1023        let mut parent_inner = self.inner_exclusive_access();
1024        let memory_set = MemorySet::from_existed_user(&parent_inner.memory_set);
1025        let trap_cx_ppn = memory_set
1026            .translate(VirtAddr::from(TRAP_CONTEXT).into())
1027            .unwrap()
1028            .ppn();
1029        let pid_handle = pid_alloc();
1030        let kernel_stack = KernelStack::new(&pid_handle);
1031        let kernel_stack_top = kernel_stack.get_top();
1032        let task_control_block = Arc::new(TaskControlBlock {
1033            pid: pid_handle,
1034            kernel_stack,
1035            inner: unsafe {
1036                UPSafeCell::new(TaskControlBlockInner {
1037                    trap_cx_ppn,
1038                    base_size: parent_inner.base_size,
1039                    task_cx: TaskContext::goto_trap_return(kernel_stack_top),
1040                    task_status: TaskStatus::Ready,
1041                    memory_set,
1042                    parent: Some(Arc::downgrade(self)),
1043                    children: Vec::new(),
1044                    exit_code: 0,
1045                })
1046            },
1047        });
1048        parent_inner.children.push(task_control_block.clone());
1049        let trap_cx = task_control_block.inner_exclusive_access().get_trap_cx();
1050        trap_cx.kernel_sp = kernel_stack_top;
1051        task_control_block
1052    }

1054    /// Exec 系统调用实现：替换进程映像
1055    ///
1056    /// 使用新的可执行文件替换当前进程的内存映像，保持相同的 PID 和内核栈，
1057    /// 但完全替换地址空间和执行上下文。这是 UNIX/Linux 系统加载新程序的标准方式。
1058    ///
1059    /// ## Arguments
1060    ///
1061    /// * `elf_data` - 新程序的 ELF 文件二进制数据
1062    ///
1063    /// ## Exec 语义
1064    ///
1065    /// **保持不变的部分**：
1066    /// - **PID**: 进程标识符不变，仍然是同一个进程
1067    /// - **内核栈**: 复用现有的内核栈，不重新分配
1068    /// - **父子关系**: 进程在进程树中的位置不变
1069    ///
1070    /// **替换的部分**：
1071    /// - **地址空间**: 完全替换为新程序的内存布局
1072    /// - **执行上下文**: 重置为新程序的入口点和初始状态
1073    /// - **用户态寄存器**: 重新初始化为新程序的启动状态
1074    ///
1075    /// ## Exec 过程详解
1076    ///
1077    /// ```text
1078    /// 1. 解析新的 ELF 文件
1079    ///    │
1080    ///    │ - 验证 ELF 文件格式
1081    ///    │ - 提取程序段信息
1082    ///    │ - 获取入口点地址
1083    ///    ▼
1084    /// 2. 创建新的地址空间
1085    ///    │
1086    ///    │ - 销毁旧的内存映射
1087    ///    │ - 建立新程序的内存布局
1088    ///    │ - 加载程序段到内存
1089    ///    ▼
1090    /// 3. 更新进程状态
1091    ///    │
1092    ///    │ - 更新内存集合
1093    ///    │ - 更新陷阱上下文页号
1094    ///    │ - 重新设置基础内存大小
1095    ///    ▼
1096    /// 4. 初始化执行环境
1097    ///    │
1098    ///    │ - 设置程序计数器为入口点
1099    ///    │ - 初始化用户栈指针
1100    ///    │ - 清零通用寄存器
1101    ///    ▼
1102    /// 5. 程序替换完成
1103    /// ```
1104    ///
1105    /// ## 内存布局变化
1106    ///
1107    /// ```text
1108    /// Exec 前 (旧程序):
1109    /// ┌─────────────────────────┐
1110    /// │      Task (PID n)       │ ← PID 保持不变
1111    /// ├─────────────────────────┤
1112    /// │     Old Program         │
1113    /// │   Address Space:        │
1114    /// │  ┌──────────────────┐   │
1115    /// │  │ Old .text/.data  │   │ ← 将被完全替换
1116    /// │  │ Old User Stack   │   │
1117    /// │  │ Old Heap         │   │
1118    /// │  └──────────────────┘   │
1119    /// └─────────────────────────┘
1120    ///
1121    /// Exec 后 (新程序):
1122    /// ┌─────────────────────────┐
1123    /// │      Task (PID n)       │ ← 同一个 PID
1124    /// ├─────────────────────────┤
1125    /// │     New Program         │
1126    /// │   Address Space:        │
1127    /// │  ┌──────────────────┐   │
1128    /// │  │ New .text/.data  │   │ ← 新程序的内存布局
1129    /// │  │ New User Stack   │   │
1130    /// │  │ New Heap         │   │
1131    /// │  └──────────────────┘   │
1132    /// └─────────────────────────┘
1133    /// ```
1134    ///
1135    /// ## 上下文重置
1136    ///
1137    /// Exec 会重置进程的执行上下文：
1138    ///
1139    /// ```text
1140    /// 重置项目                 新值
1141    /// ┌──────────────────┐    ┌─────────────────────┐
1142    /// │Program Counter   │───►│ New Entry Point     │
1143    /// ├──────────────────┤    ├─────────────────────┤
1144    /// │Stack Pointer     │───►│ New User Stack Top  │
1145    /// ├──────────────────┤    ├─────────────────────┤
1146    /// │General Registers │───►│ Zero or Init Values │
1147    /// ├──────────────────┤    ├─────────────────────┤
1148    /// │Page Table Ptr    │───►│ New Address Space   │
1149    /// └──────────────────┘    └─────────────────────┘
1150    /// ```
1151    ///
1152    /// ## 典型使用场景
1153    ///
1154    /// **Shell 命令执行**：
1155    /// ```text
1156    /// 1. Shell 进程执行 fork() 创建子进程
1157    /// 2. 子进程执行 exec() 加载目标程序
1158    /// 3. 父进程 (Shell) 执行 wait() 等待子进程完成
1159    /// ```
1160    ///
1161    /// **程序动态加载**：
1162    /// ```text
1163    /// 1. 当前进程不再需要原程序代码
1164    /// 2. 直接执行 exec() 替换为新程序
1165    /// 3. 新程序从头开始执行
1166    /// ```
1167    ///
1168    /// ## 使用示例
1169    ///
1170    /// ```rust
1171    /// // 替换当前进程为新程序
1172    /// let new_program = get_app_data("target_app");
1173    ///
1174    /// // 记录替换前的信息
1175    /// let old_pid = task.getpid();
1176    /// println!("Executing new program in PID {}", old_pid);
1177    ///
1178    /// // 执行替换
1179    /// task.exec(new_program);
1180    ///
1181    /// // 验证 PID 未变但程序已替换
1182    /// assert_eq!(task.getpid(), old_pid);
1183    ///
1184    /// // 检查新的执行状态
1185    /// {
1186    ///     let inner = task.inner_exclusive_access();
1187    ///     let trap_cx = inner.get_trap_cx();
1188    ///     println!("New entry point: 0x{:x}", trap_cx.sepc);
1189    ///     println!("New stack pointer: 0x{:x}", trap_cx.x[2]);
1190    /// }
1191    /// ```
1192    ///
1193    /// ## 安全考虑
1194    ///
1195    /// - **权限检查**: 确保有权限执行目标文件
1196    /// - **格式验证**: 验证 ELF 文件的完整性和合法性
1197    /// - **资源清理**: 确保旧程序的所有资源得到正确释放
1198    /// - **状态一致性**: 保证替换过程的原子性
1199    ///
1200    /// ## 错误处理
1201    ///
1202    /// Exec 可能在以下情况失败：
1203    /// - **文件格式错误**: ELF 文件格式无效或损坏
1204    /// - **内存不足**: 无法为新程序分配足够的内存空间
1205    /// - **权限不足**: 没有执行目标文件的权限
1206    /// - **系统资源限制**: 超出系统资源限制
1207    ///
1208    /// ## 性能特性
1209    ///
1210    /// - **内存复用**: 复用现有的 PID 和内核栈资源
1211    /// - **快速切换**: 相比 fork + exec 模式，单独 exec 更高效
1212    /// - **内存释放**: 自动释放旧程序占用的所有内存
1213    ///
1214    /// ## 与其他系统调用的关系
1215    ///
1216    /// - **fork() + exec()**: 经典的进程创建和程序加载模式
1217    /// - **wait()**: 父进程等待 exec 后的子进程完成
1218    /// - **exit()**: 进程执行完成后的正常退出
1219    pub fn exec(&self, elf_data: &[u8]) {
1220        let (memory_set, user_sp, entry_point) = MemorySet::from_elf(elf_data);
1221        let trap_cx_ppn = memory_set
1222            .translate(VirtAddr::from(TRAP_CONTEXT).into())
1223            .unwrap()
1224            .ppn();
1225        let mut inner = self.inner_exclusive_access();
1226        inner.memory_set = memory_set;
1227        inner.trap_cx_ppn = trap_cx_ppn;
1228        inner.base_size = user_sp;
1229        let trap_cx = inner.get_trap_cx();
1230        *trap_cx = TrapContext::app_init_context(
1231            entry_point,
1232            user_sp,
1233            KERNEL_SPACE.exclusive_access().token(),
1234            self.kernel_stack.get_top(),
1235            trap_handler as usize,
1236        );
1237    }
