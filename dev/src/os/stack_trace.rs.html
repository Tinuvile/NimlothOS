1//! # 栈追踪和调试模块
2//!
3//! 提供运行时栈追踪功能，用于调试 panic 和异常情况。
4//! 通过解析栈帧链表来重构函数调用链，帮助定位问题代码。
5//!
6//! ## 实现原理
7//!
8//! ### RISC-V 栈帧结构
9//!
10//! ```text
11//! 高地址 (栈顶)
12//! ┌─────────────────┐
13//! │ Local Variables │
14//! ├─────────────────┤
15//! │   saved s0      │ <- fp - 16 (前一个栈帧的 fp)
16//! ├─────────────────┤
17//! │   saved ra      │ <- fp - 8  (返回地址)
18//! ├─────────────────┤ <- 当前 fp (s0)
19//! │      ...        │
20//! └─────────────────┘
21//! 低地址 (栈底)
22//! ```
23//!
24//! ### 追踪过程
25//!
26//! 1. 从当前的 `s0` 寄存器获取栈帧指针
27//! 2. 读取栈帧中保存的返回地址 (`ra`) 和前一个栈帧指针
28//! 3. 沿着栈帧链表向上遍历，直到遇到空指针
29//! 4. 输出每一级函数调用的地址信息
30//!
31//! ## 安全注意事项
32//!
33//! - 栈追踪涉及大量 `unsafe` 指针操作
34//! - 在栈损坏或异常情况下可能导致进一步的错误
35//! - 当前实现假设栈帧结构符合标准约定
36
37use core::{arch::asm, ptr::null};
38
39use crate::println;
40
41/// 打印函数调用栈追踪信息
42///
43/// 通过遍历栈帧链表来重构函数调用链，输出每一级调用的地址信息。
44/// 主要用于调试 panic 和异常情况，帮助开发者定位问题代码的位置。
45///
46/// ## 工作原理
47///
48/// 1. **获取当前栈帧**: 从 `s0` 寄存器读取当前栈帧指针
49/// 2. **遍历栈帧链**: 沿着保存的栈帧指针向上遍历
50/// 3. **提取地址信息**: 从每个栈帧读取返回地址和前一个栈帧指针
51/// 4. **格式化输出**: 将地址信息以十六进制格式输出
52///
53/// ## 输出格式
54///
55/// ```text
56/// === Call Stack Trace Tool ===
57/// 0x0000000080200abc (0x000000008020def0)
58/// 0x0000000080201234 (0x000000008020e100)
59/// ...
60/// === End of Call Stack Trace ===
61/// ```
62///
63/// 每行包含：
64/// - 第一个地址：函数返回地址 (ra)，指向调用点之后的指令
65/// - 第二个地址：前一个栈帧指针 (fp)
66///
67/// ## Safety
68///
69/// 这是一个 `unsafe` 函数，因为它：
70/// - 直接访问和解引用栈指针
71/// - 假设栈帧结构符合 RISC-V ABI 约定
72/// - 在栈损坏的情况下可能导致内存访问错误
73///
74/// ## 栈帧布局假设
75///
76/// 该函数假设每个栈帧遵循以下布局：
77/// - `fp - 8`: 保存的返回地址 (ra)
78/// - `fp - 16`: 保存的前一个栈帧指针 (s0)
79///
80/// ## 限制
81///
82/// - 只能追踪使用标准栈帧的函数
83/// - 内联函数和优化代码可能不会出现在追踪中
84/// - 叶子函数（不调用其他函数）可能没有标准栈帧
85/// - 栈指针损坏会导致不准确或危险的输出
86///
87/// ## Usage
88///
89/// ```rust
90/// // 在 panic 处理程序中使用
91/// unsafe {
92///     print_stack_trace();
93/// }
94/// ```
95///
96/// ## Note
97///
98/// - 当前该功能在 `lang_items.rs` 中被注释掉，可根据需要启用
99/// - 在生产环境中使用时应谨慎，可能影响系统稳定性
100/// - 地址信息需要结合符号表或反汇编工具才能转换为具体的函数名
101#[allow(dead_code)]
102pub unsafe fn print_stack_trace() {
103    let mut fp: *const usize;
104
105    // 获取当前栈帧指针 (s0 寄存器)
106    unsafe {
107        asm!("mv {}, s0", out(reg) fp);
108    }
109
110    println!("=== Call Stack Trace Tool ===");
111
112    // 遍历栈帧链表
113    while fp != null() {
114        // 从栈帧中读取保存的返回地址和前一个栈帧指针
115        unsafe {
116            let saved_ra = *fp.sub(1); // fp - 8 字节处的返回地址
117            let saved_fp = *fp.sub(2); // fp - 16 字节处的栈帧指针
118
119            println!("0x{:016x} (0x{:016x})", saved_ra, saved_fp);
120
121            // 移动到前一个栈帧
122            fp = saved_fp as *const usize;
123        }
124    }
125
126    println!("=== End of Call Stack Trace ===");
127}
