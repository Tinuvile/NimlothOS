1//! # 内核堆内存分配器
2//!
3//! 提供动态内存分配功能，支持 `Box`、`Vec` 等需要堆内存的 Rust 标准库功能。
4//! 使用 Buddy System 算法进行高效的内存管理。
5//!
6//! ## 实现细节
7//!
8//! - **分配算法**: Buddy System，支持快速分配和合并
9//! - **内存大小**: 3MB (0x300000 字节)
10//! - **最大块大小**: 2^32 = 4GB
11//! - **线程安全**: 使用 `LockedHeap` 保证多线程安全
12
13use crate::config::KERNEL_HEAP_SIZE;
14use buddy_system_allocator::LockedHeap;
15use core::ptr::addr_of_mut;
16
17/// 全局堆分配器实例
18///
19/// 使用 Buddy System 分配算法，最大支持 2^32 字节的单次分配。
20/// 该分配器是线程安全的，内部使用自旋锁保护。
21#[global_allocator]
22static HEAP_ALLOCATOR: LockedHeap<32> = LockedHeap::empty();
23
24/// 内核堆内存空间
25///
26/// 静态分配的内存区域，作为动态内存分配的后备存储。
27/// 大小为 [`KERNEL_HEAP_SIZE`] (3MB)。
28static mut HEAP_SPACE: [u8; KERNEL_HEAP_SIZE] = [0; KERNEL_HEAP_SIZE];
29
30/// 初始化内核堆分配器
31///
32/// 将静态分配的 [`HEAP_SPACE`] 内存区域注册到堆分配器中，
33/// 使其可用于动态内存分配。
34///
35/// 必须在使用任何需要堆内存的功能（如 `Box`、`Vec`）之前调用。
36///
37/// ## Safety
38///
39/// 该函数使用 `unsafe` 代码访问可变静态变量 `HEAP_SPACE`。
40/// 调用者必须确保：
41/// - 只在内核初始化阶段调用一次
42/// - 调用时没有其他代码正在访问 `HEAP_SPACE`
43pub fn init_heap() {
44    unsafe {
45        HEAP_ALLOCATOR
46            .lock()
47            .init(addr_of_mut!(HEAP_SPACE) as usize, KERNEL_HEAP_SIZE);
48    }
49}
50
51/// 堆分配错误处理器
52///
53/// 当动态内存分配失败时，此函数会被调用。
54/// 由于操作系统内核中的内存分配失败通常是致命错误，
55/// 因此直接触发 panic 来终止系统运行。
56///
57/// ## Arguments
58///
59/// * `layout` - 导致分配失败的内存布局信息，包含大小和对齐要求
60///
61/// ## Panics
62///
63/// 该函数总是触发 panic，不会返回
64#[alloc_error_handler]
65pub fn alloc_error_handler(layout: core::alloc::Layout) -> ! {
66    panic!("Heap allocation error, layout = {:?}", layout);
67}
