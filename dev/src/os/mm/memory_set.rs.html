1//! # 内存集合管理模块
2//!
3//! 提供完整的地址空间管理功能，包括内存映射区域的创建、管理和地址空间切换。
4//! 支持 RISC-V SV39 分页机制，实现内核和用户程序的地址空间隔离。
5//!
6//! ## 核心概念
7//!
8//! - [`MemorySet`] - 完整的地址空间，包含页表和多个内存映射区域
9//! - [`MapArea`] - 单个连续的内存映射区域，具有统一的映射类型和权限
10//! - [`MapType`] - 映射类型：恒等映射或帧映射
11//! - [`MapPermission`] - 内存访问权限：读、写、执行、用户态访问
12//!
13//! ## 地址空间布局
14//!
15//! ### 内核地址空间 (低256G)
16//! ```text
17//! ┌──────────────────────────────────--───────────────────────────┐
18//! │                   Kernel Address Space                        │
19//! ├─────────────────┬──────────────-─┬──────────-───┬─────────────┤
20//! │  .text Section  │ .rodata Section│ .data Section│ .bss Section│
21//! │     (R+X)       │      (R)       │    (R+W)     │    (R+W)    │
22//! ├─────────────────┴───────────────-┴───────────-──┴─────────────┤
23//! │                  Physical Memory Mapping                      │
24//! │                        (R+W)                                  │
25//! └─────────────────────────────────────────────────--────────────┘
26//! ```
27//!
28//! ### 用户地址空间
29//! ```text
30//! 高地址 (TRAMPOLINE)
31//! ┌──────────────────────────────────────────────────┐
32//! │                   Trampoline                     │
33//! │                    (R+X)                         │
34//! ├──────────────────────────────────────────────────┤
35//! │                 Trap Context                     │
36//! │                    (R+W)                         │
37//! ├──────────────────────────────────────────────────┤
38//! │                  User Stack                      │
39//! │                   (R+W+U)                        │
40//! ├──────────────────────── ─────────────────────────┤
41//! │                     ...                          │
42//! ├──────────────────────────────────────────────────┤
43//! │               Program Sections                   │
44//! │            (.text/.data/.bss etc)                │
45//! │              (Based on ELF flags)                │
46//! └──────────────────────────────────────────────────┘
47//! 低地址 (0x10000)
48//! ```
49//!
50//! ## 映射类型
51//!
52//! ### Identical 映射 (恒等映射)
53//! - **特点**: 虚拟地址 = 物理地址
54//! - **用途**: 内核直接访问物理内存和设备
55//! - **优势**: 简单、高效、地址可预测
56//! - **限制**: 无法提供地址空间隔离
57//!
58//! ### Framed 映射 (帧映射)
59//! - **特点**: 每个虚拟页映射到独立分配的物理页帧
60//! - **用途**: 用户程序地址空间、内核堆
61//! - **优势**: 灵活、安全、支持地址空间隔离
62//! - **成本**: 需要额外的页表和页帧管理
63//!
64//! ## 使用示例
65//!
66//! ```rust
67//! use crate::mm::{MemorySet, MapArea, MapType, MapPermission};
68//!
69//! // 创建新的地址空间
70//! let mut memory_set = MemorySet::new_bare();
71//!
72//! // 添加内存映射区域
73//! memory_set.insert_framed_area(
74//!     VirtAddr::from(0x10000000),
75//!     VirtAddr::from(0x10001000),
76//!     MapPermission::R | MapPermission::W | MapPermission::U
77//! );
78//!
79//! // 激活地址空间
80//! memory_set.activate();
81//! ```

133/// 内存映射区域
134///
135/// 表示地址空间中一个连续的虚拟内存区域，具有统一的映射类型和访问权限。
136/// 每个 `MapArea` 管理一段虚拟页号范围，并维护其到物理页帧的映射关系。
137///
138/// ## 字段说明
139///
140/// - `vpn_range`: 虚拟页号范围，定义区域的边界
141/// - `data_frames`: 虚拟页号到物理页帧的映射表（仅用于 Framed 映射）
142/// - `map_type`: 映射类型（恒等映射或帧映射）
143/// - `map_perm`: 访问权限（读/写/执行/用户态）
144///
145/// ## 设计原理
146///
147/// `MapArea` 将连续的虚拟地址范围抽象为统一的管理单元，简化了
148/// 大块内存区域的映射和权限管理。通过 `BTreeMap` 维护页面映射关系，
149/// 既保证了查找效率，又支持有序遍历。
150pub struct MapArea {
151    vpn_range: VPNRange,
152    data_frames: BTreeMap<VirtPageNum, FrameTracker>,
153    map_type: MapType,
154    map_perm: MapPermission,
155}

157/// 内存集合（地址空间）
158///
159/// 表示一个完整的虚拟地址空间，包含页表和多个内存映射区域。
160/// 每个进程都有独立的 `MemorySet`，实现地址空间隔离。
161///
162/// ## 组成部分
163///
164/// - `page_table`: 多级页表，负责虚拟地址到物理地址的转换
165/// - `areas`: 内存映射区域列表，每个区域有独立的映射类型和权限
166///
167/// ## 生命周期管理
168///
169/// `MemorySet` 拥有其包含的所有页表和物理页帧，当对象被销毁时，
170/// 会自动释放相关的物理内存资源，确保内存安全。
171///
172/// ## 并发安全
173///
174/// 内核地址空间通过 `KERNEL_SPACE` 全局变量管理，使用 `UPSafeCell`
175/// 提供单处理器环境下的安全可变访问。
176pub struct MemorySet {
177    page_table: PageTable,
178    areas: Vec<MapArea>,
179}

181/// 内存映射类型
182///
183/// 定义虚拟页面到物理页帧的映射方式，影响地址转换的行为和性能。
184///
185/// ## 映射类型对比
186///
187/// | 类型 | 地址关系 | 用途 | 优势 | 劣势 |
188/// |------|----------|------|------|------|
189/// | Identical | VA = PA | 内核段、设备 | 简单、高效 | 无隔离 |
190/// | Framed | VA ≠ PA | 用户程序、堆 | 灵活、安全 | 复杂、开销 |
191#[derive(Copy, Clone, PartialEq, Debug)]
192pub enum MapType {
193    /// 恒等映射
194    ///
195    /// 虚拟地址直接等于物理地址，不需要额外的物理页帧分配。
196    /// 主要用于内核代码段、数据段以及设备内存映射。
197    ///
198    /// ## 特点
199    /// - 地址转换开销最小
200    /// - 地址可预测，便于调试
201    /// - 适合需要直接访问物理地址的场景
202    Identical,
203
204    /// 帧映射
205    ///
206    /// 每个虚拟页面映射到独立分配的物理页帧，提供完整的
207    /// 地址空间隔离和内存保护功能。
208    ///
209    /// ## 特点
210    /// - 支持地址空间隔离
211    /// - 可以实现写时复制、懒加载等高级功能
212    /// - 需要维护虚拟页到物理页的映射表
213    Framed,
214}

216bitflags! {
217    /// 内存映射权限
218    ///
219    /// 定义内存区域的访问权限，直接对应 RISC-V 页表项的权限位。
220    /// 权限检查由硬件 MMU 执行，违反权限将触发页面异常。
221    ///
222    /// ## 权限位布局
223    ///
224    /// ```text
225    /// 权限位布局:
226    /// ┌─────┬─────┬─────┬─────┬─────┬─────┬─────┬─────┐
227    /// │  7  │  6  │  5  │  4  │  3  │  2  │  1  │  0  │
228    /// └─────┴─────┴─────┴─────┴─────┴─────┴─────┴─────┘
229    ///                     ↑     ↑     ↑     ↑
230    ///                     U     X     W     R
231    /// ```
232    ///
233    /// ## 常用权限组合
234    ///
235    /// - `R`: 只读数据段 (.rodata)
236    /// - `R | W`: 可读写数据段 (.data, .bss)
237    /// - `R | X`: 只读代码段 (.text)
238    /// - `R | W | U`: 用户态可读写区域
239    /// - `R | W | X | U`: 用户态全权限区域
240    ///
241    /// ## 安全约束
242    ///
243    /// - 不能单独设置 `W` 权限（RISC-V 规范要求）
244    /// - `U` 权限允许用户态访问，否则仅内核态可访问
245    /// - 权限检查在每次内存访问时由硬件执行
246    pub struct MapPermission: u8 {
247        /// 可读权限 (Read)
248        ///
249        /// 允许从该内存区域读取数据。几乎所有有用的内存区域
250        /// 都需要此权限，除非是特殊的只写设备寄存器。
251        const R = 1 << 1;
252
253        /// 可写权限 (Write)
254        ///
255        /// 允许向该内存区域写入数据。根据 RISC-V 规范，
256        /// 设置写权限时必须同时设置读权限。
257        const W = 1 << 2;
258
259        /// 可执行权限 (Execute)
260        ///
261        /// 允许从该内存区域获取并执行指令。通常用于
262        /// 代码段，不应与写权限同时设置（W^X 原则）。
263        const X = 1 << 3;
264
265        /// 用户态访问权限 (User)
266        ///
267        /// 允许用户态程序访问该内存区域。不设置此权限时，
268        /// 只有内核态（S-mode）可以访问。
269        const U = 1 << 4;
270    }
271}

273impl MapArea {
274    /// 创建新的内存映射区域
275    ///
276    /// 根据指定的虚拟地址范围、映射类型和访问权限创建一个新的内存映射区域。
277    /// 地址范围会自动对齐到页边界。
278    ///
279    /// ## Arguments
280    ///
281    /// * `start_va` - 区域起始虚拟地址
282    /// * `end_va` - 区域结束虚拟地址（不包含）
283    /// * `map_type` - 映射类型（恒等映射或帧映射）
284    /// * `map_perm` - 内存访问权限
285    ///
286    /// ## Returns
287    ///
288    /// 新创建的内存映射区域，尚未进行实际的页面映射
289    ///
290    /// ## 地址对齐
291    ///
292    /// - `start_va` 向下对齐到页边界（使用 `floor()`）
293    /// - `end_va` 向上对齐到页边界（使用 `ceil()`）
294    /// - 确保整个区域以页为单位进行管理
295    ///
296    /// ## Examples
297    ///
298    /// ```rust
299    /// // 创建用户态可读写区域
300    /// let area = MapArea::new(
301    ///     VirtAddr::from(0x10000000),
302    ///     VirtAddr::from(0x10001000),
303    ///     MapType::Framed,
304    ///     MapPermission::R | MapPermission::W | MapPermission::U
305    /// );
306    ///
307    /// // 创建内核代码段（恒等映射）
308    /// let text_area = MapArea::new(
309    ///     VirtAddr::from(stext as usize),
310    ///     VirtAddr::from(etext as usize),
311    ///     MapType::Identical,
312    ///     MapPermission::R | MapPermission::X
313    /// );
314    /// ```
315    pub fn new(
316        start_va: VirtAddr,
317        end_va: VirtAddr,
318        map_type: MapType,
319        map_perm: MapPermission,
320    ) -> Self {
321        let start_vpn: VirtPageNum = start_va.floor();
322        let end_vpn: VirtPageNum = end_va.ceil();
323        Self {
324            vpn_range: VPNRange::new(start_vpn, end_vpn),
325            data_frames: BTreeMap::new(),
326            map_type,
327            map_perm,
328        }
329    }

331    /// 将整个内存区域映射到页表
332    ///
333    /// 遍历区域内的所有虚拟页号，为每个页面建立虚拟地址到物理地址的映射。
334    /// 根据映射类型的不同，采用不同的物理页面分配策略。
335    ///
336    /// ## Arguments
337    ///
338    /// * `page_table` - 目标页表的可变引用
339    ///
340    /// ## 映射行为
341    ///
342    /// - **Identical映射**: 虚拟页号直接作为物理页号使用
343    /// - **Framed映射**: 为每个虚拟页面分配新的物理页帧
344    ///
345    /// ## Panics
346    ///
347    /// 如果物理页帧分配失败（内存不足），函数会触发 panic
348    ///
349    /// ## Examples
350    ///
351    /// ```rust
352    /// let mut area = MapArea::new(
353    ///     VirtAddr::from(0x10000000),
354    ///     VirtAddr::from(0x10001000),
355    ///     MapType::Framed,
356    ///     MapPermission::R | MapPermission::W
357    /// );
358    /// let mut page_table = PageTable::new();
359    /// area.map(&mut page_table); // 建立所有页面映射
360    /// ```
361    pub fn map(&mut self, page_table: &mut PageTable) {
362        for vpn in self.vpn_range {
363            self.map_one(page_table, vpn);
364        }
365    }

367    /// 从页表中取消整个内存区域的映射
368    ///
369    /// 遍历区域内的所有虚拟页号，移除每个页面的虚拟地址到物理地址映射。
370    /// 对于 Framed 映射，还会释放对应的物理页帧。
371    ///
372    /// ## Arguments
373    ///
374    /// * `page_table` - 目标页表的可变引用
375    ///
376    /// ## 清理行为
377    ///
378    /// - **Identical映射**: 仅移除页表项，不释放物理页面
379    /// - **Framed映射**: 移除页表项并释放所有分配的物理页帧
380    ///
381    /// ## 内存安全
382    ///
383    /// 函数确保所有相关的物理内存资源得到正确释放，防止内存泄漏。
384    /// 通过 `FrameTracker` 的 RAII 机制自动管理页帧生命周期。
385    #[allow(unused)]
386    pub fn unmap(&mut self, page_table: &mut PageTable) {
387        for vpn in self.vpn_range {
388            self.unmap_one(page_table, vpn);
389        }
390    }

392    /// 将数据复制到映射区域
393    ///
394    /// 将给定的字节数据复制到已映射的虚拟内存区域中。数据会按页面大小分块复制，
395    /// 跨越多个物理页帧。仅支持 Framed 映射类型。
396    ///
397    /// ## Arguments
398    ///
399    /// * `page_table` - 用于地址转换的页表引用
400    /// * `data` - 要复制的源数据
401    ///
402    /// ## 复制过程
403    ///
404    /// 1. 从区域起始虚拟页号开始
405    /// 2. 每次复制一个页面大小的数据（最多 4KB）
406    /// 3. 通过页表转换获取物理页帧地址
407    /// 4. 直接写入物理内存
408    ///
409    /// ## Panics
410    ///
411    /// - 如果映射类型不是 `MapType::Framed`
412    /// - 如果虚拟地址转换失败
413    /// - 如果数据长度超出区域范围
414    ///
415    /// ## Examples
416    ///
417    /// ```rust
418    /// // 创建并映射区域
419    /// let mut area = MapArea::new(
420    ///     VirtAddr::from(0x10000000),
421    ///     VirtAddr::from(0x10002000),
422    ///     MapType::Framed,
423    ///     MapPermission::R | MapPermission::W
424    /// );
425    /// area.map(&mut page_table);
426    ///
427    /// // 复制 ELF 文件数据
428    /// let elf_data = &[0x7f, 0x45, 0x4c, 0x46, /* ... */];
429    /// area.copy_data(&page_table, elf_data);
430    /// ```
431    pub fn copy_data(&mut self, page_table: &PageTable, data: &[u8]) {
432        assert_eq!(self.map_type, MapType::Framed);
433        let mut start: usize = 0;
434        let mut current_vpn = self.vpn_range.get_start();
435        let len = data.len();
436        loop {
437            let src = &data[start..len.min(start + PAGE_SIZE)];
438            let dst = &mut page_table
439                .translate(current_vpn)
440                .unwrap()
441                .ppn()
442                .get_bytes_array()[..src.len()];
443            dst.copy_from_slice(src);
444            start += PAGE_SIZE;
445            if start >= len {
446                break;
447            }
448            current_vpn.step();
449        }
450    }

452    /// 映射单个虚拟页面
453    ///
454    /// 为指定的虚拟页号建立到物理页帧的映射。根据映射类型的不同，
455    /// 采用不同的物理页面分配策略。
456    ///
457    /// ## Arguments
458    ///
459    /// * `page_table` - 目标页表的可变引用
460    /// * `vpn` - 要映射的虚拟页号
461    ///
462    /// ## 映射策略
463    ///
464    /// ### Identical 映射
465    /// - 直接将虚拟页号作为物理页号使用
466    /// - 不需要分配新的物理页帧
467    /// - 不更新 `data_frames` 映射表
468    ///
469    /// ### Framed 映射
470    /// - 分配新的物理页帧
471    /// - 将虚拟页号到页帧的映射记录在 `data_frames` 中
472    /// - 通过 `FrameTracker` 管理页帧生命周期
473    ///
474    /// ## Panics
475    ///
476    /// - Framed 映射时如果物理页帧分配失败
477    /// - 权限位转换失败
478    ///
479    /// ## Examples
480    ///
481    /// ```rust
482    /// // 映射单个页面
483    /// let vpn = VirtPageNum(0x10000);
484    /// area.map_one(&mut page_table, vpn);
485    /// ```
486    pub fn map_one(&mut self, page_table: &mut PageTable, vpn: VirtPageNum) {
487        let ppn: PhysPageNum;
488        match self.map_type {
489            MapType::Identical => {
490                ppn = PhysPageNum(vpn.0);
491            }
492            MapType::Framed => {
493                let frame = frame_alloc().unwrap();
494                ppn = frame.ppn;
495                self.data_frames.insert(vpn, frame);
496            }
497        }
498        let pte_flags = PTEFlags::from_bits(self.map_perm.bits()).unwrap();
499        page_table.map(vpn, ppn, pte_flags);
500    }

504    /// 移除指定虚拟页号到物理页帧的映射。对于 Framed 映射，
505    /// 还会释放对应的物理页帧。
506    ///
507    /// ## Arguments
508    ///
509    /// * `page_table` - 目标页表的可变引用
510    /// * `vpn` - 要取消映射的虚拟页号
511    ///
512    /// ## 清理策略
513    ///
514    /// ### Identical 映射
515    /// - 仅从页表中移除页表项
516    /// - 不释放任何物理页面（因为不是独占所有）
517    ///
518    /// ### Framed 映射
519    /// - 从 `data_frames` 中移除映射记录
520    /// - 调用 `frame_dealloc()` 释放物理页帧
521    /// - 从页表中移除页表项
522    ///
523    /// ## 内存安全
524    ///
525    /// 通过 `FrameTracker` 的 RAII 机制自动管理物理页帧的生命周期，
526    /// 确保在取消映射时正确释放内存资源。
527    ///
528    /// ## Panics
529    ///
530    /// 如果 Framed 映射中找不到对应的页帧记录
531    #[allow(unused)]
532    pub fn unmap_one(&mut self, page_table: &mut PageTable, vpn: VirtPageNum) {
533        match self.map_type {
534            MapType::Identical => {}
535            MapType::Framed => {
536                let frame = self.data_frames.remove(&vpn).unwrap();
537                // frame_dealloc(frame.ppn);
538            }
539        }
540        page_table.unmap(vpn);
541    }

543    /// 缩小内存区域到指定结束位置
544    ///
545    /// 将内存区域的结束边界缩小到指定的虚拟页号，取消被移除部分的所有映射。
546    /// 主要用于内存回收和动态内存管理。
547    ///
548    /// ## Arguments
549    ///
550    /// * `page_table` - 目标页表的可变引用
551    /// * `new_end` - 新的结束虚拟页号（不包含）
552    ///
553    /// ## 操作过程
554    ///
555    /// 1. 计算需要移除的页面范围：`[new_end, old_end)`
556    /// 2. 对范围内的每个页面调用 `unmap_one()`
557    /// 3. 更新区域的 `vpn_range` 边界
558    ///
559    /// ## 内存安全
560    ///
561    /// 所有被移除的 Framed 映射页面对应的物理页帧会被自动释放，
562    /// 防止内存泄漏。
563    ///
564    /// ## Panics
565    ///
566    /// 如果 `new_end` 大于当前区域的结束位置
567    ///
568    /// ## Examples
569    ///
570    /// ```rust
571    /// // 原区域：[0x1000, 0x3000)
572    /// // 缩小到：[0x1000, 0x2000)
573    /// area.shrink_to(&mut page_table, VirtPageNum(0x2000));
574    /// // 页面 [0x2000, 0x3000) 被取消映射并释放
575    /// ```
576    #[allow(unused)]
577    pub fn shrink_to(&mut self, page_table: &mut PageTable, new_end: VirtPageNum) {
578        for vpn in VPNRange::new(new_end, self.vpn_range.get_end()) {
579            self.unmap_one(page_table, vpn)
580        }
581        self.vpn_range = VPNRange::new(self.vpn_range.get_start(), new_end);
582    }

584    /// 扩展内存区域到指定结束位置
585    ///
586    /// 将内存区域的结束边界扩展到指定的虚拟页号，为新增部分建立映射。
587    /// 主要用于动态内存分配和堆空间扩展。
588    ///
589    /// ## Arguments
590    ///
591    /// * `page_table` - 目标页表的可变引用
592    /// * `new_end` - 新的结束虚拟页号（不包含）
593    ///
594    /// ## 操作过程
595    ///
596    /// 1. 计算需要添加的页面范围：`[old_end, new_end)`
597    /// 2. 对范围内的每个页面调用 `map_one()`
598    /// 3. 更新区域的 `vpn_range` 边界
599    ///
600    /// ## 内存分配
601    ///
602    /// 对于 Framed 映射，会为每个新页面分配独立的物理页帧。
603    /// 对于 Identical 映射，直接使用对应的物理地址。
604    ///
605    /// ## Panics
606    ///
607    /// - 如果 `new_end` 小于当前区域的结束位置
608    /// - Framed 映射时如果物理页帧分配失败
609    ///
610    /// ## Examples
611    ///
612    /// ```rust
613    /// // 原区域：[0x1000, 0x2000)
614    /// // 扩展到：[0x1000, 0x3000)
615    /// area.append_to(&mut page_table, VirtPageNum(0x3000));
616    /// // 页面 [0x2000, 0x3000) 被映射并分配
617    /// ```
618    #[allow(unused)]
619    pub fn append_to(&mut self, page_table: &mut PageTable, new_end: VirtPageNum) {
620        for vpn in VPNRange::new(self.vpn_range.get_end(), new_end) {
621            self.map_one(page_table, vpn)
622        }
623        self.vpn_range = VPNRange::new(self.vpn_range.get_start(), new_end);
624    }
625}

666    /// 将内存区域添加到地址空间
667    ///
668    /// 将一个内存映射区域添加到当前地址空间中，并可选地将数据复制到该区域。
669    /// 这是地址空间构建的核心操作。
670    ///
671    /// ## Arguments
672    ///
673    /// * `map_area` - 要添加的内存映射区域
674    /// * `data` - 可选的初始化数据（如 ELF 段数据）
675    ///
676    /// ## 操作流程
677    ///
678    /// 1. **建立映射**: 调用 `map_area.map()` 在页表中建立所有页面映射
679    /// 2. **复制数据**: 如果提供了数据，将其复制到映射的内存区域
680    /// 3. **记录区域**: 将区域添加到 `areas` 列表中以便管理
681    ///
682    /// ## 内存安全
683    ///
684    /// - 区域被添加到 `areas` 列表后，其生命周期由 `MemorySet` 管理
685    /// - 当 `MemorySet` 被销毁时，所有区域的映射和页帧会自动清理
686    ///
687    /// ## 使用场景
688    ///
689    /// - 添加 ELF 文件的代码段和数据段
690    /// - 添加用户栈区域
691    /// - 添加内核的各个逻辑段
692    fn push(&mut self, mut map_area: MapArea, data: Option<&[u8]>) {
693        map_area.map(&mut self.page_table);
694        if let Some(data) = data {
695            map_area.copy_data(&self.page_table, data);
696        }
697        self.areas.push(map_area);
698    }

700    /// 插入帧映射内存区域
701    ///
702    /// 在地址空间中添加一个使用 Framed 映射类型的内存区域。
703    /// 该区域的每个页面都会分配独立的物理页帧。
704    ///
705    /// ## Arguments
706    ///
707    /// * `start_va` - 区域起始虚拟地址
708    /// * `end_va` - 区域结束虚拟地址（不包含）
709    /// * `perm` - 内存访问权限
710    ///
711    /// ## 特点
712    ///
713    /// - **独立分配**: 每个虚拟页面对应一个新分配的物理页帧
714    /// - **地址隔离**: 虚拟地址和物理地址没有固定关系
715    /// - **内存保护**: 支持完整的内存访问权限控制
716    /// - **自动清理**: 区域销毁时自动释放所有物理页帧
717    ///
718    /// ## 使用场景
719    ///
720    /// - 用户程序的堆区域
721    /// - 用户程序的栈区域
722    /// - 内核的动态内存区域
723    /// - 需要地址隔离的任何区域
724    ///
725    /// ## Examples
726    ///
727    /// ```rust
728    /// // 添加用户态可读写区域
729    /// memory_set.insert_framed_area(
730    ///     VirtAddr::from(0x10000000),
731    ///     VirtAddr::from(0x10001000),
732    ///     MapPermission::R | MapPermission::W | MapPermission::U
733    /// );
734    ///
735    /// // 添加用户栈区域
736    /// memory_set.insert_framed_area(

739    ///     MapPermission::R | MapPermission::W | MapPermission::U
740    /// );

744        start_va: VirtAddr,

751    /// 映射 Trampoline 页面
752    ///
753    /// 在地址空间的高地址区域映射 Trampoline 页面，用于内核态和用户态之间的
754    /// 上下文切换。Trampoline 是一个特殊的汇编代码段，在所有地址空间中都映射到
755    /// 同一个物理地址。
756    ///
757    /// ## 映射特点
758    ///
759    /// - **虚拟地址**: `TRAMPOLINE` 常量定义的高地址
760    /// - **物理地址**: `strampoline` 符号指向的物理地址
761    /// - **权限**: 可读 + 可执行（`R | X`）
762    /// - **共享**: 所有地址空间都映射到同一物理页面
763    ///
764    /// ## 作用
765    ///
766    /// Trampoline 允许在地址空间切换过程中继续执行代码，解决了
767    /// 在切换 `satp` 寄存器后指令取指地址变化的问题。
768    ///
769    /// ## 安全性
770    ///
771    /// Trampoline 代码经过精心设计，不会泄露内核信息给用户程序。
772    fn map_trampoline(&mut self) {
773        self.page_table.map(
774            VirtAddr::from(TRAMPOLINE).into(),
775            PhysAddr::from(strampoline as usize).into(),
776            PTEFlags::R | PTEFlags::X,
777        );
778    }

898    /// 从 ELF 文件创建用户地址空间
899    ///
900    /// 解析 ELF 文件并构建相应的用户程序地址空间，包括程序的各个段、用户栈、
901    /// Trap Context 和 Trampoline。所有用户段都使用 Framed 映射，实现地址空间隔离。
902    ///
903    /// ## Arguments
904    ///
905    /// * `elf_data` - ELF 文件的二进制数据
906    ///
907    /// ## Returns
908    ///
909    /// 返回一个三元组：
910    /// - `MemorySet`: 构建好的用户地址空间
911    /// - `usize`: 用户栈顶地址
912    /// - `usize`: 程序入口点地址
913    ///
914    /// ## 地址空间布局
915    ///
916    /// ```text
917    /// 高地址 (TRAMPOLINE)
918    /// ┌──────────────────────────────────────────────────────┐
919    /// │                   Trampoline                         │
920    /// │                    (R+X)                             │
921    /// ├──────────────────────────────────────────────────────┤
922    /// │                 Trap Context                         │
923    /// │                    (R+W)                             │
924    /// ├──────────────────────────────────────────────────────┤
925    /// │                  User Stack                          │
926    /// │                   (R+W+U)                            │
927    /// ├──────────────────────────────────────────────────────┤
928    /// │                  Guard Page                          │
929    /// ├──────────────────────────────────────────────────────┤
930    /// │               ELF Program Sections                   │
931    /// │            (.text/.data/.bss etc)                    │
932    /// │              (Based on ELF flags)                    │
933    /// └──────────────────────────────────────────────────────┘
934    /// 低地址 (0x10000)
935    /// ```
936    ///
937    /// ## ELF 解析过程
938    ///
939    /// 1. **验证 ELF 魔数**: 确保文件格式正确
940    /// 2. **解析程序头**: 遍历所有 `LOAD` 类型的程序段
941    /// 3. **权限转换**: 将 ELF 段标志转换为 `MapPermission`
942    /// 4. **段映射**: 为每个段创建 Framed 映射并复制数据
943    /// 5. **用户栈**: 在程序段之上分配用户栈空间
944    /// 6. **系统区域**: 映射 Trap Context 和 Trampoline
945    ///
946    /// ## 权限映射
947    ///
948    /// ELF 段标志到内存权限的转换：
949    /// - `PF_R` → `MapPermission::R`
950    /// - `PF_W` → `MapPermission::W`
951    /// - `PF_X` → `MapPermission::X`
952    /// - 所有用户段都包含 `MapPermission::U`
953    ///
954    /// ## 内存安全
955    ///
956    /// - 所有用户段使用 Framed 映射，与内核地址空间完全隔离
957    /// - 用户栈与程序段之间有保护页面防止栈溢出
958    /// - Trap Context 仅内核可写，用户只读
959    ///
960    /// ## Panics
961    ///
962    /// - ELF 魔数验证失败
963    /// - ELF 文件格式错误
964    /// - 物理页帧分配失败
965    ///
966    /// ## Examples
967    ///
968    /// ```rust
969    /// let app_data = get_app_data(0); // 获取应用程序 ELF 数据
970    /// let (memory_set, user_stack_top, entry_point) = MemorySet::from_elf(app_data);
971    ///
972    /// println!("Entry point: {:#x}", entry_point);
973    /// println!("User stack top: {:#x}", user_stack_top);
974    /// ```
975    pub fn from_elf(elf_data: &[u8]) -> (Self, usize, usize) {
976        let mut memory_set = Self::new_bare();
977
978        memory_set.map_trampoline();
979
980        let elf = xmas_elf::ElfFile::new(elf_data).unwrap();
981        let elf_header = elf.header;
982        // elf.header.pt1：固定格式部分，pt2：可变格式部分
983        let magic = elf_header.pt1.magic;
984        assert_eq!(magic, [0x7f, 0x45, 0x4c, 0x46], "invalid elf!");
985        let ph_count = elf_header.pt2.ph_count();
986        let mut max_end_vpn = VirtPageNum(0);
987        for i in 0..ph_count {
988            let ph = elf.program_header(i).unwrap();
989            if ph.get_type().unwrap() == xmas_elf::program::Type::Load {
990                let start_va: VirtAddr = (ph.virtual_addr() as usize).into();
991                let end_va: VirtAddr = ((ph.virtual_addr() + ph.mem_size()) as usize).into();
992                let mut map_perm = MapPermission::U;
993                let ph_flags = ph.flags();
994                if ph_flags.is_read() {
995                    map_perm |= MapPermission::R;
996                }
997                if ph_flags.is_write() {
998                    map_perm |= MapPermission::W;
999                }
1000                if ph_flags.is_execute() {
1001                    map_perm |= MapPermission::X;
1002                }
1003                let map_area = MapArea::new(start_va, end_va, MapType::Framed, map_perm);
1004                max_end_vpn = map_area.vpn_range.get_end();
1005                memory_set.push(
1006                    map_area,
1007                    Some(&elf.input[ph.offset() as usize..(ph.offset() + ph.file_size()) as usize]),
1008                );
1009            }
1010        }
1011
1012        let max_end_va: VirtAddr = max_end_vpn.into();
1013        let mut user_stack_bottom: usize = max_end_va.into();
1014
1015        // Guard Page
1016        user_stack_bottom += PAGE_SIZE;
1017        let user_stack_top: usize = user_stack_bottom + USER_STACK_SIZE;
1018        memory_set.push(
1019            MapArea::new(
1020                user_stack_bottom.into(),
1021                user_stack_top.into(),
1022                MapType::Framed,
1023                MapPermission::R | MapPermission::W | MapPermission::U,
1024            ),
1025            None,
1026        );
1027
1028        // 堆空间的初始内存区域，sbrk
1029        memory_set.push(
1030            MapArea::new(
1031                user_stack_top.into(),
1032                user_stack_top.into(),
1033                MapType::Framed,
1034                MapPermission::R | MapPermission::W | MapPermission::U,
1035            ),
1036            None,
1037        );
1038

1056    /// 激活地址空间
1057    ///
1058    /// 将当前地址空间设置为活跃的地址空间，启用该地址空间的页表进行
1059    /// 地址转换。这是地址空间切换的核心操作。
1060    ///
1061    /// ## 操作流程
1062    ///
1063    /// 1. **获取页表标识**: 调用 `page_table.token()` 获取 `satp` 寄存器值
1064    /// 2. **设置 satp**: 将页表标识写入 `satp` 寄存器
1065    /// 3. **刷新 TLB**: 执行 `sfence.vma` 指令清空 TLB 缓存
1066    ///
1067    /// ## satp 寄存器格式
1068    ///
1069    /// ```text
1070    /// ┌─────────────┬─────────────────┬─────────────────────────────────────────────┐
1071    /// │    MODE     │      ASID       │                    PPN                      │
1072    /// │   (4bit)    │     (16bit)     │                  (44bit)                    │
1073    /// └─────────────┴─────────────────┴─────────────────────────────────────────────┘
1074    /// ```
1075    ///
1076    /// ## 安全性
1077    ///
1078    /// 此操作使用 `unsafe` 代码，因为：
1079    /// - 直接操作系统寄存器
1080    /// - 执行特权指令
1081    /// - 可能影响所有后续的内存访问
1082    ///
1083    /// ## 使用场景
1084    ///
1085    /// - 系统初始化时激活内核地址空间
1086    /// - 进程切换时激活用户地址空间
1087    /// - 从用户态返回内核态时恢复内核地址空间
1088    ///
1089    /// ## Examples
1090    ///
1091    /// ```rust
1092    /// // 激活内核地址空间
1093    /// let kernel_space = KERNEL_SPACE.exclusive_access();
1094    /// kernel_space.activate();
1095    ///
1096    /// // 激活用户地址空间
1097    /// let user_space = task.get_user_space();
1098    /// user_space.activate();
1099    /// ```
1100    pub fn activate(&self) {
1101        let satp = self.page_table.token();
1102        unsafe {
1103            satp::write(satp);
1104            asm!("sfence.vma");
1105        }
1106    }

1152    /// 缩小地址空间中的内存区域
1153    ///
1154    /// 查找以指定地址开始的内存区域，并将其缩小到新的结束地址。
1155    /// 主要用于动态内存管理和堆空间回收。
1156    ///
1157    /// ## Arguments
1158    ///
1159    /// * `start` - 目标区域的起始虚拟地址
1160    /// * `new_end` - 新的结束虚拟地址（不包含）
1161    ///
1162    /// ## Returns
1163    ///
1164    /// - `true` - 成功找到并缩小了目标区域
1165    /// - `false` - 未找到以指定地址开始的区域
1166    ///
1167    /// ## 操作过程
1168    ///
1169    /// 1. **查找区域**: 遍历 `areas` 列表找到起始地址匹配的区域
1170    /// 2. **缩小区域**: 调用区域的 `shrink_to()` 方法
1171    /// 3. **清理资源**: 被移除的页面自动释放对应的物理页帧
1172    ///
1173    /// ## 地址对齐
1174    ///
1175    /// - `start` 会向下对齐到页边界进行区域匹配
1176    /// - `new_end` 会向上对齐到页边界确定新边界
1177    ///
1178    /// ## 使用场景
1179    ///
1180    /// - 堆空间收缩（`sbrk` 系统调用）
1181    /// - 内存回收和优化
1182    /// - 动态库卸载
1183    ///
1184    /// ## Examples
1185    ///
1186    /// ```rust
1187    /// // 缩小堆空间
1188    /// let heap_start = VirtAddr::from(0x10000000);
1189    /// let new_heap_end = VirtAddr::from(0x10008000); // 从 64KB 缩小到 32KB
1190    ///
1191    /// if memory_set.shrink_to(heap_start, new_heap_end) {
1192    ///     println!("Heap shrunk successfully");
1193    /// } else {
1194    ///     println!("Heap region not found");
1195    /// }
1196    /// ```
1197    #[allow(unused)]
1198    pub fn shrink_to(&mut self, start: VirtAddr, new_end: VirtAddr) -> bool {
1199        if let Some(area) = self
1200            .areas
1201            .iter_mut()
1202            .find(|area| area.vpn_range.get_start() == start.floor())
1203        {
1204            area.shrink_to(&mut self.page_table, new_end.ceil());
1205            true
1206        } else {
1207            false
1208        }
1209    }

1328/// 重映射测试函数
1329///
1330/// 验证内核地址空间中各个段的内存权限设置是否正确。通过检查内核的
1331/// .text、.rodata 和 .data 段的中间地址的权限，确保内存保护机制正常工作。
1332///
1333/// ## 测试内容
1334///
1335/// 1. **代码段权限测试** (.text)
1336///    - 验证代码段不可写（只读+可执行）
1337///    - 防止代码被恶意修改
1338///
1339/// 2. **只读数据段权限测试** (.rodata)  
1340///    - 验证只读数据段不可写
1341///    - 保护常量数据不被修改
1342///
1343/// 3. **数据段权限测试** (.data)
1344///    - 验证数据段不可执行
1345///    - 防止数据被当作代码执行（NX 保护）
1346///
1347/// ## 测试方法
1348///
1349/// - 计算每个段的中间地址作为测试点
1350/// - 通过页表转换获取对应的页表项
1351/// - 检查页表项的权限标志位
1352///
1353/// ## 安全意义
1354///
1355/// 这些权限检查实现了重要的安全机制：
1356/// - **W^X 原则**: 内存页面要么可写，要么可执行，不能同时具备
1357/// - **代码完整性**: 防止代码段被恶意修改
1358/// - **数据执行保护**: 防止缓冲区溢出等攻击
1359///
1360/// ## Panics
1361///
1362/// 如果任何权限检查失败，函数会触发 panic，表明内存保护机制存在问题
1363///
1364/// ## 使用场景
1365///
1366/// - 系统初始化后的自检
1367/// - 调试内存权限配置
1368/// - 验证页表设置的正确性
1369///
1370/// ## Examples
1371///
1372/// ```rust
1373/// // 通常在内核初始化完成后调用
1374/// remap_test(); // 如果通过，打印 "remap_test passed!"
1375/// ```
1376pub fn remap_test() {
1377    let mut kernel_space = KERNEL_SPACE.exclusive_access();
1378    let mid_text: VirtAddr = ((stext as usize + etext as usize) / 2).into();
1379    let mid_rodata: VirtAddr = ((srodata as usize + erodata as usize) / 2).into();
1380    let mid_data: VirtAddr = ((sdata as usize + edata as usize) / 2).into();
1381    assert_eq!(
1382        kernel_space
1383            .page_table
1384            .translate(mid_text.floor())
1385            .unwrap()
1386            .writable(),
1387        false
1388    );
1389    assert_eq!(
1390        kernel_space
1391            .page_table
1392            .translate(mid_rodata.floor())
1393            .unwrap()
1394            .writable(),
1395        false,
1396    );
1397    assert_eq!(
1398        kernel_space
1399            .page_table
1400            .translate(mid_data.floor())
1401            .unwrap()
1402            .executable(),
1403        false,
1404    );
1405    println!("remap_test passed!");
