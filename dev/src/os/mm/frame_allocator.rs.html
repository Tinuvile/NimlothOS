1//! # 物理页帧分配器模块
2//!
3//! 提供物理页帧的分配和回收功能，管理系统中的可用物理内存页面。
4//! 采用栈式分配策略和回收列表，支持高效的页帧分配和自动释放机制。
5//!
6//! ## 核心组件
7//!
8//! - [`StackFrameAllocator`] - 栈式物理页帧分配器实现
9//! - [`FrameTracker`] - 页帧跟踪器，提供 RAII 自动释放
10//! - [`frame_alloc()`] - 全局页帧分配接口
11//! - [`init_frame_allocator()`] - 分配器初始化函数
12//!
13//! ## 分配策略
14//!
15//! **栈式分配 (Stack Allocation)**：
16//! - 从连续的物理页号区间按递增顺序分配
17//! - 优先使用回收列表中的页帧（后进先出）
18//! - 分配失败时返回 `None`，支持内存不足处理
19//!
20//! **RAII 管理**：
21//! - 通过 [`FrameTracker`] 提供自动页帧释放
22//! - 页帧超出作用域时自动回收到分配器
23//! - 防止内存泄漏和重复释放
24//!
25//! ## 内存布局
26//!
27//! ```text
28//! 物理内存:
29//! ┌─────────────────────┬─────────────────────-┬─────────────────────┐
30//! │   Kernel Image      │  Allocatable Frames  │   Device/Reserved   │
31//! │  (Non-allocatable)  │ [ekernel,MEMORY_END) │  (Non-allocatable)  │
32//! └─────────────────────┴──────────────-───────┴─────────────────────┘
33//!                      ↑                    ↑
34//!                   起始页号              结束页号
35//! ```
36//!
37//! ## 使用示例
38//!
39//! ```rust
40//! // 初始化页帧分配器
41//! init_frame_allocator();
42//!
43//! // 分配页帧
44//! let frame1 = frame_alloc().expect("内存不足");
45//! let frame2 = frame_alloc().expect("内存不足");
46//!
47//! // 页帧自动释放
48//! drop(frame1); // 页帧回到分配器的回收列表
49//!
50//! // 再次分配可能得到刚释放的页帧
51//! let frame3 = frame_alloc(); // 可能重用 frame1 的页帧
52//! ```
53//!
54//! ## 并发安全
55//!
56//! - 使用 [`UPSafeCell`] 提供线程安全的可变访问
57//! - 支持多核环境下的页帧分配操作
58//! - 通过独占访问避免竞态条件
59

67/// 页帧分配器 trait
68///
69/// 定义物理页帧分配器的标准接口，支持页帧分配、释放和初始化操作。
70/// 所有页帧分配器实现都应该遵循此接口约定。
71///
72/// ## 设计原则
73///
74/// - **内存安全**：防止重复释放和使用已释放的页帧
75/// - **错误处理**：分配失败时返回 `None`，不触发 panic
76/// - **状态管理**：维护分配器内部状态的一致性
77///
78/// ## 实现要求
79///
80/// 实现者需要保证：
81/// - `alloc()` 返回的页号在有效范围内
82/// - `dealloc()` 只接受之前分配的页号
83/// - 不会分配同一页帧给多个调用者
84trait FrameAllocator {
85    /// 创建新的分配器实例
86    ///
87    /// 创建一个未初始化的分配器，需要调用 `init()` 方法
88    /// 设置可分配的页帧范围。
89    ///
90    /// ## Returns
91    ///
92    /// 返回新创建的分配器实例
93    fn new() -> Self;
94
95    /// 分配一个物理页帧
96    ///
97    /// 从可用页帧池中分配一个 4KB 物理页帧。优先使用
98    /// 回收列表中的页帧，其次按顺序分配新页帧。
99    ///
100    /// ## Returns
101    ///
102    /// - `Some(ppn)` - 成功分配的物理页号
103    /// - `None` - 内存不足，分配失败
104    ///
105    /// ## 分配策略
106    ///
107    /// 1. 检查回收列表，优先复用已释放的页帧
108    /// 2. 如果回收列表为空，从连续区间分配新页帧
109    /// 3. 如果所有页帧都已分配，返回 `None`
110    fn alloc(&mut self) -> Option<PhysPageNum>;
111
112    /// 释放一个物理页帧
113    ///
114    /// 将之前分配的页帧回收到分配器，使其可以被再次分配。
115    /// 释放的页帧会被添加到回收列表中。
116    ///
117    /// ## Arguments
118    ///
119    /// * `ppn` - 要释放的物理页号，必须是之前通过 `alloc()` 获得的
120    ///
121    /// ## Panics
122    ///
123    /// 以下情况会触发 panic：
124    /// - 释放未分配的页帧
125    /// - 重复释放同一页帧
126    /// - 释放无效的页号（超出分配器管理范围）
127    fn dealloc(&mut self, ppn: PhysPageNum);
128}

130/// 栈式页帧分配器
131///
132/// 采用栈式策略的物理页帧分配器实现，维护连续的可分配页帧区间
133/// 和一个回收页帧的动态列表。
134///
135/// ## 分配机制
136///
137/// **连续分配区间**：
138/// - `current`：下一个待分配的页号
139/// - `end`：分配区间的结束页号（不包含）
140/// - 新页帧按 `current` 递增顺序分配
141///
142/// **回收列表**：
143/// - `recycled`：存储已释放页帧的动态数组
144/// - 分配时优先从回收列表弹出页帧（LIFO）
145/// - 释放时将页帧推入回收列表
146///
147/// ## 内存复杂度
148///
149/// - **空间复杂度**：O(R)，R 为回收列表中的页帧数
150/// - **时间复杂度**：分配和释放均为 O(1)
151///
152/// ## 初始化状态
153///
154/// 新创建的分配器所有字段都为 0/空，需要调用 [`init()`] 方法
155/// 设置实际的页帧分配范围。
156pub struct StackFrameAllocator {
157    /// 下一个待分配的页号
158    ///
159    /// 指向连续分配区间中下一个可用的物理页号。
160    /// 每次分配后递增，直到达到 `end`。
161    current: usize,
162
163    /// 分配区间结束页号（不包含）
164    ///
165    /// 标记连续分配区间的上界，当 `current == end` 时
166    /// 表示连续区间已用完，只能从回收列表分配。
167    end: usize,
168
169    /// 回收页帧列表
170    ///
171    /// 存储已释放页帧的页号，采用 LIFO（后进先出）策略。
172    /// 分配时优先从此列表弹出页帧，提高内存局部性。
173    recycled: Vec<usize>,
174}
175
176impl StackFrameAllocator {
177    /// 初始化页帧分配器
178    ///
179    /// 设置分配器的可分配页帧范围，该范围通常是从内核镜像结束
180    /// 到物理内存结束的连续物理页面。
181    ///
182    /// ## Arguments
183    ///
184    /// * `l` - 起始物理页号（包含）
185    /// * `r` - 结束物理页号（不包含）
186    ///
187    /// ## 分配范围
188    ///
189    /// 分配器将管理区间 `[l, r)` 内的所有物理页帧：
190    /// - 总共 `r - l` 个可分配页帧
191    /// - 每个页帧大小为 4KB (PAGE_SIZE)
192    /// - 页帧按页号递增顺序分配
193    ///
194    /// ## Examples
195    ///
196    /// ```rust
197    /// let start_ppn = PhysAddr::from(0x80400000).ceil(); // 内核后第一页
198    /// let end_ppn = PhysAddr::from(0x88000000).floor();  // 物理内存最后一页
199    /// allocator.init(start_ppn, end_ppn);
200    /// // 现在可以分配 [start_ppn, end_ppn) 范围内的页帧
201    /// ```
202    ///
203    /// ## Postconditions
204    ///
205    /// 初始化后分配器状态：
206    /// - `current = l.0`：从起始页号开始分配
207    /// - `end = r.0`：设置分配上界  
208    /// - `recycled` 保持空列表
209    pub fn init(&mut self, l: PhysPageNum, r: PhysPageNum) {
210        self.current = l.0;
211        self.end = r.0;
212    }
213}

215impl FrameAllocator for StackFrameAllocator {
216    /// 创建新的栈式分配器实例
217    ///
218    /// 创建一个空的分配器实例，所有字段都初始化为 0 或空状态。
219    /// 需要调用 [`init()`] 方法才能开始分配页帧。
220    ///
221    /// ## Returns
222    ///
223    /// 返回未初始化的分配器实例
224    fn new() -> Self {
225        Self {
226            current: 0,
227            end: 0,
228            recycled: Vec::new(),
229        }
230    }
231
232    /// 分配一个物理页帧
233    ///
234    /// 实现栈式分配策略：优先从回收列表弹出页帧，如果回收列表为空
235    /// 则从连续区间分配新页帧。
236    ///
237    /// ## Returns
238    ///
239    /// - `Some(ppn)` - 成功分配的物理页号
240    /// - `None` - 内存耗尽，无可用页帧
241    ///
242    /// ## 分配逻辑
243    ///
244    /// 1. **回收列表优先**：如果 `recycled` 非空，弹出最后一个页号
245    /// 2. **连续分配**：如果回收列表为空且 `current < end`，分配 `current` 页号并递增
246    /// 3. **分配失败**：如果连续区间耗尽且回收列表为空，返回 `None`
247    ///
248    /// ## 时间复杂度
249    ///
250    /// O(1) - 所有操作都是常数时间
251    fn alloc(&mut self) -> Option<PhysPageNum> {
252        if let Some(ppn) = self.recycled.pop() {
253            Some(ppn.into())
254        } else {
255            if self.current == self.end {
256                None
257            } else {
258                self.current += 1;
259                Some((self.current - 1).into())
260            }
261        }
262    }
263
264    /// 释放一个物理页帧
265    ///
266    /// 将页帧回收到回收列表中，使其可以被再次分配。进行安全性检查
267    /// 防止释放无效或重复释放的页帧。
268    ///
269    /// ## Arguments
270    ///
271    /// * `ppn` - 要释放的物理页号
272    ///
273    /// ## 安全性检查
274    ///
275    /// - **有效性检查**：页号必须小于 `current`（已分配过）
276    /// - **重复释放检查**：页号不能已存在于回收列表中
277    ///
278    /// ## Panics
279    ///
280    /// 在以下情况下会触发 panic：
281    /// - `ppn >= current`：释放未分配的页帧
282    /// - `ppn` 已存在于回收列表：重复释放同一页帧
283    ///
284    /// ## 时间复杂度
285    ///
286    /// O(R) - R 为回收列表长度，用于重复释放检查
287    fn dealloc(&mut self, ppn: PhysPageNum) {
288        let ppn = ppn.0;
289        if ppn >= self.current || self.recycled.iter().find(|&v| *v == ppn).is_some() {
290            panic!("Frame ppn={:#x} has not been allocated", ppn);
291        }
292        self.recycled.push(ppn);
293    }
294}

322/// 初始化全局页帧分配器
323///
324/// 在系统启动时调用，设置页帧分配器的可分配物理内存范围。
325/// 分配范围从内核镜像结束处到物理内存末尾。
326///
327/// ## 分配范围计算
328///
329/// **起始地址**：`ekernel` 符号地址向上对齐到页边界
330/// - `ekernel`：链接器定义的内核镜像结束符号
331/// - [`PhysAddr::ceil()`]：向上对齐避免覆盖内核数据
332///
333/// **结束地址**：`MEMORY_END` 常量向下对齐到页边界
334/// - `MEMORY_END`：板级配置定义的物理内存结束地址
335/// - [`PhysAddr::floor()`]：向下对齐避免越界访问
336///
337/// ## 内存布局
338///
339/// ```text
340/// 物理内存:
341/// ┌─────────────────┬─────────────────┬───────────────┐
342/// │  Kernel Image   │ Allocatable PPN │ Device/Reserve│
343/// │  [0, ekernel)   │ [ekernel, END)  │  [END, MAX)   │
344/// └─────────────────┴─────────────────┴───────────────┘
345///                  ↑                 ↑
346///            ceil(ekernel)      floor(END)
347/// ```
348///
349/// ## 调用时机
350///
351/// 必须在以下之前调用：
352/// - 任何页帧分配操作 ([`frame_alloc()`])
353/// - 内存管理子系统初始化
354/// - 页表创建和映射操作
355///
356/// ## Safety
357///
358/// 使用 `unsafe extern "C"` 访问链接器符号 `ekernel`，
359/// 调用者需要确保链接脚本正确定义了该符号。
360pub fn init_frame_allocator() {
361    unsafe extern "C" {
362        safe fn ekernel();
363    }
364    FRAME_ALLOCATOR.exclusive_access().init(
365        PhysAddr::from(ekernel as usize).ceil(),
366        PhysAddr::from(MEMORY_END).floor(),
367    );
368}

370/// 页帧跟踪器 (Frame Tracker)
371///
372/// 提供物理页帧的 RAII (Resource Acquisition Is Initialization) 管理，
373/// 确保页帧在不再使用时自动释放，防止内存泄漏。
374///
375/// ## RAII 机制
376///
377/// - **获取时初始化**：创建时自动清零页面内容
378/// - **作用域管理**：离开作用域时自动释放页帧
379/// - **所有权语义**：独占拥有一个物理页帧
380///
381/// ## 生命周期
382///
383/// 1. **创建**：通过 [`frame_alloc()`] 分配并初始化
384/// 2. **使用**：通过 `ppn` 字段访问物理页号
385/// 3. **释放**：超出作用域时自动调用 [`Drop::drop()`]
386///
387/// ## 内存安全
388///
389/// - **防止泄漏**：确保每个分配的页帧都会被释放
390/// - **防止重复释放**：每个 `FrameTracker` 只释放一次
391/// - **初始化保证**：新分配的页面内容都清零
392///
393/// ## 使用示例
394///
395/// ```rust
396/// // 分配页帧，内容自动清零
397/// let frame = frame_alloc().expect("内存不足");
398///
399/// // 通过物理页号访问页面数据
400/// let page_data = frame.ppn.get_bytes_array();
401/// page_data[0] = 42;
402///
403/// // 离开作用域时页帧自动释放
404/// drop(frame); // 显式释放，也可以等待自动释放
405/// ```
406pub struct FrameTracker {
407    /// 跟踪的物理页号
408    ///
409    /// 该页帧的唯一标识，可用于访问 4KB 物理页面的内容。
410    /// 当 `FrameTracker` 被销毁时，此页号对应的页帧会被自动释放。
411    pub ppn: PhysPageNum,
412}
413
414impl FrameTracker {
415    /// 创建新的页帧跟踪器
416    ///
417    /// 接管指定物理页帧的管理权，并将页面内容全部清零。
418    /// 通常不直接调用，而是通过 [`frame_alloc()`] 间接创建。
419    ///
420    /// ## Arguments
421    ///
422    /// * `ppn` - 要跟踪的物理页号
423    ///
424    /// ## Returns
425    ///
426    /// 返回新创建的页帧跟踪器
427    ///
428    /// ## 初始化过程
429    ///
430    /// 1. 获取页面的字节数组访问权限
431    /// 2. 将 4096 字节全部设置为 0
432    /// 3. 创建跟踪器实例
433    ///
434    /// ## 时间复杂度
435    ///
436    /// O(PAGE_SIZE) - 需要清零 4096 字节
437    ///
438    /// ## Safety
439    ///
440    /// 调用者需要确保：
441    /// - `ppn` 对应有效的已分配页帧
442    /// - 没有其他代码同时访问同一页帧
443    pub fn new(ppn: PhysPageNum) -> Self {
444        let bytes_array = ppn.get_bytes_array();
445        for i in bytes_array {
446            *i = 0;
447        }
448        Self { ppn }
449    }
450}

452/// 分配一个物理页帧
453///
454/// 全局页帧分配接口，从系统页帧池中分配一个 4KB 物理页面，
455/// 并返回一个 [`FrameTracker`] 进行 RAII 管理。
456///
457/// ## Returns
458///
459/// - `Some(FrameTracker)` - 成功分配的页帧跟踪器
460/// - `None` - 系统内存不足，分配失败
461///
462/// ## 分配流程
463///
464/// 1. 获取全局分配器的独占访问权限
465/// 2. 调用底层分配器的 `alloc()` 方法
466/// 3. 如果成功，创建 `FrameTracker` 并清零页面
467/// 4. 返回跟踪器或 `None`
468///
469/// ## 并发安全
470///
471/// 通过 [`UPSafeCell::exclusive_access()`] 确保线程安全，
472/// 同一时刻只有一个线程可以访问分配器。
473///
474/// ## 使用示例
475///
476/// ```rust
477/// match frame_alloc() {
478///     Some(frame) => {
479///         println!("分配成功: {:?}", frame);
480///         // 使用页帧...
481///     }
482///     None => {
483///         panic!("系统内存不足！");
484///     }
485/// }
486/// ```
487///
488/// ## 错误处理
489///
490/// 分配失败通常表示：
491/// - 物理内存耗尽
492/// - 分配器未正确初始化
493/// - 内存碎片过多
494pub fn frame_alloc() -> Option<FrameTracker> {
495    FRAME_ALLOCATOR
496        .exclusive_access()
497        .alloc()
498        .map(|ppn| FrameTracker::new(ppn))
499}

501/// 释放物理页帧（内部函数）
502///
503/// 将指定的物理页帧回收到全局分配器中，使其可以被再次分配。
504/// 通常不直接调用，而是通过 `FrameTracker` 的 `Drop` 实现自动调用。
505///
506/// ## Arguments
507///
508/// * `ppn` - 要释放的物理页号
509///
510/// ## 并发安全
511///
512/// 通过全局分配器的独占访问机制确保线程安全。
513///
514/// ## 调用时机
515///
516/// 主要在以下场景调用：
517/// - `FrameTracker::drop()` 实现中
518/// - 错误处理路径中的手动清理
519pub fn frame_dealloc(ppn: PhysPageNum) {
520    FRAME_ALLOCATOR.exclusive_access().dealloc(ppn);
521}

523impl Drop for FrameTracker {
524    /// 页帧跟踪器析构函数
525    ///
526    /// 当 `FrameTracker` 超出作用域时自动调用，负责将管理的
527    /// 物理页帧释放回分配器，实现 RAII 的"获取即初始化"机制。
528    ///
529    /// ## 释放过程
530    ///
531    /// 1. 自动调用 [`frame_dealloc()`] 函数
532    /// 2. 将页号回收到全局分配器的回收列表
533    /// 3. 页帧可以被后续的分配请求重用
534    ///
535    /// ## 调用时机
536    ///
537    /// - 显式调用 `drop(frame)`
538    /// - 变量超出作用域
539    /// - 容器被清空 (`Vec::clear()`)
540    /// - 程序异常终止时的清理
541    ///
542    /// ## 内存安全保证
543    ///
544    /// - 每个页帧有且仅有一个 `FrameTracker` 管理
545    /// - 释放操作是原子的，不会产生竞态条件
546    /// - 释放后的页帧不会被意外访问
547    fn drop(&mut self) {
548        frame_dealloc(self.ppn);
549    }
550}
