1//! # 单处理器安全单元
2//!
3//! 提供在单处理器环境下的线程安全共享可变数据结构。
4
5use core::cell::{RefCell, RefMut};
6
7/// 单处理器安全单元 (Uniprocessor Safe Cell)
8///
9/// 在单处理器系统中，通过禁用中断可以确保原子性操作。
10/// `UPSafeCell<T>` 是 `RefCell<T>` 的一个封装，用于在内核中
11/// 安全地共享可变数据。
12///
13/// ## 使用场景
14///
15/// - 内核全局状态管理
16/// - 任务管理器等需要在中断处理程序和普通代码之间共享的数据
17/// - 需要内部可变性的静态数据结构
18///
19/// ## Safety
20///
21/// 该结构体实现了 `Sync`，但这需要调用者保证：
22/// - 在访问数据时禁用中断，确保不会被中断处理程序打断
23/// - 在单处理器环境下使用
24///
25/// ## Examples
26///
27/// ```rust
28/// use crate::sync::UPSafeCell;
29///
30/// static GLOBAL_DATA: UPSafeCell<Vec<i32>> = unsafe { UPSafeCell::new(Vec::new()) };
31///
32/// // 在中断禁用的情况下访问数据
33/// let mut data = GLOBAL_DATA.exclusive_access();
34/// data.push(42);
35/// ```
36pub struct UPSafeCell<T> {
37    /// 内部的 RefCell，提供运行时借用检查
38    inner: RefCell<T>,
39}
40
41/// 为 `UPSafeCell<T>` 实现 `Sync` trait
42///
43/// 这是一个 unsafe 实现，因为 `RefCell<T>` 本身不是 `Sync` 的。
44/// 在单处理器系统中，只要确保访问时禁用了中断，就可以安全地
45/// 在多个执行上下文之间共享 `UPSafeCell`。
46unsafe impl<T> Sync for UPSafeCell<T> {}
47
48impl<T> UPSafeCell<T> {
49    /// 创建一个新的 `UPSafeCell`
50    ///
51    /// ## Arguments
52    ///
53    /// * `value` - 要封装的初始值
54    ///
55    /// ## Returns
56    ///
57    /// 返回一个新的 `UPSafeCell` 实例
58    ///
59    /// ## Safety
60    ///
61    /// 调用者必须确保：
62    /// - 在单处理器环境下使用
63    /// - 在访问时正确禁用中断
64    pub unsafe fn new(value: T) -> Self {
65        Self {
66            inner: RefCell::new(value),
67        }
68    }
69
70    /// 获取对内部数据的独占可变引用
71    ///
72    /// 这个方法会动态检查借用规则，如果已经有其他借用存在，
73    /// 会导致 panic。在正确的使用模式下（访问时禁用中断），
74    /// 这种情况不应该发生。
75    ///
76    /// ## Returns
77    ///
78    /// 返回一个 `RefMut<'_, T>`，允许可变访问内部数据
79    ///
80    /// ## Panics
81    ///
82    /// 如果内部数据已经被借用（通过其他的 `exclusive_access` 调用），
83    /// 此方法会 panic
84    pub fn exclusive_access(&self) -> RefMut<'_, T> {
85        self.inner.borrow_mut()
86    }
87}
