1//! # 文件系统相关系统调用
2//!
3//! 实现与文件 I/O 相关的系统调用，目前主要支持向标准输出写入数据。
4
5use crate::{print, println};
6
7/// 标准输出文件描述符
8///
9/// 对应 UNIX 标准的 `stdout` 文件描述符，用于向控制台输出文本。
10const FD_STDOUT: usize = 1;
11
12/// 系统调用：向文件描述符写入数据
13///
14/// 实现 `write(2)` 系统调用，将指定的数据写入到文件描述符中。
15/// 当前实现只支持向标准输出 (stdout) 写入。
16///
17/// ## Arguments
18///
19/// * `fd` - 文件描述符，当前只支持 `FD_STDOUT` (1)
20/// * `buf` - 指向要写入数据的缓冲区指针
21/// * `len` - 要写入的字节数
22///
23/// ## Returns
24///
25/// - 成功时返回实际写入的字节数
26/// - 失败时返回负值错误码
27///
28/// ## Safety
29///
30/// 该函数使用 `unsafe` 代码：
31/// - 从原始指针 `buf` 创建切片
32/// - 假设用户提供的指针和长度是有效的
33///
34/// ## Panics
35///
36/// - 当 `fd` 不是支持的文件描述符时会 panic
37/// - 当缓冲区内容不是有效 UTF-8 字符串时会 panic
38///
39/// ## Security Note
40///
41/// 当前实现跳过了内存访问权限检查（注释掉的代码），
42/// 在生产环境中应该验证用户提供的指针是否在合法的内存范围内。
43///
44/// ## Examples
45///
46/// 从用户态调用：
47/// ```c
48/// write(1, "Hello, World!", 13);  // 向 stdout 输出字符串
49/// ```
50pub fn sys_write(fd: usize, buf: *const u8, len: usize) -> isize {
51    /*

75    match fd {
76        FD_STDOUT => {
77            // 从用户空间读取数据并输出到控制台
78            let slice = unsafe { core::slice::from_raw_parts(buf, len) };
79            let str = core::str::from_utf8(slice).unwrap();
80            print!("{}", str);
81            len as isize
82        }
83        _ => {
84            panic!("Unsupported fd in sys_write: {}!", fd);
85        }
86    }
