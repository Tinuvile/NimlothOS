10/// 标准输出文件描述符
11///
12/// 对应 UNIX 标准的 `stdout` 文件描述符，用于向控制台输出文本。
13const FD_STDIN: usize = 0;
14const FD_STDOUT: usize = 1;
15
16/// 系统调用：向文件描述符写入数据
17///
18/// 实现 `write(2)` 系统调用，将指定的数据写入到文件描述符中。
19/// 当前实现只支持向标准输出 (stdout) 写入。
20///
21/// ## Arguments
22///
23/// * `fd` - 文件描述符，当前只支持 `FD_STDOUT` (1)
24/// * `buf` - 指向要写入数据的缓冲区指针
25/// * `len` - 要写入的字节数
26///
27/// ## Returns
28///
29/// - 成功时返回实际写入的字节数
30/// - 失败时返回负值错误码
31///
32/// ## 实现原理
33///
34/// 1. **地址转换**: 使用 `translate_byte_buffer()` 将用户虚拟地址转换为内核可访问的物理地址
35/// 2. **分页处理**: 自动处理跨页面的缓冲区，将其分解为多个物理页面的切片
36/// 3. **安全输出**: 逐个处理每个物理页面的数据，确保内存访问安全
37///
38/// ## 内存管理
39///
40/// 使用新的地址空间管理机制：
41/// - 通过 `current_user_token()` 获取当前任务的页表
42/// - 使用 `translate_byte_buffer()` 进行安全的地址转换
43/// - 支持跨页面的缓冲区访问
44///
45/// ## Safety
46///
47/// 相比之前的实现，现在通过页表转换提供了更好的安全性：
48/// - 自动验证用户提供的虚拟地址是否有效
49/// - 确保只访问用户任务有权限访问的内存
50/// - 防止访问内核内存或其他任务的内存
51///
52/// ## Panics
53///
54/// - 当 `fd` 不是支持的文件描述符时会 panic
55/// - 当缓冲区内容不是有效 UTF-8 字符串时会 panic
56/// - 当用户提供的虚拟地址无效时，`translate_byte_buffer` 可能会 panic
57///
58/// ## Examples
59///
60/// 从用户态调用：
61/// ```c
62/// write(1, "Hello, World!", 13);  // 向 stdout 输出字符串
63/// ```
64pub fn sys_write(fd: usize, buf: *const u8, len: usize) -> isize {
65    match fd {
66        FD_STDOUT => {
67            // 通过页表转换获取用户缓冲区的物理地址
68            let buffers = translated_byte_buffer(current_user_token(), buf, len);
69            // 遍历所有物理页面，逐个输出数据到控制台
70            for buffer in buffers {
71                if let Ok(string) = core::str::from_utf8(buffer) {
72                    print!("{}", string);
73                } else {
74                    continue;
75                }
76            }
77            len as isize
78        }
79        _ => {
80            panic!("Unsupported fd in sys_write: {}!", fd);
81        }
82    }
83}

85/// 系统调用：从文件描述符读取数据
86///
87/// 实现 `read(2)` 系统调用，从指定的文件描述符中读取数据。
88/// 当前实现只支持从标准输入 (stdin) 读取单个字符。
89///
90/// ## Arguments
91///
92/// * `fd` - 文件描述符，当前只支持 `FD_STDIN` (0)
93/// * `buf` - 指向接收数据的缓冲区指针
94/// * `len` - 要读取的字节数，当前限制为 1
95///
96/// ## Returns
97///
98/// - 成功时返回实际读取的字节数 (当前总是返回 1)
99/// - 失败时会 panic (暂未实现错误码返回)
100///
101/// ## 实现限制
102///
103/// 当前实现有以下限制：
104/// - **单字符读取**: 只支持 `len = 1`，即每次只能读取一个字符
105/// - **仅支持 stdin**: 只能从标准输入文件描述符 (0) 读取
106/// - **阻塞读取**: 如果没有输入字符，会挂起当前任务等待输入
107///
108/// ## 阻塞机制
109///
110/// ### 字符读取逻辑
111/// - 调用 `console_getchar()` 尝试读取字符
112/// - 如果返回 0 (无字符可读)，挂起当前任务并切换到其他任务
113/// - 当有字符输入时，任务被重新调度，继续读取
114/// - 读取到有效字符后跳出循环
115///
116/// ### 任务调度协作
117/// ```rust
118/// loop {
119///     c = console_getchar();
120///     if c == 0 {
121///         suspend_current_and_run_next();  // 让出 CPU
122///         continue;                        // 被调度回来时继续尝试
123///     } else {
124///         break;                          // 成功读取字符
125///     }
126/// }
127/// ```
128///
129/// ## 内存管理
130///
131/// ### 地址转换安全性
132/// - 使用 `current_user_token()` 获取当前任务的页表
133/// - 通过 `translated_byte_buffer()` 安全地访问用户内存
134/// - 支持跨页面的缓冲区（虽然当前只读取1字节）
135///
136/// ### 内存写入
137/// - 使用 `write_volatile()` 确保写入操作不被编译器优化
138/// - 直接写入用户提供的缓冲区的第一个字节
139///
140/// ## 使用场景
141///
142/// ### 键盘输入处理
143/// ```rust
144/// // 用户态代码
145/// char ch;
146/// read(0, &ch, 1);  // 从键盘读取一个字符
147/// printf("You typed: %c\n", ch);
148/// ```
149///
150/// ### 交互式程序
151/// ```rust
152/// // 简单的交互循环
153/// loop {
154///     print!("Enter a character: ");
155///     let mut buffer = [0u8; 1];
156///     read(0, buffer.as_mut_ptr(), 1);
157///     if buffer[0] == b'q' {
158///         break;  // 输入 'q' 退出
159///     }
160///     println!("You entered: {}", buffer[0] as char);
161/// }
162/// ```
163///
164/// ## Safety
165///
166/// ### 内存安全
167/// - 通过页表转换验证用户地址的有效性
168/// - 使用 `write_volatile` 防止编译器优化导致的问题
169/// - 确保只写入用户任务有权访问的内存
170///
171/// ### 并发安全
172/// - 阻塞读取通过任务调度实现，不会阻塞整个系统
173/// - 任务挂起和恢复由调度器安全管理
174///
175/// ## 限制和待优化
176///
177/// 1. **单字符限制**: 每次只能读取一个字符，效率较低
178/// 2. **错误处理**: 当前使用 panic，应该返回适当的错误码
179/// 3. **缓冲支持**: 未来可以支持读取多个字符到缓冲区
180/// 4. **文件系统**: 目前只支持控制台输入，不支持文件读取
181///
182/// ## Panics
183///
184/// - 当 `len != 1` 时会 panic
185/// - 当 `fd` 不是 `FD_STDIN` 时会 panic
186/// - 当用户提供的虚拟地址无效时，地址转换可能会 panic
187///
188/// ## Examples
189///
190/// ### C 风格调用
191/// ```c
192/// char buffer;
193/// ssize_t result = read(0, &buffer, 1);  // 从 stdin 读取一个字符
194/// if (result == 1) {
195///     printf("Read character: %c\n", buffer);
196/// }
197/// ```
198///
199/// ### 在系统调用处理中
200/// ```rust
201/// pub fn syscall(syscall_id: usize, args: [usize; 3]) -> isize {
202///     match syscall_id {
203///         SYSCALL_READ => sys_read(args[0], args[1] as *const u8, args[2]),
204///         // ...
205///     }
206/// }
207/// ```
208pub fn sys_read(fd: usize, buf: *const u8, len: usize) -> isize {
209    match fd {
210        FD_STDIN => {
211            assert_eq!(len, 1, "Only support len = 1 in sys_read!");
212            let mut c: usize;
213            loop {
214                c = console_getchar();
215                if c == 0 {
216                    suspend_current_and_run_next();
217                    continue;
218                } else {
219                    break;
220                }
221            }
222            let ch = c as u8;
223            let mut buffers = translated_byte_buffer(current_user_token(), buf, len);
224            unsafe {
225                buffers[0].as_mut_ptr().write_volatile(ch);
226            }
227            1
228        }
229        _ => {
230            panic!("Unsupported fd in sys_read!");
231        }
232    }
