1//! # 应用程序加载器模块
2//!
3//! 提供从内核镜像中加载用户应用程序的功能。用户应用程序在构建时被链接到
4//! 内核镜像中，运行时通过此模块提供的接口进行访问和加载。
5//!
6//! ## 设计原理
7//!
8//! ### 应用程序嵌入
9//!
10//! 用户应用程序通过构建脚本 (`build.rs`) 和汇编文件 (`link_app.S`) 被嵌入到
11//! 内核镜像中。构建过程会：
12//!
13//! 1. 编译所有用户应用程序为独立的二进制文件
14//! 2. 生成包含应用程序数据的汇编代码
15//! 3. 将汇编代码链接到内核镜像中
16//!
17//! ### 内存布局
18//!
19//! 嵌入的应用程序数据在内核镜像中的布局：
20//!
21//! ```text
22//! _num_app symbol location:
23//! ┌─────────────────────────────────────────────────────────────┐
24//! │                     Number of Apps                          │
25//! │                       (8 bytes)                             │
26//! ├─────────────────────────────────────────────────────────────┤
27//! │                   App 0 Start Address                       │
28//! │                       (8 bytes)                             │
29//! ├─────────────────────────────────────────────────────────────┤
30//! │                   App 1 Start Address                       │
31//! │                       (8 bytes)                             │
32//! ├─────────────────────────────────────────────────────────────┤
33//! │                          ...                                │
34//! ├─────────────────────────────────────────────────────────────┤
35//! │                   App N End Address                         │
36//! │                       (8 bytes)                             │
37//! ├─────────────────────────────────────────────────────────────┤
38//! │                       App 0 Data                            │
39//! │                       (Variable)                            │
40//! ├─────────────────────────────────────────────────────────────┤
41//! │                       App 1 Data                            │
42//! │                       (Variable)                            │
43//! ├─────────────────────────────────────────────────────────────┤
44//! │                          ...                                │
45//! └─────────────────────────────────────────────────────────────┘
46//! ```
47//!
48//! ## 安全注意事项
49//!
50//! - 所有内存访问都使用 `read_volatile()` 避免编译器优化
51//! - 应用程序 ID 边界检查防止越界访问
52//! - 使用 `unsafe` 代码块明确标识不安全操作
53//!
54//! ## 使用示例
55//!
56//! ```rust
57//! // 获取应用程序数量
58//! let app_count = get_num_app();
59//! println!("Found {} applications", app_count);
60//!
61//! // 加载特定应用程序
62//! for i in 0..app_count {
63//!     let app_data = get_app_data(i);
64//!     println!("App {}: {} bytes", i, app_data.len());
65//!     // 将应用程序数据加载到内存中...
66//! }
67//! ```

72/// 获取嵌入的应用程序数量
73///
74/// 从内核镜像中读取应用程序数量，该数量在构建时确定并嵌入到镜像中。
75///
76/// ## 实现细节
77///
78/// 函数通过外部符号 `_num_app` 访问应用程序数量。该符号由构建脚本生成的
79/// 汇编代码定义，指向一个包含应用程序数量的内存位置。
80///
81/// ## Returns
82///
83/// 嵌入到内核镜像中的应用程序数量
84///
85/// ## Safety
86///
87/// 此函数使用 `unsafe` 代码访问外部符号和原始内存，但通过以下方式确保安全：
88/// - 使用 `read_volatile()` 防止编译器优化
89/// - `_num_app` 符号由构建系统保证有效
90/// - 读取的是构建时确定的常量值
91///
92/// ## Examples
93///
94/// ```rust
95/// let count = get_num_app();
96/// assert!(count > 0, "No applications found");
97/// ```
98pub fn get_num_app() -> usize {
99    unsafe extern "C" {
100        safe fn _num_app();
101    }
102    unsafe { (_num_app as usize as *const usize).read_volatile() }
103}

105/// 获取指定应用程序的二进制数据
106///
107/// 从内核镜像中提取指定 ID 的应用程序二进制数据，返回包含完整 ELF 文件的字节切片。
108///
109/// ## Arguments
110///
111/// * `app_id` - 应用程序标识符，范围为 `[0, get_num_app())`
112///
113/// ## Returns
114///
115/// 指向应用程序二进制数据的静态字节切片，包含完整的 ELF 文件内容
116///
117/// ## 实现原理
118///
119/// 1. **获取地址表**: 从 `_num_app` 符号开始读取地址数组
120/// 2. **边界检查**: 验证 `app_id` 在有效范围内
121/// 3. **计算范围**: 使用相邻两个地址计算应用程序数据的起始和结束位置
122/// 4. **创建切片**: 基于计算的地址范围创建字节切片
123///
124/// ## Panics
125///
126/// 如果 `app_id` 大于等于应用程序总数，函数会触发 panic
127///
128/// ## Safety
129///
130/// 函数使用大量 `unsafe` 代码，但通过以下方式确保内存安全：
131/// - 应用程序 ID 边界检查
132/// - 地址计算基于构建时生成的有效数据
133/// - 所有内存访问都在内核镜像的有效范围内
134///
135/// ## Examples
136///
137/// ```rust
138/// let app_count = get_num_app();
139/// for i in 0..app_count {
140///     let app_data = get_app_data(i);
141///     println!("Application {}: {} bytes", i, app_data.len());
142///     
143///     // 验证 ELF 魔数
144///     if app_data.len() >= 4 {
145///         assert_eq!(&app_data[0..4], b"\x7fELF");
146///     }
147/// }
148/// ```
149///
150/// ## 内存布局示意
151///
152/// ```text
153/// _num_app -> [应用数量] [app0_start] [app1_start] ... [appN_end]
154///                        ↓
155///                     [应用程序0数据] [应用程序1数据] ...
156/// ```
157pub fn get_app_data(app_id: usize) -> &'static [u8] {
158    unsafe extern "C" {
159        safe fn _num_app();
160    }
161    let num_app_ptr = _num_app as usize as *const usize;
162    let num_app = get_num_app();
163    let app_start = unsafe { core::slice::from_raw_parts(num_app_ptr.add(1), num_app + 1) };
164    assert!(app_id < num_app);
165    unsafe {
166        core::slice::from_raw_parts(
167            app_start[app_id] as *const u8,
168            app_start[app_id + 1] - app_start[app_id],
169        )
170    }
171}

173lazy_static! {
174    /// 应用程序名称缓存
175    ///
176    /// 延迟初始化的静态变量，包含所有嵌入应用程序的名称列表。应用程序名称在构建时
177    /// 确定并以 null 终止的字符串形式嵌入到内核镜像中。
178    ///
179    /// ## 设计原理
180    ///
181    /// ### 名称存储格式
182    ///
183    /// 应用程序名称通过外部符号 `_app_names` 访问，数据格式为连续的 null 终止字符串：
184    ///
185    /// ```text
186    /// _app_names symbol location:
187    /// ┌─────────────────────────────────────────────────────────────┐
188    /// │ "app0" │ \0 │ "app1" │ \0 │ "app2" │ \0 │ ... │ "appN" │ \0 │
189    /// └─────────────────────────────────────────────────────────────┘
190    /// ```
191    ///
192    /// ### 初始化过程
193    ///
194    /// 1. **获取数量**: 通过 `get_num_app()` 确定应用程序总数
195    /// 2. **遍历名称**: 从 `_app_names` 开始逐个读取 null 终止的字符串
196    /// 3. **创建切片**: 为每个名称创建字符串切片
197    /// 4. **UTF-8 转换**: 将字节切片转换为有效的 UTF-8 字符串
198    /// 5. **构建向量**: 将所有名称存储在向量中供后续查询
199    ///
200    /// ## 内存安全
201    ///
202    /// - 所有内存读取使用 `read_volatile()` 防止编译器优化
203    /// - 字符串切片基于构建时生成的有效数据
204    /// - UTF-8 转换经过验证，确保字符串有效性
205    ///
206    /// ## 性能特性
207    ///
208    /// - **延迟初始化**: 仅在首次访问时解析名称数据
209    /// - **零拷贝**: 直接引用内核镜像中的字符串数据
210    /// - **一次性成本**: 初始化后的访问为 O(1) 复杂度
211    static ref APP_NAMES: Vec<&'static str> = {
212        let num_app = get_num_app();
213        unsafe extern "C" {
214            safe fn _app_names();
215        }
216        let mut start = _app_names as usize as *const u8;
217        let mut v = Vec::new();
218        unsafe {
219            for _ in 0..num_app {
220                let mut end = start;
221                while end.read_volatile() != b'\0' {
222                    end = end.add(1)
223                }
224                let slice = core::slice::from_raw_parts(start, end as usize - start as usize);
225                let str = core::str::from_utf8(slice).unwrap();
226                v.push(str);
227                start = end.add(1);
228            }
229        }
230        v
231    };
232}

234/// 根据应用程序名称获取二进制数据
235///
236/// 通过应用程序名称查找并返回对应的二进制数据。此函数提供了比基于 ID 的
237/// `get_app_data()` 更友好的访问方式，支持通过字符串名称直接定位应用程序。
238///
239/// ## Arguments
240///
241/// * `name` - 要查找的应用程序名称，必须与构建时嵌入的名称完全匹配（区分大小写）
242///
243/// ## Returns
244///
245/// * `Some(&'static [u8])` - 如果找到匹配的应用程序，返回指向其二进制数据的切片
246/// * `None` - 如果没有找到匹配名称的应用程序
247///
248/// ## 实现细节
249///
250/// 1. **名称查找**: 在 `APP_NAMES` 向量中线性搜索匹配的名称
251/// 2. **ID 映射**: 将找到的名称索引作为应用程序 ID
252/// 3. **数据获取**: 调用 `get_app_data()` 获取对应的二进制数据
253///
254/// ## 性能考虑
255///
256/// - **时间复杂度**: O(n)，其中 n 是应用程序数量
257/// - **空间复杂度**: O(1)，不额外分配内存
258/// - **首次访问**: 触发 `APP_NAMES` 的延迟初始化
259///
260/// ## Examples
261///
262/// ```rust
263/// // 查找特定应用程序
264/// if let Some(app_data) = get_app_data_by_name("hello_world") {
265///     println!("Found hello_world: {} bytes", app_data.len());
266///     // 加载并执行应用程序...
267/// } else {
268///     println!("Application 'hello_world' not found");
269/// }
270///
271/// // 批量查找多个应用程序
272/// let apps_to_load = ["init", "shell", "user_program"];
273/// for app_name in apps_to_load.iter() {
274///     match get_app_data_by_name(app_name) {
275///         Some(data) => println!("Loaded {}: {} bytes", app_name, data.len()),
276///         None => println!("Warning: {} not found", app_name),
277///     }
278/// }
279/// ```
280///
281/// ## 使用场景
282///
283/// - **配置驱动加载**: 基于配置文件中的应用程序名称进行加载
284/// - **交互式选择**: 用户通过名称选择要运行的应用程序
285/// - **依赖解析**: 根据应用程序声明的依赖名称加载相关程序
286/// - **调试辅助**: 开发时通过可读名称快速定位特定应用程序
287#[allow(unused)]
288pub fn get_app_data_by_name(name: &str) -> Option<&'static [u8]> {
289    let num_app = get_num_app();
290    (0..num_app)
291        .find(|&i| APP_NAMES[i] == name)
292        .map(get_app_data)
293}

295/// 列出所有可用的应用程序名称
296///
297/// 打印所有嵌入到内核镜像中的应用程序名称列表，主要用于调试、诊断和
298/// 用户界面显示。输出格式为带有装饰边框的列表。
299///
300/// ## 输出格式
301///
302/// ```text
303/// /**** APPS ****
304/// app_name_1
305/// app_name_2
306/// app_name_3
307/// ...
308/// **************/
309/// ```
310///
311/// ## 使用场景
312///
313/// ### 系统诊断
314/// - **启动检查**: 验证所有预期的应用程序都已正确嵌入
315/// - **构建验证**: 确认构建过程正确处理了所有应用程序
316/// - **调试辅助**: 快速查看可用的应用程序列表
317///
318/// ### 用户交互
319/// - **菜单显示**: 为用户提供可选择的应用程序列表
320/// - **命令行帮助**: 显示可执行的程序名称
321/// - **系统信息**: 作为系统状态报告的一部分
322///
323/// ## 实现特性
324///
325/// - **延迟初始化**: 首次调用时触发 `APP_NAMES` 的初始化
326/// - **零拷贝**: 直接打印内核镜像中的字符串数据
327/// - **格式化输出**: 提供清晰的视觉分隔，便于阅读
328///
329/// ## Examples
330///
331/// ```rust
332/// // 系统启动时显示可用应用程序
333/// println!("System initialized with the following applications:");
334/// list_apps();
335///
336/// // 在交互式 shell 中使用
337/// fn show_help() {
338///     println!("Available commands:");
339///     list_apps();
340///     println!("Use 'run <app_name>' to execute an application.");
341/// }
342/// ```
343///
344/// ## 性能注意事项
345///
346/// - **输出成本**: 每个应用程序名称需要一次 `println!` 调用
347/// - **初始化成本**: 首次调用可能触发名称数据的解析
348/// - **适用场景**: 主要用于调试和低频率的用户交互
349#[allow(unused)]
350pub fn list_apps() {
351    println!("/**** APPS ****");
352    for app in APP_NAMES.iter() {
353        println!("{}", app);
354    }
355    println!("**************/");
