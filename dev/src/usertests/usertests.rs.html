29fn run_tests(tests: &[(&str, &str, &str, &str, i32)]) -> i32 {
30    let mut pass_num = 0;
31    let mut arr: [*const u8; 4] = [
32        core::ptr::null::<u8>(),
33        core::ptr::null::<u8>(),
34        core::ptr::null::<u8>(),
35        core::ptr::null::<u8>(),
36    ];
37
38    for test in tests {
39        println!("Usertests: Running {}", test.0);
40        arr[0] = test.0.as_ptr();
41        if test.1 != "\0" {
42            arr[1] = test.1.as_ptr();
43            arr[2] = core::ptr::null::<u8>();
44            arr[3] = core::ptr::null::<u8>();
45            if test.2 != "\0" {
46                arr[2] = test.2.as_ptr();
47                arr[3] = core::ptr::null::<u8>();
48                if test.3 != "\0" {
49                    arr[3] = test.3.as_ptr();
50                } else {
51                    arr[3] = core::ptr::null::<u8>();
52                }
53            } else {
54                arr[2] = core::ptr::null::<u8>();
55                arr[3] = core::ptr::null::<u8>();
56            }
57        } else {
58            arr[1] = core::ptr::null::<u8>();
59            arr[2] = core::ptr::null::<u8>();
60            arr[3] = core::ptr::null::<u8>();
61        }
62
63        let pid = fork();
64        if pid == 0 {
65            exec(test.0);
66            panic!("unreachable!");
67        } else {
68            let mut exit_code: i32 = Default::default();
69            let wait_pid = waitpid(pid as usize, &mut exit_code);
70            assert_eq!(pid, wait_pid);
71            if exit_code == test.4 {
72                // summary apps with  exit_code
73                pass_num = pass_num + 1;
74            }
75            println!(
76                "\x1b[32mUsertests: Test {} in Process {} exited with code {}\x1b[0m",
77                test.0, pid, exit_code
78            );
79        }
80    }
81    pass_num
82}

84#[unsafe(no_mangle)]
85pub fn main() -> i32 {
86    let succ_num = run_tests(SUCC_TESTS);
87    let err_num = run_tests(FAIL_TESTS);
88    if succ_num == SUCC_TESTS.len() as i32 && err_num == FAIL_TESTS.len() as i32 {
89        println!(
90            "{} of sueecssed apps, {} of failed apps run correctly. \nUsertests passed!",
91            SUCC_TESTS.len(),
92            FAIL_TESTS.len()
93        );
94        return 0;
95    }
96    if succ_num != SUCC_TESTS.len() as i32 {
97        println!(
98            "all successed app_num is  {} , but only  passed {}",
99            SUCC_TESTS.len(),
100            succ_num
101        );
102    }
103    if err_num != FAIL_TESTS.len() as i32 {
104        println!(
105            "all failed app_num is  {} , but only  passed {}",
106            FAIL_TESTS.len(),
107            err_num
108        );
109    }
110    println!(" Usertests failed!");
111    return -1;
112}
