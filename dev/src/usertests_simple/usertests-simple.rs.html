8    "exit\0",
9    "fantastic_text\0",

21use user_lib::{exec, fork, waitpid};
22
23#[unsafe(no_mangle)]
24pub fn main() -> i32 {
25    for test in TESTS {
26        println!("Usertests: Running {}", test);
27        let pid = fork();
28        if pid == 0 {
29            exec(*test);
30            panic!("unreachable!");
31        } else {
32            let mut exit_code: i32 = Default::default();
33            let wait_pid = waitpid(pid as usize, &mut exit_code);
34            assert_eq!(pid, wait_pid);
35            println!(
36                "\x1b[32mUsertests: Test {} in Process {} exited with code {}\x1b[0m",
37                test, pid, exit_code
38            );
39        }
40    }
41    println!("Usertests passed!");
42    0
43}
