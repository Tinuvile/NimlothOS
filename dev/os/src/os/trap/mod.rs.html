1//! # 陷阱和中断处理模块
2//!
3//! 处理所有从用户态到内核态的陷阱 (trap)，包括系统调用、异常和中断。
4//! 陷阱是 RISC-V 架构中用户态与内核态交互的核心机制。
5//!
6//! ## 处理的陷阱类型
7//!
8//! - **系统调用** (`UserEnvCall`): 用户程序请求内核服务
9//! - **时钟中断** (`SupervisorTimer`): 实现抢占式多任务调度
10//! - **数据访问异常** (`StoreFault`, `StorePageFault`, `LoadFault`, `LoadPageFault`): 数据内存访问违规
11//! - **指令访问异常** (`InstructionFault`, `InstructionPageFault`): 指令内存访问违规
12//! - **非法指令** (`IllegalInstruction`): 执行无效指令
13//!
14//! ## 执行流程
15//!
16//! 1. **陷阱触发**: 用户态程序执行 `ecall` 或发生异常/中断
17//! 2. **硬件切换**: CPU 自动切换到 S 模式，跳转到 `stvec` 指定的处理程序
18//! 3. **上下文保存**: `__alltraps` 保存所有寄存器到陷阱上下文
19//! 4. **处理分发**: `trap_handler` 根据陷阱类型执行相应处理
20//! 5. **上下文恢复**: `__restore` 恢复寄存器并返回用户态
21//!
22//! ## 寄存器使用
23//!
24//! - `stvec`: 陷阱向量寄存器，指向陷阱处理入口 `__alltraps`
25//! - `scause`: 陷阱原因寄存器，标识陷阱类型和具体原因
26//! - `stval`: 陷阱值寄存器，包含相关的地址或值信息
27//! - `sstatus`: 状态寄存器，控制中断使能和特权级
28//! - `sepc`: 异常程序计数器，指向触发陷阱的指令地址
29

49/// 初始化陷阱处理系统
50///
51/// 设置内核态的陷阱处理入口点，配置陷阱向量寄存器 (`stvec`)
52/// 指向内核陷阱处理函数。这是陷阱处理系统的初始化函数。
53///
54/// ## 初始化内容
55///
56/// - 设置 `stvec` 寄存器指向 `trap_from_kernel`
57/// - 配置直接模式 (`TrapMode::Direct`)
58/// - 为内核态陷阱处理做准备
59///
60/// ## 调用时机
61///
62/// 应在系统启动早期调用，在启用中断之前完成陷阱系统初始化。
63///
64/// ## Note
65///
66/// 此函数只设置内核态陷阱入口，用户态陷阱入口会在任务切换时
67/// 通过 `set_user_trap_entry()` 动态设置。
68pub fn init() {
69    set_kernel_trap_entry();
70}

72/// 启用时钟中断
73///
74/// 在监督者中断使能寄存器 (`sie`) 中启用时钟中断位，
75/// 允许时钟中断触发陷阱进入内核进行任务调度。
76///
77/// ## 功能
78///
79/// - 设置 `sie.STIE` 位，启用监督者时钟中断
80/// - 配合时钟设备实现抢占式多任务调度
81/// - 必须在时钟设备配置完成后调用
82///
83/// ## Safety
84///
85/// 此函数是安全的，但内部使用 `unsafe` 访问 CSR 寄存器。
86pub fn enable_timer_interrupt() {
87    unsafe {
88        sie::set_stimer();
89    }
90}

92/// 设置内核态陷阱入口
93///
94/// 配置 `stvec` 寄存器指向内核陷阱处理函数 `trap_from_kernel`。
95/// 当内核态发生陷阱时，硬件会跳转到此函数执行。
96///
97/// ## 安全性
98///
99/// 使用 `unsafe` 直接操作 CSR 寄存器，这是特权操作。
100fn set_kernel_trap_entry() {
101    unsafe {
102        stvec::write(trap_from_kernel as usize, TrapMode::Direct);
103    }
104}

106/// 设置用户态陷阱入口
107///
108/// 配置 `stvec` 寄存器指向用户态陷阱处理入口 `TRAMPOLINE`。
109/// 当用户态发生陷阱时，硬件会跳转到 Trampoline 页面执行。
110///
111/// ## Trampoline 机制
112///
113/// Trampoline 页面包含 `__alltraps` 和 `__restore` 汇编代码，
114/// 负责保存和恢复完整的用户态上下文。
115///
116/// ## 安全性
117///
118/// 使用 `unsafe` 直接操作 CSR 寄存器，这是特权操作。
119fn set_user_trap_entry() {
120    unsafe {
121        stvec::write(TRAMPOLINE as usize, TrapMode::Direct);
122    }
123}

125/// 内核态陷阱处理函数
126///
127/// 当内核态发生陷阱时的处理函数。在当前的简化实现中，
128/// 内核态不应该发生陷阱，因此直接触发 panic。
129///
130/// ## 设计原理
131///
132/// - 内核代码应该是可信的，不应该产生异常
133/// - 如果内核态发生陷阱，说明存在严重的内核 bug
134/// - 立即停止系统运行，避免进一步的损坏
135///
136/// ## 可能的陷阱原因
137///
138/// - 内核代码访问无效内存地址
139/// - 内核代码执行无效指令
140/// - 硬件故障或配置错误
141///
142/// ## 属性说明
143///
144/// - `#[unsafe(no_mangle)]`: 防止函数名被混淆，确保链接器能找到
145/// - `-> !`: 函数永不返回，因为会触发 panic
146#[unsafe(no_mangle)]
147pub fn trap_from_kernel() -> ! {
148    panic!("a trap {:?} from kernel!", scause::read().cause());
149}

151/// 陷阱处理主函数
152///
153/// 所有用户态陷阱的统一处理入口，根据陷阱原因分发到相应的处理逻辑。
154/// 这是陷阱处理系统的核心函数，处理系统调用、异常和中断。
155///
156/// ## 处理流程
157///
158/// 1. **设置内核陷阱入口**: 防止处理过程中的嵌套陷阱
159/// 2. **获取陷阱信息**: 读取 `scause` 和 `stval` 寄存器
160/// 3. **分发处理**: 根据陷阱类型调用相应的处理逻辑
161/// 4. **返回用户态**: 调用 `trap_return()` 恢复用户执行
162///
163/// ## 系统调用处理细节
164///
165/// 系统调用处理中有特殊的上下文管理：
166/// - 首次获取陷阱上下文以读取系统调用参数
167/// - 更新 `sepc` 寄存器指向下一条指令 (`pc += 4`)
168/// - 调用系统调用处理函数，此时可能发生任务切换
169/// - 再次获取陷阱上下文以写入返回值（因为任务切换后上下文可能变化）
170///
171/// ## 支持的陷阱类型
172///
173/// - **系统调用** (`UserEnvCall`): 处理用户程序的系统调用请求
174/// - **数据访问异常** (`StoreFault`, `StorePageFault`, `LoadFault`, `LoadPageFault`): 处理数据内存访问违规
175/// - **指令访问异常** (`InstructionFault`, `InstructionPageFault`): 处理指令内存访问违规
176/// - **非法指令** (`IllegalInstruction`): 处理无效指令执行
177/// - **时钟中断** (`SupervisorTimer`): 处理抢占式调度
178///
179/// ## 错误处理
180///
181/// - 对于致命异常（内存违规、非法指令），终止当前任务
182/// - 对于未知陷阱类型，触发 panic
183///
184/// ## 属性说明
185///
186/// - `#[unsafe(no_mangle)]`: 防止函数名被混淆，汇编代码需要调用此函数
187/// - `-> !`: 函数永不返回，总是通过 `trap_return()` 返回用户态
188#[unsafe(no_mangle)]
189pub fn trap_handler() -> ! {
190    set_kernel_trap_entry();
191    let scause = scause::read();
192    let stval = stval::read();
193    match scause.cause() {
194        Trap::Exception(Exception::UserEnvCall) => {
195            let mut cx = current_trap_cx();
196            cx.sepc += 4;
197            let result = syscall(cx.x[17], [cx.x[10], cx.x[11], cx.x[12]]);
198            cx = current_trap_cx();
199            cx.x[10] = result as usize;
200        }
201        Trap::Exception(Exception::StoreFault)
202        | Trap::Exception(Exception::StorePageFault)
203        | Trap::Exception(Exception::LoadFault)
204        | Trap::Exception(Exception::LoadPageFault)
205        | Trap::Exception(Exception::InstructionFault)
206        | Trap::Exception(Exception::InstructionPageFault) => {
207            println!(
208                "[kernel] {:?} in application, bad addr = {:#x}, bad instruction = {:#x}, kernel killed it.",
209                scause.cause(),
210                stval,
211                current_trap_cx().sepc
212            );
213            exit_current_and_run_next(-2);
214        }
215        Trap::Exception(Exception::IllegalInstruction) => {
216            println!(
217                "[kernel] IllegalInstruction in application, bad instruction = {:#x}, kernel killed it.",
218                current_trap_cx().sepc
219            );
220            exit_current_and_run_next(-3);
221        }
222        Trap::Interrupt(Interrupt::SupervisorTimer) => {
223            set_next_trigger();
224            suspend_current_and_run_next();
225        }
226        _ => {
227            panic!(
228                "Unsupported trap {:?}, stval = {:#x}!",
229                scause.cause(),
230                stval,
231            );
232        }
233    }
234    trap_return();
235}

237/// 陷阱返回函数
238///
239/// 陷阱处理完成后返回用户态的函数，负责恢复用户态执行环境并
240/// 跳转回用户程序继续执行。这是陷阱处理的最后阶段。
241///
242/// ## 执行流程
243///
244/// 1. **设置用户陷阱入口**: 配置 `stvec` 指向 Trampoline
245/// 2. **准备返回参数**: 获取陷阱上下文地址和用户页表标识符
246/// 3. **跳转到 Trampoline**: 通过内联汇编跳转到 `__restore`
247/// 4. **恢复用户状态**: `__restore` 恢复所有寄存器并执行 `sret`
248///
249/// ## 地址空间切换
250///
251/// 函数执行过程中会发生地址空间切换：
252/// - 开始：内核地址空间
253/// - 跳转到 Trampoline：仍在内核地址空间（Trampoline 在两个地址空间都有映射）
254/// - `__restore` 执行 `sret`：切换到用户地址空间
255///
256/// ## Trampoline 机制
257///
258/// ```text
259/// 地址空间切换过程:
260/// ┌─────────────────┐    ┌─────────────────┐    ┌─────────────────┐
261/// │  Kernel Space   │    │   Trampoline    │    │   User Space    │
262/// │                 │───>│    (Shared)     │───>│                 │
263/// │  trap_return()  │    │   __restore     │    │  User Program   │
264/// └─────────────────┘    └─────────────────┘    └─────────────────┘
265/// ```
266///
267/// ## 内联汇编
268///
269/// 使用内联汇编执行关键的跳转操作：
270/// - `fence.i`: 指令缓存同步
271/// - `jr {restore_va}`: 跳转到 `__restore` 函数
272/// - 传递参数：`a0` = 陷阱上下文地址，`a1` = 用户页表标识符
273///
274/// ## 安全性
275///
276/// - 使用 `unsafe` 执行特权操作和内联汇编
277/// - 地址计算确保跳转到正确的 Trampoline 位置
278/// - 参数传递确保 `__restore` 获得正确的上下文信息
279///
280/// ## 属性说明
281///
282/// - `#[unsafe(no_mangle)]`: 防止函数名被混淆，可能被其他代码调用
283/// - `-> !`: 函数永不返回，总是跳转到用户程序执行
284#[unsafe(no_mangle)]
285pub fn trap_return() -> ! {
286    set_user_trap_entry();
287    let trap_cx_ptr = TRAP_CONTEXT;
288    let user_satp = current_user_token();
289    unsafe extern "C" {
290        fn __alltraps();
291        fn __restore();
292    }
293    let restore_va = __restore as usize - __alltraps as usize + TRAMPOLINE;
294    unsafe {
295        asm!(
296            "fence.i",
297            "jr {restore_va}",
298            restore_va = in(reg) restore_va,
299            in("a0") trap_cx_ptr,
300            in("a1") user_satp,
301            options(noreturn),
302        );
303    }
