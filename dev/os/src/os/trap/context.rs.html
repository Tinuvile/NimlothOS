1//! # 陷阱上下文模块
2//!
3//! 定义陷阱上下文结构，用于在陷阱处理时保存和恢复用户态的完整 CPU 状态。
4//! 与任务上下文不同，陷阱上下文保存所有寄存器，确保用户程序状态完整性。
5
6use riscv::register::sstatus::{self, SPP, Sstatus};
7
8/// 陷阱上下文结构
9///
10/// 保存用户态程序触发陷阱时的完整 CPU 状态，包括所有通用寄存器和
11/// 关键的控制状态寄存器 (CSR)。陷阱处理完成后使用此上下文恢复用户态执行。
12///
13/// ## 内存布局
14///
15/// 使用 `#[repr(C)]` 确保内存布局与汇编代码兼容：
16/// - 通用寄存器 `x0-x31` 按顺序排列（32 × 8 字节）
17/// - `sstatus` 寄存器紧随其后（8 字节）
18/// - `sepc` 寄存器（8 字节）
19/// - `kernel_satp` 内核页表标识符（8 字节）
20/// - `kernel_sp` 内核栈指针（8 字节）
21/// - `trap_handler` 陷阱处理函数地址（8 字节）
22///
23/// 总大小：288 字节（36 × 8 字节）
24///
25/// ## 与任务上下文的区别
26///
27/// | 对比项 | 陷阱上下文 | 任务上下文 |
28/// |--------|------------|------------|
29/// | 寄存器数量 | 32个通用寄存器 | 14个被调用者保存寄存器 |
30/// | CSR寄存器 | sstatus, sepc | 无 |
31/// | 使用场景 | 用户态↔内核态 | 任务切换 |
32/// | 保存时机 | 陷阱发生时 | 任务调度时 |
33///
34/// ## 寄存器说明
35///
36/// - `x0`: 硬件零寄存器，始终为 0
37/// - `x1` (`ra`): 返回地址
38/// - `x2` (`sp`): 栈指针
39/// - `x3` (`gp`): 全局指针
40/// - `x4` (`tp`): 线程指针
41/// - `x5-x7` (`t0-t2`): 临时寄存器
42/// - `x8-x9` (`s0-s1`): 被调用者保存寄存器
43/// - `x10-x11` (`a0-a1`): 函数参数/返回值
44/// - `x12-x17` (`a2-a7`): 函数参数
45/// - `x18-x27` (`s2-s11`): 被调用者保存寄存器
46/// - `x28-x31` (`t3-t6`): 临时寄存器
47#[repr(C)]
48pub struct TrapContext {
49    /// 通用寄存器 x0-x31
50    ///
51    /// 保存用户态的所有通用寄存器状态。其中：
52    /// - `x[10]` (`a0`) 用于存储系统调用返回值
53    /// - `x[10-12]` (`a0-a2`) 用于系统调用参数
54    /// - `x[17]` (`a7`) 用于系统调用号
55    pub x: [usize; 32],
56
57    /// 监督者状态寄存器 (sstatus)
58    ///
59    /// 包含处理器状态信息，主要字段包括：
60    /// - `SPP`: 先前特权级，记录陷阱前的特权级
61    /// - `SPIE`: 先前中断使能，记录陷阱前的中断状态
62    /// - `SIE`: 监督者中断使能
63    pub sstatus: Sstatus,
64
65    /// 监督者异常程序计数器 (sepc)
66    ///
67    /// 指向触发陷阱的指令地址。对于：
68    /// - **异常**: 指向导致异常的指令
69    /// - **中断**: 指向被中断的指令
70    /// - **系统调用**: 指向 `ecall` 指令（需要手动 +4 跳过）
71    pub sepc: usize,
72
73    /// 内核页表标识符
74    ///
75    /// 存储内核地址空间的页表标识符，用于在陷阱处理时切换到内核地址空间。
76    /// 该值通常是内核页表的 `satp` 寄存器值，包含页表模式和根页表物理页号。
77    ///
78    /// ## 作用
79    ///
80    /// 在用户态触发陷阱时，需要切换到内核地址空间来执行陷阱处理代码，
81    /// 此字段保存了内核地址空间的页表信息。
82    pub kernel_satp: usize,
83
84    /// 内核栈指针
85    ///
86    /// 存储该任务在内核态执行时使用的栈指针。每个任务都有独立的内核栈，
87    /// 用于在陷阱处理时保存寄存器状态和执行内核代码。
88    ///
89    /// ## 内核栈布局
90    ///
91    /// 内核栈从高地址向低地址增长，栈顶指向 `TrapContext` 结构体：
92    /// ```text
93    /// 高地址
94    /// ┌─────────────────┐
95    /// │   Task Control  │
96    /// │     Block       │
97    /// ├─────────────────┤ ← kernel_sp 指向这里
98    /// │   TrapContext   │
99    /// ├─────────────────┤
100    /// │  Kernel Stack   │
101    /// │     Space       │
102    /// └─────────────────┘
103    /// 低地址
104    /// ```
105    pub kernel_sp: usize,
106
107    /// 陷阱处理函数地址
108    ///
109    /// 存储陷阱处理函数的入口地址，通常指向 `trap_handler` 函数。
110    /// 在陷阱发生时，硬件会跳转到此地址开始执行陷阱处理代码。
111    ///
112    /// ## 陷阱处理流程
113    ///
114    /// 1. 硬件自动保存部分状态到 CSR 寄存器
115    /// 2. 跳转到 `trap_handler` 地址执行
116    /// 3. 陷阱处理函数保存完整的用户态上下文
117    /// 4. 调用具体的陷阱处理逻辑
118    /// 5. 恢复用户态上下文并返回
119    pub trap_handler: usize,
120}
121
122impl TrapContext {
123    /// 设置栈指针寄存器
124    ///
125    /// 修改用户栈指针 (`sp` 寄存器, `x[2]`)，用于在创建新任务时
126    /// 设置用户态栈的位置。
127    ///
128    /// ## Arguments
129    ///
130    /// * `sp` - 新的栈指针值，指向用户栈顶
131    pub fn set_sp(&mut self, sp: usize) {
132        self.x[2] = sp;
133    }
134
135    /// 创建应用程序初始陷阱上下文
136    ///
137    /// 为新创建的用户应用程序创建初始的陷阱上下文，设置程序入口地址、
138    /// 用户栈指针、内核环境信息和适当的处理器状态，使应用程序能够在用户态正确运行。
139    ///
140    /// ## Arguments
141    ///
142    /// * `entry` - 应用程序入口地址，通常是应用程序的 `_start` 符号地址
143    /// * `sp` - 用户栈指针，指向为该应用程序分配的用户栈顶
144    /// * `kernel_satp` - 内核页表标识符，用于陷阱处理时的地址空间切换
145    /// * `kernel_sp` - 内核栈指针，指向该任务的内核栈顶
146    /// * `trap_handler` - 陷阱处理函数地址，用于处理该任务的陷阱
147    ///
148    /// ## Returns
149    ///
150    /// 返回配置好的 `TrapContext`，可以用于首次调度该应用程序
151    ///
152    /// ## 配置说明
153    ///
154    /// - **通用寄存器**: 全部初始化为 0
155    /// - **栈指针**: 设置为指定的用户栈地址
156    /// - **程序计数器**: 设置为应用程序入口地址
157    /// - **特权级**: 设置为用户态 (`SPP::User`)
158    /// - **内核环境**: 设置内核页表、内核栈和陷阱处理函数
159    ///
160    /// ## 执行流程
161    ///
162    /// 当此上下文被 `__restore` 恢复时：
163    /// 1. 恢复所有寄存器状态（大部分为 0）
164    /// 2. 将 `sepc` 加载到程序计数器
165    /// 3. 执行 `sret` 指令跳转到用户态
166    /// 4. 开始执行应用程序代码
167    ///
168    /// ## 陷阱处理准备
169    ///
170    /// 通过设置 `kernel_satp`、`kernel_sp` 和 `trap_handler`，确保当用户程序
171    /// 触发陷阱时能够正确切换到内核环境并执行相应的处理逻辑。
172    pub fn app_init_context(
173        entry: usize,
174        sp: usize,
175        kernel_satp: usize,
176        kernel_sp: usize,
177        trap_handler: usize,
178    ) -> Self {
179        let mut sstatus = sstatus::read();
180        sstatus.set_spp(SPP::User); // 设置返回到用户态
181        let mut cx = Self {
182            x: [0; 32],
183            sstatus,
184            sepc: entry, // 设置程序入口地址
185            kernel_satp,
186            kernel_sp,
187            trap_handler,
188        };
189        cx.set_sp(sp); // 设置用户栈指针
190        cx
191    }
192}
