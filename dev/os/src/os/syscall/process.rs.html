1//! # 进程管理相关系统调用
2//!
3//! 实现与进程生命周期管理相关的系统调用，提供进程创建、执行、退出、
4//! 等待等完整的进程管理功能。支持 Unix 风格的进程控制接口。
5//!
6//! ## 支持的系统调用
7//!
8//! - [`sys_exit`] - 进程退出
9//! - [`sys_yield`] - 让出 CPU 时间片
10//! - [`sys_time`] - 获取系统时间
11//! - [`sys_pid`] - 获取进程 PID
12//! - [`sys_fork`] - 创建子进程
13//! - [`sys_exec`] - 执行新程序
14//! - [`sys_waitpid`] - 等待子进程结束
15//!
16//! ## 进程状态管理
17//!
18//! 系统支持以下进程状态：
19//! - **Ready** - 就绪状态，等待调度
20//! - **Running** - 运行状态，正在执行
21//! - **Blocked** - 阻塞状态，等待资源
22//! - **Zombie** - 僵尸状态，已退出但未被回收
23//!
24//! ## 父子进程关系
25//!
26//! 每个进程维护一个子进程列表，支持：
27//! - 进程创建和复制（fork）
28//! - 进程替换（exec）
29//! - 进程等待和回收（waitpid）
30//! - 进程退出和清理（exit）
31

42/// 系统调用：进程退出
43///
44/// 实现 `exit(2)` 系统调用，终止当前正在运行的应用程序。
45/// 该函数会清理当前任务的资源，并调度下一个就绪任务运行。
46///
47/// ## Arguments
48///
49/// * `exit_code` - 进程退出码，用于向父进程或系统报告执行状态
50///   - 0 通常表示成功完成
51///   - 非零值通常表示出现错误或异常情况
52///
53/// ## Returns
54///
55/// 该函数实际上不会返回，因为调用进程会被终止。
56/// 返回类型 `isize` 仅用于与系统调用接口保持一致。
57///
58/// ## Behavior
59///
60/// 1. 记录进程退出信息到内核日志
61/// 2. 将当前任务标记为已退出状态
62/// 3. 调度并切换到下一个就绪任务
63/// 4. 如果没有其他任务，系统将处理所有任务完成的情况
64///
65/// ## Panics
66///
67/// 如果任务切换函数意外返回，会触发 panic（这在正常情况下不应该发生）
68pub fn sys_exit(exit_code: i32) -> isize {
69    println!("[kernel] Application exited with code {}", exit_code);
70    exit_current_and_run_next(exit_code);
71    panic!("Unreachable in sys_exit!");
72}

74/// 系统调用：让出 CPU 时间片
75///
76/// 实现 `sched_yield(2)` 系统调用，当前任务主动让出 CPU，
77/// 允许调度器选择其他就绪任务运行。这是一种协作式多任务的实现方式。
78///
79/// ## Returns
80///
81/// 成功时返回 0。当任务重新获得 CPU 时间片时，会从此处继续执行。
82///
83/// ## Behavior
84///
85/// 1. 将当前任务状态从 `Running` 改为 `Ready`
86/// 2. 调度器选择下一个就绪任务运行
87/// 3. 执行任务上下文切换
88/// 4. 当前任务将来重新被调度时，从此函数返回
89///
90/// ## Use Cases
91///
92/// - 长时间运行的任务主动让出 CPU，提高系统响应性
93/// - 等待某些条件满足时的忙等待优化
94/// - 实现用户空间的协作式多任务
95pub fn sys_yield() -> isize {
96    suspend_current_and_run_next();
97    0
98}

100/// 系统调用：获取系统时间
101///
102/// 实现 `time(2)` 系统调用的简化版本，返回系统启动以来的毫秒数。
103/// 用于用户程序的时间测量、性能分析和定时功能。
104///
105/// ## Returns
106///
107/// 系统启动以来的毫秒数（非负整数）
108///
109/// ## 时间精度
110///
111/// - 精度：毫秒级
112/// - 范围：从系统启动开始计数
113/// - 单调性：单调递增，不会回退
114///
115/// ## 使用场景
116///
117/// - 程序执行时间测量
118/// - 性能基准测试
119/// - 简单的定时器实现
120/// - 日志时间戳
121pub fn sys_time() -> isize {
122    time_ms() as isize
123}

125/// 系统调用：获取进程 PID
126///
127/// 实现 `getpid(2)` 系统调用，返回当前进程的进程标识符（PID）。
128/// PID 是系统中唯一标识进程的整数，用于进程间通信和进程管理。
129///
130/// ## Returns
131///
132/// 当前进程的 PID（非负整数）
133///
134/// ## PID 特性
135///
136/// - 唯一性：每个进程都有唯一的 PID
137/// - 持久性：进程生命周期内 PID 保持不变
138/// - 范围：通常为正整数，0 保留给内核使用
139///
140/// ## 使用场景
141///
142/// - 进程间通信（IPC）
143/// - 进程管理和监控
144/// - 日志记录和调试
145/// - 父子进程关系识别
146pub fn sys_pid() -> isize {
147    current_task().unwrap().pid.0 as isize
148}

150/// 系统调用：创建子进程（fork）
151///
152/// 实现 `fork(2)` 系统调用，创建当前进程的一个子进程。子进程将
153/// 拥有与父进程相同的地址空间内容（独立复制），并从 `fork` 返回处
154/// 开始执行。父进程得到子进程的 PID，子进程得到返回值 0。
155///
156/// ## Returns
157///
158/// - 父进程中返回新建子进程的 PID（正数）
159/// - 子进程中返回 0
160///
161/// ## 行为说明
162///
163/// 1. 复制父进程的任务控制块及地址空间（深拷贝）
164/// 2. 设置子进程 Trap 上下文的返回值 `a0 = 0`
165/// 3. 将子进程加入就绪队列等待调度
166///
167/// ## 安全与隔离
168///
169/// - 子进程拥有独立的物理页面副本，修改互不影响
170/// - Trampoline 等只读共享页面除外
171pub fn sys_fork() -> isize {
172    let current_task = current_task().unwrap();
173    let new_task = current_task.fork();
174    let new_pid = new_task.pid.0;
175    let trap_cx = new_task.inner_exclusive_access().trap_cx();
176    trap_cx.x[10] = 0; // x[10] = a0
177    add_task(new_task);
178    new_pid as isize
179}

181/// 系统调用：执行新程序（exec）
182///
183/// 实现 `execve(2)` 的简化版本，用指定的程序镜像替换当前进程的
184/// 地址空间。若加载成功，不返回；若失败，返回 -1。
185///
186/// ## Arguments
187///
188/// * `path` - 指向用户空间以 `\0` 结尾的程序名字符串
189///
190/// ## Returns
191///
192/// - 成功时返回 0（实际执行不会返回到此处，进程上下文被替换）
193/// - 失败时返回 -1（未找到指定程序）
194///
195/// ## 行为说明
196///
197/// 1. 从用户态读取程序名字符串
198/// 2. 打开文件
199/// 3. 读取文件内容
200/// 4. 调用任务的 `exec` 方法重建地址空间并跳转到新入口
201///
202/// ## 进程替换特性
203///
204/// - **地址空间替换**：完全替换当前进程的地址空间
205/// - **PID 保持**：进程 PID 保持不变
206/// - **文件描述符继承**：保持打开的文件描述符
207/// - **不返回**：成功执行后不会返回到调用点
208///
209/// ## 安全考虑
210///
211/// 通过 [`translated_str`] 安全地读取用户空间的程序路径字符串。
212pub fn sys_exec(path: *const u8) -> isize {
213    let token = current_user_token();
214    let path = translated_str(token, path);
215    if let Some(data) = open_file(path.as_str(), OpenFlags::RDONLY) {
216        let all_data = data.read_all();
217        let task = current_task().unwrap();
218        task.exec(all_data.as_slice());
219        0
220    } else {
221        -1
222    }
223}

225/// 系统调用：等待子进程结束（waitpid）
226///
227/// 实现 `waitpid(2)` 的子集，等待特定 PID 的子进程或任意子进程结束，
228/// 并将其退出码写入到用户提供的缓冲区。
229///
230/// ## Arguments
231///
232/// * `pid` - 要等待的子进程 PID；传入 `-1` 表示等待任意子进程
233/// * `exit_code_ptr` - 指向用户空间的退出码写入地址
234///
235/// ## Returns
236///
237/// - 成功时返回已回收子进程的 PID
238/// - 若没有匹配的子进程返回 -1
239/// - 若暂时没有已退出的符合条件的子进程返回 -2（可由上层重试/阻塞）
240///
241/// ## 行为说明
242///
243/// 1. 校验待等待的子进程是否存在
244/// 2. 查找符合条件且已处于 Zombie 状态的子进程
245/// 3. 回收其 Task 对象，获取退出码并写回用户缓冲区
246///
247/// ## 等待策略
248///
249/// - **阻塞等待**：如果指定 PID 的子进程尚未退出，父进程会等待
250/// - **非阻塞检查**：如果暂时没有符合条件的子进程，返回 -2
251/// - **任意子进程**：传入 `pid = -1` 等待任意子进程
252///
253/// ## 僵尸进程处理
254///
255/// 成功等待后，系统会：
256/// - 回收子进程的 Task 对象
257/// - 获取子进程的退出码
258/// - 将退出码写入用户提供的缓冲区
259/// - 清理子进程资源
260///
261/// ## Safety
262///
263/// 通过 `translated_refmut()` 将退出码写入用户空间，调用前已验证指针
264/// 在当前地址空间内有效（失败会 panic；未来可改为错误返回）。
265pub fn sys_waitpid(pid: isize, exit_code_ptr: *mut i32) -> isize {
266    let task = current_task().unwrap();
267    let mut inner = task.inner_exclusive_access();
268    if !inner
269        .children
270        .iter()
271        .any(|p| pid == -1 || pid as usize == p.getpid())
272    {
273        return -1;
274    }
275    let pair = inner.children.iter().enumerate().find(|(_, p)| {
276        p.inner_exclusive_access().is_zombie() && (pid == -1 || pid as usize == p.getpid())
277    });
278    if let Some((idx, _)) = pair {
279        let child = inner.children.remove(idx);
280        assert_eq!(Arc::strong_count(&child), 1);
281        let found_pid = child.getpid();
282        let exit_code = child.inner_exclusive_access().exit_code;
283        *translated_refmut(inner.memory_set.token(), exit_code_ptr) = exit_code;
284        found_pid as isize
285    } else {
286        -2
287    }
