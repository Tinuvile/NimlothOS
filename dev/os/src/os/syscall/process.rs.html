16/// 系统调用：进程退出
17///
18/// 实现 `exit(2)` 系统调用，终止当前正在运行的应用程序。
19/// 该函数会清理当前任务的资源，并调度下一个就绪任务运行。
20///
21/// ## Arguments
22///
23/// * `exit_code` - 进程退出码，用于向父进程或系统报告执行状态
24///   - 0 通常表示成功完成
25///   - 非零值通常表示出现错误或异常情况
26///
27/// ## Returns
28///
29/// 该函数实际上不会返回，因为调用进程会被终止。
30/// 返回类型 `isize` 仅用于与系统调用接口保持一致。
31///
32/// ## Behavior
33///
34/// 1. 记录进程退出信息到内核日志
35/// 2. 将当前任务标记为已退出状态
36/// 3. 调度并切换到下一个就绪任务
37/// 4. 如果没有其他任务，系统将处理所有任务完成的情况
38///
39/// ## Panics
40///
41/// 如果任务切换函数意外返回，会触发 panic（这在正常情况下不应该发生）
42///
43/// ## Examples
44///
45/// 从用户态调用：
46/// ```c
47/// exit(0);  // 正常退出
48/// exit(1);  // 异常退出
49/// ```
50pub fn sys_exit(exit_code: i32) -> isize {
51    println!("[kernel] Application exited with code {}", exit_code);
52    exit_current_and_run_next(exit_code);
53    panic!("Unreachable in sys_exit!");
54}

56/// 系统调用：让出 CPU 时间片
57///
58/// 实现 `sched_yield(2)` 系统调用，当前任务主动让出 CPU，
59/// 允许调度器选择其他就绪任务运行。这是一种协作式多任务的实现方式。
60///
61/// ## Returns
62///
63/// 成功时返回 0。当任务重新获得 CPU 时间片时，会从此处继续执行。
64///
65/// ## Behavior
66///
67/// 1. 将当前任务状态从 `Running` 改为 `Ready`
68/// 2. 调度器选择下一个就绪任务运行
69/// 3. 执行任务上下文切换
70/// 4. 当前任务将来重新被调度时，从此函数返回
71///
72/// ## Use Cases
73///
74/// - 长时间运行的任务主动让出 CPU，提高系统响应性
75/// - 等待某些条件满足时的忙等待优化
76/// - 实现用户空间的协作式多任务
77///
78/// ## Examples
79///
80/// 从用户态调用：
81/// ```c
82/// while (condition) {
83///     yield();  // 避免长时间占用 CPU
84/// }
85/// ```
86pub fn sys_yield() -> isize {
87    suspend_current_and_run_next();
88    0
89}

91/// 系统调用：获取系统时间
92///
93/// 实现获取系统运行时间的系统调用，返回自系统启动以来经过的毫秒数。
94/// 这是 `gettimeofday(2)` 的简化版本。
95///
96/// ## Returns
97///
98/// 返回系统启动以来经过的毫秒数（作为 `isize` 类型）
99///
100/// ## Precision
101///
102/// 时间精度取决于系统时钟频率和定时器实现，通常为毫秒级精度。
103///
104/// ## Use Cases
105///
106/// - 测量代码执行时间
107/// - 实现定时器和超时机制
108/// - 性能分析和基准测试
109/// - 随机数生成的种子
110///
111/// ## Examples
112///
113/// 从用户态调用：
114/// ```c
115/// long start_time = get_time();
116/// // 执行某些操作
117/// long end_time = get_time();
118/// long elapsed = end_time - start_time;  // 计算耗时
119/// ```
120///
121/// ## Note
122///
123/// 返回的时间值可能会在长时间运行后溢出，调用者应该考虑处理这种情况。
124pub fn sys_get_time() -> isize {
125    get_time_ms() as isize
126}

128/// 系统调用：获取进程 PID
129///
130/// 返回当前进程的进程标识符（PID）。

137/// 系统调用：创建子进程（fork）
138///
139/// 实现 `fork(2)` 系统调用，创建当前进程的一个子进程。子进程将
140/// 拥有与父进程相同的地址空间内容（独立复制），并从 `fork` 返回处
141/// 开始执行。父进程得到子进程的 PID，子进程得到返回值 0。
142///
143/// ## Returns
144///
145/// - 父进程中返回新建子进程的 PID（正数）
146/// - 子进程中返回 0
147///
148/// ## 行为说明
149///
150/// 1. 复制父进程的任务控制块及地址空间（深拷贝）
151/// 2. 设置子进程 Trap 上下文的返回值 `a0 = 0`
152/// 3. 将子进程加入就绪队列等待调度
153///
154/// ## 安全与隔离
155///
156/// - 子进程拥有独立的物理页面副本，修改互不影响
157/// - Trampoline 等只读共享页面除外
158pub fn sys_fork() -> isize {
159    let current_task = current_task().unwrap();
160    let new_task = current_task.fork();
161    let new_pid = new_task.pid.0;
162    let trap_cx = new_task.inner_exclusive_access().get_trap_cx();
163    trap_cx.x[10] = 0; // x[10] = a0
164    add_task(new_task);
165    new_pid as isize
166}

168/// 系统调用：执行新程序（exec）
169///
170/// 实现 `execve(2)` 的简化版本，用指定的程序镜像替换当前进程的
171/// 地址空间。若加载成功，不返回；若失败，返回 -1。
172///
173/// ## Arguments
174///
175/// * `path` - 指向用户空间以 `\0` 结尾的程序名字符串
176///
177/// ## Returns
178///
179/// - 成功时返回 0（实际执行不会返回到此处，进程上下文被替换）
180/// - 失败时返回 -1（未找到指定程序）
181///
182/// ## 行为说明
183///
184/// 1. 从用户态读取程序名字符串
185/// 2. 在内置应用仓库中查找对应的镜像数据
186/// 3. 调用任务的 `exec` 方法重建地址空间并跳转到新入口
187pub fn sys_exec(path: *const u8) -> isize {
188    let token = current_user_token();
189    let path = translated_str(token, path);
190    if let Some(data) = get_app_data_by_name(path.as_str()) {
191        let task = current_task().unwrap();
192        task.exec(data);
193        0
194    } else {
195        -1
196    }
197}

199/// 系统调用：等待子进程结束（waitpid）
200///
201/// 实现 `waitpid(2)` 的子集，等待特定 PID 的子进程或任意子进程结束，
202/// 并将其退出码写入到用户提供的缓冲区。
203///
204/// ## Arguments
205///
206/// * `pid` - 要等待的子进程 PID；传入 `-1` 表示等待任意子进程
207/// * `exit_code_ptr` - 指向用户空间的退出码写入地址
208///
209/// ## Returns
210///
211/// - 成功时返回已回收子进程的 PID
212/// - 若没有匹配的子进程返回 -1
213/// - 若暂时没有已退出的符合条件的子进程返回 -2（可由上层重试/阻塞）
214///
215/// ## 行为说明
216///
217/// 1. 校验待等待的子进程是否存在
218/// 2. 查找符合条件且已处于 Zombie 状态的子进程
219/// 3. 回收其 Task 对象，获取退出码并写回用户缓冲区
220///
221/// ## Safety
222///
223/// 通过 `translated_refmut()` 将退出码写入用户空间，调用前已验证指针
224/// 在当前地址空间内有效（失败会 panic；未来可改为错误返回）。
225pub fn sys_waitpid(pid: isize, exit_code_ptr: *mut i32) -> isize {
226    let task = current_task().unwrap();
227    let mut inner = task.inner_exclusive_access();
228    if !inner
229        .children
230        .iter()
231        .any(|p| pid == -1 || pid as usize == p.getpid())
232    {
233        return -1;
234    }
235    let pair = inner.children.iter().enumerate().find(|(_, p)| {
236        p.inner_exclusive_access().is_zombie() && (pid == -1 || pid as usize == p.getpid())
237    });
238    if let Some((idx, _)) = pair {
239        let child = inner.children.remove(idx);
240        assert_eq!(Arc::strong_count(&child), 1);
241        let found_pid = child.getpid();
242        let exit_code = child.inner_exclusive_access().exit_code;
243        *translated_refmut(inner.memory_set.token(), exit_code_ptr) = exit_code;
244        found_pid as isize
245    } else {
246        -2
247    }
248}
