1//! # 应用程序加载器模块
2//!
3//! 提供从内核镜像中加载用户应用程序的功能。用户应用程序在构建时被链接到
4//! 内核镜像中，运行时通过此模块提供的接口进行访问和加载。
5//!
6//! ## 设计原理
7//!
8//! ### 应用程序嵌入
9//!
10//! 用户应用程序通过构建脚本 (`build.rs`) 和汇编文件 (`link_app.S`) 被嵌入到
11//! 内核镜像中。构建过程会：
12//!
13//! 1. 编译所有用户应用程序为独立的二进制文件
14//! 2. 生成包含应用程序数据的汇编代码
15//! 3. 将汇编代码链接到内核镜像中
16//!
17//! ### 内存布局
18//!
19//! 嵌入的应用程序数据在内核镜像中的布局：
20//!
21//! ```text
22//! _num_app symbol location:
23//! ┌─────────────────────────────────────────────────────────────┐
24//! │                     Number of Apps                          │
25//! │                       (8 bytes)                             │
26//! ├─────────────────────────────────────────────────────────────┤
27//! │                   App 0 Start Address                       │
28//! │                       (8 bytes)                             │
29//! ├─────────────────────────────────────────────────────────────┤
30//! │                   App 1 Start Address                       │
31//! │                       (8 bytes)                             │
32//! ├─────────────────────────────────────────────────────────────┤
33//! │                          ...                                │
34//! ├─────────────────────────────────────────────────────────────┤
35//! │                   App N End Address                         │
36//! │                       (8 bytes)                             │
37//! ├─────────────────────────────────────────────────────────────┤
38//! │                       App 0 Data                            │
39//! │                       (Variable)                            │
40//! ├─────────────────────────────────────────────────────────────┤
41//! │                       App 1 Data                            │
42//! │                       (Variable)                            │
43//! ├─────────────────────────────────────────────────────────────┤
44//! │                          ...                                │
45//! └─────────────────────────────────────────────────────────────┘
46//! ```
47//!
48//! ## 安全注意事项
49//!
50//! - 所有内存访问都使用 `read_volatile()` 避免编译器优化
51//! - 应用程序 ID 边界检查防止越界访问
52//! - 使用 `unsafe` 代码块明确标识不安全操作
53//!
54//! ## 使用示例
55//!
56//! ```rust
57//! // 获取应用程序数量
58//! let app_count = num_app();
59//! println!("Found {} applications", app_count);
60//!
61//! // 加载特定应用程序
62//! for i in 0..app_count {
63//!     let app_data = app_data(i);
64//!     println!("App {}: {} bytes", i, app_data.len());
65//!     // 将应用程序数据加载到内存中...
66//! }
67//! ```

72/// 获取应用程序总数
73///
74/// 返回系统中已注册的应用程序数量，用于遍历所有应用程序。
75///
76/// ## Returns
77///
78/// 应用程序的总数量
79///
80/// ## Examples
81///
82/// ```
83/// let count = num_app();
84/// println!("系统中共有 {} 个应用程序", count);
85/// ```
86pub fn num_app() -> usize {
87    unsafe extern "C" {
88        fn _num_app();
89    }
90    unsafe { (_num_app as usize as *const usize).read_volatile() }
91}

93/// 获取指定应用程序的二进制数据
94///
95/// 根据应用程序 ID 获取对应的 ELF 二进制数据。
96/// 应用程序 ID 从 0 开始，范围为 `[0, num_app())`。
97///
98/// ## Arguments
99///
100/// * `app_id` - 应用程序标识符，范围为 `[0, num_app())`
101///
102/// ## Returns
103///
104/// 应用程序的 ELF 二进制数据切片
105///
106/// ## Panics
107///
108/// 如果 `app_id` 超出有效范围则 panic
109///
110/// ## Examples
111///
112/// ```
113/// let app_count = num_app();
114/// for i in 0..app_count {
115///     let app_data = app_data(i);
116///     println!("应用程序 {} 大小: {} 字节", i, app_data.len());
117/// }
118/// ```
119///
120/// ## 内存布局示意
121///
122/// ```text
123/// _num_app -> [应用数量] [app0_start] [app1_start] ... [appN_end]
124///                        ↓
125///                     [应用程序0数据] [应用程序1数据] ...
126/// ```
127pub fn app_data(app_id: usize) -> &'static [u8] {
128    unsafe extern "C" {
129        safe fn _num_app();
130    }
131    let num_app_ptr = _num_app as usize as *const usize;
132    let num_app = num_app();
133    let app_start = unsafe { core::slice::from_raw_parts(num_app_ptr.add(1), num_app + 1) };
134    assert!(app_id < num_app);
135    unsafe {
136        core::slice::from_raw_parts(
137            app_start[app_id] as *const u8,
138            app_start[app_id + 1] - app_start[app_id],
139        )
140    }
141}

143lazy_static! {
144    /// 应用程序名称缓存
145    ///
146    /// 延迟初始化的静态变量，包含所有嵌入应用程序的名称列表。应用程序名称在构建时
147    /// 确定并以 null 终止的字符串形式嵌入到内核镜像中。
148    ///
149    /// ## 设计原理
150    ///
151    /// ### 名称存储格式
152    ///
153    /// 应用程序名称通过外部符号 `_app_names` 访问，数据格式为连续的 null 终止字符串：
154    ///
155    /// ```text
156    /// _app_names symbol location:
157    /// ┌─────────────────────────────────────────────────────────────┐
158    /// │ "app0" │ \0 │ "app1" │ \0 │ "app2" │ \0 │ ... │ "appN" │ \0 │
159    /// └─────────────────────────────────────────────────────────────┘
160    /// ```
161    ///
162    /// ### 初始化过程
163    ///
164    /// 1. **获取数量**: 通过 `num_app()` 确定应用程序总数
165    /// 2. **遍历名称**: 从 `_app_names` 开始逐个读取 null 终止的字符串
166    /// 3. **创建切片**: 为每个名称创建字符串切片
167    /// 4. **UTF-8 转换**: 将字节切片转换为有效的 UTF-8 字符串
168    /// 5. **构建向量**: 将所有名称存储在向量中供后续查询
169    ///
170    /// ## 内存安全
171    ///
172    /// - 所有内存读取使用 `read_volatile()` 防止编译器优化
173    /// - 字符串切片基于构建时生成的有效数据
174    /// - UTF-8 转换经过验证，确保字符串有效性
175    ///
176    /// ## 性能特性
177    ///
178    /// - **延迟初始化**: 仅在首次访问时解析名称数据
179    /// - **零拷贝**: 直接引用内核镜像中的字符串数据
180    /// - **一次性成本**: 初始化后的访问为 O(1) 复杂度
181    static ref APP_NAMES: Vec<&'static str> = {
182        let num_app = num_app();
183        unsafe extern "C" {
184            fn _app_names();
185        }
186        let mut start = _app_names as usize as *const u8;
187        let mut v = Vec::new();
188        unsafe {
189            for _ in 0..num_app {
190                let mut end = start;
191                while end.read_volatile() != b'\0' {
192                    end = end.add(1)
193                }
194                let slice = core::slice::from_raw_parts(start, end as usize - start as usize);
195                let str = core::str::from_utf8(slice).unwrap();
196                v.push(str);
197                start = end.add(1);
198            }
199        }
200        v
201    };
202}

204/// 根据名称获取应用程序数据
205///
206/// 通过应用程序名称获取对应的 ELF 二进制数据。
207/// 这是对 `app_data()` 更友好的访问方式，支持通过字符串名称直接定位应用程序。
208///
209/// ## Arguments
210///
211/// * `name` - 应用程序名称
212///
213/// ## Returns
214///
215/// - `Some(data)` - 找到应用程序，返回其二进制数据
216/// - `None` - 未找到指定名称的应用程序
217///
218/// ## 查找过程
219///
220/// 1. **获取数量**: 通过 `num_app()` 确定应用程序总数
221/// 2. **名称匹配**: 遍历所有应用程序名称，查找匹配项
222/// 3. **数据获取**: 调用 `app_data()` 获取对应的二进制数据
223///
224/// ## Examples
225///
226/// ```
227/// // 直接获取指定应用程序
228/// if let Some(app_data) = app_data_by_name("hello_world") {
229///     println!("找到 hello_world 应用程序，大小: {} 字节", app_data.len());
230/// }
231///
232/// // 批量处理应用程序
233/// let apps_to_load = ["init", "shell", "user_program"];
234/// for app_name in apps_to_load.iter() {
235///     match app_data_by_name(app_name) {
236///         Some(data) => println!("加载应用程序: {}", app_name),
237///         None => println!("未找到应用程序: {}", app_name),
238///     }
239/// }
240/// ```
241pub fn app_data_by_name(name: &str) -> Option<&'static [u8]> {
242    let num_app = num_app();
243    (0..num_app).find(|&i| APP_NAMES[i] == name).map(app_data)
244}

246/// 列出所有可用的应用程序名称
247///
248/// 打印所有嵌入到内核镜像中的应用程序名称列表，主要用于调试、诊断和
249/// 用户界面显示。输出格式为带有装饰边框的列表。
250///
251/// ## 输出格式
252///
253/// ```text
254/// /**** APPS ****
255/// app_name_1
256/// app_name_2
257/// app_name_3
258/// ...
259/// **************/
260/// ```
261///
262/// ## 使用场景
263///
264/// ### 系统诊断
265/// - **启动检查**: 验证所有预期的应用程序都已正确嵌入
266/// - **构建验证**: 确认构建过程正确处理了所有应用程序
267/// - **调试辅助**: 快速查看可用的应用程序列表
268///
269/// ### 用户交互
270/// - **菜单显示**: 为用户提供可选择的应用程序列表
271/// - **命令行帮助**: 显示可执行的程序名称
272/// - **系统信息**: 作为系统状态报告的一部分
273///
274/// ## 实现特性
275///
276/// - **延迟初始化**: 首次调用时触发 `APP_NAMES` 的初始化
277/// - **零拷贝**: 直接打印内核镜像中的字符串数据
278/// - **格式化输出**: 提供清晰的视觉分隔，便于阅读
279///
280/// ## Examples
281///
282/// ```rust
283/// // 系统启动时显示可用应用程序
284/// println!("System initialized with the following applications:");
285/// list_apps();
286///
287/// // 在交互式 shell 中使用
288/// fn show_help() {
289///     println!("Available commands:");
290///     list_apps();
291///     println!("Use 'run <app_name>' to execute an application.");
292/// }
293/// ```
294///
295/// ## 性能注意事项
296///
297/// - **输出成本**: 每个应用程序名称需要一次 `println!` 调用
298/// - **初始化成本**: 首次调用可能触发名称数据的解析
299/// - **适用场景**: 主要用于调试和低频率的用户交互
300#[allow(unused)]
301pub fn list_apps() {
302    println!("/**** APPS ****");
303    for app in APP_NAMES.iter() {
304        println!("{}", app);
305    }
306    println!("**************/");
