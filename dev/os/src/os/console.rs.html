1//! # 控制台输出模块
2//!
3//! 提供格式化文本输出功能，实现类似标准库的 `print!` 和 `println!` 宏。
4//! 通过 SBI 接口与底层硬件交互，将文本输出到控制台。
5//!
6//! ## 功能特性
7//!
8//! - **格式化输出**: 支持 Rust 标准的格式化字符串语法
9//! - **SBI 集成**: 通过 SBI 调用实现跨平台控制台输出
10//! - **宏接口**: 提供便捷的 `print!` 和 `println!` 宏
11//! - **UTF-8 支持**: 完全支持 Unicode 字符输出
12//!
13//! ## 使用示例
14//!
15//! ```rust
16//! println!("Hello, world!");
17//! print!("Answer: {}", 42);
18//! println!("Debug info: {:?}", some_struct);
19//! ```
20
21use crate::sbi::console_putchar;
22use core::fmt::{self, Write};
23
24/// 标准输出结构体
25///
26/// 实现了 `Write` trait，将格式化的文本通过 SBI 接口输出到控制台。
27/// 这是一个零大小类型 (ZST)，不占用内存空间。
28struct Stdout;
29
30impl Write for Stdout {
31    /// 将字符串写入标准输出
32    ///
33    /// 遍历字符串中的每个字符，通过 SBI 的 `console_putchar` 函数
34    /// 逐个输出到控制台。
35    ///
36    /// ## Arguments
37    ///
38    /// * `s` - 要输出的字符串切片
39    ///
40    /// ## Returns
41    ///
42    /// 总是返回 `Ok(())`，因为 SBI 调用不会失败
43    fn write_str(&mut self, s: &str) -> fmt::Result {
44        for c in s.chars() {
45            console_putchar(c as usize);
46        }
47        Ok(())
48    }
49}
50
51/// 格式化输出函数
52///
53/// 接受格式化参数并输出到控制台，是 `print!` 和 `println!` 宏的底层实现。
54///
55/// ## Arguments
56///
57/// * `args` - 格式化参数，由 `format_args!` 宏生成
58///
59/// ## Panics
60///
61/// 如果格式化或写入过程中发生错误会 panic，但在正常情况下不会发生
62pub fn print(args: fmt::Arguments) {
63    Stdout.write_fmt(args).unwrap();
64}
65
66/// 格式化打印宏（不换行）
67///
68/// 类似于标准库的 `print!` 宏，将格式化的文本输出到控制台，
69/// 不会在末尾添加换行符。
70///
71/// ## 语法
72///
73/// ```rust
74/// print!("format string", arg1, arg2, ...);
75/// ```
76///
77/// ## 格式化支持
78///
79/// 支持 Rust 标准的格式化语法：
80/// - `{}` - 默认格式化
81/// - `{:?}` - Debug 格式化
82/// - `{:x}` - 十六进制格式化
83/// - `{:>10}` - 右对齐，宽度为10
84///
85/// ## Examples
86///
87/// ```rust
88/// print!("Hello, ");
89/// print!("world!");        // 输出: Hello, world!
90/// print!("Value: {}", 42); // 输出: Value: 42
91/// ```
92#[macro_export]
93macro_rules! print {
94    ($fmt: literal $(, $($arg: tt)+)?) => {
95        $crate::console::print(format_args!($fmt $(, $($arg)+)?));
96    }
97}
98
99/// 格式化打印宏（带换行）
100///
101/// 类似于标准库的 `println!` 宏，将格式化的文本输出到控制台，
102/// 并在末尾自动添加换行符。
103///
104/// ## 语法
105///
106/// ```rust
107/// println!("format string", arg1, arg2, ...);
108/// ```
109///
110/// ## 与 print! 的区别
111///
112/// - `print!` - 不添加换行符
113/// - `println!` - 自动添加换行符 (`\n`)
114///
115/// ## Examples
116///
117/// ```rust
118/// println!("Hello, world!");           // 输出: Hello, world!\n
119/// println!("Answer: {}", 42);          // 输出: Answer: 42\n
120/// println!("Debug: {:?}", some_value); // 输出: Debug: SomeValue\n
121/// ```
122///
123/// ## Implementation
124///
125/// 内部通过字符串连接将 `\n` 添加到格式字符串末尾，
126/// 然后调用 [`print!`] 宏进行实际输出。
127#[macro_export]
128macro_rules! println {
129    ($fmt: literal $(, $($arg: tt)+)?) => {
130        $crate::console::print(format_args!(concat!($fmt, "\n") $(, $($arg)+)?));
131    }
132}
