1//! # 文件系统模块
2//!
3//! 提供操作系统内核的文件系统抽象层，包括文件操作接口、inode 管理、
4//! 标准输入输出等功能。为上层应用提供统一的文件访问接口。
5//!
6//! ## 模块组织
7//!
8//! - [`inode`] - 文件 inode 管理，提供文件读写和元数据操作
9//! - [`stdio`] - 标准输入输出设备，包括 stdin 和 stdout
10//!
11//! ## 设计目标
12//!
13//! - **统一接口**: 为不同类型的文件提供统一的 `File` trait 接口
14//! - **类型安全**: 通过 trait 约束确保文件操作的类型安全
15//! - **并发支持**: 支持多线程并发访问文件
16//! - **抽象层次**: 屏蔽底层文件系统的复杂性
17//!
18//! ## 核心组件
19//!
20//! ### 文件接口
21//! - [`File`] - 文件抽象 trait，定义基本的文件操作
22//! - [`OSInode`] - 操作系统级别的 inode 封装
23//! - [`Stdin`] / [`Stdout`] - 标准输入输出设备
24//!
25//! ### 文件操作
26//! - [`open_file`] - 打开文件的统一接口
27//! - [`list_apps`] - 列出应用程序列表
28//! - [`OpenFlags`] - 文件打开标志位
29//!
30//! ## 文件系统特性
31//!
32//! - **多种文件类型**: 支持普通文件、目录、设备文件等
33//! - **权限控制**: 支持读写权限的独立控制
34//! - **位置管理**: 支持文件偏移量的自动管理
35//! - **缓冲优化**: 支持用户缓冲区的批量处理
36//!
37//! ## 使用示例
38//!
39//! ```rust
40//! use crate::fs::{File, open_file, OpenFlags};
41//!
42//! // 打开文件进行读写
43//! let file = open_file("test.txt", OpenFlags::RDWR).unwrap();
44//!
45//! // 读取文件内容
46//! let mut buf = [0u8; 1024];
47//! let user_buf = UserBuffer::new(&mut buf);
48//! let bytes_read = file.read(user_buf);
49//!
50//! // 写入文件内容
51//! let data = b"Hello, World!";
52//! let user_buf = UserBuffer::new(data);
53//! let bytes_written = file.write(user_buf);
54//! ```

66/// 文件抽象接口
67///
68/// 定义文件的基本操作接口，为不同类型的文件提供统一的抽象。
69/// 该 trait 是文件系统抽象层的核心，支持普通文件、设备文件等各种文件类型。
70///
71/// ## 实现要求
72///
73/// 实现者需要确保：
74/// - **线程安全**: 支持多线程并发访问
75/// - **原子操作**: 单个读写操作应该是原子的
76/// - **错误处理**: 在文件错误时应该有适当的错误处理机制
77/// - **权限检查**: 根据文件权限正确实现读写操作
78///
79/// ## 文件操作语义
80///
81/// - `read`: 从文件读取数据到用户缓冲区
82/// - `write`: 将用户缓冲区数据写入文件
83/// - `readable` / `writable`: 检查文件的读写权限
84///
85/// ## 性能考虑
86///
87/// - 实现应该尽可能减少内存拷贝
88/// - 支持批量操作以提高性能
89/// - 考虑缓存策略以优化频繁访问的文件
90///
91/// ## 注意事项
92///
93/// - 缓冲区大小应该合理，避免过大的内存分配
94/// - 实现应该处理边界情况和错误状态
95/// - 写入操作可能需要考虑持久化策略
96pub trait File: Send + Sync {
97    /// 从文件读取数据
98    ///
99    /// 将文件中的数据读取到提供的用户缓冲区中。该操作是原子的，
100    /// 要么读取请求的数据，要么返回已读取的字节数。
101    ///
102    /// ## Arguments
103    /// * `buf` - 用户缓冲区，用于存储读取的数据
104    ///
105    /// ## Returns
106    /// 实际读取的字节数，0 表示已到达文件末尾
107    ///
108    /// ## 行为
109    /// - 从当前文件偏移量开始读取数据
110    /// - 将数据写入提供的用户缓冲区
111    /// - 更新文件偏移量
112    /// - 返回实际读取的字节数
113    ///
114    /// ## 错误处理
115    /// - 如果文件不可读，行为由具体实现定义
116    /// - 如果缓冲区为空，返回 0
117    /// - 如果到达文件末尾，返回 0
118    fn read(&self, buf: UserBuffer) -> usize;
119
120    /// 向文件写入数据
121    ///
122    /// 将用户缓冲区中的数据写入文件。该操作是原子的，
123    /// 要么完全写入成功，要么返回已写入的字节数。
124    ///
125    /// ## Arguments
126    /// * `buf` - 用户缓冲区，包含要写入的数据
127    ///
128    /// ## Returns
129    /// 实际写入的字节数
130    ///
131    /// ## 行为
132    /// - 从当前文件偏移量开始写入数据
133    /// - 将用户缓冲区数据写入文件
134    /// - 更新文件偏移量
135    /// - 返回实际写入的字节数
136    ///
137    /// ## 错误处理
138    /// - 如果文件不可写，行为由具体实现定义
139    /// - 如果缓冲区为空，返回 0
140    /// - 如果磁盘空间不足，返回已写入的字节数
141    fn write(&self, buf: UserBuffer) -> usize;
142
143    /// 检查文件是否可读
144    ///
145    /// ## Returns
146    /// 如果文件可读返回 `true`，否则返回 `false`
147    fn readable(&self) -> bool;
148
149    /// 检查文件是否可写
150    ///
151    /// ## Returns
152    /// 如果文件可写返回 `true`，否则返回 `false`
153    fn writable(&self) -> bool;
154}
