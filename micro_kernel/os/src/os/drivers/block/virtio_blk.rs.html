1//! # VirtIO 块设备驱动实现
2//!
3//! 实现基于 VirtIO 协议的块设备驱动，提供高性能的虚拟化存储设备访问。
4//! 支持 QEMU 等虚拟化环境中的虚拟磁盘设备，为文件系统提供底层存储能力。
5//!
6//! ## VirtIO 协议特性
7//!
8//! - **标准化接口**: 基于 VirtIO 1.1 规范的标准设备接口
9//! - **高性能**: 支持 DMA 和中断机制，减少 CPU 开销
10//! - **虚拟化友好**: 专为虚拟化环境优化的设备协议
11//! - **可扩展性**: 支持多种 VirtIO 设备类型
12//!
13//! ## 核心组件
14//!
15//! - [`VirtIOBlock`] - VirtIO 块设备的主要接口结构
16//! - [`VirtioHal`] - 硬件抽象层实现，提供内存管理接口
17//! - [`QUEUE_FRAMES`] - 队列帧管理，用于 DMA 缓冲区
18//!
19//! ## 内存管理
20//!
21//! - **DMA 分配**: 通过 `dma_alloc` 分配连续的物理页面
22//! - **地址转换**: 支持虚拟地址到物理地址的转换
23//! - **帧跟踪**: 自动跟踪分配的物理帧，确保正确释放
24//!
25//! ## 使用示例
26//!
27//! ```rust
28//! use crate::drivers::block::VirtIOBlock;
29//!
30//! // 创建 VirtIO 块设备实例
31//! let virtio_block = VirtIOBlock::new();
32//!
33//! // 读取块数据
34//! let mut buf = [0u8; 512];
35//! virtio_block.read_block(0, &mut buf);
36//!
37//! // 写入块数据
38//! let data = [0x42u8; 512];
39//! virtio_block.write_block(0, &data);
40//! ```

52/// VirtIO 块设备在内存映射 I/O 中的基地址
53///
54/// 该地址是 VirtIO 设备在 QEMU 虚拟化环境中的标准映射地址。
55/// 设备寄存器和其他控制结构都位于此地址附近。
56#[allow(unused)]
57const VIRTIO0: usize = 0x10001000;

59/// VirtIO 块设备驱动结构
60///
61/// 封装 VirtIO 块设备的功能，提供线程安全的块设备访问接口。
62/// 使用 `UPSafeCell` 确保内部可变性，支持多线程并发访问。
63///
64/// ## 内部结构
65///
66/// 包含一个 `VirtIOBlk` 实例，该实例实现了 VirtIO 块设备的具体功能。
67/// 通过 `UPSafeCell` 提供内部可变性，允许在不可变引用上修改内部状态。
68///
69/// ## 线程安全
70///
71/// 该结构是线程安全的，多个线程可以同时访问块设备进行读写操作。
72/// 并发控制通过 `UPSafeCell` 和内部的锁机制实现。
73///
74/// ## 生命周期管理
75///
76/// 设备实例的生命周期与系统运行时间相同，在系统启动时初始化，
77/// 在系统关闭时自动清理。
78pub struct VirtIOBlock(UPSafeCell<VirtIOBlk<'static, VirtioHal>>);
79
80lazy_static! {
81    /// 队列帧管理器
82    ///
83    /// 管理 VirtIO 设备队列使用的物理帧，确保 DMA 缓冲区的正确分配和释放。
84    /// 使用 `UPSafeCell` 提供线程安全的帧管理。
85    ///
86    /// ## 帧管理策略
87    ///
88    /// - **连续分配**: 确保分配的物理帧在地址空间中是连续的
89    /// - **自动跟踪**: 记录所有分配的帧，便于后续释放
90    /// - **批量操作**: 支持一次性分配多个帧
91    ///
92    /// ## 内存安全
93    ///
94    /// 该管理器确保分配的物理帧在设备使用期间不会被意外释放，
95    /// 并在设备不再需要时正确回收内存。
96    static ref QUEUE_FRAMES: UPSafeCell<Vec<FrameTracker>> = unsafe { UPSafeCell::new(Vec::new()) };
97}

99impl BlockDevice for VirtIOBlock {
100    /// 从块设备读取数据
101    ///
102    /// 从指定的块 ID 读取数据到提供的缓冲区中。该操作是原子的，
103    /// 要么读取整个块的数据，要么失败。
104    ///
105    /// ## Arguments
106    ///
107    /// * `block_id` - 要读取的块 ID
108    /// * `buf` - 用于存储读取数据的缓冲区
109    ///
110    /// ## 行为
111    ///
112    /// - 从 VirtIO 块设备读取指定块的数据
113    /// - 将数据写入提供的缓冲区
114    /// - 如果缓冲区大小小于块大小，只读取缓冲区能容纳的数据
115    /// - 如果缓冲区大小大于块大小，多余部分保持不变
116    ///
117    /// ## 错误处理
118    ///
119    /// 如果读取操作失败，会触发 panic 并显示错误信息。
120    /// 这通常表示设备硬件错误或协议错误。
121    ///
122    /// ## 性能说明
123    ///
124    /// 该操作通过 VirtIO 协议进行，支持 DMA 传输，具有较高的性能。
125    /// 读取操作是同步的，会阻塞直到数据传输完成。
126    fn read_block(&self, block_id: usize, buf: &mut [u8]) {
127        self.0
128            .exclusive_access()
129            .read_block(block_id, buf)
130            .expect("Error when reading VirtIOBlk");
131    }
132
133    /// 向块设备写入数据
134    ///
135    /// 将缓冲区中的数据写入指定的块 ID。该操作是原子的，
136    /// 要么完全写入成功，要么完全失败。
137    ///
138    /// ## Arguments
139    ///
140    /// * `block_id` - 要写入的块 ID
141    /// * `buf` - 包含要写入数据的缓冲区
142    ///
143    /// ## 行为
144    ///
145    /// - 将缓冲区数据写入指定的块
146    /// - 如果缓冲区大小小于块大小，块中未覆盖的部分保持不变
147    /// - 如果缓冲区大小大于块大小，只写入块能容纳的数据
148    /// - 写入操作完成后，数据立即可用于后续的读取操作
149    ///
150    /// ## 错误处理
151    ///
152    /// 如果写入操作失败，会触发 panic 并显示错误信息。
153    /// 这通常表示设备硬件错误、存储空间不足或协议错误。
154    ///
155    /// ## 持久化
156    ///
157    /// 写入的数据会立即持久化到存储设备，在系统重启后仍然可用。
158    /// 写入操作会刷新设备缓存，确保数据安全。
159    fn write_block(&self, block_id: usize, buf: &[u8]) {
160        self.0
161            .exclusive_access()
162            .write_block(block_id, buf)
163            .expect("Error when writing VirtIOBlk");
164    }
165}

167impl VirtIOBlock {
168    /// 创建新的 VirtIO 块设备实例
169    ///
170    /// 初始化 VirtIO 块设备，建立与硬件设备的连接。
171    /// 该函数会配置设备寄存器，建立 VirtIO 队列，并完成设备初始化。
172    ///
173    /// ## 初始化过程
174    ///
175    /// 1. **设备发现**: 通过内存映射 I/O 访问设备寄存器
176    /// 2. **协议协商**: 与设备协商 VirtIO 协议版本和功能特性
177    /// 3. **队列设置**: 建立 VirtIO 队列用于数据传输
178    /// 4. **驱动激活**: 激活设备，使其准备好处理 I/O 请求
179    ///
180    /// ## Returns
181    ///
182    /// 返回新创建的 `VirtIOBlock` 实例，包装在 `UPSafeCell` 中以支持内部可变性
183    ///
184    /// ## Safety
185    ///
186    /// 该函数使用 `unsafe` 代码访问硬件寄存器，调用者必须确保：
187    /// - 硬件设备已正确初始化并可用
188    /// - 内存映射 I/O 地址有效且可访问
189    /// - 没有其他代码同时访问同一设备
190    ///
191    /// ## Panics
192    ///
193    /// 如果设备初始化失败（如设备不存在、协议不兼容等），会触发 panic。
194    /// 这通常表示硬件配置错误或设备驱动问题。
195    ///
196    /// ## Examples
197    ///
198    /// ```
199    /// let virtio_block = VirtIOBlock::new();
200    /// // 现在可以使用 virtio_block 进行块设备操作
201    /// ```
202    #[allow(unused)]
203    pub fn new() -> Self {
204        unsafe {
205            Self(UPSafeCell::new(
206                VirtIOBlk::<VirtioHal>::new(&mut *(VIRTIO0 as *mut VirtIOHeader)).unwrap(),
207            ))
208        }
209    }
210}

212/// VirtIO 硬件抽象层实现
213///
214/// 为 VirtIO 设备驱动提供硬件抽象接口，包括内存管理、地址转换等功能。
215/// 该结构实现了 `Hal` trait，为 VirtIO 驱动提供必要的硬件服务。
216///
217/// ## 主要功能
218///
219/// - **DMA 内存管理**: 分配和释放用于 DMA 传输的物理内存
220/// - **地址转换**: 在虚拟地址和物理地址之间进行转换
221/// - **内存映射**: 提供内存映射 I/O 支持
222///
223/// ## 内存管理策略
224///
225/// - **连续分配**: 确保 DMA 缓冲区在物理内存中连续
226/// - **自动跟踪**: 自动跟踪分配的物理帧，确保正确释放
227/// - **页面对齐**: 所有分配都按页面边界对齐
228pub struct VirtioHal;
229
230impl Hal for VirtioHal {
231    /// 分配 DMA 内存
232    ///
233    /// 为 VirtIO 设备分配连续的物理页面，用于 DMA 传输。
234    /// 分配的页面必须是连续的，以满足 DMA 传输的要求。
235    ///
236    /// ## Arguments
237    ///
238    /// * `pages` - 要分配的页面数量
239    ///
240    /// ## Returns
241    ///
242    /// 返回分配的内存的物理地址
243    ///
244    /// ## 分配策略
245    ///
246    /// 1. **连续分配**: 确保分配的页面在物理地址空间中连续
247    /// 2. **帧跟踪**: 将分配的帧记录到 `QUEUE_FRAMES` 中
248    /// 3. **地址计算**: 返回第一个页面的物理地址
249    ///
250    /// ## 内存管理
251    ///
252    /// 分配的页面由 `QUEUE_FRAMES` 管理器跟踪，确保在设备不再需要时正确释放。
253    /// 这防止了内存泄漏，确保系统的内存使用效率。
254    ///
255    /// ## Panics
256    ///
257    /// 如果没有足够的连续物理页面可用，会触发 panic。
258    /// 这通常表示系统内存不足或内存碎片化严重。
259    fn dma_alloc(pages: usize) -> usize {
260        let mut ppn_base = PhysPageNum(0);
261        for i in 0..pages {
262            let frame = frame_alloc().unwrap();
263            if i == 0 {
264                ppn_base = frame.ppn;
265            }
266            assert_eq!(frame.ppn.0, ppn_base.0 + i);
267            QUEUE_FRAMES.exclusive_access().push(frame);
268        }
269        let pa: PhysAddr = ppn_base.into();
270        pa.0
271    }
272
273    /// 释放 DMA 内存
274    ///
275    /// 释放之前通过 `dma_alloc` 分配的 DMA 内存。
276    /// 该函数会回收所有相关的物理帧，并将它们返回到空闲帧池。
277    ///
278    /// ## Arguments
279    ///
280    /// * `pa` - 要释放的内存的物理地址
281    /// * `pages` - 要释放的页面数量
282    ///
283    /// ## Returns
284    ///
285    /// 总是返回 0，表示操作成功
286    ///
287    /// ## 释放过程
288    ///
289    /// 1. **地址转换**: 将物理地址转换为物理页号
290    /// 2. **批量释放**: 逐个释放所有相关的物理帧
291    /// 3. **帧回收**: 将释放的帧返回到空闲帧池
292    ///
293    /// ## 内存安全
294    ///
295    /// 该函数确保释放的内存不会与其他分配冲突，并正确更新内存管理器的状态。
296    /// 释放后的内存可以重新分配给其他用途。
297    fn dma_dealloc(pa: usize, pages: usize) -> i32 {
298        let pa = PhysAddr::from(pa);
299        let mut ppn_base: PhysPageNum = pa.into();
300        for _ in 0..pages {
301            frame_dealloc(ppn_base);
302            ppn_base.step();
303        }
304        0
305    }
306
307    /// 物理地址到虚拟地址的转换
308    ///
309    /// 在当前的实现中，物理地址和虚拟地址是相同的。
310    /// 这是因为内核运行在直接映射模式下，物理地址直接对应虚拟地址。
311    ///
312    /// ## Arguments
313    ///
314    /// * `addr` - 物理地址
315    ///
316    /// ## Returns
317    ///
318    /// 返回对应的虚拟地址（在当前实现中与物理地址相同）
319    ///
320    /// ## 实现说明
321    ///
322    /// 该函数在直接映射模式下是恒等映射，即 `phys_to_virt(addr) == addr`。
323    /// 这种设计简化了地址转换，提高了性能。
324    fn phys_to_virt(addr: usize) -> usize {
325        addr
326    }
327
328    /// 虚拟地址到物理地址的转换
329    ///
330    /// 通过内核页表将虚拟地址转换为物理地址。
331    /// 该函数使用内核的页表进行地址转换，支持复杂的虚拟内存映射。
332    ///
333    /// ## Arguments
334    ///
335    /// * `vaddr` - 虚拟地址
336    ///
337    /// ## Returns
338    ///
339    /// 返回对应的物理地址
340    ///
341    /// ## 转换过程
342    ///
343    /// 1. **页表查找**: 使用内核页表查找虚拟地址对应的页表项
344    /// 2. **地址计算**: 从页表项中提取物理页号
345    /// 3. **偏移计算**: 结合页内偏移计算最终的物理地址
346    ///
347    /// ## 错误处理
348    ///
349    /// 如果虚拟地址没有有效的页表映射，会触发 panic。
350    /// 这通常表示内存访问错误或页表配置问题。
351    ///
352    /// ## 性能说明
353    ///
354    /// 该操作涉及页表查找，可能触发 TLB 缺失，有一定的性能开销。
355    /// 对于频繁访问的地址，建议缓存转换结果。
356    fn virt_to_phys(vaddr: usize) -> usize {
357        let result = PageTable::from_token(kernel_token())
358            .translate_va(VirtAddr::from(vaddr))
359            .unwrap()
360            .0;
361        result
362    }
