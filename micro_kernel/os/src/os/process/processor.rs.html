1//! # 处理器管理模块
2//!
3//! 提供处理器状态管理和进程调度的核心功能，实现抢占式多进程调度系统。
4//! 管理当前正在执行的进程，协调进程切换和处理器资源分配。
5//!
6//! ## 核心组件
7//!
8//! - [`Processor`] - 处理器状态管理器，维护当前进程和空闲上下文
9//! - [`PROCESSOR`] - 全局处理器实例，系统唯一的处理器管理器
10//! - [`run_processs`] - 主调度循环，负责进程分发和执行
11//! - [`schedule`] - 进程调度函数，实现进程上下文切换
12//!
13//! ## 调度机制
14//!
15//! ### 调度策略
16//! 采用**协作式调度**与**抢占式调度**相结合的混合调度模式：
17//! - **时间片轮转**: 基于时钟中断的抢占式调度
18//! - **主动让出**: 进程可以主动调用 `yield` 让出 CPU
19//! - **阻塞调度**: I/O 等待时自动切换到其他进程
20//!
21//! ### 进程状态转换
22//! ```text
23//! ┌─────────────┐   Scheduler    ┌─────────────┐
24//! │    Ready    │ ─────────────► │   Running   │
25//! └─────────────┘   Selection    └──────┬──────┘
26//!       ▲                               │
27//!       │                               │
28//!       │    Timeout/Yield/Block        │
29//!       └───────────────────────────────┘
30//! ```
31//!
32//! ## 处理器架构
33//!
34//! ```text
35//! ┌─────────────────────────────────────────────────────────┐
36//! │                    CPU Core                             │
37//! │  ┌─────────────┐                    ┌─────────────┐     │
38//! │  │   Current   │                    │  Idle Process  │     │
39//! │  │    Process     │ ◄──── Switch ────► │   Context   │     │
40//! │  │   Context   │                    │             │     │
41//! │  └─────────────┘                    └─────────────┘     │
42//! └─────────────────────────────────────────────────────────┘
43//!              ▲                                ▲
44//!              │                                │
45//!        ┌─────────────┐                ┌─────────────┐
46//!        │ User Space  │                │ Scheduler   │
47//!        │   Processs     │                │   Loop      │
48//!        └─────────────┘                └─────────────┘
49//! ```
50//!
51//! ## 上下文切换流程
52//!
53//! ```text
54//! Process A Run ──► Interrupt/Syscall ──► Save Context A ──► Load Context B ──► Process B Run
55//!    │                               │                 │             │
56//!    │                               ▼                 ▼             │
57//!    │                        ┌─────────────┐ ┌─────────────┐        │
58//!    │                        │ ProcessContext │ │ ProcessContext │        │
59//!    │                        │     A       │ │     B       │        │
60//!    │                        └─────────────┘ └─────────────┘        │
61//!    │                                                               │
62//!    └─────────────────── Time Slice Rotation ◄──────────────────────┘
63//! ```
64//!
65//! ## 使用示例
66//!
67//! ```rust
68//! // 系统初始化后启动调度器
69//! run_processs(); // 进入主调度循环，永不返回
70//!
71//! // 在中断处理中进行进程切换
72//! pub fn handle_timer_interrupt() {
73//!     // 处理时钟中断...
74//!     schedule(current_process_cx_ptr); // 切换到其他进程
75//! }
76//!
77//! // 获取当前进程信息
78//! if let Some(process) = current_process() {
79//!     println!("Current PID: {}", process.getpid());
80//! }
81//! ```

92/// 处理器状态管理器
93///
94/// 管理单个 CPU 核心的执行状态，包括当前正在执行的进程和空闲进程上下文。
95/// 作为调度器的核心数据结构，协调进程之间的 CPU 资源分配。
96///
97/// ## 设计原理
98///
99/// ### 双上下文模型
100/// 处理器维护两种执行上下文：
101/// - **进程上下文**: 用户进程的执行状态（寄存器、栈指针等）
102/// - **空闲上下文**: 调度器的执行状态，用于进程切换时的中转
103///
104/// ### 所有权管理
105/// 通过 `Option<Arc<ProcessControlBlock>>` 管理当前进程：
106/// - `Some(process)`: CPU 正在执行该进程
107/// - `None`: CPU 处于空闲状态，等待新进程调度
108///
109/// ## 状态转换
110///
111/// ```text
112/// 处理器状态转换:
113///
114///    Idle State            Running State
115/// ┌─────────────┐         ┌─────────────┐
116/// │   Idle      │ ──────► │   Running   │
117/// │ current=None│Schedule │current=Some │
118/// └─────────────┘         └──────┬──────┘
119///       ▲                        │
120///       │  Process Complete/Switch  │
121///       └────────────────────────┘
122/// ```
123///
124/// ## 内存布局
125///
126/// ```text
127/// Processor 内存结构:
128/// ┌─────────────────────────────────────┐
129/// │           Processor                 │
130/// ├─────────────────────────────────────┤
131/// │ current: Option<Arc<TCB>>           │
132/// │  └─ Some: Points to Current TCB     │
133/// │  └─ None: Processor Idle            │
134/// ├─────────────────────────────────────┤
135/// │ idle_process_cx: ProcessContext           │
136/// │  └─ ra: Return Address              │
137/// │  └─ sp: Stack Pointer               │
138/// │  └─ s0-s11: Saved Registers         │
139/// └─────────────────────────────────────┘
140/// ```
141///
142/// ## 并发安全
143///
144/// `Processor` 本身不提供线程安全保证，必须通过 `UPSafeCell` 包装：
145/// - 单处理器系统中通过禁用中断保证原子性
146/// - 多处理器系统需要额外的同步机制
147///
148/// ## 性能特征
149///
150/// - **内存占用**: 约 264 bytes（ProcessContext ≈ 264 bytes + Arc 指针）
151/// - **切换开销**: O(1) 常数时间的上下文切换
152/// - **缓存友好**: 紧凑的内存布局提供良好的缓存局部性
153pub struct Processor {
154    /// 当前正在执行的进程
155    ///
156    /// - `Some(process)`: 指向正在 CPU 上执行的进程控制块
157    /// - `None`: CPU 处于空闲状态，调度器正在寻找下一个可运行进程
158    ///
159    /// 使用 `Arc` 允许进程控制块在调度器和其他组件之间共享所有权。
160    current: Option<Arc<ProcessControlBlock>>,
161
162    /// 空闲进程上下文
163    ///
164    /// 当没有用户进程运行时，CPU 执行调度器循环的上下文。
165    /// 包含调度器的寄存器状态，用于在进程切换时保存/恢复调度器状态。
166    ///
167    /// ## 作用
168    /// - **进程切换中转**: 从进程A切换到进程B时的中间状态
169    /// - **调度器状态**: 保存调度器循环的执行状态
170    /// - **空闲处理**: CPU 空闲时的默认执行上下文
171    idle_process_cx: ProcessContext,
172}

174impl Processor {
175    /// 创建新的处理器实例
176    ///
177    /// 初始化一个空闲状态的处理器，准备接受进程调度。
178    /// 处理器创建时没有当前进程，空闲上下文被清零。
179    ///
180    /// ## Returns
181    ///
182    /// 新创建的处理器实例，处于空闲状态
183    ///
184    /// ## 初始状态
185    ///
186    /// - `current`: `None` - 没有正在执行的进程
187    /// - `idle_process_cx`: 零初始化的进程上下文
188    ///
189    /// ## 使用场景
190    ///
191    /// - 系统初始化时创建全局处理器
192    /// - 多处理器系统中创建额外的处理器核心
193    ///
194    /// ## Examples
195    ///
196    /// ```rust
197    /// let processor = Processor::new();
198    /// assert!(processor.current().is_none()); // 初始状态为空闲
199    /// ```
200    pub fn new() -> Self {
201        Self {
202            current: None,
203            idle_process_cx: ProcessContext::zero_init(),
204        }
205    }

207    /// 获取当前正在执行的进程（克隆引用）
208    ///
209    /// 返回当前进程控制块的克隆引用，保持原始引用不变。
210    /// 适用于需要访问当前进程信息但不需要取走所有权的场景。
211    ///
212    /// ## Returns
213    ///
214    /// - `Some(Arc<ProcessControlBlock>)` - 当前进程的克隆引用
215    /// - `None` - 处理器当前处于空闲状态
216    ///
217    /// ## 引用计数
218    ///
219    /// 此函数会增加进程控制块的引用计数：
220    /// ```text
221    /// 调用前: Arc::strong_count = n
222    /// 调用后: Arc::strong_count = n + 1
223    /// ```
224    ///
225    /// ## 使用场景
226    ///
227    /// - **信息查询**: 获取当前进程的 PID、状态等信息
228    /// - **权限检查**: 验证当前进程的访问权限
229    /// - **上下文共享**: 在多个组件间共享进程引用
230    ///
231    /// ## Examples
232    ///
233    /// ```rust
234    /// if let Some(process) = processor.current() {
235    ///     println!("Current PID: {}", process.getpid());
236    ///     // 进程引用在作用域结束时自动释放
237    /// } else {
238    ///     println!("No process currently running");
239    /// }
240    /// ```
241    ///
242    /// ## 性能考虑
243    ///
244    /// - 引用计数操作开销很小（原子操作）
245    /// - 适合频繁的只读访问场景
246    /// - 避免不必要的所有权转移
247    pub fn current(&self) -> Option<Arc<ProcessControlBlock>> {
248        self.current.as_ref().map(|process| Arc::clone(process))
249    }
250
251    /// 取出当前正在执行的进程（转移所有权）
252    ///
253    /// 移除并返回当前进程控制块，将处理器设置为空闲状态。
254    /// 这是一个所有权转移操作，调用后处理器不再持有进程引用。
255    ///
256    /// ## Returns
257    ///
258    /// - `Some(Arc<ProcessControlBlock>)` - 被取出的进程控制块
259    /// - `None` - 处理器已经处于空闲状态
260    ///
261    /// ## 状态变化
262    ///
263    /// ```text
264    /// 调用前: current = Some(process), 处理器运行状态
265    /// 调用后: current = None,       处理器空闲状态
266    /// ```
267    ///
268    /// ## 使用场景
269    ///
270    /// - **进程切换**: 在调度新进程前取出当前进程
271    /// - **进程完成**: 进程退出时清理处理器状态
272    /// - **进程挂起**: 将进程转移到等待队列
273    ///
274    /// ## 调度流程中的作用
275    ///
276    /// ```text
277    /// 1. take_current_process() ──► 取出当前进程
278    /// 2. 进程状态处理 ────────► 加入就绪队列或退出
279    /// 3. fetch_process() ───────► 获取新进程
280    /// 4. 设置新的 current ───► 开始执行新进程
281    /// ```
282    ///
283    /// ## Examples
284    ///
285    /// ```rust
286    /// // 在进程调度中使用
287    /// if let Some(old_process) = processor.process_current() {
288    ///     // 处理旧进程（加入就绪队列、退出等）
289    ///     handle_process_switch(old_process);
290    /// }
291    /// // 现在处理器处于空闲状态，可以调度新进程
292    /// ```
293    ///
294    /// ## 注意事项
295    ///
296    /// - 调用后处理器立即进入空闲状态
297    /// - 必须确保在合适的时机调用，避免进程丢失
298    /// - 通常与进程调度算法配合使用
299    pub fn process_current(&mut self) -> Option<Arc<ProcessControlBlock>> {
300        self.current.take()
301    }
302
303    /// 获取空闲进程上下文的可变指针
304    ///
305    /// 返回指向空闲进程上下文的原始指针，用于底层的上下文切换操作。
306    /// 这是一个私有方法，只在内部上下文切换时使用。
307    ///
308    /// ## Returns
309    ///
310    /// `*mut ProcessContext` - 指向空闲上下文的可变原始指针
311    ///
312    /// ## 安全性考虑
313    ///
314    /// 此方法返回原始指针，调用者必须确保：
315    /// - 指针在使用期间保持有效
316    /// - 不会导致数据竞争或内存安全问题
317    /// - 只在适当的同步保护下使用
318    ///
319    /// ## 使用场景
320    ///
321    /// - **上下文切换**: 在 `__switch` 汇编函数中使用
322    /// - **调度器循环**: 保存/恢复调度器状态
323    /// - **进程切换中转**: 作为进程间切换的中介
324    ///
325    /// ## 上下文切换流程
326    ///
327    /// ```text
328    /// 1. 保存当前进程上下文到进程控制块
329    /// 2. 恢复空闲上下文 ────────────► 返回调度器
330    /// 3. 调度器选择新进程
331    /// 4. 保存空闲上下文 ────────────► 为下次切换准备
332    /// 5. 恢复新进程上下文 ─────────► 执行新进程
333    /// ```
334    ///
335    /// ## 内存布局
336    ///
337    /// 返回的指针指向 `idle_process_cx` 字段的内存地址，
338    /// 包含完整的 RISC-V 寄存器上下文信息。
339    fn idle_process_cx_ptr(&mut self) -> *mut ProcessContext {
340        &mut self.idle_process_cx as *mut _
341    }
342}

345    /// 全局处理器实例
346    ///
347    /// 系统中唯一的处理器管理器，负责协调所有的进程调度和 CPU 资源分配。
348    /// 使用 `UPSafeCell` 提供单处理器环境下的安全可变访问。
349    ///
350    /// ## 设计特点
351    ///
352    /// ### 单例模式
353    /// - **全局唯一**: 整个系统只有一个处理器管理器实例
354    /// - **延迟初始化**: 在首次访问时才进行初始化
355    /// - **生命周期管理**: 伴随程序整个生命周期存在
356    ///
357    /// ### 并发安全
358    /// - **UPSafeCell 保护**: 单处理器环境下的内部可变性
359    /// - **互斥访问**: 通过 `exclusive_access()` 获取独占访问权
360    /// - **中断安全**: 在中断处理期间禁用抢占保证原子性
361    ///
362    /// ### 内存布局
363    /// ```text
364    /// 全局内存区域:
365    /// ┌─────────────────────────────────────────┐
366    /// │             PROCESSOR                   │
367    /// │ ┌─────────────────────────────────────┐ │
368    /// │ │        UPSafeCell<Processor>        │ │
369    /// │ │ ┌─────────────────────────────────┐ │ │
370    /// │ │ │         Processor               │ │ │
371    /// │ │ │  ┌───────────────────────────┐  │ │ │
372    /// │ │ │  │ current: Option<Arc<TCB>> │  │ │ │
373    /// │ │ │  ├───────────────────────────┤  │ │ │
374    /// │ │ │  │ idle_process_cx: ProcessContext │  │ │ │
375    /// │ │ │  └───────────────────────────┘  │ │ │
376    /// │ │ └─────────────────────────────────┘ │ │
377    /// │ └─────────────────────────────────────┘ │
378    /// └─────────────────────────────────────────┘
379    /// ```
380    ///
381    /// ## 访问模式
382    ///
383    /// ### 推荐方式（通过全局函数）
384    /// ```rust
385    /// // 安全的高层接口
386    /// let current_process = current_process();
387    /// let user_token = current_user_token();
388    /// ```
389    ///
390    /// ### 直接访问（需要小心）
391    /// ```rust
392    /// // 低层接口，需要手动管理锁
393    /// let processor = PROCESSOR.exclusive_access();
394    /// let process = processor.current();
395    /// drop(processor); // 及时释放锁
396    /// ```
397    ///
398    /// ## 初始化时机
399    ///
400    /// - **首次访问**: 在第一次调用相关函数时初始化
401    /// - **系统启动**: 通常在调度器启动前完成初始化
402    /// - **一次性**: 初始化后不会重复执行
403    ///
404    /// ## 性能影响
405    ///
406    /// - **全局访问**: 不涉及页表切换，访问速度快
407    /// - **锁开销**: 在单核系统中开销极小
408    /// - **缓存友好**: 全局变量在内存中位置固定
409    ///
410    /// ## 线程安全保证
411    ///
412    /// - **原子访问**: `exclusive_access()` 保证同一时间只有一个访问者
413    /// - **中断禁用**: 访问期间自动禁用中断避免竞争
414    /// - **RAII 管理**: 作用域结束时自动释放访问权限
415    ///
416    /// ## 使用示例
417    ///
418    /// ```rust
419    /// // 系统初始化时
420    /// pub fn init_scheduler() {
421    ///     // PROCESSOR 在首次访问时自动初始化
422    ///     run_processs(); // 启动主调度循环
423    /// }
424    ///
425    /// // 中断处理中
426    /// pub fn handle_timer_interrupt() {
427    ///     // 通过全局函数安全访问
428    ///     if let Some(process) = current_process() {
429    ///         // 处理当前进程的时间片...
430    ///     }
431    /// }
432    /// ```
433    pub static ref PROCESSOR: UPSafeCell<Processor> = unsafe { UPSafeCell::new(Processor::new()) };
434}

436/// 取出当前正在执行的进程
437///
438/// 从全局处理器中移除并返回当前进程，将处理器设置为空闲状态。
439/// 这是一个高级封装函数，提供线程安全的进程取出操作。
440///
441/// ## Returns
442///
443/// - `Some(Arc<ProcessControlBlock>)` - 成功取出的当前进程
444/// - `None` - 处理器当前没有运行任何进程
445///
446/// ## 线程安全
447///
448/// 此函数通过 `PROCESSOR.exclusive_access()` 确保原子性操作：
449/// - 获取全局处理器的独占访问权
450/// - 安全地取出当前进程
451/// - 自动释放访问权限
452///
453/// ## 使用场景
454///
455/// ### 进程调度
456/// ```rust
457/// // 在调度器中取出当前进程进行状态管理
458/// if let Some(process) = take_current_process() {
459///     // 根据进程状态决定后续处理
460///     match process.inner_exclusive_access().process_status {
461///         ProcessStatus::Running => {
462///             // 进程主动让出 CPU，加入就绪队列
463///             add_process(process);
464///         }
465///         ProcessStatus::Zombie => {
466///             // 进程已退出，不需要重新调度
467///             println!("Process {} exited", process.getpid());
468///         }
469///         _ => {
470///             // 其他状态处理...
471///         }
472///     }
473/// }
474/// ```
475///
476/// ### 系统调用处理
477/// ```rust
478/// // 在 exit 系统调用中
479/// pub fn sys_exit(exit_code: i32) -> ! {
480///     let process = take_current_process().unwrap();
481///     // 设置进程为僵尸状态...
482///     // 调度到其他进程...
483/// }
484/// ```
485///
486/// ## 性能特征
487///
488/// - **时间复杂度**: O(1) 常数时间操作
489/// - **锁开销**: 单核系统中开销极小
490/// - **原子性**: 通过独占访问保证操作的原子性
491///
492/// ## 注意事项
493///
494/// - 调用后处理器立即进入空闲状态
495/// - 必须确保有适当的进程调度机制跟进
496/// - 避免长时间持有取出的进程而不进行处理
497pub fn take_current_process() -> Option<Arc<ProcessControlBlock>> {
498    PROCESSOR.exclusive_access().process_current()
499}

501/// 获取当前正在执行的进程（只读访问）
502///
503/// 返回当前进程的克隆引用，不改变处理器状态。适用于需要查询
504/// 当前进程信息但不需要修改进程状态的场景。
505///
506/// ## Returns
507///
508/// - `Some(Arc<ProcessControlBlock>)` - 当前进程的克隆引用
509/// - `None` - 处理器当前没有运行任何进程
510///
511/// ## 引用语义
512///
513/// 返回的是 `Arc` 的克隆，增加引用计数但不转移所有权：
514/// ```text
515/// 处理器状态保持不变：current = Some(process)
516/// 返回值：Arc::clone(process) - 新的引用
517/// ```
518///
519/// ## 使用场景
520///
521/// ### 信息查询
522/// ```rust
523/// // 获取当前进程信息
524/// if let Some(process) = current_process() {
525///     println!("Current PID: {}", process.getpid());
526///     println!("Process status: {:?}", process.inner_exclusive_access().process_status);
527/// } else {
528///     println!("No process currently running");
529/// }
530/// ```
531///
532/// ### 权限检查
533/// ```rust
534/// // 验证系统调用权限
535/// pub fn sys_read(fd: usize, buf: *mut u8, len: usize) -> isize {
536///     if let Some(process) = current_process() {
537///         // 验证文件描述符属于当前进程...
538///         // 验证内存访问权限...
539///     } else {
540///         return -1; // 无当前进程
541///     }
542/// }
543/// ```
544///
545/// ### 上下文获取
546/// ```rust
547/// // 获取当前进程的内存地址空间
548/// let current_memory_set = current_process()
549///     .unwrap()
550///     .inner_exclusive_access()
551///     .memory_set
552///     .token();
553/// ```
554///
555/// ## 性能考虑
556///
557/// - **引用计数开销**: Arc::clone 涉及原子操作，开销很小
558/// - **锁开销**: 短暂的独占访问，影响极小  
559/// - **内存友好**: 不复制进程数据，只复制引用
560///
561/// ## 线程安全
562///
563/// 通过 `PROCESSOR.exclusive_access()` 保证读取的原子性，
564/// 避免在读取过程中进程被切换导致的竞争条件。
565pub fn current_process() -> Option<Arc<ProcessControlBlock>> {
566    PROCESSOR.exclusive_access().current()
567}

569/// 获取当前进程的用户地址空间标识符
570///
571/// 返回当前用户进程的页表标识符（satp 寄存器值），用于地址空间管理
572/// 和内存访问控制。这是一个便捷函数，封装了获取用户空间令牌的复杂过程。
573///
574/// ## Returns
575///
576/// `usize` - 当前进程的用户地址空间标识符
577///
578/// ## Panics
579///
580/// 如果当前没有运行进程，函数会 panic。调用前应确保有进程在运行。
581///
582/// ## satp 寄存器格式
583///
584/// 返回值符合 RISC-V satp 寄存器格式：
585/// ```text
586/// ┌────────────┬────────────────┬──────────────────────────────────────────────┐
587/// │    MODE    │      ASID      │                    PPN                       │
588/// │   (4bit)   │     (16bit)    │                  (44bit)                     │
589/// └────────────┴────────────────┴──────────────────────────────────────────────┘
590/// ```
591///
592/// ## 使用场景
593///
594/// ### 地址空间切换
595/// ```rust
596/// // 在用户态陷阱处理中保存/恢复地址空间
597/// pub fn handle_user_trap() {
598///     let user_satp = current_user_token();
599///     
600///     // 切换到内核地址空间处理陷阱...
601///     
602///     // 恢复用户地址空间
603///     unsafe {
604///         satp::write(user_satp);
605///         asm!("sfence.vma");
606///     }
607/// }
608/// ```
609///
610/// ### 内存访问验证
611/// ```rust
612/// // 验证用户地址是否属于当前进程
613/// pub fn validate_user_address(addr: VirtAddr) -> bool {
614///     let user_token = current_user_token();
615///     // 使用用户页表进行地址转换验证...
616/// }
617/// ```
618///
619/// ### 系统调用参数访问
620/// ```rust
621/// // 安全地访问用户空间数据
622/// pub fn copy_from_user(user_ptr: *const u8, len: usize) -> Vec<u8> {
623///     let user_token = current_user_token();
624///     // 使用用户页表访问用户内存...
625/// }
626/// ```
627///
628/// ## 调用链
629///
630/// ```text
631/// current_user_token()
632///     ↓
633/// current_process().unwrap()
634///     ↓
635/// process.inner_exclusive_access()
636///     ↓
637/// inner.user_token()
638///     ↓
639/// memory_set.token()
640/// ```
641///
642/// ## 性能特征
643///
644/// - **多级访问**: 需要穿越多层封装获取令牌
645/// - **锁开销**: 涉及进程控制块的互斥访问
646/// - **缓存友好**: 令牌通常会被频繁使用，具有良好的时间局部性
647///
648/// ## 安全考虑
649///
650/// 返回的令牌代表了完整的用户地址空间访问权限，使用时需要注意：
651/// - 确保在正确的上下文中使用
652/// - 避免跨进程使用其他进程的令牌
653/// - 在地址空间切换后及时更新
654pub fn current_user_token() -> usize {
655    let process = current_process().unwrap();
656    let token = process.inner_exclusive_access().user_token();
657    token
658}

660/// 获取当前进程的陷阱上下文
661///
662/// 返回指向当前进程陷阱上下文的可变引用，用于中断和系统调用处理。
663/// 陷阱上下文包含了用户程序在陷入内核时的完整 CPU 状态。
664///
665/// ## Returns
666///
667/// `&'static mut TrapContext` - 指向当前进程陷阱上下文的可变引用
668///
669/// ## Panics
670///
671/// 如果当前没有运行进程，函数会 panic。通常在中断/异常处理上下文中调用。
672///
673/// ## 陷阱上下文结构
674///
675/// 返回的陷阱上下文包含：
676/// ```text
677/// TrapContext {
678///     x[0..32]:  通用寄存器（x0-x31）
679///     sstatus:   处理器状态寄存器
680///     sepc:      异常程序计数器
681///     kernel_satp: 内核页表标识符
682///     kernel_sp:   内核栈指针
683///     trap_handler: 陷阱处理函数地址
684/// }
685/// ```
686///
687/// ## 使用场景
688///
689/// ### 系统调用处理
690/// ```rust
691/// // 获取系统调用参数和设置返回值
692/// pub fn syscall(syscall_id: usize, args: [usize; 3]) -> isize {
693///     let cx = current_trap_cx();
694///     
695///     // 读取寄存器参数
696///     let arg0 = cx.x[10]; // a0 寄存器
697///     let arg1 = cx.x[11]; // a1 寄存器
698///     
699///     let result = match syscall_id {
700///         // 处理各种系统调用...
701///     };
702///     
703///     // 设置返回值
704///     cx.x[10] = result as usize; // 通过 a0 寄存器返回
705/// }
706/// ```
707///
708/// ### 中断处理
709/// ```rust
710/// // 时钟中断处理
711/// pub fn handle_timer_interrupt() {
712///     let cx = current_trap_cx();
713///     
714///     // 保存中断时的程序计数器
715///     let user_pc = cx.sepc;
716///     println!("Timer interrupt at PC: {:#x}", user_pc);
717///     
718///     // 设置时间片到期标志...
719/// }
720/// ```
721///
722/// ### 程序状态检查
723/// ```rust
724/// // 检查用户程序执行状态
725/// pub fn check_user_state() {
726///     let cx = current_trap_cx();
727///     
728///     if cx.sstatus.spie() {
729///         println!("User program had interrupts enabled");
730///     }
731///     
732///     println!("User PC: {:#x}", cx.sepc);
733///     println!("User SP: {:#x}", cx.x[2]); // sp 寄存器
734/// }
735/// ```
736///
737/// ## 内存布局
738///
739/// 陷阱上下文位于用户地址空间的固定位置：
740/// ```text
741/// 用户地址空间高地址区域:
742/// ┌──────────────────────────────────┐ ← TRAMPOLINE
743/// │         Trampoline Page          │
744/// ├──────────────────────────────────┤ ← TRAP_CONTEXT  
745/// │        Trap Context              │ ← current_trap_cx() 返回
746/// │  ┌────────────────────────────┐  │
747/// │  │ x[0..32]: General Registers│  │
748/// │  ├────────────────────────────┤  │
749/// │  │ sstatus: Status Register   │  │
750/// │  ├────────────────────────────┤  │
751/// │  │ sepc: Program Counter      │  │
752/// │  └────────────────────────────┘  │
753/// └──────────────────────────────────┘
754/// ```
755///
756/// ## 生命周期
757///
758/// 返回的引用具有 `'static` 生命周期，因为陷阱上下文在进程的整个
759/// 生命周期内都有效，并且存储在固定的虚拟地址位置。
760///
761/// ## 安全性考虑
762///
763/// - 返回可变引用，可以直接修改用户程序状态
764/// - 仅应在内核态的中断/系统调用处理中使用
765/// - 修改陷阱上下文会直接影响用户程序的执行
766pub fn current_trap_cx() -> &'static mut TrapContext {
767    current_process()
768        .unwrap()
769        .inner_exclusive_access()
770        .trap_cx()
771}

773/// 主调度循环 - 系统调度器的核心
774///
775/// 这是操作系统的心脏，负责持续地从就绪队列中获取进程并执行。
776/// 函数永不返回，构成了系统的主调度循环，确保 CPU 始终在执行有用的工作。
777///
778/// ## 调度算法
779///
780/// 采用 **FIFO (First In, First Out)** 调度策略：
781/// - 按照进程进入就绪队列的顺序进行调度
782/// - 保证调度的公平性和可预测性
783/// - 避免进程饥饿问题
784///
785/// ## 执行流程
786///
787/// ```text
788/// 主调度循环流程:
789///
790/// ┌───────────────────────────────────────────────────────────┐
791/// │                    run_processs()                            │
792/// │  ┌─────────────┐                                          │
793/// │  │   Get       │ ──► PROCESSOR.exclusive_access()         │
794/// │  │ Exclusive   │                                          │
795/// │  │ Processor   │                                          │
796/// │  └─────────────┘                                          │
797/// │         │                                                 │
798/// │         ▼                                                 │
799/// │  ┌─────────────┐    Yes  ┌────────-───────────────────┐   │
800/// │  │   Fetch     │ ──────► │     Prepare Process Switch    │   │
801/// │  │ Next Ready  │         │  1. Get Idle Context Ptr   │   │
802/// │  │   Process      │         │  2. Get Process Context Ptr   │   │
803/// │  └─────────────┘         │  3. Set Process Status Running│   │
804/// │         │                │  4. Set as Current Process    │   │
805/// │         │ No             └────────────────────────────┘   │
806/// │         ▼                               │                 │
807/// │  ┌─────────────┐                        ▼                 │
808/// │  │ Idle Wait   │         ┌─────────────────────────────┐  │
809/// │  │ CPU Halt    │         │    Execute Context Switch   │  │
810/// │  └─────────────┘         │  __switch(idle_cx, process_cx) │  │
811/// │         │                └─────────────────────────────┘  │
812/// │         │                               │                 │
813/// │         │                               ▼                 │
814/// │         │                ┌─────────────────────────────┐  │
815/// │         │                │     Process Start Running      │  │
816/// │         │                │  (Switch to User Mode)      │  │
817/// │         │                └─────────────────────────────┘  │
818/// │         │                               │                 │
819/// │         │                               │                 │
820/// │         └◄──────── Continue Loop ◄──────┘                 │
821/// └───────────────────────────────────────────────────────────┘
822/// ```
823///
824/// ## 上下文切换机制
825///
826/// ### 双上下文模型
827/// ```text
828/// 切换前:                切换后:
829/// ┌────────────┐      ┌─-────────────┐
830/// │Idle Context│ ──►  │ Process Context │
831/// │(Scheduler) │ ◄─── │(User Program)│
832/// └────────────┘      └──-───────────┘
833/// ```
834///
835/// ### 切换步骤
836/// 1. **保存调度器状态**: 将当前寄存器保存到 `idle_process_cx`
837/// 2. **恢复进程状态**: 从 `process.process_cx` 恢复进程寄存器
838/// 3. **跳转执行**: CPU 开始执行用户进程代码
839/// 4. **进程返回**: 进程通过中断/系统调用返回时恢复调度器状态
840///
841/// ## 内存管理
842///
843/// 在进程切换时自动处理：
844/// - **地址空间切换**: 每个进程有独立的虚拟地址空间
845/// - **页表切换**: 通过 `satp` 寄存器切换页表
846/// - **内核栈**: 每个进程有独立的内核栈用于系统调用处理
847///
848/// ## 性能特征
849///
850/// - **调度延迟**: O(1) - 常数时间进程获取和切换
851/// - **上下文切换开销**: ~100-200 CPU 周期（寄存器保存/恢复）
852/// - **内存开销**: 每进程约 4KB 内核栈 + 页表开销
853/// - **公平性**: FIFO 策略保证所有进程获得公平的 CPU 时间
854///
855/// ## 使用场景
856///
857/// ### 系统启动
858/// ```rust
859/// pub fn main() {
860///     // 系统初始化...
861///     init_scheduler();
862///     
863///     // 启动调度器（永不返回）
864///     run_processs();
865/// }
866/// ```
867///
868/// ### 空闲状态处理
869/// ```rust
870/// // 当没有就绪进程时，CPU 进入空闲循环
871/// // run_processs() 会持续检查新的就绪进程
872/// // 通常配合中断机制唤醒进程
873/// ```
874///
875/// ## 并发安全
876///
877/// - **原子操作**: 通过 `exclusive_access()` 保证进程获取的原子性
878/// - **中断安全**: 在关键区域禁用中断防止竞争条件
879/// - **状态一致性**: 确保进程状态转换的一致性
880///
881/// ## 调试特征
882///
883/// 可以添加调试代码监控调度行为：
884/// ```rust
885/// // 在实际代码中可以添加调试输出
886/// println!("Switching to process PID: {}", process.getpid());
887/// ```
888///
889/// ## 注意事项
890///
891/// - 此函数永不返回，是系统的主循环
892/// - 必须在系统初始化完成后调用
893/// - 调用前应确保有初始进程在就绪队列中
894/// - 如果没有进程，会在循环中等待（可能需要中断唤醒）
895pub fn run_processs() {
896    loop {
897        let mut processor = PROCESSOR.exclusive_access();
898        if let Some(process) = fetch_process() {
899            let idle_process_cx_ptr = processor.idle_process_cx_ptr();
900            let mut process_inner = process.inner_exclusive_access();
901            let next_process_cx_ptr = &process_inner.process_cx as *const ProcessContext;
902            process_inner.process_status = ProcessStatus::Running;
903            drop(process_inner);
904            processor.current = Some(process);
905            drop(processor);
906            unsafe {
907                __switch(idle_process_cx_ptr, next_process_cx_ptr);
908            }
909        }
910    }
911}

913/// 进程调度函数 - 从当前进程切换回调度器
914///
915/// 将执行控制权从当前正在运行的进程转移回调度器的核心函数。
916/// 通常在时间片到期、进程主动让出 CPU 或进程阻塞时调用。
917///
918/// ## Arguments
919///
920/// * `switched_process_cx_ptr` - 指向当前进程上下文的可变指针，用于保存进程状态
921///
922/// ## 调度时机
923///
924/// ### 抢占式调度
925/// ```rust
926/// // 时钟中断处理中
927/// pub fn handle_timer_interrupt() {
928///     // 处理时间片到期
929///     set_next_trigger();
930///     
931///     // 切换到调度器
932///     schedule(current_process_cx_ptr);
933/// }
934/// ```
935///
936/// ### 协作式调度  
937/// ```rust
938/// // yield 系统调用
939/// pub fn sys_yield() -> isize {
940///     // 进程主动让出 CPU
941///     schedule(current_process_cx_ptr);
942///     0
943/// }
944/// ```
945///
946/// ### 阻塞调度
947/// ```rust
948/// // 进程等待资源时
949/// pub fn sys_wait_for_resource() -> isize {
950///     // 将进程加入等待队列
951///     add_to_wait_queue(current_process());
952///     
953///     // 切换到其他进程
954///     schedule(current_process_cx_ptr);
955/// }
956/// ```
957///
958/// ## 执行流程
959///
960/// ```text
961/// schedule() 执行流程:
962///
963/// ┌─────────────────────────────────────────────────────────────┐
964/// │                   schedule()                                │
965/// │                                                             │
966/// │  ┌─────────────────┐                                        │
967/// │  │ Save Current    │ ──► Caller already saved process state    │
968/// │  │ Process State to   │     to context                         │
969/// │  │ Context         │                                        │
970/// │  └─────────────────┘                                        │
971/// │           │                                                 │
972/// │           ▼                                                 │
973/// │  ┌─────────────────┐                                        │
974/// │  │ Get Exclusive   │ ──► PROCESSOR.exclusive_access()       │
975/// │  │ Processor       │                                        │
976/// │  │ Access          │                                        │
977/// │  └─────────────────┘                                        │
978/// │           │                                                 │
979/// │           ▼                                                 │
980/// │  ┌─────────────────┐                                        │
981/// │  │ Get Idle        │ ──► processor.idle_process_cx_ptr()   │
982/// │  │ Context         │                                        │
983/// │  │ Pointer         │                                        │
984/// │  └─────────────────┘                                        │
985/// │           │                                                 │
986/// │           ▼                                                 │
987/// │  ┌─────────────────┐                                        │
988/// │  │ Release         │ ──► drop(processor)                    │
989/// │  │ Processor       │                                        │
990/// │  │ Access          │                                        │
991/// │  └─────────────────┘                                        │
992/// │           │                                                 │
993/// │           ▼                                                 │
994/// │  ┌─────────────────┐                                        │
995/// │  │ Execute Context │ ──► __switch(process_cx, idle_cx)         │
996/// │  │ Switch Return   │                                        │
997/// │  │ to Scheduler    │                                        │
998/// │  └─────────────────┘                                        │
999/// │           │                                                 │
1000/// │           ▼                                                 │
1001/// │  ┌─────────────────┐                                        │
1002/// │  │ Scheduler       │ ──► Return to run_processs() loop         │
1003/// │  │ Continues       │                                        │
1004/// │  │ Execution       │                                        │
1005/// │  └─────────────────┘                                        │
1006/// └─────────────────────────────────────────────────────────────┘
1007/// ```
1008///
1009/// ## 上下文切换详解
1010///
1011/// ### 切换方向
1012/// ```text
1013/// 进程执行中 ──► schedule() ──► 调度器循环
1014///     │                          │
1015///     │                          ▼
1016///     │                    选择新进程
1017///     │                          │
1018///     │                          ▼
1019///     └◄─── 新进程执行 ◄─── run_processs()
1020/// ```
1021///
1022/// ### 内存状态变化
1023/// ```text
1024/// 切换前:                 切换后:
1025/// ┌─────────────┐         ┌─────────────┐
1026/// │   Process A    │ ──────► │  Scheduler  │
1027/// │  (Running)  │         │  (Active)   │  
1028/// └─────────────┘         └─────────────┘
1029///       │                       │
1030///       ▼                       ▼
1031/// ┌─────────────┐         ┌─────────────┐
1032/// │ Process Context│         │ Idle Context│
1033/// │  (Saved)    │         │  (Restored) │
1034/// └─────────────┘         └─────────────┘
1035/// ```
1036///
1037/// ## 调用约定
1038///
1039/// ### 调用前要求
1040/// - 当前进程的上下文必须已经保存到 `switched_process_cx_ptr` 指向的位置
1041/// - 进程状态应该已经适当更新（Ready, Blocked 等）
1042/// - 如需要，进程应该已经加入相应的队列（就绪队列、等待队列等）
1043///
1044/// ### 调用后保证
1045/// - 控制权转移到调度器
1046/// - 调度器会选择下一个进程执行
1047/// - 当前进程可能稍后被重新调度
1048///
1049/// ## 性能考虑
1050///
1051/// - **切换开销**: 约 100-200 CPU 周期
1052/// - **内存访问**: 主要是寄存器保存/恢复
1053/// - **缓存影响**: 可能导致缓存未命中
1054/// - **TLB 影响**: 如果切换到不同进程，可能导致 TLB 刷新
1055///
1056/// ## 使用示例
1057///
1058/// ### 时间片轮转
1059/// ```rust
1060/// // 在时钟中断处理函数中
1061/// pub extern "C" fn timer_interrupt_handler() {
1062///     // 更新时间片计数...
1063///     
1064///     // 时间片到期，调度其他进程
1065///     let current_cx_ptr = current_process_context_ptr();
1066///     
1067///     // 将当前进程重新加入就绪队列
1068///     if let Some(process) = take_current_process() {
1069///         add_process(process);
1070///     }
1071///     
1072///     // 切换到调度器选择新进程
1073///     schedule(current_cx_ptr);
1074/// }
1075/// ```
1076///
1077/// ## 安全性考虑
1078///
1079/// - 使用 `unsafe` 代码进行底层上下文切换
1080/// - 必须确保传入的指针有效且指向正确的上下文结构
1081/// - 调用时必须在适当的内核态上下文中
1082///
1083/// ## 与 run_processs() 的协作
1084///
1085/// `schedule()` 和 `run_processs()` 构成完整的调度循环：
1086/// - `run_processs()`: 从调度器切换到进程
1087/// - `schedule()`: 从进程切换回调度器
1088pub fn schedule(switched_process_cx_ptr: *mut ProcessContext) {
1089    let mut processor = PROCESSOR.exclusive_access();
1090    let idle_process_cx_ptr = processor.idle_process_cx_ptr();
1091    drop(processor);
1092    unsafe {
1093        __switch(switched_process_cx_ptr, idle_process_cx_ptr);
1094    }
1095}
