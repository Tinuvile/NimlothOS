1//! # 进程管理器模块
2//!
3//! 提供进程调度和管理的核心功能，实现就绪队列的维护和进程分发。
4//! 采用多级反馈队列 (MLFQ) 调度策略，实现可降低优先级的抢占式调度。
5//!
6//! ## 核心组件
7//!
8//! - [`ProcessManager`] - 进程管理器，维护就绪进程队列
9//! - [`PROCESS_MANAGER`] - 全局进程管理器实例
10//! - [`add_process`] - 向就绪队列添加进程的全局接口
11//! - [`fetch_process`] - 从就绪队列获取进程的全局接口
12//!
13//! ## 设计原理
14//!
15//! ### 调度策略
16//!
17//! 采用 **多级反馈队列 (MLFQ)** 调度算法：
18//! - **优先级分级**：维护多个优先级不同的就绪队列
19//! - **动态降级**：进程用完时间片后降级到低优先级队列
20//! - **响应性优化**：新进程和I/O密集型进程优先调度
21//! - **公平性保证**：避免长期饥饿，底层队列也会被调度
22//!
23//! ### 数据结构选择
24//!
25//! 使用 `Vec<VecDeque<Arc<ProcessControlBlock>>>` 作为多级就绪队列：
26//! - **多级队列**：每个优先级都有独立的双端队列
27//! - **优先级调度**：总是从最高优先级非空队列取进程
28//! - **动态时间片**：不同优先级队列使用不同的时间片长度
29//! - **时间复杂度**：入队和出队操作均为 O(1)
30//!
31//! ### 并发安全
32//!
33//! 通过 `UPSafeCell` 实现单处理器环境下的线程安全：
34//! - **互斥访问**：确保同一时间只有一个线程可以修改就绪队列
35//! - **内部可变性**：允许在不可变引用下修改内部数据
36//! - **无锁设计**：在单核环境下避免昂贵的锁操作
37//!
38//! ## 调度流程
39//!
40//! ```text
41//! Process Creation/Wake-up    Process Scheduling
42//!         │                           │
43//!         ▼                           ▼
44//! ┌──────────────┐             ┌──────────────┐
45//! │  add_process │             │ fetch_process│
46//! │  (Enqueue)   │             │  (Dequeue)   │
47//! └──────┬───────┘             └──────┬───────┘
48//!        │                            │
49//!        ▼                            ▼
50//! ┌───────────────────────────────────────-----──----─┐
51//! │            Ready Queue                            │
52//! │  ┌───---──┐  ┌────---─┐  ┌──---───┐  ┌─────---┐   │
53//! │  │Process │◄─│Process │◄─│Process │◄─│Process │   │
54//! │  │    A   │  │    B   │  │    C   │  │    D   │   │
55//! │  └────---─┘  └---─────┘  └─────---┘  └──---───┘   │
56//! │    ▲                                       │      │
57//! │    │ Add New Process       Get Next Process│      │
58//! │    └──────────────────────------------──-──┘      │
59//! └────────────────────────────────---─────────------─┘
60//! ```
61//!
62//! ## 性能特征
63//!
64//! - **入队延迟**: O(1) - 常数时间复杂度
65//! - **出队延迟**: O(1) - 常数时间复杂度  
66//! - **内存开销**: 每个进程一个 `Arc` 指针 (8 bytes)
67//! - **缓存友好性**: 连续内存布局提供良好的缓存局部性
68//!
69//! ## 使用示例
70//!
71//! ```rust
72//! use crate::process::manager::{add_process, fetch_process};
73//! use crate::process::process::ProcessControlBlock;
74//! use alloc::sync::Arc;
75//!
76//! // 添加新进程到就绪队列
77//! let new_process = Arc::new(ProcessControlBlock::new(app_data));
78//! add_process(new_process);
79//!
80//! // 调度器获取下一个进程
81//! if let Some(process) = fetch_process() {
82//!     println!("调度进程 PID: {}", process.getpid());
83//!     // 执行进程...
84//! } else {
85//!     println!("没有就绪进程，进入空闲状态");
86//! }
87//! ```

98/// 进程管理器
99///
100/// 负责维护系统中所有就绪态进程的队列，提供高效的进程调度服务。
101/// 采用 FIFO 调度策略，确保进程调度的公平性和可预测性。
102///
103/// ## 设计特点
104///
105/// ### 数据结构
106/// - **就绪队列**: 使用 `VecDeque` 实现双端队列，支持 O(1) 的队首删除和队尾插入
107/// - **进程引用**: 存储 `Arc<ProcessControlBlock>` 而非进程本身，支持共享所有权
108/// - **内存布局**: 连续内存分配提供良好的缓存局部性
109///
110/// ### 调度算法
111/// ```text
112/// Schedule Order: Process1 → Process2 → Process3 → Process4
113///          ┌──---───┐  ┌────---─┐  ┌─---────┐  ┌───---──┐
114/// Enqueue ►│Process4│◄─│Process3│◄─│Process2│◄─│Process1│◄─── Dequeue
115///          └─────---┘  └───---──┘  └────---─┘  └---─────┘
116///         (最新)                      (最老)
117/// ```
118///
119/// ### 线程安全
120/// 虽然 `ProcessManager` 本身不提供线程安全保证，但通过全局的 `UPSafeCell`
121/// 包装器确保在单处理器环境下的并发安全访问。
122///
123/// ## 使用模式
124///
125/// 通常不直接实例化 `ProcessManager`，而是通过全局函数接口使用：
126/// ```rust
127/// // 推荐方式：通过全局接口使用
128/// add_process(new_process);
129/// let next_process = fetch_process();
130///
131/// // 不推荐：直接操作实例
132/// let mut manager = ProcessManager::new();
133/// manager.add(process);
134/// ```
135///
136/// ## 性能保证
137///
138/// - **入队延迟**: O(1) 常数时间
139/// - **出队延迟**: O(1) 常数时间
140/// - **内存开销**: 8 bytes per process (仅存储 Arc 指针)
141/// - **扩容策略**: 按需自动扩容，避免频繁内存分配
142pub struct ProcessManager {
143    /// 多级就绪进程队列
144    ///
145    /// 使用多个双端队列实现不同优先级的进程调度。
146    /// 索引 0 为最高优先级队列，索引越大优先级越低。
147    /// 调度器总是从最高优先级非空队列取进程执行。
148    ///
149    /// ## 队列特性
150    /// - **优先级调度**: 高优先级队列优先被调度
151    /// - **动态降级**: 用完时间片的进程降级到下一队列
152    /// - **时间片分配**: 不同队列有不同的时间片长度
153    /// - **高效操作**: 每个队列的入队和出队均为 O(1)
154    ready_queues: Vec<VecDeque<Arc<ProcessControlBlock>>>,
155
156    /// 每个队列的时间片长度（时钟周期数）
157    ///
158    /// 优先级越高的队列时间片越短，保证响应性；
159    /// 优先级越低的队列时间片越长，提高吞吐量。
160    time_slices: Vec<usize>,
161
162    /// 队列数量
163    queue_count: usize,
164}
165
166impl ProcessManager {
167    /// 创建新的进程管理器实例
168    ///
169    /// 初始化一个空的就绪队列，准备接收和调度进程。
170    /// 新创建的管理器不包含任何进程，队列容量为 0。
171    ///
172    /// ## 返回值
173    ///
174    /// 返回一个新的 `ProcessManager` 实例，包含：
175    /// - 空的就绪队列（容量为 0）
176    /// - 所有计数器重置为 0
177    ///
178    /// ## 性能特征
179    ///
180    /// - **时间复杂度**: O(1) - 常数时间初始化
181    /// - **内存分配**: 不进行堆内存分配
182    /// - **初始容量**: 0，按需扩容以优化内存使用
183    ///
184    /// ## 使用说明
185    ///
186    /// 通常不需要直接调用此函数，而是通过全局 `PROCESS_MANAGER` 实例使用：
187    ///
188    /// ```rust
189    /// // 推荐：使用全局实例
190    /// use crate::process::manager::{add_process, fetch_process};
191    ///
192    /// // 不推荐：直接创建实例
193    /// let manager = ProcessManager::new();
194    /// ```
195    ///
196    /// ## 线程安全
197    ///
198    /// `new()` 函数本身是线程安全的，但返回的实例需要通过 `UPSafeCell`
199    /// 等同步原语保护才能在多线程环境中安全使用。
200    pub fn new() -> Self {
201        use crate::config::{MLFQ_BASE_TIME_SLICE, MLFQ_QUEUE_COUNT};
202
203        let mut ready_queues = Vec::new();
204        let mut time_slices = Vec::new();
205
206        // 初始化多级队列，时间片按优先级递增
207        for i in 0..MLFQ_QUEUE_COUNT {
208            ready_queues.push(VecDeque::new());
209            // 时间片：10ms, 20ms, 40ms, 80ms
210            time_slices.push(MLFQ_BASE_TIME_SLICE * (1 << i));
211        }
212
213        Self {
214            ready_queues,
215            time_slices,
216            queue_count: MLFQ_QUEUE_COUNT,
217        }
218    }
219
220    /// 向就绪队列添加进程
221    ///
222    /// 将指定的进程控制块添加到就绪队列的尾部，使其可以被调度器选中执行。
223    /// 新添加的进程将按照 FIFO 顺序等待调度。
224    ///
225    /// ## 参数
226    ///
227    /// * `process` - 要添加到队列的进程控制块，包装在 `Arc` 中支持共享所有权
228    ///
229    /// ## 调度策略
230    ///
231    /// 进程按照加入队列的顺序被调度：
232    ///
233    /// ```text
234    /// 调度顺序 (先进先出):
235    ///
236    /// add(Process1) ──► add(Process2) ──► add(Process3)
237    ///      │             │             │
238    ///      ▼             ▼             ▼
239    /// ┌─────────┐   ┌─────────┐   ┌─────────┐
240    /// │Process1 │◄──│ Process2│◄──│ Process3│
241    /// │ (First) │   │(Second) │   │ (Last)  │
242    /// └─────────┘   └─────────┘   └─────────┘
243    ///      ▲                           ▲
244    ///      │                           │
245    ///   fetch()                     最新添加
246    /// ```
247    ///
248    /// ## 性能特征
249    ///
250    /// - **时间复杂度**: O(1) 摊还常数时间
251    /// - **内存分配**: 队列满时可能触发 O(n) 的扩容操作
252    /// - **扩容策略**: 通常按 2 倍容量扩展
253    ///
254    /// ## 进程状态要求
255    ///
256    /// 添加到就绪队列的进程通常应该满足：
257    /// - 进程状态为 `ProcessStatus::Ready`
258    /// - 拥有有效的内存地址空间
259    /// - 陷阱上下文已正确初始化
260    ///
261    /// ## 使用示例
262    ///
263    /// ```rust
264    /// use alloc::sync::Arc;
265    /// use crate::process::process::ProcessControlBlock;
266    ///
267    /// let process = Arc::new(ProcessControlBlock::new(app_data));
268    /// manager.add(process);
269    ///
270    /// // 进程现在在就绪队列中等待调度
271    /// println!("进程已添加到就绪队列");
272    /// ```
273    ///
274    /// ## 并发考虑
275    ///
276    /// 在并发环境中，应通过适当的同步机制保护此操作：
277    ///
278    /// ```rust
279    /// // 全局接口已提供同步保护
280    /// add_process(process); // 推荐方式
281    ///
282    /// // 直接使用需要手动同步
283    /// PROCESS_MANAGER.exclusive_access().add(process);
284    /// ```
285    /// 向指定优先级队列添加进程
286    ///
287    /// 将进程添加到指定优先级的就绪队列中。如果优先级超出范围，
288    /// 则添加到最低优先级队列。
289    ///
290    /// ## 参数
291    /// * `process` - 要添加的进程控制块
292    /// * `priority` - 目标优先级队列（0为最高优先级）
293    pub fn add(&mut self, process: Arc<ProcessControlBlock>, priority: usize) {
294        let queue_idx = priority.min(self.queue_count - 1);
295        self.ready_queues[queue_idx].push_back(process);
296    }
297
298    /// 向最高优先级队列添加新进程
299    ///
300    /// 新创建的进程默认进入最高优先级队列（队列0），
301    /// 保证新进程的响应性。
302    ///
303    /// ## 参数
304    /// * `process` - 要添加的进程控制块
305    #[allow(unused)]
306    pub fn add_new(&mut self, process: Arc<ProcessControlBlock>) {
307        self.add(process, 0);
308    }
309
310    /// 从就绪队列获取下一个待调度进程
311    ///
312    /// 从队列头部移除并返回最早加入的进程，实现 FIFO 调度策略。
313    /// 如果队列为空，返回 `None`。
314    ///
315    /// ## 返回值
316    ///
317    /// * `Some(Arc<ProcessControlBlock>)` - 成功获取到进程控制块
318    /// * `None` - 就绪队列为空，没有可调度的进程
319    ///
320    /// ## 调度行为
321    ///
322    /// 按照进程加入队列的顺序进行调度：
323    ///
324    /// ```text
325    /// 队列状态变化:
326    ///
327    /// 调用前:
328    /// ┌─────────┐   ┌─────────┐   ┌─────────┐
329    /// │ Process1│◄──│Process2 │◄──│ Process3│
330    /// │(Oldest) │   │         │   │(Latest) │
331    /// └─────────┘   └─────────┘   └─────────┘
332    ///      ▲
333    ///   fetch() 返回 Process1
334    ///
335    /// 调用后:
336    /// ┌────---─────┐   ┌─────────┐
337    /// │  Process2  │◄──│ Process3│
338    /// │(Now Oldest)│   │(Latest) │
339    /// └───────---──┘   └─────────┘
340    /// ```
341    ///
342    /// ## 性能特征
343    ///
344    /// - **时间复杂度**: O(1) 常数时间
345    /// - **内存操作**: 仅移动指针，不涉及数据拷贝
346    /// - **缓存效率**: 访问队列头部具有良好的缓存局部性
347    ///
348    /// ## 空队列处理
349    ///
350    /// 当队列为空时安全返回 `None`，调用者应适当处理：
351    ///
352    /// ```rust
353    /// match manager.fetch() {
354    ///     Some(process) => {
355    ///         println!("获取到进程 PID: {}", process.getpid());
356    ///         // 执行进程调度...
357    ///     }
358    ///     None => {
359    ///         println!("没有就绪进程，CPU 进入空闲状态");
360    ///         // 可能触发节能模式或等待中断
361    ///     }
362    /// }
363    /// ```
364    ///
365    /// ## 进程生命周期
366    ///
367    /// 从队列中取出的进程通常会：
368    /// 1. 状态从 `Ready` 转变为 `Running`
369    /// 2. 被加载到 CPU 执行
370    /// 3. 根据执行结果重新加入队列或结束
371    ///
372    /// ## 调度公平性
373    ///
374    /// FIFO 策略保证了调度的公平性：
375    /// - 所有进程都有相同的调度机会
376    /// - 不会出现进程饥饿问题
377    /// - 调度行为完全可预测
378    ///
379    /// ## 使用示例
380    ///
381    /// ```rust
382    /// // 调度器主循环
383    /// loop {
384    ///     if let Some(process) = manager.fetch() {
385    ///         // 切换到进程执行
386    ///         run_process(process);
387    ///     } else {
388    ///         // 没有进程，等待事件
389    ///         wait_for_interrupt();
390    ///     }
391    /// }
392    /// ```
393    /// 从最高优先级非空队列获取进程
394    ///
395    /// 按优先级顺序遍历所有队列，从第一个非空队列取出进程。
396    /// 实现严格的优先级调度策略。
397    ///
398    /// ## 返回值
399    /// * `Some(Arc<ProcessControlBlock>)` - 成功获取到进程
400    /// * `None` - 所有队列都为空
401    pub fn fetch(&mut self) -> Option<Arc<ProcessControlBlock>> {
402        for queue in &mut self.ready_queues {
403            if let Some(process) = queue.pop_front() {
404                return Some(process);
405            }
406        }
407        None
408    }
409
410    /// 获取指定优先级队列的时间片长度
411    ///
412    /// ## 参数
413    /// * `priority` - 优先级队列索引
414    ///
415    /// ## 返回值
416    /// 该优先级队列的时间片长度（时钟周期数）
417    pub fn get_time_slice(&self, priority: usize) -> usize {
418        let queue_idx = priority.min(self.queue_count - 1);
419        self.time_slices[queue_idx]
420    }
421}

423lazy_static! {
424    /// 全局进程管理器实例
425    ///
426    /// 系统唯一的进程管理器实例，负责维护所有就绪进程的全局状态。
427    /// 使用 `UPSafeCell` 包装以提供在单处理器环境下的线程安全访问。
428    ///
429    /// ## 设计特点
430    ///
431    /// ### 单例模式
432    /// - **全局唯一**: 整个系统只有一个进程管理器实例
433    /// - **集中管理**: 所有进程调度操作都通过此实例进行
434    /// - **状态一致**: 保证系统进程调度状态的一致性
435    ///
436    /// ### 并发安全
437    /// - **UPSafeCell**: 单处理器环境下的内部可变性
438    /// - **互斥访问**: 通过 `exclusive_access()` 获得互斥访问权
439    /// - **无竞争**: 在单核系统中避免复杂的锁机制
440    ///
441    /// ### 延迟初始化
442    /// - **lazy_static**: 在首次访问时才执行初始化
443    /// - **零成本**: 不会在启动时增加领外开销
444    /// - **线程安全**: 初始化过程本身由 lazy_static 保证安全
445    ///
446    /// ## 内存布局
447    ///
448    /// ```text
449    /// 全局内存区域:
450    /// ┌────────────────────────────────────────┐
451    /// │            PROCESS_MANAGER             │
452    /// ├────────────────────────────────────────┤
453    /// │        UPSafeCell<ProcessManager>      │
454    /// │  ┌───────────────────────────-─────-┐  │
455    /// │  │          ProcessManager          │  │
456    /// │  │ ┌──────────────────────────────┐ │  │
457    /// │  │ │      VecDeque<Arc<TCB>>      │ │  │
458    /// │  │ │    [Process1][Process2] ...  │ │  │
459    /// │  │ └──────────────────────────────┘ │  │
460    /// │  └─────────────────────────────-───-┘  │
461    /// └────────────────────────────────────────┘
462    /// ```
463    ///
464    /// ## 访问模式
465    ///
466    /// **直接访问**（不推荐）：
467    /// ```rust
468    /// // 危险！需要手动管理锁
469    /// let mut manager = PROCESS_MANAGER.exclusive_access();
470    /// manager.add(process);
471    /// drop(manager); // 必须显式释放
472    /// ```
473    ///
474    /// **推荐方式**（通过全局函数）：
475    /// ```rust
476    /// // 安全！自动管理锁生命周期
477    /// add_process(process);
478    /// let next_process = fetch_process();
479    /// ```
480    ///
481    /// ## 初始化时机
482    ///
483    /// - **首次访问**: 在第一次调用 `add_process` 或 `fetch_process` 时初始化
484    /// - **延迟加载**: 不会影响系统启动速度
485    /// - **一次性**: 初始化完成后不会重复执行
486    ///
487    /// ## 性能影响
488    ///
489    /// - **全局访问**: 不涉及页表切换，访问速度很快
490    /// - **单点争用**: 所有调度操作都会经过此实例
491    /// - **缓存局部性**: 全局变量在内存中位置相对固定
492    ///
493    /// ## 线程安全保证
494    ///
495    /// - **互斥访问**: `exclusive_access()` 保证同一时间只有一个线程可以修改
496    /// - **内存顺序**: 在单核系统中保证操作的原子性
497    /// - **无死锁**: 简单的所有权模型避免了复杂的锁依赖
498    pub static ref PROCESS_MANAGER: UPSafeCell<ProcessManager> =
499        unsafe { UPSafeCell::new(ProcessManager::new()) };
500
501    pub static ref PID2TCB: UPSafeCell<BTreeMap<usize, Arc<ProcessControlBlock>>> =
502        unsafe { UPSafeCell::new(BTreeMap::new()) };
503}

505/// 向全局就绪队列添加进程
506///
507/// - 将 `process` 写入 `PID2TCB` 映射，便于通过 PID 查找 TCB
508/// - 将进程入队到全局 `PROCESS_MANAGER` 的就绪队列，等待调度
509///
510/// ## 参数
511/// * `process` - 待加入调度的进程控制块
512///
513/// ## 复杂度
514/// - O(1) 摊还时间；队列扩容时可能触发 O(n) 拷贝
515/// 向指定优先级队列添加进程
516///
517/// 将进程添加到指定优先级的就绪队列中，同时更新 PID 映射。
518///
519/// ## 参数
520/// * `process` - 待加入调度的进程控制块
521/// * `priority` - 目标优先级队列（0为最高优先级）
522pub fn add_process_with_priority(process: Arc<ProcessControlBlock>, priority: usize) {
523    PID2TCB
524        .exclusive_access()
525        .insert(process.getpid(), Arc::clone(&process));
526    PROCESS_MANAGER.exclusive_access().add(process, priority);
527}

529/// 向最高优先级队列添加新进程
530///
531/// 新创建的进程默认进入最高优先级队列，保证响应性。
532/// 兼容原有的接口，保持向后兼容性。
533///
534/// ## 参数
535/// * `process` - 待加入调度的进程控制块
536pub fn add_process(process: Arc<ProcessControlBlock>) {
537    add_process_with_priority(process, 0);
538}

540/// 通过 PID 查询进程控制块
541///
542/// ## 参数
543/// * `pid` - 目标进程标识符
544///
545/// ## 返回
546/// - `Some(Arc<ProcessControlBlock>)`：找到对应进程
547/// - `None`：不存在该 PID 的进程（可能已退出并被回收）
548pub fn pid2process(pid: usize) -> Option<Arc<ProcessControlBlock>> {
549    let map = PID2TCB.exclusive_access();
550    map.get(&pid).map(Arc::clone)
551}

553/// 从全局 PID → TCB 映射中移除进程
554///
555/// 典型调用点：
556/// - 进程退出路径（如 `exit_current_and_run_next` / `waitpid` 回收）
557///
558/// ## 参数
559/// * `pid` - 将要移除的进程标识符
560///
561/// ## 行为
562/// - 若不存在该 PID，触发 panic，用于暴露流程一致性问题
563pub fn remove_from_pid2process(pid: usize) {
564    let mut map = PID2TCB.exclusive_access();
565    if map.remove(&pid).is_none() {
566        panic!("cannot find pid {} in pid2process!", pid);
567    }
568}

570/// 从全局进程管理器获取下一个待调度进程
571///
572/// 调度器核心函数，从就绪队列中获取最早加入的进程进行执行。
573/// 实现 FIFO 调度策略，确保进程调度的公平性。
574///
575/// ## 返回值
576///
577/// * `Some(Arc<ProcessControlBlock>)` - 成功获取到一个就绪进程
578/// * `None` - 就绪队列为空，没有可调度的进程
579///
580/// ## 调度策略
581///
582/// 采用先进先出 (FIFO) 策略：
583///
584/// ```text
585/// 调度顺序示意图:
586///
587/// 时间轴: T1    T2    T3    T4
588///         │     │     │     │
589///         ▼     ▼     ▼     ▼
590///      add(A) add(B) add(C) add(D)
591///         │     │     │     │
592///         ▼     ▼     ▼     ▼
593/// 队列: [A] ─► [B|A] [C|B|A] [D|C|B|A]
594///         │           │         │
595///         ▼           ▼         ▼
596///     fetch()    fetch()   fetch()
597///      返回A       返回B     返回C
598/// ```
599///
600/// ## 使用场景
601///
602/// ### 主调度循环
603/// ```rust
604/// use crate::process::manager::fetch_process;
605///
606/// loop {
607///     match fetch_process() {
608///         Some(process) => {
609///             println!("调度进程 PID: {}", process.getpid());
610///             run_process(process);
611///         }
612///         None => {
613///             println!("CPU 进入空闲状态");
614///             wait_for_interrupt();
615///         }
616///     }
617/// }
618/// ```
619///
620/// ### 批量处理
621/// ```rust
622/// // 一次取出多个进程进行批量处理
623/// let mut processs = Vec::new();
624/// while let Some(process) = fetch_process() {
625///     processs.push(process);
626///     if processs.len() >= BATCH_SIZE {
627///         break;
628///     }
629/// }
630/// process_processs_batch(processs);
631/// ```
632///
633/// ### 空闲检测
634/// ```rust
635/// if fetch_process().is_none() {
636///     println!("系统无进程，可以进入节能模式");
637///     enter_power_save_mode();
638/// }
639/// ```
640///
641/// ## 执行流程
642///
643/// ```text
644/// fetch_process 调用流程:
645///
646/// Scheduler Code
647///      │
648///      ▼
649/// ┌─────────────────┐
650/// │ fetch_process() │ ─── 全局函数接口
651/// └────────┬────────┘
652///          │
653///          ▼
654/// ┌─────────────────-┐
655/// │exclusive_access()│ ─── 获取互斥访问权
656/// └────────┬────────-┘
657///          │
658///          ▼
659/// ┌─────────────────┐
660/// │ manager.fetch() │ ─── 从队列头部取出
661/// └────────┬────────┘
662///          │
663///          ▼
664/// ┌─────────────────┐
665/// │ automatic unlock│ ─── 返回后自动释放
666/// └─────────────────┘
667///          │
668///          ▼
669///   Option<Arc<TCB>>
670/// ```
671///
672/// ## 性能特征
673///
674/// - **时间复杂度**: O(1) 常数时间操作
675/// - **内存开销**: 仅转移 Arc 所有权，不复制数据
676/// - **锁竞争**: 在单核系统中几乎无开销
677/// - **缓存友好**: 访问顺序性数据结构
678///
679/// ## 空队列处理策略
680///
681/// 当没有可调度进程时的常见处理方式：
682///
683/// ```rust
684/// match fetch_process() {
685///     Some(process) => {
686///         // 正常调度流程
687///         switch_to_process(process);
688///     }
689///     None => {
690///         // 选择合适的空闲处理策略
691///         idle_process();           // 执行空闲进程
692///         // halt_cpu();         // 挂起 CPU 等待中断
693///         // power_management();  // 进入节能模式
694///     }
695/// }
696/// ```
697///
698/// ## 调度公平性
699///
700/// FIFO 策略保证了：
701/// - **无饥饿**: 没有进程会被无限期推迟
702/// - **公平性**: 所有进程都有相等的被调度机会
703/// - **可预测**: 调度顺序完全可预测
704///
705/// ## 错误处理
706///
707/// 此函数不会返回错误，但可能在以下情况下 panic：
708/// - 全局管理器未正确初始化
709/// - 队列数据结构损坏（极罕见）
710///
711/// ## 线程安全
712///
713/// - **自动锁管理**: 函数进入时自动加锁，返回时自动解锁
714/// - **互斥性**: 与 `add_process` 互斥，不会产生竞争条件
715/// - **异常安全**: 在 panic 发生时也能正确释放锁
716///
717/// ## 性能优化建议
718///
719/// 对于高频调用场景，考虑批量获取优化：
720///
721/// ```rust
722/// // 简单但低效的方式
723/// for _ in 0..100 {
724///     if let Some(process) = fetch_process() {
725///         process_process(process);
726///     }
727/// }
728///
729/// // 更高效的批量处理
730/// let processs = {
731///     let mut manager = PROCESS_MANAGER.exclusive_access();
732///     let mut batch = Vec::new();
733///     while let Some(process) = manager.fetch() {
734///         batch.push(process);
735///         if batch.len() >= 100 { break; }
736///     }
737///     batch
738/// };
739/// for process in processs {
740///     process_process(process);
741/// }
742/// ```
743pub fn fetch_process() -> Option<Arc<ProcessControlBlock>> {
744    PROCESS_MANAGER.exclusive_access().fetch()
745}

747/// 获取指定优先级队列的时间片长度
748///
749/// ## 参数
750/// * `priority` - 优先级队列索引（0为最高优先级）
751///
752/// ## 返回值
753/// 该优先级队列的时间片长度（时钟周期数）
754pub fn get_time_slice(priority: usize) -> usize {
755    PROCESS_MANAGER.exclusive_access().get_time_slice(priority)
756}

758/// 提升进程优先级（用于 I/O 操作后）
759///
760/// 当进程完成 I/O 操作或从阻塞状态恢复时，将其提升到最高优先级队列，
761/// 实现 MLFQ 的 I/O 优化策略。
762///
763/// ## 参数
764/// * `process` - 要提升优先级的进程控制块
765#[allow(unused)]
766pub fn boost_process_priority(process: Arc<ProcessControlBlock>) {
767    // 重置为最高优先级
768    {
769        let mut inner = process.inner_exclusive_access();
770        inner.priority = 0;
771        inner.time_slice_used = 0;
772        inner.time_slice_limit = get_time_slice(0);
773    }
774    // 重新加入最高优先级队列
775    add_process_with_priority(process, 0);
776}
