1//! # 进程控制块模块
2//!
3//! 提供进程控制块 (Process Control Block, TCB) 的实现，管理进程的完整生命周期。
4//! 支持进程创建、fork、exec、状态管理等核心操作，是操作系统进程管理的核心数据结构。
5//!
6//! ## 核心组件
7//!
8//! - [`ProcessControlBlock`] - 进程控制块，包含进程的所有信息
9//! - [`ProcessControlBlockInner`] - TCB 内部可变部分，受互斥锁保护
10//! - [`ProcessStatus`] - 进程状态枚举，表示进程的运行状态
11//!
12//! ## 标准文件描述符
13//!
14//! 每个进程创建时自动分配三个标准文件描述符：
15//! - **标准输入 (stdin, fd=0)**: 程序的标准输入源，实现为 [`Stdin`]
16//! - **标准输出 (stdout, fd=1)**: 程序的正常输出目标，实现为 [`Stdout`]
17//! - **标准错误 (stderr, fd=2)**: 程序的错误输出目标，实现为 [`Stderr`]
18//!
19//! 这些标准文件描述符遵循POSIX标准，提供基本的输入输出功能。
20//! 在fork操作中，子进程会继承父进程的所有文件描述符。
21//!
22//! ## 设计原理
23//!
24//! ### 分离设计
25//! TCB 采用内外分离的设计模式：
26//! - **不变部分**：PID 和内核栈在进程生命周期内保持不变
27//! - **可变部分**：进程状态、内存集合等需要互斥保护的数据
28//!
29//! ### 进程层次结构
30//! 支持完整的进程树结构：
31//! - **父子关系**：通过 `parent` 和 `children` 字段维护
32//! - **引用计数**：使用 `Arc` 和 `Weak` 防止循环引用
33//! - **孤儿进程处理**：父进程退出时子进程重新指向 init 进程
34//!
35//! ## 内存布局
36//!
37//! 每个进程的内存空间布局：
38//!
39//! ```text
40//! 高地址 TRAMPOLINE (0x3ffffff000)
41//! ┌──────────────────────────────────────────────────────┐
42//! │                   Trampoline                         │
43//! │                    (R+X)                             │
44//! ├──────────────────────────────────────────────────────┤
45//! │                 Trap Context                         │
46//! │                    (R+W)                             │
47//! ├──────────────────────────────────────────────────────┤
48//! │                  User Stack                          │
49//! │                   (R+W+U)                            │
50//! ├──────────────────────────────────────────────────────┤
51//! │               Program Sections                       │
52//! │            (.text/.data/.bss etc)                    │
53//! │              (Based on ELF flags)                    │
54//! └──────────────────────────────────────────────────────┘
55//! 低地址 (0x10000)
56//!
57//! 内核空间 - 每个进程独立的内核栈：
58//! ┌──────────────────────────────────────────────────────┐
59//! │             Process N Kernel Stack                   │
60//! ├──────────────────────────────────────────────────────┤
61//! │                  Guard Page                          │
62//! ├──────────────────────────────────────────────────────┤
63//! │           Process N-1 Kernel Stack                   │
64//! └──────────────────────────────────────────────────────┘
65//! ```
66//!
67//! ## 进程状态转换
68//!
69//! ```text
70//!           fork/exec
71//!         ┌──────────┐
72//!         │          │
73//!         ▼          │
74//!     ┌───────┐      │      Scheduler Selection
75//!     │ Ready │◄─────┘ ◄─────────────┐
76//!     └───┬───┘                      │
77//!         │                          │
78//!         │ Scheduler Selection      │ Timeslice Expire/Yield
79//!         │                          │
80//!         ▼                          │
81//!    ┌─────────┐                     │
82//!    │ Running │───────────────────-─┘
83//!    └────┬────┘
84//!         │
85//!         │ exit() System Call
86//!         │
87//!         ▼
88//!     ┌───────┐
89//!     │Zombie │ ◄── Wait for Parent wait()
90//!     └───────┘
91//! ```
92//!
93//! ## 并发安全
94//!
95//! - **互斥保护**：可变部分使用 `UPSafeCell` 保证线程安全
96//! - **原子操作**：PID 分配和内核栈管理通过全局锁保护
97//! - **引用计数**：`Arc` 确保进程对象的生命周期管理
98//!
99//! ## 使用示例
100//!
101//! ```rust
102//! use alloc::sync::Arc;
103//!
104//! // 创建新进程
105//! let elf_data = app_data(0);
106//! let process = Arc::new(ProcessControlBlock::new(elf_data));
107//!
108//! // 检查标准文件描述符
109//! {
110//!     let inner = process.inner_exclusive_access();
111//!     // 验证标准文件描述符已正确初始化
112//!     assert!(inner.fd_table[0].is_some()); // stdin
113//!     assert!(inner.fd_table[1].is_some()); // stdout  
114//!     assert!(inner.fd_table[2].is_some()); // stderr
115//! }
116//!
117//! // Fork 子进程
118//! let child_process = process.fork();
119//!
120//! // 检查子进程继承了父进程的文件描述符
121//! {
122//!     let child_inner = child_process.inner_exclusive_access();
123//!     assert_eq!(child_inner.fd_table.len(), process.inner_exclusive_access().fd_table.len());
124//! }
125//!
126//! // 执行新程序
127//! process.exec(new_elf_data);
128//!
129//! // 获取进程状态
130//! let inner = process.inner_exclusive_access();
131//! println!("Process status: {:?}", inner.process_status);
132//! ```

150/// 进程控制块 (Process Control Block)
151///
152/// 操作系统中每个进程的核心数据结构，包含进程运行所需的全部信息。
153/// TCB 采用内外分离设计，将不变和可变部分分开管理，提高并发访问效率。
154///
155/// ## 结构组成
156///
157/// ### 不变部分（直接字段）
158/// - `pid`: 进程标识符句柄，进程生命周期内唯一且不变
159/// - `kernel_stack`: 内核栈，用于系统调用和中断处理
160///
161/// ### 可变部分（受保护字段）
162/// - `inner`: 包含所有可变状态，使用 [`UPSafeCell`] 进行互斥保护
163///
164/// ## 设计优势
165///
166/// **性能优化**：
167/// - 不变字段可以直接访问，避免锁开销
168/// - 可变字段集中保护，减少锁的粒度和争用
169///
170/// **并发安全**：
171/// - [`UPSafeCell`] 提供单处理器环境下的安全可变访问
172/// - 通过 [`Arc`] 支持多处理器环境下的引用计数管理
173/// - 通过 [`Weak`] 避免循环引用，防止内存泄漏
174///
175/// **资源管理**：
176/// - PID 和内核栈通过 RAII 自动管理生命周期
177/// - 内存集合在进程退出时自动清理
178///
179/// ## 生命周期
180///
181/// ```text
182/// 创建阶段:
183/// ┌────────────-─┐    ┌─-────────────--┐    ┌────────────┐
184/// │ Allocate PID │───►│ Create K-Stack │───►│ Parse ELF  │
185/// └──────────-───┘    └-────────────-─-┘    └────────────┘
186///                                                 │
187///                                                 ▼
188/// ┌────────────┐    ┌─-────────-────┐      ┌──────────--───┐
189/// │ Init State │◄───│ Setup AddrSpc │  ◄─  │ Setup TrapCtx │
190/// └-───────────┘    └-─────────────-┘      └────────────--─┘
191///
192/// 运行阶段：
193/// Ready ──Schedule──► Running ──Timeslice/Yield──► Ready
194///                       │
195///                       │ exit()
196///                       ▼
197///                    Zombie ──wait()──► Destroy
198///
199/// 销毁阶段:
200/// ┌─────────────┐    ┌───────────-──┐    ┌──────────┐
201/// │ Free Memory │───►│ Free K-Stack │───►│ Free PID │
202/// └─────────────┘    └─────────────-┘    └──────────┘
203/// ```
204///
205/// ## 使用模式
206///
207/// TCB 通常包装在 [`Arc`] 中使用，支持多所有者场景：
208/// - 进程管理器持有引用进行调度
209/// - 父进程持有子进程引用进行管理
210/// - 处理器持有当前运行进程的引用
211///
212/// ## Examples
213///
214/// ```rust
215/// use alloc::sync::Arc;
216///
217/// // 创建新进程
218/// let elf_data = include_bytes!("user_program.elf");
219/// let process = Arc::new(ProcessControlBlock::new(elf_data));
220///
221/// // 访问不变字段（无需锁）
222/// println!("Process PID: {}", process.getpid());
223///
224/// // 访问可变字段（需要获取锁）
225/// {
226///     let inner = process.inner_exclusive_access();
227///     println!("Process status: {:?}", inner.process_status);
228/// } // 锁在此处自动释放
229///
230/// // Fork 创建子进程
231/// let child_process = process.fork();
232/// ```
233pub struct ProcessControlBlock {
234    /// 进程标识符句柄
235    ///
236    /// 包含系统唯一的进程 ID，通过 RAII 机制自动管理 PID 的分配和回收。
237    /// 在进程的整个生命周期中保持不变，可以安全地并发访问。
238    pub pid: PidHandle,
239
240    /// 内核栈
241    ///
242    /// 每个进程在内核空间中的独立栈空间，用于：
243    /// - 系统调用处理时的临时数据存储
244    /// - 中断和异常处理时的上下文保存
245    /// - 进程切换时的寄存器保存
246    ///
247    /// 通过 RAII 机制自动管理内核栈的分配、映射和回收。
248    pub kernel_stack: KernelStack,
249
250    /// 内部可变状态
251    ///
252    /// 包含所有需要在运行时修改的进程状态信息，使用 [`UPSafeCell`]
253    /// 提供线程安全的可变访问。包括进程状态、内存集合、上下文等。
254    inner: UPSafeCell<ProcessControlBlockInner>,
255}
256
257/// 进程控制块内部可变状态
258///
259/// 包含进程的所有可变状态信息，需要在运行时进行修改的字段都集中在此结构中。
260/// 通过 [`UPSafeCell`] 进行互斥保护，确保并发访问的安全性。
261///
262/// ## 字段说明
263///
264/// ### 运行时状态
265/// - `process_status`: 进程当前状态（Ready/Running/Zombie）
266/// - `process_cx`: 进程上下文，保存寄存器状态用于进程切换
267/// - `exit_code`: 进程退出码，用于父进程获取子进程执行结果
268///
269/// ### 内存管理
270/// - `memory_set`: 进程的完整地址空间，包含所有内存映射区域
271/// - `trap_cx_ppn`: 陷阱上下文的物理页号，用于用户态/内核态切换
272/// - `base_size`: 进程初始堆栈大小，用于内存分配决策
273///
274/// ### 进程关系
275/// - `parent`: 父进程的弱引用，避免循环引用导致内存泄漏
276/// - `children`: 子进程列表，维护进程树结构
277///
278/// ## 设计考虑
279///
280/// **并发安全**：
281/// - 所有字段都受到外层 [`UPSafeCell`] 的保护
282/// - 访问时需要获取独占锁，避免数据竞争
283/// - 锁的持有时间应当尽可能短，避免性能影响
284///
285/// **内存管理**：
286/// - [`MemorySet`] 自动管理进程地址空间的生命周期
287/// - 使用 [`Arc`] 和 [`Weak`] 管理进程间的引用关系
288/// - 进程退出时自动清理相关资源
289///
290/// **状态一致性**：
291/// - 进程状态与上下文信息保持同步
292/// - 父子关系的双向引用保持一致
293/// - 内存映射与陷阱上下文匹配
294pub struct ProcessControlBlockInner {
295    /// 进程当前状态
296    ///
297    /// 表示进程在操作系统中的当前状态，影响调度器的调度决策。
298    /// 状态转换遵循严格的状态机规则。
299    pub process_status: ProcessStatus,

305    /// - 栈指针 (sp): 内核栈的栈顶地址
306    /// - 被调用者保存寄存器 (s0-s11): 函数调用约定要求保存的寄存器
307    pub process_cx: ProcessContext,
308
309    /// 进程地址空间
310    ///
311    /// 管理进程的完整虚拟地址空间，包括：
312    /// - 代码段、数据段、堆段、栈段的映射
313    /// - 页表管理和地址转换
314    /// - 内存权限控制和保护
315    pub memory_set: MemorySet,
316
317    /// 陷阱上下文物理页号
318    ///
319    /// 指向保存陷阱上下文的物理页面，陷阱上下文包含：
320    /// - 用户态所有寄存器的值
321    /// - 系统调用参数和返回值
322    /// - 异常处理相关信息
323    pub trap_cx_ppn: PhysPageNum,
324
325    /// 进程基础内存大小
326    ///
327    /// 记录进程初始化时的内存使用情况，用于：
328    /// - 内存分配和回收的决策参考
329    /// - 进程资源使用统计
330    /// - 堆空间管理的基准值

333    /// 父进程引用

341    ///
342    /// 维护当前进程的所有子进程，用于：
343    /// - 实现 wait 系统调用，等待子进程退出
344    /// - 进程退出时处理孤儿进程，重新指向 init 进程
345    /// - 信号传递和进程组管理
346    pub children: Vec<Arc<ProcessControlBlock>>,

349    ///
350    /// 记录进程的退出状态，供父进程通过 wait 系统调用获取。

352    pub exit_code: i32,

354    /// 文件描述符表
355    ///
356    /// 维护进程打开的所有文件描述符，每个元素对应一个文件描述符：
357    /// - `Some(file)` - 文件描述符已打开，指向对应的文件对象
358    /// - `None` - 文件描述符未使用，可以被重新分配
359    ///
360    /// ## 标准文件描述符
361    ///
362    /// 进程创建时自动分配以下标准文件描述符：
363    /// - `fd_table[0]` - 标准输入 (stdin) - [`Stdin`]
364    /// - `fd_table[1]` - 标准输出 (stdout) - [`Stdout`]
365    /// - `fd_table[2]` - 标准错误 (stderr) - [`Stderr`]
366    ///
367    /// 这些标准文件描述符在进程创建时自动初始化，用户程序无需手动打开。
368    /// 它们提供了基本的输入输出功能，遵循POSIX标准。
369    ///
370    /// ## 文件描述符分配
371    ///
372    /// - 新文件描述符从索引 0 开始查找第一个 `None` 位置
373    /// - 如果所有位置都被占用，则扩展表大小
374    /// - 文件描述符关闭时设置为 `None`，可以被重用
375    ///
376    /// ## 并发安全
377    ///
378    /// 文件描述符表通过外层的 [`UPSafeCell`] 保护，确保并发访问安全。
379    /// 文件对象本身通过 [`Arc`] 实现引用计数，支持多进程共享。
380    ///
381    /// ## 使用场景
382    ///
383    /// - 系统调用中查找和验证文件描述符
384    /// - 进程 fork 时复制文件描述符表
385    /// - 进程退出时关闭所有打开的文件
386    /// - 文件描述符的分配和回收管理
387    ///
388    /// ## 标准文件描述符的语义
389    ///
390    /// ### 标准输入 (stdin, fd=0)
391    /// - **用途**: 程序的标准输入源
392    /// - **实现**: [`Stdin`] 结构体
393    /// - **特性**: 只读，阻塞式读取，支持单字符输入
394    /// - **行为**: 当没有输入时会让出CPU，等待用户输入
395    ///
396    /// ### 标准输出 (stdout, fd=1)
397    /// - **用途**: 程序的正常输出目标
398    /// - **实现**: [`Stdout`] 结构体
399    /// - **特性**: 只写，实时输出，支持UTF-8编码
400    /// - **行为**: 立即将输出显示到控制台
401    ///
402    /// ### 标准错误 (stderr, fd=2)
403    /// - **用途**: 程序的错误信息和诊断输出
404    /// - **实现**: [`Stderr`] 结构体
405    /// - **特性**: 只写，实时输出，支持UTF-8编码
406    /// - **行为**: 与标准输出类似，但语义上区分用途
407    ///
408    /// ## 文件描述符继承
409    ///
410    /// 在fork操作中，子进程会继承父进程的所有文件描述符：
411    /// - 标准文件描述符 (0, 1, 2) 被复制到子进程
412    /// - 用户打开的文件描述符也被复制
413    /// - 父子进程共享相同的文件对象引用
414    ///
415    /// ## 错误处理
416    ///
417    /// 文件描述符操作可能出现的错误：

420    /// - **EFAULT**: 用户缓冲区地址无效

428    /// assert!(inner.fd_table[1].is_some()); // stdout
429    /// assert!(inner.fd_table[2].is_some()); // stderr

434    /// ```

462    pub time_slice_limit: usize,

465/// 进程状态枚举
466///
467/// 定义进程在操作系统中的各种运行状态，遵循经典的进程状态模型。
468/// 状态转换受到严格的状态机规则约束，确保系统的一致性和可预测性。
469///
470/// ## 状态转换图
471///
472/// ```text
473///                Process Creation
474///                      │
475///                      ▼
476///               ┌─────────────┐
477///               │    Ready    │ ◄─────────────────┐
478///               │  (Ready)    │                   │
479///               └──────┬──────┘                   │
480///                      │                          │
481///                      │ Scheduler Selection      │
482///                      ▼                          │
483///               ┌─────────────┐                   │
484///               │   Running   │                   │
485///               │  (Running)  │                   │
486///               └──────┬──────┘                   │
487///                      │                          │
488///            ┌─────────┴─────────┐                │
489///            │                   │                │
490///            ▼                   ▼                │
491///   Timeslice Expire/Yield   exit() System Call   │
492///            │                   │                │
493///            └───────────────────┼───────────────-┘
494///                                ▼
495///                        ┌─────────────┐
496///                        │   Zombie    │
497///                        │  (Zombie)   │
498///                        └─────────────┘
499///                                │
500///                                │ Parent wait()
501///                                ▼
502///                        Process Destruction
503/// ```
504///
505/// ## 状态详细说明
506///
507/// ### Ready (就绪)
508/// - **含义**: 进程已准备好运行，等待 CPU 分配
509/// - **特征**: 所有运行条件已满足，仅等待调度器调度
510/// - **转入**: 进程创建完成、时间片用完、系统调用返回
511/// - **转出**: 被调度器选中执行
512///
513/// ### Running (运行)
514/// - **含义**: 进程正在 CPU 上执行
515/// - **特征**: 拥有 CPU 控制权，正在执行指令
516/// - **转入**: 从就绪队列被调度器选中
517/// - **转出**: 时间片用完、主动让出 CPU、进程退出
518///
519/// ### Zombie (僵尸)
520/// - **含义**: 进程已执行完毕，等待父进程收集退出信息
521/// - **特征**: 保留进程控制块，但不再调度执行
522/// - **转入**: 进程调用 exit 系统调用
523/// - **转出**: 父进程调用 wait 系统调用回收
524///
525/// ## 使用示例
526///
527/// ```rust
528/// // 创建进程后的状态检查
529/// let process = ProcessControlBlock::new(elf_data);
530/// {
531///     let inner = process.inner_exclusive_access();
532///     assert_eq!(inner.process_status, ProcessStatus::Ready);
533/// }
534///
535/// // 状态转换示例
536/// match current_status {
537///     ProcessStatus::Ready => {
538///         // 可以被调度器选中
539///         scheduler.add_to_running_queue(process);
540///     }
541///     ProcessStatus::Running => {
542///         // 正在执行，可能需要时间片管理
543///         if time_slice_expired() {
544///             process.suspend_and_yield();
545///         }
546///     }
547///     ProcessStatus::Zombie => {
548///         // 等待父进程回收
549///         if parent.is_waiting() {
550///             parent.collect_child_exit_code(process.exit_code);
551///         }
552///     }
553/// }
554/// ```
555///
556/// ## 状态约束
557///
558/// - **互斥性**: 同一时刻进程只能处于一种状态
559/// - **有序性**: 状态转换必须遵循预定义的转换路径
560/// - **原子性**: 状态转换是原子操作，不存在中间状态
561/// - **一致性**: 状态转换与系统其他部分的状态保持一致
562#[derive(Copy, Clone, PartialEq, Debug)]
563pub enum ProcessStatus {
564    /// 就绪状态
565    ///
566    /// 进程已准备好执行，所有资源已分配，等待调度器分配 CPU 时间。
567    /// 处于此状态的进程位于就绪队列中，按调度策略等待执行机会。
568    Ready,
569
570    /// 运行状态
571    ///
572    /// 进程正在 CPU 上执行指令，拥有处理器控制权。在单处理器系统中，
573    /// 同一时刻只有一个进程可以处于运行状态。
574    Running,
575
576    /// 僵尸状态
577    ///
578    /// 进程已执行完毕并退出，但进程控制块仍然保留，等待父进程
579    /// 通过 wait 系统调用收集其退出状态信息。
580    Zombie,
581}

677    /// 分配新的文件描述符
678    ///
679    /// 在文件描述符表中查找第一个可用的位置，并返回对应的文件描述符编号。
680    /// 如果所有现有位置都被占用，则扩展表大小并返回新的文件描述符。
681    ///
682    /// ## Returns
683    ///
684    /// 返回新分配的文件描述符编号（非负整数）
685    ///
686    /// ## 分配策略
687    ///
688    /// 1. **查找策略**: 从索引 0 开始查找第一个 `None` 位置
689    /// 2. **扩展策略**: 如果所有位置都被占用，则向表末尾添加新的 `None` 条目
690    /// 3. **重用策略**: 优先重用已关闭的文件描述符，减少表大小增长
691    ///
692    /// ## 分配过程
693    ///
694    /// ```text
695    /// 文件描述符表状态示例:
696    /// ┌-─────-┬──-───--┬──--───-┬-─────-┬─────┐
697    /// │  fd0  │  fd1   │   fd2  │  fd3  │ fd4 │
698    /// ├──-───-┼─-────--┼──-──-─-┼──-───-┼─────┤
699    /// │ stdin │ stdout │ stderr │ file1 │ None│ ← 返回 fd4
700    /// └─-────-┴──-───--┴─-─--───┴─-────-┴─────┘
701    ///
702    /// 关闭 fd1 后:
703    /// ┌-─────-┬──-───--┬──--───-┬-─────-┬─────┐
704    /// │  fd0  │  fd1   │   fd2  │  fd3  │ fd4 │
705    /// ├──-───-┼─-────--┼──-──-─-┼──-───-┼─────┤
706    /// │ stdin │ stdout │ stderr │ file1 │ None│ ← 返回 fd1
707    /// └─-────-┴──-───--┴─-─--───┴─-────-┴─────┘
708    /// ```
709    ///
710    /// ## 使用场景
711    ///
712    /// - **open 系统调用**: 打开新文件时分配文件描述符
713    /// - **dup 系统调用**: 复制文件描述符时分配新的编号
714    /// - **pipe 系统调用**: 创建管道时分配读写端文件描述符
715    /// - **socket 系统调用**: 创建套接字时分配文件描述符
716    ///
717    /// ## 并发安全
718    ///
719    /// 此方法需要可变访问权限，调用者必须持有 [`UPSafeCell`] 的独占锁。
720    /// 分配过程是原子的，不会与其他进程的文件描述符分配产生冲突。
721    ///
722    /// ## Examples
723    ///
724    /// ```rust
725    /// // 在 open 系统调用中使用
726    /// let mut inner = process.inner_exclusive_access();
727    /// let new_fd = inner.alloc_fd();
728    /// inner.fd_table[new_fd] = Some(file_object);
729    /// return new_fd as isize;
730    ///
731    /// // 检查分配结果
732    /// println!("Allocated file descriptor: {}", new_fd);
733    /// assert!(new_fd < inner.fd_table.len());
734    /// ```
735    ///
736    /// ## 性能特性
737    ///
738    /// - **时间复杂度**: O(n)，其中 n 是文件描述符表的当前大小
739    /// - **空间复杂度**: 最坏情况下需要扩展表大小
740    /// - **内存效率**: 优先重用已关闭的描述符，减少内存浪费
741    pub fn alloc_fd(&mut self) -> usize {
742        if let Some(fd) = (0..self.fd_table.len()).find(|fd| self.fd_table[*fd].is_none()) {
743            fd
744        } else {
745            self.fd_table.push(None);
746            self.fd_table.len() - 1
747        }
748    }

753    ///

764    ///
765    /// ## 初始化过程
766    ///
767    /// ```text

787    ///    ▼

789    /// ```

792    ///
793    /// 创建后的进程具有以下内存布局：
794    ///
795    /// ```text
796    /// 用户地址空间:
797    /// 高地址 TRAMPOLINE
798    /// ┌────────────────────┐
799    /// │    Trampoline      │ ← 用户态/内核态切换代码
800    /// ├────────────────────┤
801    /// │   Trap Context     │ ← 用户态寄存器状态
802    /// ├────────────────────┤
803    /// │    User Stack      │ ← 用户态栈空间
804    /// ├────────────────────┤
805    /// │    Guard Page      │ ← 防止栈溢出
806    /// ├────────────────────┤
807    /// │  Program Sections  │ ← .text/.data/.bss
808    /// └────────────────────┘
809    /// 低地址 0x10000
810    ///
811    /// 内核地址空间:
812    /// ┌────────────────────┐
813    /// │   Kernel Stack     │ ← 系统调用处理时使用

832    ///

871        let kernel_stack_top = kernel_stack.top();
872        let process_control_block = Self {
873            pid: pid_handle,
874            kernel_stack,
875            inner: unsafe {
876                UPSafeCell::new(ProcessControlBlockInner {
877                    process_status: ProcessStatus::Ready,
878                    process_cx: ProcessContext::goto_trap_return(kernel_stack_top),
879                    memory_set,
880                    trap_cx_ppn,
881                    base_size: user_sp,
882                    parent: None,
883                    children: Vec::new(),
884                    exit_code: 0,
885                    fd_table: vec![
886                        Some(Arc::new(Stdin)),
887                        Some(Arc::new(Stdout)),
888                        Some(Arc::new(Stderr)),
889                    ],
890                    signals: SignalFlags::empty(),
891                    signal_mask: SignalFlags::empty(),
892                    handling_sig: -1,
893                    signal_actions: SignalActions::default(),
894                    killed: false,
895                    frozen: false,
896                    trap_ctx_backup: None,
897                    priority: 0, // 新进程从最高优先级开始
898                    time_slice_used: 0,
899                    time_slice_limit: {
900                        use crate::config::MLFQ_BASE_TIME_SLICE;
901                        MLFQ_BASE_TIME_SLICE
902                    },
903                })
904            },
905        };
906        let trap_cx = process_control_block.inner_exclusive_access().trap_cx();

908            entry_point,
909            user_sp,
910            KERNEL_SPACE.exclusive_access().token(),
911            kernel_stack_top,
912            trap_handler as usize,
913        );
914        process_control_block
915    }

1006    /// - **不变性**: PID 在进程生命周期内不变
1007    /// - **递增性**: PID 按创建顺序递增分配（在回收重用前）

1015    /// - **资源管理**: 进程资源统计和限制
1016    /// - **安全检查**: 权限验证和访问控制

1018    /// ## 与 POSIX 兼容性
1019    ///
1020    /// 返回的 PID 值遵循 POSIX 约定：

1071    /// Fork 系统调用实现：创建子进程
1072    ///
1073    /// 复制当前进程创建一个新的子进程，子进程继承父进程的地址空间、文件描述符等资源。
1074    /// 这是 UNIX/Linux 系统中创建新进程的主要方式，遵循经典的 fork 语义。
1075    ///
1076    /// ## Arguments
1077    ///
1078    /// * `self` - 父进程的 Arc 引用，用于建立父子关系
1079    ///
1080    /// ## Returns
1081    ///
1082    /// 返回新创建的子进程的 [`Arc<ProcessControlBlock>`]
1083    ///
1084    /// ## Fork 语义
1085    ///
1086    /// **父子进程差异**：
1087    /// - **PID**: 子进程分配新的 PID
1088    /// - **内核栈**: 子进程分配独立的内核栈
1089    /// - **返回值**: 在父进程中返回子进程 PID，在子进程中返回 0
1090    ///
1091    /// **共享与复制**：
1092    /// - **地址空间**: 子进程获得父进程地址空间的完整副本
1093    /// - **寄存器状态**: 子进程继承父进程当前的所有寄存器值
1094    /// - **文件描述符**: 子进程继承父进程打开的文件描述符
1095    /// - **工作目录**: 子进程继承父进程的工作目录
1096    ///
1097    /// ## Fork 过程详解
1098    ///
1099    /// ```text
1100    /// 1. 复制地址空间
1101    ///    │
1102    ///    │ - 创建新的内存集合
1103    ///    │ - 逐页复制父进程的所有内存内容
1104    ///    │ - 建立独立的页表结构
1105    ///    ▼
1106    /// 2. 分配系统资源
1107    ///    │
1108    ///    │ - 分配新的 PID
1109    ///    │ - 创建独立的内核栈
1110    ///    │ - 分配新的 TCB
1111    ///    ▼
1112    /// 3. 建立父子关系
1113    ///    │
1114    ///    │ - 子进程记录父进程弱引用
1115    ///    │ - 父进程添加子进程到 children 列表
1116    ///    │ - 维护进程树结构
1117    ///    ▼
1118    /// 4. 初始化子进程状态
1119    ///    │
1120    ///    │ - 设置进程状态为 Ready
1121    ///    │ - 复制进程上下文
1122    ///    │ - 调整内核栈指针
1123    ///    ▼
1124    /// 5. 子进程创建完成
1125    /// ```
1126    ///
1127    /// ## 内存布局对比
1128    ///
1129    /// ```text
1130    /// Fork 前 (父进程):
1131    /// ┌─────────────────┐
1132    /// │   Parent Process   │
1133    /// │     (PID n)     │
1134    /// ├─────────────────┤
1135    /// │  Address Space  │ ← 原始地址空间
1136    /// │    Virtual      │
1137    /// │     Memory      │
1138    /// └─────────────────┘
1139    ///
1140    /// Fork 后:
1141    /// ┌─────────────────┐    ┌─────────────────┐
1142    /// │   Parent Process   │    │   Child Process    │
1143    /// │     (PID n)     │    │    (PID n+1)    │
1144    /// ├─────────────────┤    ├─────────────────┤
1145    /// │  Address Space  │    │  Address Space  │ ← 完整副本
1146    /// │    (Original)   │    │     (Copy)      │
1147    /// │     Memory      │    │     Memory      │
1148    /// └─────────────────┘    └─────────────────┘
1149    ///            │                      ▲
1150    ///            └──────────────────────┘
1151    ///                 父子关系
1152    /// ```
1153    ///
1154    /// ## 系统调用返回值处理
1155    ///
1156    /// Fork 的一个重要特征是一次调用两次返回：
1157    ///
1158    /// ```text
1159    /// 调用时机：                父进程返回          子进程返回
1160    /// ┌──────────┐             ┌──────────┐        ┌──────────┐
1161    /// │ fork()   │────────────►│child_pid │        │    0     │
1162    /// │ SysCall  │             │          │        │          │
1163    /// └──────────┘             └──────────┘        └──────────┘
1164    /// ```
1165    ///
1166    /// 返回值设置在陷阱上下文中的 `x[10]` 寄存器（RISC-V ABI 返回值寄存器）。
1167    ///
1168    /// ## 使用示例
1169    ///
1170    /// ```rust
1171    /// use alloc::sync::Arc;
1172    ///
1173    /// // 父进程执行 fork
1174    /// let parent_process = Arc::new(ProcessControlBlock::new(parent_elf));
1175    /// let child_process = parent_process.fork();
1176    ///
1177    /// // 检查父子关系
1178    /// {
1179    ///     let child_inner = child_process.inner_exclusive_access();
1180    ///     assert!(child_inner.parent.is_some());
1181    /// }
1182    ///
1183    /// {
1184    ///     let parent_inner = parent_process.inner_exclusive_access();
1185    ///     assert_eq!(parent_inner.children.len(), 1);
1186    ///     assert_eq!(parent_inner.children[0].getpid(), child_process.getpid());
1187    /// }
1188    ///
1189    /// println!("Parent PID: {}, Child PID: {}",
1190    ///          parent_process.getpid(), child_process.getpid());
1191    /// ```
1192    ///
1193    /// ## 错误处理
1194    ///
1195    /// Fork 可能在以下情况失败：
1196    /// - **内存不足**: 无法分配子进程的地址空间或内核栈
1197    /// - **PID 耗尽**: 系统 PID 资源已用完
1198    /// - **系统限制**: 达到进程数量或内存使用限制
1199    ///
1200    /// ## 性能考虑
1201    ///
1202    /// - **写时复制 (COW)**: 某些系统实现 COW 优化，本实现为完整复制
1203    /// - **内存开销**: 完整复制地址空间会消耗大量内存
1204    /// - **时间开销**: 复制过程的时间与父进程地址空间大小成正比
1205    ///
1206    /// ## RISC-V 特定处理
1207    ///
1208    /// - **内核栈指针**: 更新子进程陷阱上下文中的内核栈指针
1209    /// - **寄存器继承**: 子进程继承父进程的所有通用寄存器值
1210    /// - **特权模式**: 子进程在用户模式下开始执行
1211    ///
1212    /// ## 相关系统调用
1213    ///
1214    /// - [`exec()`] - 替换进程映像，通常与 fork 配合使用
1215    /// - [`wait()`] - 父进程等待子进程退出
1216    /// - [`exit()`] - 进程正常退出
1217    ///
1218    /// ## 返回
1219    /// - 父进程：返回新建子进程的 `Arc<ProcessControlBlock>`（随后入队由调度器运行）
1220    /// - 子进程：调度运行后，从系统调用返回 `0`
1221    pub fn fork(self: &Arc<Self>) -> Arc<Self> {
1222        let mut parent_inner = self.inner_exclusive_access();
1223        let memory_set = MemorySet::from_existed_user(&parent_inner.memory_set);
1224        let trap_cx_ppn = memory_set
1225            .translate(VirtAddr::from(TRAP_CONTEXT).into())
1226            .unwrap()
1227            .ppn();
1228        let pid_handle = pid_alloc();
1229        let kernel_stack = KernelStack::new(&pid_handle);
1230        let kernel_stack_top = kernel_stack.top();
1231        let mut new_fd_table: Vec<Option<Arc<dyn File + Send + Sync>>> = Vec::new();
1232        for fd in parent_inner.fd_table.iter() {
1233            if let Some(file) = fd {
1234                new_fd_table.push(Some(file.clone()));
1235            } else {
1236                new_fd_table.push(None);
1237            }
1238        }
1239
1240        let process_control_block = Arc::new(ProcessControlBlock {
1241            pid: pid_handle,
1242            kernel_stack,
1243            inner: unsafe {
1244                UPSafeCell::new(ProcessControlBlockInner {
1245                    trap_cx_ppn,
1246                    base_size: parent_inner.base_size,
1247                    process_cx: ProcessContext::goto_trap_return(kernel_stack_top),
1248                    process_status: ProcessStatus::Ready,
1249                    memory_set,
1250                    parent: Some(Arc::downgrade(self)),
1251                    children: Vec::new(),
1252                    exit_code: 0,
1253                    fd_table: new_fd_table,
1254                    signals: SignalFlags::empty(),
1255                    signal_mask: parent_inner.signal_mask,
1256                    handling_sig: -1,
1257                    signal_actions: parent_inner.signal_actions.clone(),
1258                    killed: false,
1259                    frozen: false,
1260                    trap_ctx_backup: None,
1261                    // 子进程继承父进程的优先级信息
1262                    priority: parent_inner.priority,
1263                    time_slice_used: 0, // 重置时间片使用计数
1264                    time_slice_limit: parent_inner.time_slice_limit,
1265                })
1266            },
1267        });
1268        parent_inner.children.push(process_control_block.clone());
1269        let trap_cx = process_control_block.inner_exclusive_access().trap_cx();
1270        trap_cx.kernel_sp = kernel_stack_top;
1271        process_control_block
1272    }

1274    /// Exec 系统调用实现：替换进程映像
1275    ///
1276    /// 使用新的可执行文件替换当前进程的内存映像，保持相同的 PID 和内核栈，
1277    /// 但完全替换地址空间和执行上下文。这是 UNIX/Linux 系统加载新程序的标准方式。
1278    ///
1279    /// ## Arguments
1280    ///
1281    /// * `elf_data` - 新程序的 ELF 文件二进制数据
1282    ///
1283    /// ## Exec 语义
1284    ///
1285    /// **保持不变的部分**：
1286    /// - **PID**: 进程标识符不变，仍然是同一个进程
1287    /// - **内核栈**: 复用现有的内核栈，不重新分配
1288    /// - **父子关系**: 进程在进程树中的位置不变
1289    ///
1290    /// **替换的部分**：
1291    /// - **地址空间**: 完全替换为新程序的内存布局
1292    /// - **执行上下文**: 重置为新程序的入口点和初始状态
1293    /// - **用户态寄存器**: 重新初始化为新程序的启动状态
1294    ///
1295    /// ## Exec 过程详解
1296    ///
1297    /// ```text
1298    /// 1. 解析新的 ELF 文件
1299    ///    │
1300    ///    │ - 验证 ELF 文件格式
1301    ///    │ - 提取程序段信息
1302    ///    │ - 获取入口点地址
1303    ///    ▼
1304    /// 2. 创建新的地址空间
1305    ///    │
1306    ///    │ - 销毁旧的内存映射
1307    ///    │ - 建立新程序的内存布局
1308    ///    │ - 加载程序段到内存
1309    ///    ▼
1310    /// 3. 更新进程状态
1311    ///    │
1312    ///    │ - 更新内存集合
1313    ///    │ - 更新陷阱上下文页号
1314    ///    │ - 重新设置基础内存大小
1315    ///    ▼
1316    /// 4. 初始化执行环境
1317    ///    │
1318    ///    │ - 设置程序计数器为入口点
1319    ///    │ - 初始化用户栈指针
1320    ///    │ - 清零通用寄存器
1321    ///    ▼
1322    /// 5. 程序替换完成
1323    /// ```
1324    ///
1325    /// ## 内存布局变化
1326    ///
1327    /// ```text
1328    /// Exec 前 (旧程序):
1329    /// ┌─────────────────────────┐
1330    /// │      Process (PID n)       │ ← PID 保持不变
1331    /// ├─────────────────────────┤
1332    /// │     Old Program         │
1333    /// │   Address Space:        │
1334    /// │  ┌──────────────────┐   │
1335    /// │  │ Old .text/.data  │   │ ← 将被完全替换
1336    /// │  │ Old User Stack   │   │
1337    /// │  │ Old Heap         │   │
1338    /// │  └──────────────────┘   │
1339    /// └─────────────────────────┘
1340    ///
1341    /// Exec 后 (新程序):
1342    /// ┌─────────────────────────┐
1343    /// │      Process (PID n)       │ ← 同一个 PID
1344    /// ├─────────────────────────┤
1345    /// │     New Program         │
1346    /// │   Address Space:        │
1347    /// │  ┌──────────────────┐   │
1348    /// │  │ New .text/.data  │   │ ← 新程序的内存布局
1349    /// │  │ New User Stack   │   │
1350    /// │  │ New Heap         │   │
1351    /// │  └──────────────────┘   │
1352    /// └─────────────────────────┘
1353    /// ```
1354    ///
1355    /// ## 上下文重置
1356    ///
1357    /// Exec 会重置进程的执行上下文：
1358    ///
1359    /// ```text
1360    /// 重置项目                 新值
1361    /// ┌──────────────────┐    ┌─────────────────────┐
1362    /// │Program Counter   │───►│ New Entry Point     │
1363    /// ├──────────────────┤    ├─────────────────────┤
1364    /// │Stack Pointer     │───►│ New User Stack Top  │
1365    /// ├──────────────────┤    ├─────────────────────┤
1366    /// │General Registers │───►│ Zero or Init Values │
1367    /// ├──────────────────┤    ├─────────────────────┤
1368    /// │Page Table Ptr    │───►│ New Address Space   │
1369    /// └──────────────────┘    └─────────────────────┘
1370    /// ```
1371    ///
1372    /// ## 典型使用场景
1373    ///
1374    /// **Shell 命令执行**：
1375    /// ```text
1376    /// 1. Shell 进程执行 fork() 创建子进程
1377    /// 2. 子进程执行 exec() 加载目标程序
1378    /// 3. 父进程 (Shell) 执行 wait() 等待子进程完成
1379    /// ```
1380    ///
1381    /// **程序动态加载**：
1382    /// ```text
1383    /// 1. 当前进程不再需要原程序代码
1384    /// 2. 直接执行 exec() 替换为新程序
1385    /// 3. 新程序从头开始执行
1386    /// ```
1387    ///
1388    /// ## 使用示例
1389    ///
1390    /// ```rust
1391    /// // 替换当前进程为新程序
1392    /// let new_program = app_data("target_app");
1393    ///
1394    /// // 记录替换前的信息
1395    /// let old_pid = process.getpid();
1396    /// println!("Executing new program in PID {}", old_pid);
1397    ///
1398    /// // 执行替换
1399    /// process.exec(new_program);
1400    ///
1401    /// // 验证 PID 未变但程序已替换
1402    /// assert_eq!(process.getpid(), old_pid);
1403    ///
1404    /// // 检查新的执行状态
1405    /// {
1406    ///     let inner = process.inner_exclusive_access();
1407    ///     let trap_cx = inner.trap_cx();
1408    ///     println!("New entry point: 0x{:x}", trap_cx.sepc);
1409    ///     println!("New stack pointer: 0x{:x}", trap_cx.x[2]);
1410    /// }
1411    /// ```
1412    ///
1413    /// ## 安全考虑
1414    ///
1415    /// - **权限检查**: 确保有权限执行目标文件
1416    /// - **格式验证**: 验证 ELF 文件的完整性和合法性
1417    /// - **资源清理**: 确保旧程序的所有资源得到正确释放
1418    /// - **状态一致性**: 保证替换过程的原子性
1419    ///
1420    /// ## 错误处理
1421    ///
1422    /// Exec 可能在以下情况失败：
1423    /// - **文件格式错误**: ELF 文件格式无效或损坏
1424    /// - **内存不足**: 无法为新程序分配足够的内存空间
1425    /// - **权限不足**: 没有执行目标文件的权限
1426    /// - **系统资源限制**: 超出系统资源限制
1427    ///
1428    /// ## 性能特性
1429    ///
1430    /// - **内存复用**: 复用现有的 PID 和内核栈资源
1431    /// - **快速切换**: 相比 fork + exec 模式，单独 exec 更高效
1432    /// - **内存释放**: 自动释放旧程序占用的所有内存
1433    ///
1434    /// ## 与其他系统调用的关系
1435    ///
1436    /// - **fork() + exec()**: 经典的进程创建和程序加载模式
1437    /// - **wait()**: 父进程等待 exec 后的子进程完成
1438    /// - **exit()**: 进程执行完成后的正常退出
1439    pub fn exec(&self, elf_data: &[u8], args: Vec<String>) {
1440        let (memory_set, mut user_sp, entry_point) = MemorySet::from_elf(elf_data);
1441        let trap_cx_ppn = memory_set
1442            .translate(VirtAddr::from(TRAP_CONTEXT).into())
1443            .unwrap()
1444            .ppn();
1445
1446        user_sp -= (args.len() + 1) * core::mem::size_of::<usize>();
1447        let argv_base = user_sp;
1448        let mut argv: Vec<_> = (0..=args.len())
1449            .map(|arg| {
1450                translated_refmut(
1451                    memory_set.token(),
1452                    (argv_base + arg * core::mem::size_of::<usize>()) as *mut usize,
1453                )
1454            })
1455            .collect();
1456        *argv[args.len()] = 0;
1457        for i in 0..args.len() {
1458            user_sp -= args[i].len() + 1;
1459            *argv[i] = user_sp;
1460            let mut p = user_sp;
1461            for c in args[i].as_bytes() {
1462                *translated_refmut(memory_set.token(), p as *mut u8) = *c;
1463                p += 1;
1464            }
1465            *translated_refmut(memory_set.token(), p as *mut u8) = 0;
1466        }
1467        user_sp -= user_sp % core::mem::size_of::<usize>();
1468
1469        let mut inner = self.inner_exclusive_access();
1470        inner.memory_set = memory_set;
1471        inner.trap_cx_ppn = trap_cx_ppn;
1472        let mut trap_cx = TrapContext::app_init_context(
1473            entry_point,
1474            user_sp,
1475            KERNEL_SPACE.exclusive_access().token(),
1476            self.kernel_stack.top(),
1477            trap_handler as usize,
1478        );
1479        trap_cx.x[10] = args.len();
1480        trap_cx.x[11] = argv_base;
1481        *inner.trap_cx() = trap_cx;
1482    }
