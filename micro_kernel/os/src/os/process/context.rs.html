1//! # 进程上下文模块
2//!
3//! 定义进程上下文结构，用于在进程切换时保存和恢复 CPU 寄存器状态。
4//! 进程上下文包含了进程切换时需要保存的所有寄存器信息。
5use crate::trap::trap_return;
6
7/// 进程上下文结构
8///
9/// 存储进程在被切换时的 CPU 寄存器状态。根据 RISC-V 调用约定，
10/// 只需要保存调用者保存寄存器 (callee-saved registers)。
11///
12/// ## 内存布局
13///
14/// 该结构体使用 `#[repr(C)]` 确保字段按 C 语言结构体布局排列，
15/// 这样汇编代码可以通过固定偏移访问各个字段。
16///
17/// ## 寄存器说明
18///
19/// - `ra`: 返回地址寄存器，存储函数返回地址
20/// - `sp`: 栈指针寄存器，指向当前栈顶
21/// - `s0-s11`: 被调用者保存寄存器，需要在函数调用间保持不变
22///
23/// ## 切换原理
24///
25/// 在进程切换时：
26/// 1. 将当前进程的这些寄存器值保存到其 ProcessContext 中
27/// 2. 从目标进程的 ProcessContext 恢复寄存器值
28/// 3. 通过 `ret` 指令跳转到目标进程的 `ra` 地址继续执行
29#[repr(C)]
30#[derive(Copy, Clone)]
31pub struct ProcessContext {
32    /// 返回地址寄存器 (ra)
33    ///
34    /// 存储进程恢复执行时应该跳转到的地址。
35    /// 对于新创建的进程，通常指向 `__restore` 函数。
36    ra: usize,
37
38    /// 栈指针寄存器 (sp)
39    ///
40    /// 指向进程的内核栈顶。进程切换时会恢复此栈指针，
41    /// 确保进程在正确的栈上继续执行。
42    sp: usize,
43
44    /// 被调用者保存寄存器 (s0-s11)
45    ///
46    /// RISC-V 调用约定中的被调用者保存寄存器，函数调用时
47    /// 被调用函数需要保证这些寄存器的值不变。
48    s: [usize; 12],
49}
50
51impl ProcessContext {
52    /// 创建零初始化的进程上下文
53    ///
54    /// 创建一个所有寄存器都为零的进程上下文，通常用作临时变量
55    /// 或占位符，不代表真实的进程状态。
56    ///
57    /// ## Returns
58    ///
59    /// 返回一个所有字段都为 0 的 `ProcessContext` 实例
60    ///
61    /// ## Use Cases
62    ///
63    /// - 在 `run_first_process` 中作为当前上下文的占位符
64    /// - 初始化进程数组时的默认值
65    pub fn zero_init() -> Self {
66        Self {
67            ra: 0,
68            sp: 0,
69            s: [0; 12],
70        }
71    }
72
73    /// 创建指向陷阱返回的进程上下文
74    ///
75    /// 为新进程创建一个进程上下文，设置返回地址指向 `trap_return` 函数。
76    /// 这使得当进程首次被调度时，会跳转到 `trap_return` 执行，从而
77    /// 恢复陷阱上下文并进入用户态开始执行。
78    ///
79    /// ## Arguments
80    ///
81    /// * `kstack_ptr` - 内核栈指针，指向该进程的内核栈顶
82    ///
83    /// ## Returns
84    ///
85    /// 配置好的进程上下文，可用于进程首次调度
86    ///
87    /// ## 执行流程
88    ///
89    /// 当此上下文被 `__switch` 恢复时：
90    /// 1. 恢复栈指针到指定的内核栈
91    /// 2. 通过 `ret` 指令跳转到 `trap_return` 函数
92    /// 3. `trap_return` 恢复陷阱上下文并切换到用户态
93    /// 4. 用户程序开始执行
94    ///
95    /// ## 与 `zero_init()` 的区别
96    ///
97    /// | 方法 | 返回地址 | 用途 |
98    /// |------|----------|------|
99    /// | `zero_init()` | 0 | 临时占位符 |
100    /// | `goto_trap_return()` | `trap_return` | 新进程初始化 |
101    ///
102    /// ## 设计原理
103    ///
104    /// 新创建的进程需要一个特殊的启动流程：
105    /// - 不能直接跳转到用户代码（缺少完整的用户态上下文）
106    /// - 必须通过 `trap_return` 建立完整的用户态环境
107    /// - 这种设计统一了进程切换和陷阱返回的机制
108    ///
109    /// ## Examples
110    ///
111    /// ```rust
112    /// // 为新进程创建上下文
113    /// let process_cx = ProcessContext::goto_trap_return(kernel_stack_top);
114    ///
115    /// // 进程首次调度时的执行路径：
116    /// // __switch -> trap_return -> __restore -> 用户程序
117    /// ```
118    pub fn goto_trap_return(kstack_ptr: usize) -> Self {
119        Self {
120            ra: trap_return as usize,
121            sp: kstack_ptr,
122            s: [0; 12],
123        }
124    }
125}
