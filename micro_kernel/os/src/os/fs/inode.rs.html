1//! # 文件 Inode 管理模块
2//!
3//! 提供操作系统级别的文件 inode 管理功能，包括文件读写、元数据操作、
4//! 文件系统访问等功能。封装底层 Easy-FS 文件系统，为上层提供统一的文件接口。
5//!
6//! ## 核心组件
7//!
8//! - [`OSInode`] - 操作系统级别的 inode 封装，提供文件操作接口
9//! - [`OSInodeInner`] - inode 的内部状态管理
10//! - [`OpenFlags`] - 文件打开标志位，控制文件的打开模式
11//! - [`ROOT_INODE`] - 全局根目录 inode 实例
12//!
13//! ## 文件操作特性
14//!
15//! - **读写权限**: 支持独立的读写权限控制
16//! - **位置管理**: 自动管理文件偏移量
17//! - **批量操作**: 支持一次性读取整个文件内容
18//! - **文件创建**: 支持新文件的创建和现有文件的截断
19//!
20//! ## 使用示例
21//!
22//! ```rust
23//! use crate::fs::{open_file, OpenFlags};
24//!
25//! // 打开文件进行读写
26//! let file = open_file("test.txt", OpenFlags::RDWR).unwrap();
27//!
28//! // 读取整个文件内容
29//! let content = file.read_all();
30//!
31//! // 列出应用程序
32//! list_apps();
33//! ```

46/// OSInode 的内部状态结构
47///
48/// 管理文件 inode 的内部状态，包括文件偏移量和底层 inode 引用。
49/// 该结构被封装在 `UPSafeCell` 中，提供线程安全的内部可变性。
50///
51/// ## 字段说明
52///
53/// - `offset` - 当前文件偏移量，表示下次读写操作的位置
54/// - `inode` - 底层 Easy-FS inode 的引用计数智能指针
55///
56/// ## 状态管理
57///
58/// - 文件偏移量在每次读写操作后自动更新
59/// - inode 引用确保底层文件系统资源不会被过早释放
60/// - 通过 `UPSafeCell` 提供线程安全的内部状态修改
61pub struct OSInodeInner {
62    offset: usize,
63    inode: Arc<Inode>,
64}

66/// 操作系统级别的文件 Inode
67///
68/// 封装底层 Easy-FS 文件系统的 inode，提供操作系统级别的文件操作接口。
69/// 支持读写权限控制、文件偏移量管理和批量文件操作。
70///
71/// ## 内部结构
72///
73/// 包含读写权限标志和内部状态管理结构，通过 `UPSafeCell` 提供线程安全。
74///
75/// ## 线程安全
76///
77/// 该结构是线程安全的，多个线程可以同时访问同一个文件进行读写操作。
78/// 并发控制通过 `UPSafeCell` 和内部的锁机制实现。
79///
80/// ## 生命周期管理
81///
82/// 文件实例的生命周期由引用计数管理，当所有引用都被释放时，
83/// 底层文件系统资源会被自动清理。
84pub struct OSInode {
85    readable: bool,
86    writable: bool,
87    inner: UPSafeCell<OSInodeInner>,
88}

90impl OSInode {
91    /// 创建新的 OSInode 实例
92    ///
93    /// 根据指定的读写权限和底层 inode 创建新的文件实例。
94    /// 新创建的文件偏移量初始化为 0。
95    ///
96    /// ## Arguments
97    ///
98    /// * `readable` - 文件是否可读
99    /// * `writable` - 文件是否可写
100    /// * `inode` - 底层 Easy-FS inode 的引用
101    ///
102    /// ## Returns
103    ///
104    /// 新创建的 `OSInode` 实例
105    ///
106    /// ## 初始化状态
107    ///
108    /// - 文件偏移量设置为 0
109    /// - 读写权限根据参数设置
110    /// - 内部状态通过 `UPSafeCell` 保护
111    ///
112    /// ## Examples
113    ///
114    /// ```
115    /// let inode = get_some_inode();
116    /// let file = OSInode::new(true, false, inode); // 只读文件
117    /// ```
118    pub fn new(readable: bool, writable: bool, inode: Arc<Inode>) -> Self {
119        Self {
120            readable,
121            writable,
122            inner: unsafe { UPSafeCell::new(OSInodeInner { offset: 0, inode }) },
123        }
124    }
125
126    /// 读取整个文件内容
127    ///
128    /// 从文件开头开始读取所有内容，直到文件末尾。
129    /// 该操作会重置文件偏移量到开头，然后连续读取直到文件结束。
130    ///
131    /// ## Returns
132    ///
133    /// 包含整个文件内容的字节向量
134    ///
135    /// ## 读取策略
136    ///
137    /// 1. **重置偏移量**: 将文件偏移量设置为 0
138    /// 2. **批量读取**: 使用 512 字节的缓冲区进行批量读取
139    /// 3. **连续读取**: 循环读取直到到达文件末尾
140    /// 4. **内容收集**: 将所有读取的内容合并到一个向量中
141    ///
142    /// ## 性能说明
143    ///
144    /// 该操作会读取整个文件，对于大文件可能消耗较多内存和时间。
145    /// 建议只对小文件使用此方法，大文件应该使用流式读取。
146    ///
147    /// ## 内存使用
148    ///
149    /// 返回的向量大小等于文件大小，会占用相应的内存空间。
150    /// 调用者负责管理返回的内存。
151    ///
152    /// ## Examples
153    ///
154    /// ```
155    /// let content = file.read_all();
156    /// println!("文件大小: {} 字节", content.len());
157    /// ```
158    pub fn read_all(&self) -> Vec<u8> {
159        let mut inner = self.inner.exclusive_access();
160        let mut buffer = [0u8; 512];
161        let mut v: Vec<u8> = Vec::new();
162        loop {
163            let len = inner.inode.read_at(inner.offset, &mut buffer);
164            if len == 0 {
165                break;
166            }
167            inner.offset += len;
168            v.extend_from_slice(&buffer[..len]);
169        }
170        v
171    }
172}

174impl File for OSInode {
175    /// 从文件读取数据到用户缓冲区
176    ///
177    /// 从当前文件偏移量开始读取数据，支持跨页面的用户缓冲区。
178    /// 该操作会更新文件偏移量，为下次读取做准备。
179    ///
180    /// ## Arguments
181    ///
182    /// * `buf` - 用户缓冲区，可能跨越多个页面
183    ///
184    /// ## Returns
185    ///
186    /// 实际读取的字节数，0 表示已到达文件末尾
187    ///
188    /// ## 读取过程
189    ///
190    /// 1. **权限检查**: 验证文件是否可读
191    /// 2. **缓冲区遍历**: 遍历用户缓冲区的所有页面
192    /// 3. **数据读取**: 从当前偏移量读取数据到每个页面
193    /// 4. **偏移量更新**: 更新文件偏移量
194    /// 5. **提前结束**: 如果到达文件末尾则停止读取
195    ///
196    /// ## 错误处理
197    ///
198    /// 如果文件不可读，行为由具体实现定义。
199    /// 读取过程中如果遇到错误，会返回已读取的字节数。
200    fn read(&self, mut buf: UserBuffer) -> usize {
201        let mut inner = self.inner.exclusive_access();
202        let mut total_read_size = 0usize;
203        for slice in buf.buffers.iter_mut() {
204            let read_size = inner.inode.read_at(inner.offset, *slice);
205            if read_size == 0 {
206                break;
207            }
208            inner.offset += read_size;
209            total_read_size += read_size;
210        }
211        total_read_size
212    }
213
214    /// 向文件写入数据
215    ///
216    /// 从当前文件偏移量开始写入数据，支持跨页面的用户缓冲区。
217    /// 该操作会更新文件偏移量，为下次写入做准备。
218    ///
219    /// ## Arguments
220    ///
221    /// * `buf` - 用户缓冲区，包含要写入的数据
222    ///
223    /// ## Returns
224    ///
225    /// 实际写入的字节数
226    ///
227    /// ## 写入过程
228    ///
229    /// 1. **权限检查**: 验证文件是否可写
230    /// 2. **缓冲区遍历**: 遍历用户缓冲区的所有页面
231    /// 3. **数据写入**: 将每个页面的数据写入文件
232    /// 4. **偏移量更新**: 更新文件偏移量
233    /// 5. **完整性检查**: 确保写入的字节数与缓冲区大小一致
234    ///
235    /// ## 错误处理
236    ///
237    /// 如果文件不可写，会触发 panic。
238    /// 如果写入的字节数与缓冲区大小不一致，会触发 panic。
239    fn write(&self, buf: UserBuffer) -> usize {
240        let mut inner = self.inner.exclusive_access();
241        let mut total_write_size = 0usize;
242        for slice in buf.buffers.iter() {
243            let write_size = inner.inode.write_at(inner.offset, *slice);
244            assert_eq!(write_size, slice.len());
245            inner.offset += write_size;
246            total_write_size += write_size;
247        }
248        total_write_size
249    }
250
251    /// 检查文件是否可读
252    ///
253    /// ## Returns
254    ///
255    /// 如果文件可读返回 `true`，否则返回 `false`
256    fn readable(&self) -> bool {
257        self.readable
258    }
259
260    /// 检查文件是否可写
261    ///
262    /// ## Returns
263    ///
264    /// 如果文件可写返回 `true`，否则返回 `false`
265    fn writable(&self) -> bool {
266        self.writable
267    }
268}

270lazy_static! {
271    /// 全局根目录 Inode 实例
272    ///
273    /// 使用 `lazy_static` 实现全局单例模式，确保整个系统共享同一个根目录 inode。
274    /// 该实例在首次访问时初始化，提供对文件系统根目录的访问接口。
275    ///
276    /// ## 初始化过程
277    ///
278    /// 1. **文件系统打开**: 通过 `EasyFileSystem::open()` 打开底层文件系统
279    /// 2. **根目录获取**: 调用 `EasyFileSystem::root_inode()` 获取根目录 inode
280    /// 3. **引用包装**: 将根目录 inode 包装在 `Arc` 中以支持多线程共享访问
281    ///
282    /// ## 使用场景
283    ///
284    /// - 文件系统根目录的访问
285    /// - 应用程序列表的获取
286    /// - 新文件的创建和查找
287    /// - 文件系统遍历的起点
288    ///
289    /// ## 线程安全
290    ///
291    /// 该实例是线程安全的，多个线程可以同时访问根目录进行文件操作。
292    /// 具体的并发控制由底层 Easy-FS 文件系统实现。
293    ///
294    /// ## 生命周期
295    ///
296    /// 根目录 inode 的生命周期与系统运行时间相同，在系统启动时初始化，
297    /// 在系统关闭时自动清理。
298    ///
299    /// ## Examples
300    ///
301    /// ```
302    /// use crate::fs::ROOT_INODE;
303    ///
304    /// // 列出根目录下的所有文件
305    /// let files = ROOT_INODE.ls();
306    /// for file in files {
307    ///     println!("文件: {}", file);
308    /// }
309    ///
310    /// // 在根目录下查找文件
311    /// if let Some(file) = ROOT_INODE.find("test.txt") {
312    ///     println!("找到文件: test.txt");
313    /// }
314    /// ```
315    pub static ref ROOT_INODE: Arc<Inode> = {
316        let efs = EasyFileSystem::open(BLOCK_DEVICE.clone());
317        Arc::new(EasyFileSystem::root_inode(&efs))
318    };
319}

365bitflags! {
366    /// 文件打开标志位
367    ///
368    /// 定义文件打开时的各种标志位，控制文件的打开模式和行为。
369    /// 使用 `bitflags` 宏实现，支持标志位的组合和检查。
370    ///
371    /// ## 标志位说明
372    ///
373    /// - `RDONLY` - 只读模式，文件只能读取不能写入
374    /// - `WRONLY` - 只写模式，文件只能写入不能读取
375    /// - `RDWR` - 读写模式，文件既可以读取也可以写入
376    /// - `CREATE` - 创建标志，如果文件不存在则创建新文件
377    /// - `TRUNC` - 截断标志，如果文件存在则清空文件内容
378    ///
379    /// ## 组合使用
380    ///
381    /// 标志位可以组合使用，例如：
382    /// - `RDWR | CREATE` - 读写模式，如果文件不存在则创建
383    /// - `WRONLY | CREATE | TRUNC` - 只写模式，创建新文件或清空现有文件
384    ///
385    /// ## 默认行为
386    ///
387    /// 如果不指定任何标志位（空标志），默认为只读模式。
388    pub struct OpenFlags: u32 {
389        const RDONLY = 0;
390        const WRONLY = 1 << 0;
391        const RDWR = 1 << 1;
392        const CREATE = 1 << 9;
393        const TRUNC = 1 << 10;
394    }
395}

397impl OpenFlags {
398    /// 解析读写权限
399    ///
400    /// 根据标志位解析文件的读写权限，返回一个元组 `(readable, writable)`。
401    ///
402    /// ## Returns
403    ///
404    /// 返回 `(readable, writable)` 元组：
405    /// - `readable` - 文件是否可读
406    /// - `writable` - 文件是否可写
407    ///
408    /// ## 解析规则
409    ///
410    /// - 空标志位：`(true, false)` - 只读模式
411    /// - `WRONLY`：`(false, true)` - 只写模式
412    /// - `RDWR` 或其他组合：`(true, true)` - 读写模式
413    ///
414    /// ## Examples
415    ///
416    /// ```
417    /// let flags = OpenFlags::RDWR | OpenFlags::CREATE;
418    /// let (readable, writable) = flags.read_write();
419    /// assert_eq!(readable, true);
420    /// assert_eq!(writable, true);
421    /// ```
422    pub fn read_write(&self) -> (bool, bool) {
423        if self.is_empty() {
424            (true, false)
425        } else if self.contains(Self::WRONLY) {
426            (false, true)
427        } else {
428            (true, true)
429        }
430    }
431}

433/// 打开文件
434///
435/// 根据文件名和打开标志位打开文件，返回文件的操作接口。
436/// 该函数是文件系统的主要入口点，支持文件的创建、查找和权限控制。
437///
438/// ## Arguments
439///
440/// * `name` - 文件名
441/// * `flags` - 文件打开标志位
442///
443/// ## Returns
444///
445/// - `Some(file)` - 成功打开文件，返回文件操作接口
446/// - `None` - 文件不存在且未指定 `CREATE` 标志，或创建失败
447///
448/// ## 打开流程
449///
450/// 1. **权限解析**: 根据标志位解析读写权限
451/// 2. **文件查找**: 在根目录下查找指定文件
452/// 3. **文件创建**: 如果指定 `CREATE` 标志且文件不存在，则创建新文件
453/// 4. **文件截断**: 如果指定 `TRUNC` 标志，则清空现有文件内容
454/// 5. **接口创建**: 创建 `OSInode` 实例并返回
455///
456/// ## 标志位处理
457///
458/// - `CREATE`: 如果文件不存在则创建，如果存在则打开现有文件
459/// - `TRUNC`: 清空文件内容，将文件大小设置为 0
460/// - 权限标志：控制返回文件的读写权限
461///
462/// ## 错误处理
463///
464/// - 文件不存在且未指定 `CREATE` 标志：返回 `None`
465/// - 文件创建失败：返回 `None`
466/// - 权限不足：返回 `None`（当前实现中未检查权限）
467///
468/// ## Examples
469///
470/// ```
471/// use crate::fs::{open_file, OpenFlags};
472///
473/// // 只读打开现有文件
474/// let file = open_file("config.txt", OpenFlags::RDONLY);
475///
476/// // 读写打开，如果不存在则创建
477/// let file = open_file("data.txt", OpenFlags::RDWR | OpenFlags::CREATE);
478///
479/// // 只写打开，清空现有内容
480/// let file = open_file("log.txt", OpenFlags::WRONLY | OpenFlags::TRUNC);
481/// ```
482pub fn open_file(name: &str, flags: OpenFlags) -> Option<Arc<OSInode>> {
483    let (readable, writable) = flags.read_write();
484    if flags.contains(OpenFlags::CREATE) {
485        if let Some(inode) = ROOT_INODE.find(name) {
486            inode.clear();
487            Some(Arc::new(OSInode::new(readable, writable, inode)))
488        } else {
489            ROOT_INODE
490                .create(name)
491                .map(|inode| Arc::new(OSInode::new(readable, writable, inode)))
492        }
493    } else {
494        ROOT_INODE.find(name).map(|inode| {
495            if flags.contains(OpenFlags::TRUNC) {
496                inode.clear();
497            }
498            Arc::new(OSInode::new(readable, writable, inode))
499        })
500    }
