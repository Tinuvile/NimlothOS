1//! # 陷阱和中断处理模块
2//!
3//! 处理所有从用户态到内核态的陷阱 (trap)，包括系统调用、异常和中断。
4//! 当前实现将“同步异常转信号”的语义融入到陷阱处理流程中：对内存访问/非法指令
5//! 等异常不再直接打印并杀死进程，而是先投递相应信号，随后统一在信号处理阶段
6//! 执行默认/用户自定义动作。
7//!
8//! ## 处理的陷阱类型
9//!
10//! - **系统调用** (`UserEnvCall`): 用户程序请求内核服务
11//! - **时钟中断** (`SupervisorTimer`): 实现抢占式多任务调度
12//! - **数据访问异常** (`StoreFault`, `StorePageFault`, `LoadFault`, `LoadPageFault`): 数据内存访问违规
13//! - **指令访问异常** (`InstructionFault`, `InstructionPageFault`): 指令内存访问违规
14//! - **非法指令** (`IllegalInstruction`): 执行无效指令
15//!
16//! ## 执行流程（更新）
17//!
18//! 1. **陷阱触发**: 用户态程序执行 `ecall` 或发生异常/中断
19//! 2. **硬件切换**: CPU 自动切换到 S 模式，跳转到 `stvec` 指定的处理程序
20//! 3. **上下文保存**: `__alltraps` 保存所有寄存器到陷阱上下文
21//! 4. **处理分发**: `trap_handler` 根据陷阱类型执行相应处理（见下）
22//! 5. **信号阶段**: 调用 `handle_signals()` 检查/进入用户信号处理；
23//!    对致命信号，`check_signals_error_of_current()` 会返回标准退出码并退出进程
24//! 6. **上下文恢复**: `trap_return()` → `__restore` 恢复寄存器并返回用户态
25//!
26//! ## 寄存器使用
27//!
28//! - `stvec`: 陷阱向量寄存器，指向陷阱处理入口 `__alltraps`
29//! - `scause`: 陷阱原因寄存器，标识陷阱类型和具体原因
30//! - `stval`: 陷阱值寄存器，包含相关的地址或值信息
31//! - `sstatus`: 状态寄存器，控制中断使能和特权级
32//! - `sepc`: 异常程序计数器，指向触发陷阱的指令地址
33

56/// 初始化陷阱处理系统
57///
58/// 设置内核态的陷阱处理入口点，配置陷阱向量寄存器 (`stvec`)
59/// 指向内核陷阱处理函数。这是陷阱处理系统的初始化函数。
60///
61/// ## 初始化内容
62///
63/// - 设置 `stvec` 寄存器指向 `trap_from_kernel`
64/// - 配置直接模式 (`TrapMode::Direct`)
65/// - 为内核态陷阱处理做准备
66///
67/// ## 调用时机
68///
69/// 应在系统启动早期调用，在启用中断之前完成陷阱系统初始化。
70///
71/// ## Note
72///
73/// 此函数只设置内核态陷阱入口，用户态陷阱入口会在任务切换时
74/// 通过 `set_user_trap_entry()` 动态设置。
75pub fn init() {
76    set_kernel_trap_entry();
77    enable_timer_interrupt();
78}

80/// 启用时钟中断
81///
82/// 在监督者中断使能寄存器 (`sie`) 中启用时钟中断位，
83/// 允许时钟中断触发陷阱进入内核进行任务调度。
84///
85/// ## 功能
86///
87/// - 设置 `sie.STIE` 位，启用监督者时钟中断
88/// - 配合时钟设备实现抢占式多任务调度
89/// - 必须在时钟设备配置完成后调用
90///
91/// ## Safety
92///
93/// 此函数是安全的，但内部使用 `unsafe` 访问 CSR 寄存器。
94pub fn enable_timer_interrupt() {
95    unsafe {
96        sie::set_stimer();
97    }
98}

100/// 设置内核态陷阱入口
101///
102/// 配置 `stvec` 寄存器指向内核陷阱处理函数 `trap_from_kernel`。
103/// 当内核态发生陷阱时，硬件会跳转到此函数执行。
104///
105/// ## 安全性
106///
107/// 使用 `unsafe` 直接操作 CSR 寄存器，这是特权操作。
108fn set_kernel_trap_entry() {
109    unsafe {
110        stvec::write(trap_from_kernel as usize, TrapMode::Direct);
111    }
112}

114/// 设置用户态陷阱入口
115///
116/// 配置 `stvec` 寄存器指向用户态陷阱处理入口 `TRAMPOLINE`。
117/// 当用户态发生陷阱时，硬件会跳转到 Trampoline 页面执行。
118///
119/// ## Trampoline 机制
120///
121/// Trampoline 页面包含 `__alltraps` 和 `__restore` 汇编代码，
122/// 负责保存和恢复完整的用户态上下文。
123///
124/// ## 安全性
125///
126/// 使用 `unsafe` 直接操作 CSR 寄存器，这是特权操作。
127fn set_user_trap_entry() {
128    unsafe {
129        stvec::write(TRAMPOLINE as usize, TrapMode::Direct);
130    }
131}

133#[unsafe(no_mangle)]
134pub fn trap_from_kernel() -> ! {
135    use riscv::register::sepc;
136    println!("stval = {:#x}, sepc = {:#x}", stval::read(), sepc::read());
137    panic!("a trap {:?} from kernel!", scause::read().cause());
138}

140/// 陷阱处理主函数
141///
142/// 用户态陷阱统一入口：
143/// - 系统调用：两次获取/写回 Trap 上下文（因期间可能发生任务切换）
144/// - 访问/执行异常：不直接终止，改为投递 `SIGSEGV`/`SIGILL` 等信号
145/// - 时钟中断：设置下一次触发并让出 CPU
146/// 处理完毕后进入信号阶段，必要时退出当前进程，然后返回用户态继续执行。
147///
148/// ## 处理流程
149///
150/// 1. **设置内核陷阱入口**: 防止处理过程中的嵌套陷阱（`stvec` 指向内核）
151/// 2. **获取陷阱信息**: 读取 `scause` 和 `stval` 寄存器
152/// 3. **分发处理**: 根据陷阱类型执行：系统调用/异常转信号/时钟中断让出
153/// 4. **信号处理**: `handle_signals()` 进入/完成用户处理；对致命信号退出
154/// 5. **返回用户态**: `trap_return()` 恢复用户执行
155///
156/// ## 系统调用处理细节
157///
158/// 系统调用处理中有特殊的上下文管理：
159/// - 首次获取陷阱上下文以读取参数
160/// - 提前移动 `sepc += 4` 跳过 `ecall`
161/// - 调用系统调用处理函数（期间可能调度）
162/// - 再次获取陷阱上下文写回返回值（避免因调度导致的上下文失配）
163///
164/// ## 支持的陷阱类型
165///
166/// - **系统调用** (`UserEnvCall`): 处理用户程序的系统调用请求
167/// - **数据访问异常** (`StoreFault`, `StorePageFault`, `LoadFault`, `LoadPageFault`): 处理数据内存访问违规
168/// - **指令访问异常** (`InstructionFault`, `InstructionPageFault`): 处理指令内存访问违规
169/// - **非法指令** (`IllegalInstruction`): 处理无效指令执行
170/// - **时钟中断** (`SupervisorTimer`): 处理抢占式调度
171///
172/// ## 错误/信号处理
173///
174/// - 访问/执行异常：改为投递信号，由信号阶段决定是否终止或进入用户处理
175/// - 未知陷阱类型：触发 panic
176///
177/// ## 属性说明
178///
179/// - `#[unsafe(no_mangle)]`: 防止函数名被混淆，汇编代码需要调用此函数
180/// - `-> !`: 函数永不返回，总是通过 `trap_return()` 返回用户态
181#[unsafe(no_mangle)]
182pub fn trap_handler() -> ! {
183    set_kernel_trap_entry();
184    let scause = scause::read();
185    let stval = stval::read();
186    match scause.cause() {
187        Trap::Exception(Exception::UserEnvCall) => {
188            let mut cx = current_trap_cx();
189            cx.sepc += 4;
190            let result = syscall(cx.x[17], [cx.x[10], cx.x[11], cx.x[12]]);
191            cx = current_trap_cx();
192            cx.x[10] = result as usize;
193        }
194        Trap::Exception(Exception::StoreFault)
195        | Trap::Exception(Exception::StorePageFault)
196        | Trap::Exception(Exception::LoadFault)
197        | Trap::Exception(Exception::LoadPageFault)
198        | Trap::Exception(Exception::InstructionFault)
199        | Trap::Exception(Exception::InstructionPageFault) => {
200            // println!(
201            //     "[kernel] {:?} in application, bad addr = {:#x}, bad instruction = {:#x}, kernel killed it.",
202            //     scause.cause(),
203            //     stval,
204            //     current_trap_cx().sepc
205            // );
206            current_add_signal(SignalFlags::SIGSEGV);
207        }
208        Trap::Exception(Exception::IllegalInstruction) => {
209            // println!(
210            //     "[kernel] IllegalInstruction in application, bad instruction = {:#x}, kernel killed it.",
211            //     current_trap_cx().sepc
212            // );
213            current_add_signal(SignalFlags::SIGILL);
214        }
215        Trap::Interrupt(Interrupt::SupervisorTimer) => {
216            next_trigger();
217            suspend_current_and_run_next();
218        }
219        _ => {
220            panic!(
221                "Unsupported trap {:?}, stval = {:#x}!",
222                scause.cause(),
223                stval,
224            );
225        }
226    }
227
228    handle_signals();
229
230    if let Some((errno, msg)) = check_signals_error_of_current() {
231        println!("[kernel] {}", msg);
232        exit_current_and_run_next(errno);
233    }
234
235    trap_return();
236}

238/// 陷阱返回函数
239///
240/// 陷阱处理完成后返回用户态的函数，负责恢复用户态执行环境并
241/// 跳转回用户程序继续执行。这是陷阱处理的最后阶段。
242///
243/// ## 执行流程
244///
245/// 1. **设置用户陷阱入口**: 配置 `stvec` 指向 Trampoline
246/// 2. **准备返回参数**: 获取陷阱上下文地址和用户页表标识符
247/// 3. **跳转到 Trampoline**: 通过内联汇编跳转到 `__restore`
248/// 4. **恢复用户状态**: `__restore` 恢复所有寄存器并执行 `sret`
249///
250/// ## 地址空间切换
251///
252/// 函数执行过程中会发生地址空间切换：
253/// - 开始：内核地址空间
254/// - 跳转到 Trampoline：仍在内核地址空间（Trampoline 在两个地址空间都有映射）
255/// - `__restore` 执行 `sret`：切换到用户地址空间
256///
257/// ## Trampoline 机制
258///
259/// ```text
260/// 地址空间切换过程:
261/// ┌─────────────────┐    ┌─────────────────┐    ┌─────────────────┐
262/// │  Kernel Space   │    │   Trampoline    │    │   User Space    │
263/// │                 │───>│    (Shared)     │───>│                 │
264/// │  trap_return()  │    │   __restore     │    │  User Program   │
265/// └─────────────────┘    └─────────────────┘    └─────────────────┘
266/// ```
267///
268/// ## 内联汇编
269///
270/// 使用内联汇编执行关键的跳转操作：
271/// - `fence.i`: 指令缓存同步
272/// - `jr {restore_va}`: 跳转到 `__restore` 函数
273/// - 传递参数：`a0` = 陷阱上下文地址，`a1` = 用户页表标识符
274///
275/// ## 安全性
276///
277/// - 使用 `unsafe` 执行特权操作和内联汇编
278/// - 地址计算确保跳转到正确的 Trampoline 位置
279/// - 参数传递确保 `__restore` 获得正确的上下文信息
280///
281/// ## 属性说明
282///
283/// - `#[unsafe(no_mangle)]`: 防止函数名被混淆，可能被其他代码调用
284/// - `-> !`: 函数永不返回，总是跳转到用户程序执行
285#[unsafe(no_mangle)]
286pub fn trap_return() -> ! {
287    set_user_trap_entry();
288    let trap_cx_ptr = TRAP_CONTEXT;
289    let user_satp = current_user_token();
290    unsafe extern "C" {
291        fn __alltraps();
292        fn __restore();
293    }
294    let restore_va = __restore as usize - __alltraps as usize + TRAMPOLINE;
295    unsafe {
296        asm!(
297            "fence.i",
298            "jr {restore_va}",
299            restore_va = in(reg) restore_va,
300            in("a0") trap_cx_ptr,
301            in("a1") user_satp,
302            options(noreturn),
303        );
304    }
305}
