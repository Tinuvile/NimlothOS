1//! # 文件系统布局模块
2//!
3//! 定义了简单文件系统（Micro File System）的磁盘布局和数据结构。
4//! 包含超级块、磁盘 inode、目录项等核心数据结构的定义和操作。
5//!
6//! ## 文件系统布局
7//!
8//! ```text
9//! ┌─────────────────────────────────────────────────────────────┐
10//! │                        SuperBlock                           │
11//! ├─────────────────────────────────────────────────────────────┤
12//! │                       Inode Bitmap                          │
13//! ├─────────────────────────────────────────────────────────────┤
14//! │                        Inode Area                           │
15//! ├─────────────────────────────────────────────────────────────┤
16//! │                        Data Bitmap                          │
17//! ├─────────────────────────────────────────────────────────────┤
18//! │                         Data Area                           │
19//! └─────────────────────────────────────────────────────────────┘
20//! ```
21//!
22//! ## 核心组件
23//!
24//! - [`SuperBlock`] - 文件系统超级块，包含文件系统元数据
25//! - [`DiskInode`] - 磁盘 inode，管理文件的数据块分配
26//! - [`DirEntry`] - 目录项，用于目录文件的组织
27//!
28//! ## 数据块分配策略
29//!
30//! 采用三级索引结构：
31//! - **直接块**：前 28 个数据块直接存储在 inode 中
32//! - **一级间接块**：第 29-1024 个数据块通过一级间接块索引
33//! - **二级间接块**：第 1025 个及以后的数据块通过二级间接块索引
34//!
35//! ## 使用示例
36//!
37//! ```rust
38//! use micro_fs::layout::{SuperBlock, DiskInode, DirEntry};
39//!
40//! // 初始化超级块
41//! let mut sb = SuperBlock::default();
42//! sb.initialize(1024, 1, 8, 1, 100);
43//!
44//! // 创建文件 inode
45//! let mut inode = DiskInode::default();
46//! inode.initialize(DiskInodeType::File);
47//!
48//! // 创建目录项
49//! let entry = DirEntry::new("test.txt", 1);
50//! ```
51//!

56/// 文件系统魔数，用于标识 Micro File System
57const MFS_MAGIC: u32 = 0x3b800001;
58
59/// 文件名长度限制（不包括结尾的 null 字符）
60const NAME_LENGTH_LIMIT: usize = 27;
61
62/// 直接数据块数量
63const INODE_DIRECT_COUNT: usize = 27;
64
65/// 直接块边界
66const DIRECT_BOUND: usize = INODE_DIRECT_COUNT;
67
68/// 一级间接块能索引的数据块数量
69const INODE_INDIRECT1_COUNT: usize = BLOCK_SZ / 4;
70
71/// 一级间接块边界
72const INDIRECT1_BOUND: usize = DIRECT_BOUND + INODE_INDIRECT1_COUNT;
73
74/// 二级间接块能索引的数据块数量
75const INODE_INDIRECT2_COUNT: usize = INODE_INDIRECT1_COUNT * INODE_INDIRECT1_COUNT;
76
77/// 三级间接块能索引的数据块数量
78const INODE_INDIRECT3_COUNT: usize = INODE_INDIRECT2_COUNT * INODE_INDIRECT1_COUNT;
79
80/// 二级间接块边界
81#[allow(unused)]
82const INDIRECT2_BOUND: usize = INDIRECT1_BOUND + INODE_INDIRECT2_COUNT;
83
84/// 目录项大小（字节）
85pub const DIRENT_SZ: usize = 32;

87/// 间接块类型，存储块 ID 数组
88type IndirectBlock = [u32; BLOCK_SZ / 4];

90/// 数据块类型，存储文件数据

93/// 文件系统超级块
94///
95/// 存储文件系统的全局元数据，包括文件系统大小、各个区域的块数等信息。
96/// 超级块是文件系统的核心数据结构，用于文件系统的识别和初始化。
97///
98/// ## 布局结构
99///
100/// ```text
101/// ┌─────────┬─────────────────┬─────────────────┬─────────────────┬─────────────────┐
102/// │  Magic  │  Total Blocks   │  Inode Bitmap   │   Inode Area    │   Data Bitmap   │
103/// │  (4B)   │      (4B)       │      (4B)       │      (4B)       │       (4B)      │
104/// └─────────┴─────────────────┴─────────────────┴─────────────────┴─────────────────┘
105/// ```
106///
107/// ## 字段说明
108///
109/// - `magic` - 文件系统魔数，用于验证文件系统格式
110/// - `total_blocks` - 文件系统总块数
111/// - `inode_bitmap_blocks` - inode 位图占用的块数
112/// - `inode_area_blocks` - inode 区域占用的块数
113/// - `data_bitmap_blocks` - 数据位图占用的块数
114/// - `data_area_blocks` - 数据区域占用的块数
115#[repr(C)]
116pub struct SuperBlock {
117    magic: u32,
118    pub total_blocks: u32,
119    pub inode_bitmap_blocks: u32,
120    pub inode_area_blocks: u32,
121    pub data_bitmap_blocks: u32,
122    pub data_area_blocks: u32,
123}

125/// 磁盘 inode 结构
126///
127/// 存储文件或目录的元数据，包括文件大小、数据块分配信息等。
128/// 支持三级索引结构，能够管理大文件的数据块分配。
129///
130/// ## 数据块索引结构
131///
132/// ```text
133/// ┌───────────┬───────────────────┬────────────────┬────────────────┬────────────────┐
134/// │ Size (4B) │ Direct[27] (108B) │ Indirect1 (4B) │ Indirect2 (4B) │ Indirect3 (4B) │
135/// └───────────┴───────────────────┴────────────────┴────────────────┴────────────────┘
136/// ```
137///
138/// ## 索引策略
139///
140/// - **直接块**：前 27 个数据块直接存储在 `direct` 数组中
141/// - **一级间接块**：通过 `indirect1` 指向的块存储块 ID 数组
142/// - **二级间接块**：通过 `indirect2` 指向的块存储一级间接块的块 ID 数组
143/// - **三级间接块**：通过 `indirect3` 指向的块存储二级间接块的块 ID 数组
144///
145/// ## 最大文件大小
146///
147/// 理论上支持的最大文件大小：
148/// - 直接块：27 × 512B = 13.5KB
149/// - 一级间接块：128 × 512B = 64KB
150/// - 二级间接块：128 × 128 × 512B = 8MB
151/// - 三级间接块：128 × 128 × 128 × 512B = 1GB
152/// - 总计：约 1GB
153#[repr(C)]
154pub struct DiskInode {
155    pub size: u32,
156    pub direct: [u32; INODE_DIRECT_COUNT],
157    pub indirect1: u32,
158    pub indirect2: u32,
159    pub indirect3: u32,
160    type_: DiskInodeType,
161}
162

249    /// 获取指定逻辑块号对应的物理块 ID
250    ///
251    /// 根据三级索引结构查找逻辑块号对应的物理块 ID。
252    /// 支持直接块、一级间接块和二级间接块的查找。
253    ///
254    /// ## Arguments
255    /// * `inner_id` - 逻辑块号（从 0 开始）
256    /// * `block_device` - 块设备引用
257    ///
258    /// ## Returns
259    /// 对应的物理块 ID
260    ///
261    /// ## 索引策略
262    /// - 0-27：直接块，从 `direct` 数组获取
263    /// - 28-1023：一级间接块，通过 `indirect1` 查找
264    /// - 1024+：二级间接块，通过 `indirect2` 查找
265    pub fn block_id(&self, inner_id: u32, block_device: &Arc<dyn BlockDevice>) -> u32 {
266        let inner_id = inner_id as usize;
267        if inner_id < INODE_DIRECT_COUNT {
268            self.direct[inner_id]
269        } else if inner_id < INDIRECT1_BOUND {
270            block_cache(self.indirect1 as usize, Arc::clone(block_device))
271                .lock()
272                .read(0, |indirect_block: &IndirectBlock| {
273                    indirect_block[inner_id - INODE_DIRECT_COUNT]
274                })
275        } else if inner_id < INDIRECT2_BOUND {
276            let last = inner_id - INDIRECT1_BOUND;
277            let indirect1 = block_cache(self.indirect2 as usize, Arc::clone(block_device))
278                .lock()
279                .read(0, |indirect2: &IndirectBlock| {
280                    indirect2[last / INODE_INDIRECT1_COUNT]
281                });
282            block_cache(indirect1 as usize, Arc::clone(block_device))
283                .lock()
284                .read(0, |indirect1: &IndirectBlock| {
285                    indirect1[last % INODE_INDIRECT1_COUNT]
286                })
287        } else {
288            let last = inner_id - INDIRECT2_BOUND;
289            let indirect1 = block_cache(self.indirect3 as usize, Arc::clone(block_device))
290                .lock()
291                .read(0, |indirect3: &IndirectBlock| {
292                    indirect3[last / INODE_INDIRECT2_COUNT]
293                });
294            let indirect2 = block_cache(indirect1 as usize, Arc::clone(block_device))
295                .lock()
296                .read(0, |indirect2: &IndirectBlock| {
297                    indirect2[(last % INODE_INDIRECT2_COUNT) / INODE_INDIRECT1_COUNT]
298                });
299            block_cache(indirect2 as usize, Arc::clone(block_device))
300                .lock()
301                .read(0, |indirect1: &IndirectBlock| {
302                    indirect1[(last % INODE_INDIRECT2_COUNT) % INODE_INDIRECT1_COUNT]
303                })
304        }
305    }

307    /// 计算文件占用的数据块数量
308    ///

315    }

317    /// 根据文件大小计算数据块数量
318    ///
319    /// ## Arguments
320    /// * `size` - 文件大小（字节）
321    ///
322    /// ## Returns
323    /// 需要的数据块数量
324    fn _data_blocks(size: u32) -> u32 {
325        (size + BLOCK_SZ as u32 - 1) / BLOCK_SZ as u32
326    }

328    /// 计算文件占用的总块数（包括索引块）
329    ///
330    /// 计算文件占用的所有块数，包括数据块和索引块。
331    ///
332    /// ## Arguments
333    /// * `size` - 文件大小（字节）
334    ///
335    /// ## Returns
336    /// 文件占用的总块数
337    pub fn total_blocks(size: u32) -> u32 {
338        let data_blocks = Self::_data_blocks(size) as usize;
339        let mut total = data_blocks;
340        // indirect1
341        if data_blocks > INODE_DIRECT_COUNT {
342            total += 1;
343        }
344        // indirect2
345        if data_blocks > INDIRECT1_BOUND {
346            total += 1;
347            let level2_blocks =
348                (data_blocks - INDIRECT1_BOUND + INODE_INDIRECT1_COUNT - 1) / INODE_INDIRECT1_COUNT;
349            total += level2_blocks.min(INODE_INDIRECT1_COUNT);
350        }
351        // indirect3
352        if data_blocks > INDIRECT2_BOUND {
353            total += 1;
354            // sub indirect2
355            total +=
356                (data_blocks - INDIRECT2_BOUND + INODE_INDIRECT2_COUNT - 1) / INODE_INDIRECT2_COUNT;
357            // sub indirect1
358            total +=
359                (data_blocks - INDIRECT2_BOUND + INODE_INDIRECT1_COUNT - 1) / INODE_INDIRECT1_COUNT;
360        }
361        total as u32
362    }

364    /// 计算扩展文件需要的新块数
365    ///
366    /// 计算将文件从当前大小扩展到新大小需要的新块数。
367    ///
368    /// ## Arguments
369    /// * `new_size` - 新的文件大小
370    ///
371    /// ## Returns
372    /// 需要的新块数
373    ///
374    /// ## Panics
375    /// 如果 `new_size` 小于当前大小则 panic
376    pub fn blocks_num_needed(&self, new_size: u32) -> u32 {
377        assert!(new_size >= self.size);
378        Self::total_blocks(new_size) - Self::total_blocks(self.size)
379    }

381    /// 扩展文件大小并分配新块
382    ///
383    /// 将文件扩展到指定大小，并分配新的数据块。支持三级索引结构的
384    /// 块分配，自动处理直接块、一级间接块和二级间接块的分配。
385    ///
386    /// ## Arguments
387    /// * `new_size` - 新的文件大小
388    /// * `new_blocks` - 新分配的块 ID 列表
389    /// * `block_device` - 块设备引用
390    ///
391    /// ## 分配策略
392    /// 1. 优先使用直接块（前 28 个）
393    /// 2. 当直接块用完时，分配一级间接块
394    /// 3. 当一级间接块用完时，分配二级间接块
395    ///
396    /// ## 注意事项
397    /// - `new_blocks` 的长度必须等于 `blocks_num_needed(new_size)`
398    /// - 扩展操作是原子的，要么完全成功，要么完全失败
399    pub fn increase_size(
400        &mut self,
401        new_size: u32,
402        new_blocks: Vec<u32>,
403        block_device: &Arc<dyn BlockDevice>,
404    ) {
405        let mut current_blocks = self.data_blocks();
406        self.size = new_size;
407        let mut total_blocks = self.data_blocks();
408        let mut new_blocks = new_blocks.into_iter();
409
410        while current_blocks < total_blocks.min(INODE_DIRECT_COUNT as u32) {
411            self.direct[current_blocks as usize] = new_blocks.next().unwrap();
412            current_blocks += 1;
413        }
414
415        if total_blocks > INODE_DIRECT_COUNT as u32 {
416            if current_blocks == INODE_DIRECT_COUNT as u32 {
417                self.indirect1 = new_blocks.next().unwrap();
418            }
419            current_blocks -= INODE_DIRECT_COUNT as u32;
420            total_blocks -= INODE_DIRECT_COUNT as u32;
421        } else {
422            return;
423        }
424        block_cache(self.indirect1 as usize, Arc::clone(block_device))
425            .lock()
426            .modify(0, |indirect1: &mut IndirectBlock| {
427                while current_blocks < total_blocks.min(INODE_INDIRECT1_COUNT as u32) {
428                    indirect1[current_blocks as usize] = new_blocks.next().unwrap();
429                    current_blocks += 1;
430                }
431            });
432
433        if total_blocks > INODE_INDIRECT1_COUNT as u32 {
434            if current_blocks == INODE_INDIRECT1_COUNT as u32 {
435                self.indirect2 = new_blocks.next().unwrap();
436            }
437            current_blocks -= INODE_INDIRECT1_COUNT as u32;
438            total_blocks -= INODE_INDIRECT1_COUNT as u32;
439        } else {
440            return;
441        }
442        let mut a0 = current_blocks as usize / INODE_INDIRECT1_COUNT;
443        let mut b0 = current_blocks as usize % INODE_INDIRECT1_COUNT;
444        let a1 = total_blocks as usize / INODE_INDIRECT1_COUNT;
445        let b1 = total_blocks as usize % INODE_INDIRECT1_COUNT;
446        block_cache(self.indirect2 as usize, Arc::clone(block_device))
447            .lock()
448            .modify(0, |indirect2: &mut IndirectBlock| {
449                while (a0 < a1) || (a0 == a1 && b0 < b1) {
450                    if b0 == 0 {
451                        indirect2[a0] = new_blocks.next().unwrap();
452                    }
453                    block_cache(indirect2[a0] as usize, Arc::clone(block_device))
454                        .lock()
455                        .modify(0, |indirect1: &mut IndirectBlock| {
456                            indirect1[b0] = new_blocks.next().unwrap();
457                        });
458                    b0 += 1;
459                    if b0 == INODE_INDIRECT1_COUNT {
460                        b0 = 0;
461                        a0 += 1;
462                    }
463                }
464            });
465        // 处理三级间接索引：更新计数并按需分配/填充 indirect3
466        if total_blocks > INODE_INDIRECT2_COUNT as u32 {
467            if current_blocks == INODE_INDIRECT2_COUNT as u32 {
468                self.indirect3 = new_blocks.next().unwrap();
469            }
470            current_blocks -= INODE_INDIRECT2_COUNT as u32;
471            total_blocks -= INODE_INDIRECT2_COUNT as u32;
472        } else {
473            return;
474        }
475        // 填充 indirect3 (a0, b0, c0) -> (a1, b1, c1)
476        let mut a0 = current_blocks as usize / INODE_INDIRECT2_COUNT;
477        let mut b0 = (current_blocks as usize % INODE_INDIRECT2_COUNT) / INODE_INDIRECT1_COUNT;
478        let mut c0 = (current_blocks as usize % INODE_INDIRECT2_COUNT) % INODE_INDIRECT1_COUNT;
479        let a1 = total_blocks as usize / INODE_INDIRECT2_COUNT;
480        let b1 = (total_blocks as usize % INODE_INDIRECT2_COUNT) / INODE_INDIRECT1_COUNT;
481        let c1 = (total_blocks as usize % INODE_INDIRECT2_COUNT) % INODE_INDIRECT1_COUNT;
482        // 从 低层 -> 高层
483        block_cache(self.indirect3 as usize, Arc::clone(block_device))
484            .lock()
485            .modify(0, |indirect3: &mut IndirectBlock| {
486                while (a0 < a1) || (a0 == a1 && b0 < b1) || (a0 == a1 && b0 == b1 && c0 < c1) {
487                    if b0 == 0 {
488                        indirect3[a0] = new_blocks.next().unwrap();
489                    }
490                    // 填充当前 indirect3 的 a0 层对应的 indirect2
491                    block_cache(indirect3[a0] as usize, Arc::clone(block_device))
492                        .lock()
493                        .modify(0, |indirect2: &mut IndirectBlock| {
494                            if c0 == 0 {
495                                indirect2[b0] = new_blocks.next().unwrap();
496                            }
497                            // 填充当前 indirect2 的 b0 层对应的 indirect1
498                            block_cache(indirect2[b0] as usize, Arc::clone(block_device))
499                                .lock()
500                                .modify(0, |indirect1: &mut IndirectBlock| {
501                                    indirect1[c0] = new_blocks.next().unwrap();
502                                });
503                            // 移动到下一个需要分配的 block
504                            c0 += 1;
505                            if c0 == INODE_INDIRECT1_COUNT {
506                                c0 = 0;
507                                b0 += 1;
508                            }
509                        });
510                    // 移动到下一个需要分配的 block
511                    if b0 == INODE_INDIRECT2_COUNT {
512                        b0 = 0;
513                        a0 += 1;
514                    }
515                }
516            });
517    }

519    /// 清空文件内容并回收所有数据块
520    ///
521    /// 将文件大小重置为 0，清空所有数据块引用，并返回所有被释放的块 ID。
522    /// 支持三级索引结构的块回收。
523    ///
524    /// ## Arguments
525    /// * `block_device` - 块设备引用
526    ///
527    /// ## Returns
528    /// 被释放的块 ID 列表
529    ///
530    /// ## 回收策略
531    /// 1. 回收直接块
532    /// 2. 回收一级间接块及其指向的数据块
533    /// 3. 回收二级间接块及其指向的所有块
534    pub fn clear_size(&mut self, block_device: &Arc<dyn BlockDevice>) -> Vec<u32> {
535        let mut v = Vec::new();
536        let mut data_blocks = self.data_blocks() as usize;
537        self.size = 0;
538        let mut current_blocks = 0usize;
539
540        while current_blocks < data_blocks.min(INODE_DIRECT_COUNT) {
541            v.push(self.direct[current_blocks]);
542            self.direct[current_blocks] = 0;
543            current_blocks += 1;
544        }
545
546        if data_blocks > INODE_DIRECT_COUNT {
547            v.push(self.indirect1);
548            data_blocks -= INODE_DIRECT_COUNT;
549            current_blocks = 0;
550        } else {
551            return v;
552        }
553        block_cache(self.indirect1 as usize, Arc::clone(block_device))
554            .lock()
555            .modify(0, |indirect1: &mut IndirectBlock| {
556                while current_blocks < data_blocks.min(INODE_INDIRECT1_COUNT) {
557                    v.push(indirect1[current_blocks]);
558                    current_blocks += 1;
559                }
560            });
561        self.indirect1 = 0;
562
563        if data_blocks > INODE_INDIRECT1_COUNT {
564            v.push(self.indirect2);
565            data_blocks -= INODE_INDIRECT1_COUNT;
566        } else {
567            return v;
568        }
569        assert!(data_blocks <= INODE_INDIRECT2_COUNT);
570        let a1 = data_blocks / INODE_INDIRECT1_COUNT;
571        let b1 = data_blocks % INODE_INDIRECT1_COUNT;
572        block_cache(self.indirect2 as usize, Arc::clone(block_device))
573            .lock()
574            .modify(0, |indirect2: &mut IndirectBlock| {
575                for entry in indirect2.iter_mut().take(a1) {
576                    v.push(*entry);
577                    block_cache(*entry as usize, Arc::clone(block_device))
578                        .lock()
579                        .modify(0, |indirect1: &mut IndirectBlock| {
580                            for entry in indirect1.iter() {
581                                v.push(*entry);
582                            }
583                        });
584                }
585                if b1 > 0 {
586                    v.push(indirect2[a1]);
587                    block_cache(indirect2[a1] as usize, Arc::clone(block_device))
588                        .lock()
589                        .modify(0, |indirect1: &mut IndirectBlock| {
590                            for entry in indirect1.iter().take(b1) {
591                                v.push(*entry);
592                            }
593                        });
594                }
595            });
596        self.indirect2 = 0;
597        assert!(data_blocks <= INODE_INDIRECT3_COUNT);
598        if data_blocks > INODE_INDIRECT2_COUNT {
599            v.push(self.indirect3);
600            data_blocks -= INODE_INDIRECT2_COUNT;
601        } else {
602            return v;
603        }
604        // 释放 indirect3
605        assert!(data_blocks <= INODE_INDIRECT3_COUNT);
606        let a2 = data_blocks / INODE_INDIRECT2_COUNT;
607        let b2 = (data_blocks % INODE_INDIRECT2_COUNT) / INODE_INDIRECT1_COUNT;
608        let c2 = (data_blocks % INODE_INDIRECT2_COUNT) % INODE_INDIRECT1_COUNT;
609        block_cache(self.indirect3 as usize, Arc::clone(block_device))
610            .lock()
611            .modify(0, |indirect3: &mut IndirectBlock| {
612                // 满的 indirect2 释放
613                for entry in indirect3.iter().take(a2) {
614                    v.push(*entry);
615                    block_cache(*entry as usize, Arc::clone(block_device))
616                        .lock()
617                        .modify(0, |indirect2: &mut IndirectBlock| {
618                            // 满的 indirect1 释放
619                            for entry in indirect2.iter() {
620                                v.push(*entry);
621                                block_cache(*entry as usize, Arc::clone(block_device))
622                                    .lock()
623                                    .modify(0, |indirect1: &mut IndirectBlock| {
624                                        for entry in indirect1.iter() {
625                                            v.push(*entry);
626                                        }
627                                    });
628                            }
629                        });
630                    // 最后一个 indirect2 释放
631                    if b2 > 0 {
632                        v.push(indirect3[a2]);
633                        block_cache(indirect3[a2] as usize, Arc::clone(block_device))
634                            .lock()
635                            .modify(0, |indirect2: &mut IndirectBlock| {
636                                // 满的 indirect1 释放
637                                for entry in indirect2.iter().take(b2) {
638                                    v.push(*entry);
639                                    block_cache(*entry as usize, Arc::clone(block_device))
640                                        .lock()
641                                        .modify(0, |indirect1: &mut IndirectBlock| {
642                                            for entry in indirect1.iter() {
643                                                v.push(*entry);
644                                            }
645                                        });
646                                }
647                                // 最后一个 indirect1 释放
648                                if c2 > 0 {
649                                    v.push(indirect2[b2]);
650                                    block_cache(indirect2[b2] as usize, Arc::clone(block_device))
651                                        .lock()
652                                        .modify(0, |indirect1: &mut IndirectBlock| {
653                                            for entry in indirect1.iter().take(c2) {
654                                                v.push(*entry);
655                                            }
656                                        });
657                                }
658                            });
659                    }
660                }
661            });
662        self.indirect3 = 0;
663        v
664    }

666    /// 从指定偏移量读取文件数据
667    ///
668    /// 从文件的指定偏移量开始读取数据到缓冲区中。支持跨块读取，
669    /// 自动处理块边界和数据对齐。
670    ///
671    /// ## Arguments
672    /// * `offset` - 读取起始偏移量
673    /// * `buf` - 数据缓冲区
674    /// * `block_device` - 块设备引用
675    ///
676    /// ## Returns
677    /// 实际读取的字节数
678    ///
679    /// ## 读取策略
680    /// 1. 计算起始块号和结束块号
681    /// 2. 逐块读取数据
682    /// 3. 处理块内偏移和跨块读取
683    ///
684    /// ## 边界处理
685    /// - 如果偏移量超出文件大小，返回 0
686    /// - 如果缓冲区大小超出文件剩余部分，只读取到文件末尾
687    pub fn read_at(
688        &self,
689        offset: usize,
690        buf: &mut [u8],
691        block_device: &Arc<dyn BlockDevice>,
692    ) -> usize {
693        let mut start = offset;
694        let end = (offset + buf.len()).min(self.size as usize);
695        if start >= end {
696            return 0;
697        }
698
699        let mut start_block = start / BLOCK_SZ;
700        let mut read_size = 0usize;
701        loop {
702            let mut end_current_block = (start / BLOCK_SZ + 1) * BLOCK_SZ;
703            end_current_block = end_current_block.min(end);
704
705            let block_read_size = end_current_block - start;
706            let dst = &mut buf[read_size..read_size + block_read_size];
707            block_cache(
708                self.block_id(start_block as u32, block_device) as usize,
709                Arc::clone(block_device),
710            )
711            .lock()
712            .read(0, |data_block: &DataBlock| {
713                let src = &data_block[start % BLOCK_SZ..start % BLOCK_SZ + block_read_size];
714                dst.copy_from_slice(src);
715            });
716            read_size += block_read_size;
717            if end_current_block == end {
718                break;
719            }
720            start_block += 1;
721            start = end_current_block;
722        }
723        read_size
724    }
725
726    /// 向指定偏移量写入文件数据
727    ///
728    /// 将缓冲区中的数据写入文件的指定偏移量位置。支持跨块写入，
729    /// 自动处理块边界和数据对齐。
730    ///
731    /// ## Arguments
732    /// * `offset` - 写入起始偏移量
733    /// * `buf` - 要写入的数据
734    /// * `block_device` - 块设备引用
735    ///
736    /// ## Returns
737    /// 实际写入的字节数
738    ///
739    /// ## 写入策略
740    /// 1. 计算起始块号和结束块号
741    /// 2. 逐块写入数据
742    /// 3. 处理块内偏移和跨块写入
743    ///
744    /// ## 注意事项
745    /// - 写入操作会修改文件内容
746    /// - 如果写入位置超出当前文件大小，需要先扩展文件
747    /// - 写入是原子的，要么完全成功，要么完全失败
748    pub fn write_at(
749        &mut self,
750        offset: usize,
751        buf: &[u8],
752        block_device: &Arc<dyn BlockDevice>,
753    ) -> usize {
754        let mut start = offset;
755        let end = (offset + buf.len()).min(self.size as usize);
756        assert!(start <= end);
757
758        let mut start_block = start / BLOCK_SZ;
759        let mut write_size = 0usize;
760
761        loop {
762            let mut end_current_block = (start / BLOCK_SZ + 1) * BLOCK_SZ;
763            end_current_block = end_current_block.min(end);
764
765            let block_write_size = end_current_block - start;
766            block_cache(
767                self.block_id(start_block as u32, block_device) as usize,
768                Arc::clone(block_device),
769            )
770            .lock()
771            .modify(0, |data_block: &mut DataBlock| {
772                let src = &buf[write_size..write_size + block_write_size];
773                let dst = &mut data_block[start % BLOCK_SZ..start % BLOCK_SZ + block_write_size];
774                dst.copy_from_slice(src);
775            });
776            write_size += block_write_size;
777            if end_current_block == end {
778                break;
779            }
780            start_block += 1;
781            start = end_current_block;
782        }
783        write_size
784    }
785}

787/// 目录项结构
788///
789/// 用于在目录文件中存储文件和子目录的信息。每个目录项包含文件名
790/// 和对应的 inode 号，支持目录的层次结构组织。
791///
792/// ## 布局结构
793///
794/// ```text
795/// ┌────────────┬─────────────┐
796/// │ Name (28B) │ Inode (4B)  │
797/// └────-───────┴─────────────┘
798/// ```
799///
800/// ## 字段说明
801///
802/// - `name` - 文件名，固定长度 28 字节，以 null 结尾
803/// - `inode_number` - 对应的 inode 号
804///
805/// ## 大小限制
806///
807/// - 文件名最大长度：27 个字符（不包括结尾的 null）
808/// - 目录项总大小：32 字节
809#[repr(C)]
810pub struct DirEntry {
811    name: [u8; NAME_LENGTH_LIMIT + 1],
812    inode_number: u32,
813}
814
815impl DirEntry {
816    /// 创建空的目录项
817    ///
818    /// 创建一个未使用的目录项，文件名为空，inode 号为 0。
819    ///
820    /// ## Returns
821    /// 空的目录项
822    pub fn empty() -> Self {
823        Self {
824            name: [0u8; NAME_LENGTH_LIMIT + 1],
825            inode_number: 0,
826        }
827    }
828
829    /// 创建新的目录项
830    ///
831    /// 使用指定的文件名和 inode 号创建目录项。
832    ///
833    /// ## Arguments
834    /// * `name` - 文件名
835    /// * `inode_number` - 对应的 inode 号
836    ///
837    /// ## 注意事项
838    /// - 文件名长度不能超过 27 个字符
839    /// - 如果文件名过长，会被截断
840    pub fn new(name: &str, inode_number: u32) -> Self {
841        let mut bytes = [0u8; NAME_LENGTH_LIMIT + 1];
842        bytes[..name.len()].copy_from_slice(name.as_bytes());
843        Self {
844            name: bytes,
845            inode_number,
846        }
847    }
848
849    /// 获取目录项的字节表示
850    ///
851    /// 将目录项转换为字节切片，用于磁盘存储。
852    ///
853    /// ## Returns
854    /// 目录项的字节表示
855    ///
856    /// ## Safety
857    /// 返回的字节切片直接对应结构体的内存布局
858    pub fn as_bytes(&self) -> &[u8] {
859        unsafe { core::slice::from_raw_parts(self as *const _ as usize as *const u8, DIRENT_SZ) }
860    }
861
862    /// 获取目录项的可变字节表示
863    ///
864    /// 将目录项转换为可变字节切片，用于磁盘写入。
865    ///
866    /// ## Returns
867    /// 目录项的可变字节表示
868    ///
869    /// ## Safety
870    /// 返回的字节切片直接对应结构体的内存布局
871    pub fn as_bytes_mut(&mut self) -> &mut [u8] {
872        unsafe { core::slice::from_raw_parts_mut(self as *mut _ as usize as *mut u8, DIRENT_SZ) }
873    }
874
875    /// 获取文件名
876    ///
877    /// 从目录项中提取文件名，自动处理 null 结尾。
878    ///
879    /// ## Returns
880    /// 文件名字符串
881    ///
882    /// ## 注意事项
883    /// - 文件名以 null 字符结尾
884    /// - 返回的字符串不包含 null 字符
885    pub fn name(&self) -> &str {
886        let len = (0usize..).find(|i| self.name[*i] == 0).unwrap();
887        core::str::from_utf8(&self.name[..len]).unwrap()
888    }
889
890    /// 获取 inode 号
891    ///
892    /// ## Returns
893    /// 对应的 inode 号
894    pub fn inode_number(&self) -> u32 {
895        self.inode_number
896    }
