1//! # 虚拟文件系统接口模块
2//!
3//! 提供 Micro File System 的虚拟文件系统（VFS）接口，为用户程序提供统一的文件操作 API。
4//! 封装了底层文件系统的复杂性，提供简洁易用的文件管理接口。
5//!
6//! ## 架构设计
7//!
8//! ```text
9//! ┌─────────────────────────────────────────────────────────────┐
10//! │                     User Application                        │
11//! ├─────────────────────────────────────────────────────────────┤
12//! │                     VFS Interface                           │
13//! │  ┌─────────────┬─────────────┬─────────────┬─────────────┐  │
14//! │  │    find     │    create   │   read_at   │  write_at   │  │
15//! │  └─────────────┴─────────────┴─────────────┴─────────────┘  │
16//! ├─────────────────────────────────────────────────────────────┤
17//! │                     Micro File System                        │
18//! └─────────────────────────────────────────────────────────────┘
19//! ```
20//!
21//! ## 核心功能
22//!
23//! - **文件查找**: 在目录中查找指定名称的文件或子目录
24//! - **文件创建**: 创建新的文件或目录
25//! - **文件读写**: 支持随机访问的文件读写操作
26//! - **目录管理**: 列出目录内容，管理目录结构
27//! - **文件清理**: 清空文件内容，回收存储空间
28//!
29//! ## 使用示例
30//!
31//! ```rust
32//! use micro_fs::vfs::Inode;
33//!
34//! // 查找文件
35//! if let Some(file) = root_inode.find("test.txt") {
36//!     // 读取文件内容
37//!     let mut buf = [0u8; 1024];
38//!     let bytes_read = file.read_at(0, &mut buf);
39//! }
40//!
41//! // 创建新文件
42//! if let Some(new_file) = root_inode.create("new.txt") {
43//!     // 写入数据
44//!     new_file.write_at(0, b"Hello, World!");
45//! }
46//!
47//! // 列出目录内容
48//! let files = root_inode.ls();
49//! for file in files {
50//!     println!("文件: {}", file);
51//! }
52//! ```

61/// 文件系统 inode 接口
62///
63/// 表示文件系统中的一个文件或目录，提供统一的文件操作接口。
64/// 封装了底层磁盘 inode 的复杂性，提供类型安全和易用的 API。
65///
66/// ## 内存布局
67///
68/// ```text
69/// ┌─────────────┬─────────────────┬─────────────────┬─────────────────┐
70/// │  Block ID   │   Block Offset  │   File System   │   Block Device  │
71/// │   usize     │     usize       │      Arc        │      Arc        │
72/// └─────────────┴─────────────────┴─────────────────┴─────────────────┘
73/// ```
74///
75/// ## 字段说明
76///
77/// - `block_id` - 磁盘 inode 所在的块号
78/// - `block_offset` - 磁盘 inode 在块内的偏移量
79/// - `fs` - 文件系统实例的引用计数智能指针
80/// - `block_device` - 块设备接口的引用计数智能指针
81///
82/// ## 生命周期管理
83///
84/// 使用引用计数智能指针管理文件系统和块设备的生命周期，
85/// 确保在 inode 存在期间相关资源不会被释放。
86///
87/// ## 并发安全
88///
89/// 通过文件系统内部的互斥锁保证并发访问的安全性。
90/// 多个线程可以同时访问不同的 inode，但同一 inode 的并发访问会被序列化。
91pub struct Inode {
92    block_id: usize,
93    block_offset: usize,
94    fs: Arc<Mutex<MicroFileSystem>>,
95    block_device: Arc<dyn BlockDevice>,
96}

132    /// 读取磁盘 inode 数据
133    ///
134    /// 从块缓存中读取磁盘 inode 数据，并通过闭包函数进行处理。
135    /// 提供类型安全的磁盘 inode 访问接口。
136    ///
137    /// ## Type Parameters
138    ///
139    /// * `V` - 闭包函数的返回类型
140    ///
141    /// ## Arguments
142    ///
143    /// * `f` - 处理磁盘 inode 数据的闭包函数
144    ///
145    /// ## Returns
146    ///
147    /// 闭包函数的返回值
148    ///
149    /// ## Examples
150    ///
151    /// ```
152    /// let size = self.read_disk_inode(|disk_inode| {
153    ///     disk_inode.size
154    /// });
155    /// ```
156    fn read_disk_inode<V>(&self, f: impl FnOnce(&DiskInode) -> V) -> V {
157        block_cache(self.block_id, Arc::clone(&self.block_device))
158            .lock()
159            .read(self.block_offset, f)
160    }

163    ///
164    /// 从块缓存中获取磁盘 inode 的可变引用，并通过闭包函数进行修改。
165    /// 提供类型安全的磁盘 inode 修改接口。
166    ///
167    /// ## Type Parameters
168    ///
169    /// * `V` - 闭包函数的返回类型
170    ///
171    /// ## Arguments
172    ///
173    /// * `f` - 修改磁盘 inode 数据的闭包函数
174    ///
175    /// ## Returns
176    ///
177    /// 闭包函数的返回值
178    ///
179    /// ## 副作用
180    ///
181    /// 调用此方法会修改磁盘 inode 的内容，修改会被缓存并最终写回磁盘
182    ///
183    /// ## Examples
184    ///
185    /// ```
186    /// self.modify_disk_inode(|disk_inode| {
187    ///     disk_inode.size = new_size;
188    /// });
189    /// ```
190    fn modify_disk_inode<V>(&self, f: impl FnOnce(&mut DiskInode) -> V) -> V {
191        block_cache(self.block_id, Arc::clone(&self.block_device))
192            .lock()
193            .modify(self.block_offset, f)
194    }

196    /// 查找指定名称的文件或目录
197    ///
198    /// 在当前目录中查找指定名称的文件或子目录。
199    /// 如果找到，返回对应的 inode 接口对象。
200    ///
201    /// ## Arguments
202    ///
203    /// * `name` - 要查找的文件或目录名称
204    ///
205    /// ## Returns
206    ///
207    /// - `Some(inode)` - 找到文件或目录，返回对应的 inode
208    /// - `None` - 未找到指定名称的文件或目录
209    ///
210    /// ## 查找过程
211    ///
212    /// 1. 获取文件系统锁，确保并发安全
213    /// 2. 读取当前目录的磁盘 inode
214    /// 3. 遍历目录项，查找匹配的文件名
215    /// 4. 如果找到，创建并返回对应的 inode 接口
216    ///
217    /// ## 注意事项
218    ///
219    /// - 当前 inode 必须是目录类型
220    /// - 文件名区分大小写
221    /// - 返回的 inode 共享相同的文件系统和块设备引用
222    ///
223    /// ## Examples
224    ///
225    /// ```
226    /// if let Some(file) = root_inode.find("config.txt") {
227    ///     println!("找到文件: config.txt");
228    ///     // 使用 file 进行文件操作
229    /// }
230    /// ```
231    pub fn find(&self, name: &str) -> Option<Arc<Inode>> {
232        let fs = self.fs.lock();
233        self.read_disk_inode(|disk_inode| {
234            self.find_inode_id(name, disk_inode).map(|inode_id| {
235                let (block_id, block_offset) = fs.disk_inode_pos(inode_id);
236                Arc::new(Self::new(
237                    block_id,
238                    block_offset,
239                    self.fs.clone(),
240                    self.block_device.clone(),
241                ))
242            })
243        })
244    }

246    /// 在目录中查找指定名称的 inode ID
247    ///
248    /// 遍历目录项，查找指定名称对应的 inode 号。
249    /// 这是 `find` 方法的内部实现。
250    ///
251    /// ## Arguments
252    ///
253    /// * `name` - 要查找的文件或目录名称
254    /// * `disk_inode` - 当前目录的磁盘 inode
255    ///
256    /// ## Returns
257    ///
258    /// - `Some(inode_id)` - 找到对应的 inode 号
259    /// - `None` - 未找到指定名称的文件或目录
260    ///
261    /// ## 查找算法
262    ///
263    /// 1. 计算目录中的文件数量
264    /// 2. 遍历所有目录项
265    /// 3. 比较目录项名称与目标名称
266    /// 4. 找到匹配项时返回对应的 inode 号
267    ///
268    /// ## Panics
269    ///
270    /// 如果当前 inode 不是目录类型，则触发 panic
271    fn find_inode_id(&self, name: &str, disk_inode: &DiskInode) -> Option<u32> {
272        assert!(disk_inode.dir());
273        let file_count = (disk_inode.size as usize) / DIRENT_SZ;
274        let mut dirent = DirEntry::empty();
275        for i in 0..file_count {
276            assert_eq!(
277                disk_inode.read_at(DIRENT_SZ * i, dirent.as_bytes_mut(), &self.block_device,),
278                DIRENT_SZ,
279            );
280            if dirent.name() == name {
281                return Some(dirent.inode_number() as u32);
282            }
283        }
284        None
285    }

287    /// 列出目录内容
288    ///
289    /// 返回当前目录中所有文件和子目录的名称列表。
290    ///
291    /// ## Returns
292    ///
293    /// 包含所有文件和目录名称的字符串向量
294    ///
295    /// ## 列出过程
296    ///
297    /// 1. 获取文件系统锁，确保并发安全
298    /// 2. 读取当前目录的磁盘 inode
299    /// 3. 计算目录中的文件数量
300    /// 4. 遍历所有目录项，提取文件名
301    /// 5. 返回文件名列表
302    ///
303    /// ## 注意事项
304    ///
305    /// - 当前 inode 必须是目录类型
306    /// - 返回的名称不包含路径信息
307    /// - 名称按目录项存储顺序返回
308    ///
309    /// ## Examples
310    ///
311    /// ```
312    /// let files = root_inode.ls();
313    /// for file in files {
314    ///     println!("文件: {}", file);
315    /// }
316    /// ```
317    pub fn ls(&self) -> Vec<String> {
318        let _fs = self.fs.lock();
319        self.read_disk_inode(|disk_inode| {
320            let file_count = (disk_inode.size as usize) / DIRENT_SZ;
321            let mut v: Vec<String> = Vec::new();
322            for i in 0..file_count {
323                let mut dirent = DirEntry::empty();
324                assert_eq!(
325                    disk_inode.read_at(i * DIRENT_SZ, dirent.as_bytes_mut(), &self.block_device,),
326                    DIRENT_SZ,
327                );
328                v.push(String::from(dirent.name()));
329            }
330            v
331        })
332    }

334    /// 创建新文件
335    ///
336    /// 在当前目录中创建指定名称的新文件。
337    /// 如果文件已存在，返回 `None`。
338    ///
339    /// ## Arguments
340    ///
341    /// * `name` - 新文件的名称
342    ///
343    /// ## Returns
344    ///
345    /// - `Some(inode)` - 成功创建文件，返回新文件的 inode
346    /// - `None` - 文件已存在，创建失败
347    ///
348    /// ## 创建过程
349    ///
350    /// 1. 检查文件是否已存在
351    /// 2. 分配新的 inode
352    /// 3. 初始化新 inode 为文件类型
353    /// 4. 在当前目录中添加目录项
354    /// 5. 同步缓存到磁盘
355    /// 6. 返回新文件的 inode
356    ///
357    /// ## 注意事项
358    ///
359    /// - 当前 inode 必须是目录类型
360    /// - 文件名不能超过 27 个字符
361    /// - 创建操作是原子的，要么完全成功，要么完全失败
362    ///
363    /// ## Examples
364    ///
365    /// ```
366    /// if let Some(new_file) = root_inode.create("new.txt") {
367    ///     println!("成功创建文件: new.txt");
368    ///     // 使用 new_file 进行文件操作
369    /// } else {
370    ///     println!("文件已存在或创建失败");
371    /// }
372    /// ```
373    pub fn create(&self, name: &str) -> Option<Arc<Inode>> {
374        let mut fs = self.fs.lock();
375        let op = |root_inode: &DiskInode| {
376            assert!(root_inode.dir());
377            self.find_inode_id(name, root_inode)
378        };
379        if self.read_disk_inode(op).is_some() {
380            return None;
381        }
382
383        let new_inode_id = fs.alloc_inode();
384        let (new_inode_block_id, new_inode_block_offset) = fs.disk_inode_pos(new_inode_id);
385        block_cache(new_inode_block_id as usize, Arc::clone(&self.block_device))
386            .lock()
387            .modify(new_inode_block_offset, |new_inode: &mut DiskInode| {
388                new_inode.initialize(DiskInodeType::File);
389            });
390        self.modify_disk_inode(|root_inode| {
391            let file_count = (root_inode.size as usize) / DIRENT_SZ;
392            let new_size = (file_count + 1) * DIRENT_SZ;
393            self.increase_size(new_size as u32, root_inode, &mut fs);
394            let dirent = DirEntry::new(name, new_inode_id);
395            root_inode.write_at(
396                file_count * DIRENT_SZ,
397                dirent.as_bytes(),
398                &self.block_device,
399            );
400        });
401
402        let (block_id, block_offset) = fs.disk_inode_pos(new_inode_id);
403        block_cache_sync_all();
404        Some(Arc::new(Self::new(
405            block_id,
406            block_offset,
407            self.fs.clone(),
408            self.block_device.clone(),
409        )))
410    }

412    /// 扩展文件大小
413    ///
414    /// 将文件扩展到指定大小，并分配必要的数据块。
415    /// 这是文件创建和写入操作的内部辅助方法。
416    ///
417    /// ## Arguments
418    ///
419    /// * `new_size` - 新的文件大小
420    /// * `disk_inode` - 要扩展的磁盘 inode
421    /// * `fs` - 文件系统的可变引用
422    ///
423    /// ## 扩展过程
424    ///
425    /// 1. 检查新大小是否大于当前大小
426    /// 2. 计算需要的新块数
427    /// 3. 分配新的数据块
428    /// 4. 更新磁盘 inode 的块分配信息
429    ///
430    /// ## 注意事项
431    ///
432    /// - 只有在需要扩展时才执行分配操作
433    /// - 分配操作通过文件系统管理器进行
434    /// - 扩展操作是原子的
435    fn increase_size(
436        &self,
437        new_size: u32,
438        disk_inode: &mut DiskInode,
439        fs: &mut MutexGuard<MicroFileSystem>,
440    ) {
441        if new_size < disk_inode.size {
442            return;
443        }
444        let blocks_needed = disk_inode.blocks_num_needed(new_size);
445        let mut v: Vec<u32> = Vec::new();
446        for _ in 0..blocks_needed {
447            v.push(fs.alloc_data());
448        }
449        disk_inode.increase_size(new_size, v, &self.block_device);
450    }

452    /// 清空文件内容
453    ///
454    /// 将文件大小重置为 0，清空所有数据块，并回收存储空间。
455    /// 文件本身不会被删除，只是内容被清空。
456    ///
457    /// ## 清空过程
458    ///
459    /// 1. 获取文件系统锁
460    /// 2. 读取当前文件大小
461    /// 3. 清空磁盘 inode 的块分配信息
462    /// 4. 回收所有数据块（当前实现中未实际回收）
463    /// 5. 同步缓存到磁盘
464    ///
465    /// ## 注意事项
466    ///
467    /// - 清空操作是原子的
468    /// - 文件类型和权限保持不变
469    /// - 当前实现中数据块回收功能未完全实现
470    ///
471    /// ## Examples
472    ///
473    /// ```
474    /// file.clear(); // 清空文件内容
475    /// ```
476    pub fn clear(&self) {
477        let mut fs = self.fs.lock();
478        self.modify_disk_inode(|disk_inode| {
479            let size = disk_inode.size;
480            let data_blocks_dealloc = disk_inode.clear_size(&self.block_device);
481            assert!(data_blocks_dealloc.len() == DiskInode::total_blocks(size) as usize);
482            for data_block in data_blocks_dealloc.into_iter() {
483                fs.dealloc_data(data_block);
484            }
485        });
486        block_cache_sync_all();
487    }

489    /// 从指定偏移量读取文件数据
490    ///
491    /// 从文件的指定偏移量开始读取数据到缓冲区中。
492    /// 支持跨块读取，自动处理块边界。
493    ///
494    /// ## Arguments
495    ///
496    /// * `offset` - 读取起始偏移量
497    /// * `buf` - 数据缓冲区
498    ///
499    /// ## Returns
500    ///
501    /// 实际读取的字节数
502    ///
503    /// ## 读取策略
504    ///
505    /// 1. 获取文件系统锁，确保并发安全
506    /// 2. 读取磁盘 inode 获取文件大小
507    /// 3. 计算起始块号和结束块号
508    /// 4. 逐块读取数据到缓冲区
509    /// 5. 处理块内偏移和跨块读取
510    ///
511    /// ## 边界处理
512    ///
513    /// - 如果偏移量超出文件大小，返回 0
514    /// - 如果缓冲区大小超出文件剩余部分，只读取到文件末尾
515    /// - 支持读取 0 字节（返回 0）
516    ///
517    /// ## Examples
518    ///
519    /// ```
520    /// let mut buf = [0u8; 1024];
521    /// let bytes_read = file.read_at(0, &mut buf);
522    /// println!("读取了 {} 字节", bytes_read);
523    /// ```
524    pub fn read_at(&self, offset: usize, buf: &mut [u8]) -> usize {
525        let _fs = self.fs.lock();
526        self.read_disk_inode(|disk_inode| disk_inode.read_at(offset, buf, &self.block_device))
527    }

529    /// 向指定偏移量写入文件数据
530    ///
531    /// 将缓冲区中的数据写入文件的指定偏移量位置。
532    /// 支持跨块写入，自动扩展文件大小。
533    ///
534    /// ## Arguments
535    ///
536    /// * `offset` - 写入起始偏移量
537    /// * `buf` - 要写入的数据
538    ///
539    /// ## Returns
540    ///
541    /// 实际写入的字节数
542    ///
543    /// ## 写入策略
544    ///
545    /// 1. 获取文件系统锁
546    /// 2. 计算新的文件大小
547    /// 3. 如果需要，扩展文件大小并分配新块
548    /// 4. 逐块写入数据
549    /// 5. 同步缓存到磁盘
550    ///
551    /// ## 扩展行为
552    ///
553    /// - 如果写入位置超出当前文件大小，自动扩展文件
554    /// - 扩展操作会分配必要的数据块
555    /// - 写入操作是原子的
556    ///
557    /// ## 注意事项
558    ///
559    /// - 写入操作会修改文件内容
560    /// - 支持写入 0 字节（返回 0）
561    /// - 写入完成后会自动同步到磁盘
562    ///
563    /// ## Examples
564    ///
565    /// ```
566    /// let data = b"Hello, World!";
567    /// let bytes_written = file.write_at(0, data);
568    /// println!("写入了 {} 字节", bytes_written);
569    /// ```
570    pub fn write_at(&self, offset: usize, buf: &[u8]) -> usize {
571        let mut fs = self.fs.lock();
572        let size = self.modify_disk_inode(|disk_inode| {
573            self.increase_size((offset + buf.len()) as u32, disk_inode, &mut fs);
574            disk_inode.write_at(offset, buf, &self.block_device)
575        });
576        block_cache_sync_all();
577        size
578    }
