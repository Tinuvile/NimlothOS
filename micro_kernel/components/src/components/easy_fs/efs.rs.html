1//! # Easy File System 核心模块
2//!
3//! 实现了简单文件系统（Easy File System）的核心功能，包括文件系统的创建、
4//! 打开、inode 和数据块的管理等。该模块是文件系统的核心，负责协调各个
5//! 组件的工作。
6//!
7//! ## 文件系统结构
8//!
9//! ```text
10//! ┌─────────────────────────────────────────────────────────────┐
11//! │                        SuperBlock                           │
12//! ├─────────────────────────────────────────────────────────────┤
13//! │                       Inode Bitmap                          │
14//! ├─────────────────────────────────────────────────────────────┤
15//! │                        Inode Area                           │
16//! ├─────────────────────────────────────────────────────────────┤
17//! │                        Data Bitmap                          │
18//! ├─────────────────────────────────────────────────────────────┤
19//! │                         Data Area                           │
20//! └─────────────────────────────────────────────────────────────┘
21//! ```
22//!
23//! ## 核心组件
24//!
25//! - [`EasyFileSystem`] - 文件系统主体，管理所有文件系统操作
26//! - 位图管理：inode 位图和数据位图，用于资源分配
27//! - 区域管理：inode 区域和数据区域的位置计算
28//!
29//! ## 主要功能
30//!
31//! - **文件系统创建**：格式化块设备，创建文件系统结构
32//! - **文件系统打开**：从现有块设备加载文件系统
33//! - **资源分配**：inode 和数据块的分配与回收
34//! - **根目录管理**：提供根目录的访问接口
35//!
36//! ## 使用示例
37//!
38//! ```rust
39//! use easy_fs::{EasyFileSystem, BlockDevice};
40//!
41//! // 创建新的文件系统
42//! let efs = EasyFileSystem::create(block_device, 1024, 1);
43//!
44//! // 打开现有文件系统
45//! let efs = EasyFileSystem::open(block_device);
46//!
47//! // 获取根目录
48//! let root_inode = EasyFileSystem::root_inode(&efs);
49//! ```
50//!

59/// Easy File System 主体结构
60///
61/// 管理整个文件系统的状态和操作，包括位图管理、区域定位、资源分配等。
62/// 该结构是文件系统的核心，所有文件系统操作都通过它进行。
63///
64/// ## 字段说明
65///
66/// - `block_device` - 底层块设备，提供存储能力
67/// - `inode_bitmap` - inode 位图，管理 inode 的分配状态
68/// - `data_bitmap` - 数据位图，管理数据块的分配状态
69/// - `inode_area_start_block` - inode 区域的起始块号
70/// - `data_area_start_block` - 数据区域的起始块号
71///
72/// ## 生命周期
73///
74/// 文件系统的生命周期包括：
75/// 1. **创建阶段**：格式化块设备，初始化所有数据结构
76/// 2. **运行阶段**：处理文件操作，管理资源分配
77/// 3. **关闭阶段**：同步缓存，确保数据持久化
78pub struct EasyFileSystem {
79    pub block_device: Arc<dyn BlockDevice>,
80    pub inode_bitmap: Bitmap,
81    pub data_bitmap: Bitmap,
82    inode_area_start_block: u32,
83    data_area_start_block: u32,
84}
85
86impl EasyFileSystem {
87    /// 创建新的文件系统
88    ///
89    /// 在指定的块设备上创建并格式化一个新的 Easy File System。
90    /// 该操作会清空块设备上的所有数据，并初始化文件系统结构。
91    ///
92    /// ## Arguments
93    /// * `block_device` - 要格式化的块设备
94    /// * `total_blocks` - 文件系统总块数
95    /// * `inode_bitmap_blocks` - inode 位图占用的块数
96    ///
97    /// ## 创建过程
98    /// 1. 计算各个区域的大小和位置
99    /// 2. 初始化位图结构
100    /// 3. 清空所有数据块
101    /// 4. 创建超级块
102    /// 5. 创建根目录 inode
103    /// 6. 同步所有缓存到磁盘
104    ///
105    /// ## Returns
106    /// 新创建的文件系统实例，包装在 `Arc<Mutex<>>` 中以支持并发访问
107    ///
108    /// ## 注意事项
109    /// - 此操作会清空块设备上的所有现有数据
110    /// - 创建完成后会自动分配根目录 inode（ID 为 0）
111    /// - 文件系统创建后立即可用
112    pub fn create(
113        block_device: Arc<dyn BlockDevice>,
114        total_blocks: u32,
115        inode_bitmap_blocks: u32,
116    ) -> Arc<Mutex<Self>> {
117        let inode_bitmap = Bitmap::new(1, inode_bitmap_blocks as usize);
118        let inode_num = inode_bitmap.maximum();
119        let inode_area_blocks =
120            ((inode_num * core::mem::size_of::<DiskInode>() + BLOCK_SZ - 1) / BLOCK_SZ) as u32;
121        let inode_total_blocks = inode_bitmap_blocks + inode_area_blocks;
122        let data_total_blocks = total_blocks - 1 - inode_total_blocks;
123        let data_bitmap_blocks = (data_total_blocks + 4096) / 4097;
124        let data_area_blocks = data_total_blocks - data_bitmap_blocks;
125        let data_bitmap = Bitmap::new(
126            (1 + inode_total_blocks) as usize,
127            data_bitmap_blocks as usize,
128        );
129        let mut efs = Self {
130            block_device: Arc::clone(&block_device),
131            inode_bitmap,
132            data_bitmap,
133            inode_area_start_block: 1 + inode_bitmap_blocks,
134            data_area_start_block: 1 + inode_total_blocks + data_bitmap_blocks,
135        };
136        for i in 0..total_blocks {
137            block_cache(i as usize, Arc::clone(&block_device))
138                .lock()
139                .modify(0, |data_block: &mut DataBlock| {
140                    for byte in data_block.iter_mut() {
141                        *byte = 0;
142                    }
143                });
144        }
145        block_cache(0, Arc::clone(&block_device)).lock().modify(
146            0,
147            |super_block: &mut SuperBlock| {
148                super_block.initialize(
149                    total_blocks,
150                    inode_bitmap_blocks,
151                    inode_area_blocks,
152                    data_bitmap_blocks,
153                    data_area_blocks,
154                );
155            },
156        );
157        assert_eq!(efs.alloc_inode(), 0);
158        let (root_inode_block_id, root_inode_offset) = efs.disk_inode_pos(0);
159        block_cache(root_inode_block_id as usize, Arc::clone(&block_device))
160            .lock()
161            .modify(root_inode_offset, |disk_inode: &mut DiskInode| {
162                disk_inode.initialize(DiskInodeType::Directory);
163            });
164        block_cache_sync_all();
165        Arc::new(Mutex::new(efs))
166    }

168    /// 打开现有文件系统
169    ///
170    /// 从指定的块设备上加载已存在的 Easy File System。
171    /// 该操作会读取超级块信息，并恢复文件系统的状态。
172    ///
173    /// ## Arguments
174    /// * `block_device` - 包含文件系统的块设备
175    ///
176    /// ## 加载过程
177    /// 1. 读取块设备第 0 块的超级块
178    /// 2. 验证文件系统魔数
179    /// 3. 根据超级块信息重建位图结构
180    /// 4. 计算各个区域的位置
181    ///
182    /// ## Returns
183    /// 加载的文件系统实例，包装在 `Arc<Mutex<>>` 中以支持并发访问
184    ///
185    /// ## Panics
186    /// 如果块设备不包含有效的 Easy File System 则 panic
187    ///
188    /// ## 注意事项
189    /// - 块设备必须包含有效的文件系统
190    /// - 加载过程不会修改块设备上的数据
191    /// - 文件系统加载后立即可用
192    pub fn open(block_device: Arc<dyn BlockDevice>) -> Arc<Mutex<Self>> {
193        block_cache(0, Arc::clone(&block_device))
194            .lock()
195            .read(0, |super_block: &SuperBlock| {
196                assert!(super_block.valid(), "Error loading EFS!");
197                let inode_total_blocks =
198                    super_block.inode_bitmap_blocks + super_block.inode_area_blocks;
199                let efs = Self {
200                    block_device,
201                    inode_bitmap: Bitmap::new(1, super_block.inode_bitmap_blocks as usize),
202                    data_bitmap: Bitmap::new(
203                        (1 + inode_total_blocks) as usize,
204                        super_block.data_bitmap_blocks as usize,
205                    ),
206                    inode_area_start_block: 1 + super_block.inode_bitmap_blocks,
207                    data_area_start_block: 1 + inode_total_blocks + super_block.data_bitmap_blocks,
208                };
209                Arc::new(Mutex::new(efs))
210            })
211    }

213    /// 获取指定 inode 在磁盘上的位置
214    ///
215    /// 根据 inode ID 计算对应的磁盘 inode 在块设备上的具体位置。
216    /// 返回块号和块内偏移量。
217    ///
218    /// ## Arguments
219    /// * `inode_id` - inode 的 ID
220    ///
221    /// ## Returns
222    /// `(block_id, offset)` 元组，表示 inode 所在的块号和块内偏移量
223    ///
224    /// ## 计算原理
225    /// - 每个块可以存储多个 inode（取决于 `DiskInode` 的大小）
226    /// - 块号 = inode 区域起始块号 + inode_id / 每块 inode 数量
227    /// - 偏移量 = (inode_id % 每块 inode 数量) × inode 大小
228    pub fn disk_inode_pos(&self, inode_id: u32) -> (u32, usize) {
229        let inode_size = core::mem::size_of::<DiskInode>();
230        let inodes_per_block = (BLOCK_SZ / inode_size) as u32;
231        let block_id = self.inode_area_start_block + inode_id / inodes_per_block;
232        (
233            block_id,
234            (inode_id % inodes_per_block) as usize * inode_size,
235        )
236    }

238    /// 获取数据块在磁盘上的实际块号
239    ///
240    /// 将逻辑数据块 ID 转换为在块设备上的实际块号。
241    /// 数据块 ID 是相对于数据区域的，需要加上数据区域的起始块号。
242    ///
243    /// ## Arguments
244    /// * `data_block_id` - 逻辑数据块 ID
245    ///
246    /// ## Returns
247    /// 数据块在块设备上的实际块号
248    pub fn data_block_id(&self, data_block_id: u32) -> u32 {
249        self.data_area_start_block + data_block_id
250    }

252    /// 分配一个新的 inode
253    ///
254    /// 从 inode 位图中分配一个可用的 inode ID。
255    /// 该操作会更新位图并返回新分配的 inode ID。
256    ///
257    /// ## Returns
258    /// 新分配的 inode ID
259    ///
260    /// ## Panics
261    /// 如果没有可用的 inode 则 panic
262    ///
263    /// ## 注意事项
264    /// - 分配的 inode 需要后续初始化才能使用
265    /// - inode ID 从 0 开始连续分配
266    /// - 根目录的 inode ID 固定为 0
267    pub fn alloc_inode(&mut self) -> u32 {
268        self.inode_bitmap.alloc(&self.block_device).unwrap() as u32
269    }

271    /// 分配一个新的数据块
272    ///
273    /// 从数据位图中分配一个可用的数据块，并返回其在块设备上的实际块号。
274    /// 该操作会更新位图并返回新分配的数据块号。
275    ///
276    /// ## Returns
277    /// 新分配的数据块在块设备上的实际块号
278    ///
279    /// ## Panics
280    /// 如果没有可用的数据块则 panic
281    ///
282    /// ## 注意事项
283    /// - 返回的是绝对块号，可以直接用于块设备操作
284    /// - 分配的数据块内容未初始化
285    /// - 数据块分配后需要手动初始化内容
286    pub fn alloc_data(&mut self) -> u32 {
287        self.data_bitmap.alloc(&self.block_device).unwrap() as u32 + self.data_area_start_block
288    }

326    /// 释放指定的数据块
327    ///
328    /// 将指定的数据块标记为可用，并清空其内容。
329    /// 该操作会更新位图并重置数据块内容。
330    ///
331    /// ## Arguments
332    /// * `block_id` - 要释放的数据块在块设备上的实际块号
333    ///
334    /// ## 操作过程
335    /// 1. 清空数据块的所有字节为 0
336    /// 2. 更新数据位图，标记块为可用
337    /// 3. 将绝对块号转换为相对块号进行位图操作
338    ///
339    /// ## 注意事项
340    /// - `block_id` 必须是绝对块号（相对于整个块设备）
341    /// - 释放操作会清空数据块内容
342    /// - 释放后的数据块可以重新分配使用
343    pub fn dealloc_data(&mut self, block_id: u32) {
344        block_cache(block_id as usize, Arc::clone(&self.block_device))
345            .lock()
346            .modify(0, |data_block: &mut DataBlock| {
347                data_block.iter_mut().for_each(|p| {
348                    *p = 0;
349                })
350            });
351        self.data_bitmap.dealloc(
352            &self.block_device,
353            (block_id - self.data_area_start_block) as usize,
354        )
355    }

357    /// 获取文件系统的根目录 inode
358    ///
359    /// 创建并返回根目录的 inode 对象，提供对根目录的访问接口。
360    /// 根目录是文件系统的入口点，所有其他文件和目录都从根目录开始访问。
361    ///
362    /// ## Arguments
363    /// * `efs` - 文件系统实例的引用
364    ///
365    /// ## Returns
366    /// 根目录的 inode 对象
367    ///
368    /// ## 根目录特性
369    /// - 根目录的 inode ID 固定为 0
370    /// - 根目录在文件系统创建时自动初始化
371    /// - 根目录始终存在且可访问
372    /// - 根目录支持子文件和子目录的创建
373    ///
374    /// ## 使用示例
375    /// ```rust
376    /// let root = EasyFileSystem::root_inode(&efs);
377    /// let file = root.create("test.txt").unwrap();
378    /// ```
379    pub fn root_inode(efs: &Arc<Mutex<Self>>) -> Inode {
380        let block_device = Arc::clone(&efs.lock().block_device);
381        let (block_id, block_offset) = efs.lock().disk_inode_pos(0);
382        Inode::new(block_id, block_offset, Arc::clone(efs), block_device)
383    }
