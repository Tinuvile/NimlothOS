1//! # 简化日志消息格式
2//!
3//! 定义微内核日志系统中使用的简单消息格式。
4
5use crate::log::{MAX_MESSAGE_LEN, MAX_MODULE_LEN};
6use alloc::string::{String, ToString};
7use core::fmt;
8
9/// 日志级别定义（与原系统保持一致）
10#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
11#[repr(u8)]
12pub enum LogLevel {
13    Error = 1,
14    Warn = 2,
15    Info = 3,
16    Debug = 4,
17    Trace = 5,
18}
19
20/// 简化的日志消息结构
21///
22/// 消息格式：
23/// ```text
24/// ┌─────────┬─────────┬─────────┬─────────┬─────────────┬─────────────┐
25/// │ Level   │ PID     │ CPU     │ Time    │   Module    │   Message   │
26/// │ (1B)    │ (4B)    │ (1B)    │ (4B)    │  Variable   │  Variable   │
27/// └─────────┴─────────┴─────────┴─────────┴─────────────┴─────────────┘
28/// ```
29#[derive(Debug, Clone)]
30pub struct LogMessage {
31    /// 日志级别
32    pub level: LogLevel,
33    /// 进程ID
34    pub pid: u32,
35    /// CPU ID
36    pub cpu_id: u8,
37    /// 时间戳
38    pub timestamp: u32,
39    /// 模块名
40    pub module: String,
41    /// 消息内容
42    pub message: String,
43}
44
45impl LogLevel {
46    /// 获取日志级别名称
47    pub fn as_str(&self) -> &'static str {
48        match self {
49            LogLevel::Error => "ERROR",
50            LogLevel::Warn => "WARN",
51            LogLevel::Info => "INFO",
52            LogLevel::Debug => "DEBUG",
53            LogLevel::Trace => "TRACE",
54        }
55    }
56
57    /// 获取日志级别对应的ANSI颜色代码
58    pub fn color_code(&self) -> u8 {
59        match self {
60            LogLevel::Error => 31, // 红色
61            LogLevel::Warn => 93,  // 亮黄色
62            LogLevel::Info => 34,  // 蓝色
63            LogLevel::Debug => 32, // 绿色
64            LogLevel::Trace => 90, // 暗灰色
65        }
66    }
67}
68
69impl LogMessage {
70    /// 创建新的日志消息
71    pub fn new(
72        level: LogLevel,
73        pid: u32,
74        cpu_id: u8,
75        timestamp: u32,
76        module: &str,
77        message: &str,
78    ) -> Option<Self> {
79        // 验证长度限制
80        if module.len() > MAX_MODULE_LEN || message.len() > MAX_MESSAGE_LEN {
81            return None;
82        }
83
84        Some(Self {
85            level,
86            pid,
87            cpu_id,
88            timestamp,
89            module: module.to_string(),
90            message: message.to_string(),
91        })
92    }
93
94    /// 序列化为字节流（简单版本）
95    pub fn serialize(&self) -> alloc::vec::Vec<u8> {
96        let mut buffer = alloc::vec::Vec::new();
97
98        // 写入基本信息
99        buffer.push(self.level as u8);
100        buffer.extend_from_slice(&self.pid.to_le_bytes());
101        buffer.push(self.cpu_id);
102        buffer.extend_from_slice(&self.timestamp.to_le_bytes());
103
104        // 写入长度信息
105        buffer.push(self.module.len() as u8);
106        buffer.extend_from_slice(&(self.message.len() as u16).to_le_bytes());
107
108        // 写入字符串内容
109        buffer.extend_from_slice(self.module.as_bytes());
110        buffer.extend_from_slice(self.message.as_bytes());
111
112        buffer
113    }
114
115    /// 从字节流反序列化
116    pub fn deserialize(data: &[u8]) -> Option<Self> {
117        if data.len() < 12 {
118            // 最小头部大小
119            return None;
120        }
121
122        let mut offset = 0;
123
124        // 读取基本信息
125        let level = match data[offset] {
126            1 => LogLevel::Error,
127            2 => LogLevel::Warn,
128            3 => LogLevel::Info,
129            4 => LogLevel::Debug,
130            5 => LogLevel::Trace,
131            _ => return None,
132        };
133        offset += 1;
134
135        let pid = u32::from_le_bytes([
136            data[offset],
137            data[offset + 1],
138            data[offset + 2],
139            data[offset + 3],
140        ]);
141        offset += 4;
142
143        let cpu_id = data[offset];
144        offset += 1;
145
146        let timestamp = u32::from_le_bytes([
147            data[offset],
148            data[offset + 1],
149            data[offset + 2],
150            data[offset + 3],
151        ]);
152        offset += 4;
153
154        // 读取长度信息
155        if offset + 3 > data.len() {
156            return None;
157        }
158
159        let module_len = data[offset] as usize;
160        offset += 1;
161
162        let message_len = u16::from_le_bytes([data[offset], data[offset + 1]]) as usize;
163        offset += 2;
164
165        // 读取字符串内容
166        if offset + module_len + message_len > data.len() {
167            return None;
168        }
169
170        let module = String::from_utf8_lossy(&data[offset..offset + module_len]).to_string();
171        offset += module_len;
172
173        let message = String::from_utf8_lossy(&data[offset..offset + message_len]).to_string();
174
175        Some(Self {
176            level,
177            pid,
178            cpu_id,
179            timestamp,
180            module,
181            message,
182        })
183    }
184
185    /// 格式化输出（与原系统保持一致的格式）
186    pub fn format(&self) -> String {
187        alloc::format!(
188            "\u{1B}[{}m{:>5} [T{:>4}] [CPU{}] [{}] {}\u{1B}[0m",
189            self.level.color_code(),
190            self.level.as_str(),
191            self.timestamp,
192            self.cpu_id,
193            self.module,
194            self.message
195        )
196    }
197}
198
199impl fmt::Display for LogLevel {
200    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
201        write!(f, "{}", self.as_str())
202    }
203}
204
205impl fmt::Display for LogMessage {
206    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
207        write!(f, "{}", self.format())
208    }
209}
